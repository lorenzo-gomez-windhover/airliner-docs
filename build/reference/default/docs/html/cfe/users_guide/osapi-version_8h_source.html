36 /*
37  * Note about the OS_REVISION value:
38  * During development of the "next" version of OSAL (whatever number it might be), the
39  * OS_REVISION value should be something high (>90) to indicate the fact that this is a development
40  * version and not an official release version. For instance, at the time of this writing
41  * the official version is 4.1.1 and the next version would likely be 4.2.0. So anyone
42  * who uses the "bleeding edge" development branch before 4.2.0 is officially released will
43  * get a version that reads e.g. "4.1.91" to show that this is a pre-4.2 build.
44  *
45  * This still leaves room for patches to be applied to 4.1.1 to produce 4.1.2 if needed.
46  *
47  * During development, if an API-affecting change is introduced then the OS_REVISION value should
48  * be incremented with it to indicate this. Client code that depends on the new API can do a "#if"
49  * against the development version number e.g. "#if OSAL_API_VERSION >= 40191" and this check
50  * will still be valid after the version becomes 40200 (although at some point it can be cleaned
51  * up to reflect the official version number, it is not going to break the build in the meantime).
52  */
53 
54 #define OS_MAJOR_VERSION 4
55 #define OS_MINOR_VERSION 2
56 #define OS_REVISION 1
57 #define OS_MISSION_REV 0

63 #define OSAL_API_VERSION ((OS_MAJOR_VERSION * 10000) + (OS_MINOR_VERSION * 100) + OS_REVISION)
