78 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
79 int32 CFE_ES_ShellOutputCommand(const char * CmdString, const char *Filename)
80 {
81  int32 Result;
82  int32 ReturnCode = CFE_SUCCESS;
83  int32 fd;
84  int32 FileSize;
85  int32 CurrFilePtr;
86  uint32 i;
87 
88  /* the extra 1 added for the \0 char */
89  char CheckCmd [CFE_ES_CHECKSIZE + 1];
90  char Cmd [CFE_ES_MAX_SHELL_CMD];
91  char OutputFilename [OS_MAX_PATH_LEN];
92 
93  /* Use default filename if not provided */
94  if (Filename[0] == '\0')
95  {
96  (void) CFE_SB_MessageStringGet(OutputFilename, CFE_ES_DEFAULT_SHELL_FILENAME, NULL,
97  sizeof(OutputFilename), sizeof(CFE_ES_DEFAULT_SHELL_FILENAME));
98  }
99  else
100  {
101  (void) CFE_SB_MessageStringGet(OutputFilename, Filename, NULL, sizeof(OutputFilename), strlen(Filename));
102  }
103 
104  /* Make sure string is null terminated */
105  OutputFilename[OS_MAX_PATH_LEN - 1] = '\0';
106 
107  /* Remove previous version of output file */
108  OS_remove(OutputFilename);
109 
110  fd = OS_creat(OutputFilename, OS_READ_WRITE);
111 
112  if (fd < OS_FS_SUCCESS)
113  {
114  Result = OS_FS_ERROR;
115  }
116 
117  else
118  {
119  (void) CFE_SB_MessageStringGet(CheckCmd, CmdString, NULL, sizeof(CheckCmd), CFE_ES_MAX_SHELL_CMD);
120 
121  CheckCmd[CFE_ES_CHECKSIZE] = '\0';
122 
123  (void) CFE_SB_MessageStringGet(Cmd, CmdString, NULL, sizeof(Cmd), CFE_ES_MAX_SHELL_CMD);
124 
125  /* We need to check if this command is directed at ES, or at the
126  operating system */
127 
128  if (strncmp(CheckCmd,"ES_",CFE_ES_CHECKSIZE) == 0)
129  {
130  /* This list can be expanded to include other ES functionality */
131  if ( strncmp(Cmd,CFE_ES_LIST_APPS_CMD,strlen(CFE_ES_LIST_APPS_CMD) )== 0)
132  {
133  Result = CFE_ES_ListApplications(fd);
134  }
135  else if ( strncmp(Cmd,CFE_ES_LIST_TASKS_CMD,strlen(CFE_ES_LIST_TASKS_CMD) )== 0)
136  {
137  Result = CFE_ES_ListTasks(fd);
138  }
139  else if ( strncmp(Cmd,CFE_ES_LIST_RESOURCES_CMD,strlen(CFE_ES_LIST_RESOURCES_CMD) )== 0)
140  {
141  Result = CFE_ES_ListResources(fd);
142  }
143 
144  /* default if there is not an ES command that matches */
145  else
146  {
147  Result = CFE_ES_ERR_SHELL_CMD;
148  CFE_ES_WriteToSysLog("There is no ES Shell command that matches %s \n",Cmd);
149  }
150 
151  }
152  /* if the command is not directed at ES, pass it through to the
153  * underlying OS */
154  else
155  {
156  Result = OS_ShellOutputToFile(Cmd,fd);
157  }
158 
159  /* seek to the end of the file to get it's size */
160  FileSize = OS_lseek(fd,0,OS_SEEK_END);
161 
162  if (FileSize == OS_FS_ERROR)
163  {
164  OS_close(fd);
165  CFE_ES_WriteToSysLog("OS_lseek call failed from CFE_ES_ShellOutputCmd 1\n");
166  Result = OS_FS_ERROR;
167  }
168 
169 
170 
171  /* We want to add 3 characters at the end of the telemetry,'\n','$','\0'.
172  * To do this we need to make sure there are at least 3 empty characters at
173  * the end of the last CFE_ES_MAX_SHELL_PKT so we don't over write any data. If
174  * the current file has only 0,1, or 2 free spaces at the end, we want to
175  * make the file longer to start a new tlm packet of size CFE_ES_MAX_SHELL_PKT.
176  * This way we will get a 'blank' packet with the correct 3 characters at the end.
177  */
178 
179  else
180  {
181  /* if we are within 2 bytes of the end of the packet*/
182  if ( FileSize % CFE_ES_MAX_SHELL_PKT > (CFE_ES_MAX_SHELL_PKT - 3))
183  {
184  /* add enough bytes to start a new packet */
185  for (i = 0; i < CFE_ES_MAX_SHELL_PKT - (FileSize % CFE_ES_MAX_SHELL_PKT) + 1 ; i++)
186  {
187  OS_write(fd," ",1);
188  }
189  }
190  else
191  {
192  /* we are exactly at the end */
193  if( FileSize % CFE_ES_MAX_SHELL_PKT == 0)
194  {
195  OS_write(fd," ",1);
196  }
197  }
198 
199  /* seek to the end of the file again to get it's new size */
200  FileSize = OS_lseek(fd,0,OS_SEEK_END);
201 
202  if (FileSize == OS_FS_ERROR)
203  {
204  OS_close(fd);
205  CFE_ES_WriteToSysLog("OS_lseek call failed from CFE_ES_ShellOutputCmd 2\n");
206  Result = OS_FS_ERROR;
207  }
208 
209 
210  else
211  {
212  /* set the file back to the beginning */
213  OS_lseek(fd,0,OS_SEEK_SET);
214 
215 
216  /* start processing the chunks. We want to have one packet left so we are sure this for loop
217  * won't run over */
218 
219  for (CurrFilePtr=0; CurrFilePtr < (FileSize - CFE_ES_MAX_SHELL_PKT); CurrFilePtr += CFE_ES_MAX_SHELL_PKT)
220  {
221  OS_read(fd, CFE_ES_TaskData.ShellPacket.Payload.ShellOutput, CFE_ES_MAX_SHELL_PKT);
222 
223  /* Send the packet */
224  CFE_SB_TimeStampMsg((CFE_SB_Msg_t *) &CFE_ES_TaskData.ShellPacket);
225  CFE_SB_SendMsg((CFE_SB_Msg_t *) &CFE_ES_TaskData.ShellPacket);
226  /* delay to not flood the pipe on large messages */
227  OS_TaskDelay(200);
228  }
229 
230  /* finish off the last portion of the file */
231  /* over write the last packet with spaces, then it will get filled
232  * in with the correct info below. This assures that the last non full
233  * part of the packet will be spaces */
234  for (i =0; i < CFE_ES_MAX_SHELL_PKT; i++)
235  {
236  CFE_ES_TaskData.ShellPacket.Payload.ShellOutput[i] = ' ';
237  }
238 
239  OS_read(fd, CFE_ES_TaskData.ShellPacket.Payload.ShellOutput, ( FileSize - CurrFilePtr));
240 
241  /* From our check above, we are assured that there are at least 3 free
242  * characters to write our data into at the end of this last packet
243  *
244  * The \n assures we are on a new line, the $ gives us our prompt, and the
245  * \0 assures we are null terminalted.
246  */
247 
248 
249  CFE_ES_TaskData.ShellPacket.Payload.ShellOutput[ CFE_ES_MAX_SHELL_PKT - 3] = '\n';
250  CFE_ES_TaskData.ShellPacket.Payload.ShellOutput[ CFE_ES_MAX_SHELL_PKT - 2] = '$';
251  CFE_ES_TaskData.ShellPacket.Payload.ShellOutput[ CFE_ES_MAX_SHELL_PKT - 1] = '\0';
252 
253  /* Send the last packet */
254  CFE_SB_TimeStampMsg((CFE_SB_Msg_t *) &CFE_ES_TaskData.ShellPacket);
255  CFE_SB_SendMsg((CFE_SB_Msg_t *) &CFE_ES_TaskData.ShellPacket);
256 
257  /* Close the file descriptor */
258  OS_close(fd);
259  } /* if FilseSize == OS_FS_ERROR */
260  } /* if FileSeize == OS_FS_ERROR */
261  }/* if fd < OS_FS_SUCCESS */
262 
263 
264  /* cppcheck-suppress duplicateExpression */
265  if (Result != OS_SUCCESS && Result != CFE_SUCCESS )
266  {
267  ReturnCode = CFE_ES_ERR_SHELL_CMD;
268  CFE_ES_WriteToSysLog("OS_ShellOutputToFile call failed from CFE_ES_ShellOutputCommand\n");
269  }
270  else
271  {
272  ReturnCode = CFE_SUCCESS;
273  }
274 
275  return ReturnCode;
276 }
