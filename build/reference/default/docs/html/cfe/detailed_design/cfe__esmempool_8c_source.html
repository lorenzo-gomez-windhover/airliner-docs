147 int32 CFE_ES_PoolCreateEx(CFE_ES_MemHandle_t *HandlePtr,
148  uint8 *MemPtr,
149  uint32 Size,
150  uint32 NumBlockSizes,
151  uint32 *BlockSizes,
152  uint16 UseMutex )
153 {
154  char MutexName[OS_MAX_API_NAME];
155  cpuaddr MemAddr;
156  uint32 i;
157  uint32 j;
158  uint32 k;
159  uint32 *BlockSizeArrayPtr;
160  uint32 BlockSizeArraySize;
161  uint32 MinBlockSize;
162 
163  /*
164  ** Local Variables
165  */
166  Pool_t *PoolPtr = (Pool_t *)MemPtr;
167  MemAddr = (cpuaddr)MemPtr;
168  *HandlePtr = MemAddr;
169 
170  /* Force the size given to be 32 bit aligned */
171  Size &= 0xFFFFFFFC;
172 
173 #ifdef CFE_ES_MEMPOOL_ALIGNED
174  /* Determine if the memory pool address is 32-bit aligned */
175  if ((MemAddr & 0x00000003) != 0)
176  {
177  CFE_ES_WriteToSysLog("CFE_ES:poolCreate Pool Address(0x%08lX) is not 32-bit aligned.\n", (unsigned long)MemAddr);
178  return(CFE_ES_BAD_ARGUMENT);
179  }
180 #endif
181 
182  /* If too many sizes are specified, return an error */
183  if (NumBlockSizes > CFE_ES_MAX_MEMPOOL_BLOCK_SIZES)
184  {
185  CFE_ES_WriteToSysLog("CFE_ES:poolCreate Num Block Sizes (%d) greater than max (%d)\n",
186  (int)NumBlockSizes, CFE_ES_MAX_MEMPOOL_BLOCK_SIZES);
187  return(CFE_ES_BAD_ARGUMENT);
188  }
189 
190  if ((UseMutex != CFE_ES_USE_MUTEX) && (UseMutex != CFE_ES_NO_MUTEX))
191  {
192  CFE_ES_WriteToSysLog("CFE_ES:poolCreate Invalid Mutex Usage Option (%d), must be %d or %d\n",
193  UseMutex, CFE_ES_NO_MUTEX, CFE_ES_USE_MUTEX);
194  return(CFE_ES_BAD_ARGUMENT);
195  }
196 
197  if (UseMutex == CFE_ES_USE_MUTEX)
198  {
199  /*
200  ** Construct a name for the Mutex from the address
201  ** This is needed only because OS_MutSemCreate requires
202  ** a unique name for each semaphore created.
203  */
204  snprintf(MutexName, OS_MAX_API_NAME, "%08lX", (unsigned long)MemPtr);
205 
206  /* create a semphore to protect this memory pool */
207  OS_MutSemCreate(&(PoolPtr->MutexId), MutexName, 0);
208 
209  /* Take the semaphore to ensure the mem pool is not being used during its creation */
210  OS_MutSemTake(PoolPtr->MutexId);
211  }
212 
213  PoolPtr->Start = (cpuaddr *)*HandlePtr;
214  PoolPtr->End = MemAddr + Size;
215  PoolPtr->Size = Size;
216  PoolPtr->Current = (uint32 *)(MemAddr + sizeof(Pool_t));
217  PoolPtr->SizeDescPtr = NULL;
218 
219  PoolPtr->CheckErrCntr = 0;
220  PoolPtr->RequestCntr = 0;
221 
222  PoolPtr->UseMutex = UseMutex;
223 
224  for (i=0; i<CFE_ES_MAX_MEMPOOL_BLOCK_SIZES; i++)
225  {
226  PoolPtr->SizeDesc[i].NumCreated = 0;
227  PoolPtr->SizeDesc[i].NumFree = 0;
228  PoolPtr->SizeDesc[i].MaxSize = 0;
229  PoolPtr->SizeDesc[i].Top = NULL;
230  }
231 
232  /* Use default block sizes if none or too many sizes are specified */
233  if ((NumBlockSizes == 0) || (BlockSizes == NULL))
234  {
235  BlockSizeArrayPtr = &CFE_ES_MemPoolDefSize[0];
236  BlockSizeArraySize = CFE_ES_MAX_MEMPOOL_BLOCK_SIZES;
237  }
238  else
239  {
240  BlockSizeArrayPtr = BlockSizes;
241  BlockSizeArraySize = NumBlockSizes;
242  }
243 
244  /* Use specified block sizes but make sure they are ordered largest to smallest */
245  MinBlockSize = 0xffffffff;
246  for (i=0; i<BlockSizeArraySize; i++)
247  {
248  if ((BlockSizeArrayPtr[i] < MinBlockSize) && (BlockSizeArrayPtr[i] != 0))
249  {
250  MinBlockSize = BlockSizeArrayPtr[i];
251  }
252 
253  j = 0;
254  while (j<CFE_ES_MAX_MEMPOOL_BLOCK_SIZES)
255  {
256  if (BlockSizeArrayPtr[i] > PoolPtr->SizeDesc[j].MaxSize)
257  {
258  /* Make space for new size */
259  for (k=i; k>j; k--)
260  {
261  PoolPtr->SizeDesc[k].MaxSize = PoolPtr->SizeDesc[k-1].MaxSize;
262  }
263 
264  /* Insert the size in the correct location */
265  PoolPtr->SizeDesc[j].MaxSize = BlockSizeArrayPtr[i];
266  j = CFE_ES_MAX_MEMPOOL_BLOCK_SIZES;
267  }
268  else
269  {
270  j++;
271  }
272  }
273  }
274 
275  if (Size <= (sizeof(Pool_t) + MinBlockSize + sizeof(BD_t) ) )
276  { /* must be able make Pool management structure and at least one of the smallest blocks */
277  CFE_ES_WriteToSysLog("CFE_ES:poolCreate Pool size(%d) too small for management structure, need >=(%d)\n",
278  (int)Size,
279  (int)(sizeof(Pool_t) + MinBlockSize + sizeof(BD_t)) );
280 
281  if (UseMutex == CFE_ES_USE_MUTEX)
282  {
283  /* Mutex is no longer needed with error in initialization. Give it and delete it */
284  OS_MutSemGive(PoolPtr->MutexId);
285  OS_MutSemDelete(PoolPtr->MutexId);
286  }
287  return(CFE_ES_BAD_ARGUMENT);
288  }
289 
290  if (UseMutex == CFE_ES_USE_MUTEX)
291  {
292  OS_MutSemGive(PoolPtr->MutexId);
293  }
294 
295  return(CFE_SUCCESS);
296 }

298 /*
299 ** Function:
300 ** CFE_ES_GetPoolBuf
301 **
302 ** Purpose:
303 ** CFE_ES_GetPoolBuf allocates a block from the memory pool.
304 */
305 int32 CFE_ES_GetPoolBuf(uint32 **BufPtr,
306  CFE_ES_MemHandle_t Handle,
307  uint32 Size )
308 {
309  Pool_t * PoolPtr = (Pool_t *)Handle;
310  uint32 Block;
311  BD_t * BdPtr;
312  uint32 AppId= 0xFFFFFFFF;
313 
314  if (PoolPtr != NULL)
315  {
316  if (Handle != (CFE_ES_MemHandle_t)PoolPtr->Start)
317  {
318  CFE_ES_GetAppID(&AppId);
319  CFE_ES_WriteToSysLog("CFE_ES:getPoolBuf err:Bad handle(0x%08lX) AppId=%d\n",(unsigned long)Handle,(int)AppId);
320  return(CFE_ES_ERR_MEM_HANDLE);
321  }
322  }
323  else
324  {
325  CFE_ES_GetAppID(&AppId);
326  CFE_ES_WriteToSysLog("CFE_ES:getPoolBuf err:Bad handle(0x%08lX) AppId=%d\n",(unsigned long)Handle,(int)AppId);
327  return(CFE_ES_ERR_MEM_HANDLE);
328  }
329 
330 
331  if (PoolPtr->UseMutex == CFE_ES_USE_MUTEX)
332  {
333  OS_MutSemTake(PoolPtr->MutexId);
334  }
335 
336  *BufPtr = NULL;
337 
338  Block = CFE_ES_GetBlockSize(PoolPtr, Size);
339  if (Block == 0xFFFFFFFF)
340  {
341  CFE_ES_WriteToSysLog("CFE_ES:getPoolBuf err:size(%d) > max(%d).\n",(int)Size,(int)PoolPtr->SizeDesc[0].MaxSize);
342  if (PoolPtr->UseMutex == CFE_ES_USE_MUTEX)
343  {
344  OS_MutSemGive(PoolPtr->MutexId);
345  }
346  return(CFE_ES_ERR_MEM_BLOCK_SIZE);
347  }
348 
349  /*
350  ** Check if any of the requested size are available
351  */
352  if (PoolPtr->SizeDescPtr->Top != NULL) /* Set by GetBlockSize call */
353  {
354  /*
355  ** Get it off the top on the list
356  */
357  BdPtr = PoolPtr->SizeDescPtr->Top;
358 
359  BdPtr->CheckBits = CFE_ES_CHECK_PATTERN;
360  BdPtr->Allocated = CFE_ES_MEMORY_ALLOCATED; /* Flag memory block as allocated */
361  BdPtr->Size = Block;
362  PoolPtr->SizeDescPtr->Top = (BD_t *)BdPtr->Next;
363  PoolPtr->SizeDescPtr->NumFree--;
364  BdPtr->Next = NULL;
365  *BufPtr = (uint32 *)(BdPtr + 1);
366  }
367  else /* go make one */
368  {
369  BdPtr = (BD_t *)PoolPtr->Current; /* point to new memory block */
370  if ( ((cpuaddr)BdPtr + sizeof(BD_t) + Block ) >= PoolPtr->End ){ /* can't fit in remaing mem */
371  CFE_ES_WriteToSysLog("CFE_ES:getPoolBuf err:Request won't fit in remaining memory\n");
372  if (PoolPtr->UseMutex == CFE_ES_USE_MUTEX)
373  {
374  OS_MutSemGive(PoolPtr->MutexId);
375  }
376  return(CFE_ES_ERR_MEM_BLOCK_SIZE);
377  }
378 
379  PoolPtr->SizeDescPtr->NumCreated++;
380  PoolPtr->RequestCntr++;
381 
382  /*
383  ** create the buffer descriptor at the front of it
384  */
385  BdPtr->CheckBits = CFE_ES_CHECK_PATTERN;
386  BdPtr->Allocated = CFE_ES_MEMORY_ALLOCATED; /* Flag memory block as allocated */
387  BdPtr->Size = Block;
388  BdPtr->Next = NULL;
389  *BufPtr = (uint32 *)(BdPtr + 1);
390 
391  /*
392  ** adjust pool current pointer
393  */
394  PoolPtr->Current = (uint32 *)( (uint8 *)PoolPtr->Current
395  + Block
396  + sizeof(BD_t) );
397  }
398 
399  if (PoolPtr->UseMutex == CFE_ES_USE_MUTEX)
400  {
401  OS_MutSemGive(PoolPtr->MutexId);
402  }
403  return (int32)Block;
404 }

407 ** CFE_ES_GetPoolBufInfo gets the size of the specified block (if it exists).
408 */
409 int32 CFE_ES_GetPoolBufInfo(CFE_ES_MemHandle_t Handle,
410  uint32 * BufPtr)
411 {
412  Pool_t *PoolPtr = (Pool_t *)Handle;
413  BD_t *BdPtr = (BD_t *) ((uint8 *)BufPtr - sizeof(BD_t));
414 
415  if (PoolPtr != NULL)
416  {
417  if ( ((cpuaddr)BdPtr < Handle) || ((cpuaddr)BdPtr >= (PoolPtr->End - sizeof(BD_t))) )
418  {
419  /* sanity check */
420  return(CFE_ES_BUFFER_NOT_IN_POOL);
421  }
422  }
423  else
424  {
425  /* bad handle */
426  return(CFE_ES_ERR_MEM_HANDLE);
427  }
428 
429  if (PoolPtr->UseMutex == CFE_ES_USE_MUTEX)
430  {
431  OS_MutSemTake(PoolPtr->MutexId);
432  }
433 
434  /*
435  ** Simple sanity checks for descriptor
436  */
437 
438  /* If a block is no longer allocated, report an error */
439  if (BdPtr->Allocated != CFE_ES_MEMORY_ALLOCATED)
440  {
441  if (PoolPtr->UseMutex == CFE_ES_USE_MUTEX)
442  {
443  OS_MutSemGive(PoolPtr->MutexId);
444  }
445  return(CFE_ES_ERR_MEM_HANDLE);
446  }
447 
448  if (BdPtr->CheckBits != CFE_ES_CHECK_PATTERN)
449  {
450  if (PoolPtr->UseMutex == CFE_ES_USE_MUTEX)
451  {
452  OS_MutSemGive(PoolPtr->MutexId);
453  }
454  return(CFE_ES_ERR_MEM_HANDLE);
455  }
456 
457 
458  if (PoolPtr->UseMutex == CFE_ES_USE_MUTEX)
459  {
460  OS_MutSemGive(PoolPtr->MutexId);
461  }
462  return ((int32) (BdPtr->Size));
463 }

466 ** CFE_ES_putPoolBuf returns a block back to the memory pool.
467 */
468 int32 CFE_ES_PutPoolBuf(CFE_ES_MemHandle_t Handle,
469  uint32 * BufPtr)
470 {
471  Pool_t *PoolPtr = (Pool_t *)Handle;
472  BD_t *BdPtr = (BD_t *) ((uint8 *)BufPtr - sizeof(BD_t));
473  uint32 Block;
474  cpuaddr BdAddr;
475 
476  BdAddr = (cpuaddr)BdPtr;
477 
478  if (PoolPtr != NULL)
479  {
480  if ( (BdAddr < Handle) || (BdAddr >= (PoolPtr->End - sizeof(BD_t))) )
481  {
482  /* sanity check */
483  CFE_ES_WriteToSysLog("CFE_ES:putPoolBuf err:Invalid Memory Handle (0x%08lX) or memory block (0x%08lX).\n",
484  (unsigned long) Handle, (unsigned long)BdAddr);
485  return(CFE_ES_ERR_MEM_HANDLE);
486  }
487  }
488  else
489  {
490  /* sanity check */
491  CFE_ES_WriteToSysLog("CFE_ES:putPoolBuf err:Invalid Memory Handle (0x%08lX).\n", (unsigned long) Handle);
492  return(CFE_ES_ERR_MEM_HANDLE);
493  }
494 
495 
496  if (PoolPtr->UseMutex == CFE_ES_USE_MUTEX)
497  {
498  OS_MutSemTake(PoolPtr->MutexId);
499  }
500 
501  /*
502  ** Simple sanity checks for descriptor
503  */
504 
505  /* Don't allow a block that has already been deallocated to be deallocated again */
506  if (BdPtr->Allocated != CFE_ES_MEMORY_ALLOCATED)
507  {
508  PoolPtr->CheckErrCntr++;
509  CFE_ES_WriteToSysLog("CFE_ES:putPoolBuf err:Deallocating unallocated memory block @ 0x%08lX\n", (unsigned long)BdAddr);
510  if (PoolPtr->UseMutex == CFE_ES_USE_MUTEX)
511  {
512  OS_MutSemGive(PoolPtr->MutexId);
513  }
514  return(CFE_ES_ERR_MEM_HANDLE);
515  }
516 
517  if (BdPtr->CheckBits != CFE_ES_CHECK_PATTERN)
518  {
519  PoolPtr->CheckErrCntr++;
520  CFE_ES_WriteToSysLog("CFE_ES:putPoolBuf err:Invalid/Corrupted Memory descriptor @ 0x%08lX\n", (unsigned long)BdAddr);
521  if (PoolPtr->UseMutex == CFE_ES_USE_MUTEX)
522  {
523  OS_MutSemGive(PoolPtr->MutexId);
524  }
525  return(CFE_ES_ERR_MEM_HANDLE);
526  }
527 
528  Block = CFE_ES_GetBlockSize(PoolPtr, BdPtr->Size);
529 
530  if (Block == 0xFFFFFFFF)
531  {
532  PoolPtr->CheckErrCntr++;
533  CFE_ES_WriteToSysLog("CFE_ES:putPoolBuf err:size(%d) > max(%d).\n",(int)BdPtr->Size,(int)PoolPtr->SizeDesc[0].MaxSize);
534  if (PoolPtr->UseMutex == CFE_ES_USE_MUTEX)
535  {
536  OS_MutSemGive(PoolPtr->MutexId);
537  }
538  return(CFE_ES_ERR_MEM_HANDLE);
539  }
540 
541  BdPtr->Allocated = CFE_ES_MEMORY_DEALLOCATED;
542  BdPtr->Next = (uint32 *)PoolPtr->SizeDescPtr->Top; /* Set by GetBlockSize call */
543  PoolPtr->SizeDescPtr->Top = BdPtr;
544  PoolPtr->SizeDescPtr->NumFree++;
545 
546  if (PoolPtr->UseMutex == CFE_ES_USE_MUTEX)
547  {
548  OS_MutSemGive(PoolPtr->MutexId);
549  }
550  return (int32)Block;
551 }

554 ** Function:
555 ** ES_GetBlockSize
556 **
557 ** Purpose:
558 **
559 */
560 uint32 CFE_ES_GetBlockSize(Pool_t *PoolPtr, uint32 Size)
561 {
562  uint32 i=0;
563 
564  PoolPtr->SizeDescPtr = NULL;
565 
566  /* If caller is requesting a size larger than the largest allowed */
567  /* then return an error code */
568  if (Size > PoolPtr->SizeDesc[0].MaxSize)
569  {
570  return(0xFFFFFFFF);
571  }
572 
573  /* Locate the smallest size that holds the desired size */
574  while ((i < (CFE_ES_MAX_MEMPOOL_BLOCK_SIZES-1)) &&
575  (Size <= PoolPtr->SizeDesc[i+1].MaxSize))
576  {
577  i++;
578  }
579 
580  PoolPtr->SizeDescPtr = &PoolPtr->SizeDesc[i];
581  return(PoolPtr->SizeDesc[i].MaxSize);
582 }
