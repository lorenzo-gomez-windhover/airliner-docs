104 /******************************************************************************
105 ** Function: CFE_SB_EarlyInit()
106 **
107 ** Purpose:
108 ** Initialize the Software Bus routing tables.
109 **
110 ** Arguments:
111 **
112 ** Notes:
113 ** This function MUST be called before any SB API's are called.
114 **
115 ** Return:
116 ** CFE_SUCCESS
117 */
118 int32 CFE_SB_EarlyInit (void) {
119 
120  int32 Stat = CFE_SUCCESS;
121 
122  CFE_SB_Default_Qos.Priority = CFE_SB_QOS_LOW_PRIORITY;
123  CFE_SB_Default_Qos.Reliability = CFE_SB_QOS_LOW_RELIABILITY;
124 
125  Stat = OS_MutSemCreate(&CFE_SB.SharedDataMutexId, "CFE_SB_DataMutex", 0);
126  if(Stat != OS_SUCCESS){
127  CFE_ES_WriteToSysLog("SB shared data mutex creation failed! RC=0x%08x\n",(unsigned int)Stat);
128  return Stat;
129  }/* end if */
130 
131  /* Initialize the state of susbcription reporting */
132  CFE_SB.SubscriptionReporting = CFE_SB_DISABLE;
133 
134  /* Initialize the state of sender reporting */
135  CFE_SB.SenderReporting = CFE_SB_DEFAULT_REPORT_SENDER;
136 
137  /* Initialize memory partition. */
138  Stat = CFE_SB_InitBuffers();
139  if(Stat != CFE_SUCCESS){
140  /* error reported in CFE_SB_InitBuffers */
141  return Stat;
142  }/* end if */
143 
144  /* Initialize the pipe table. */
145  CFE_SB_InitPipeTbl();
146 
147  /* Initialize the routing index look up table */
148  CFE_SB_InitMsgMap();
149 
150  /* Initialize the routing table. */
151  CFE_SB_InitRoutingTbl();
152 
153  /* Initialize the SB Statistics Pkt */
154  CFE_SB_InitMsg(&CFE_SB.StatTlmMsg.Hdr.Pri,
155  CFE_SB_STATS_TLM_MID,
156  sizeof(CFE_SB_StatMsg_t),
157  TRUE);
158 
159  CFE_SB.ZeroCopyTail = NULL;
160 
161  return Stat;
162 
163 }/* end CFE_SB_EarlyInit */
