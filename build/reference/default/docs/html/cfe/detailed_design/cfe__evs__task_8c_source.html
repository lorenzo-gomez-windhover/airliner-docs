288 ** Function Prologue
289 **
290 ** Function Name: EVS_TaskMain
291 **
292 ** Purpose: This is the main EVS task process loop.
293 **
294 ** Assumptions and Notes:
295 **
296 */
297 void CFE_EVS_TaskMain(void)
298 {
299  int32 Status;
300  CFE_SB_MsgPtr_t EVS_MsgPtr; /* Pointer to SB message */
301 
302  CFE_ES_PerfLogEntry(CFE_EVS_MAIN_PERF_ID);
303 
304  Status = CFE_EVS_TaskInit();
305 
306  if(Status != CFE_SUCCESS)
307  {
308  CFE_ES_WriteToSysLog("EVS:Application Init Failed,RC=0x%08X\n", (unsigned int)Status);
309  CFE_ES_PerfLogExit(CFE_EVS_MAIN_PERF_ID);
310  /* Note: CFE_ES_ExitApp will not return */
311  CFE_ES_ExitApp(CFE_ES_RUNSTATUS_CORE_APP_INIT_ERROR);
312  }/* end if */
313 
314  /*
315  * Wait for other apps to start.
316  * It is important that the core apps are present before this starts receiving
317  * messages from the command pipe, as some of those handlers might depend on
318  * the other core apps.
319  */
320  CFE_ES_WaitForStartupSync(CFE_CORE_MAX_STARTUP_MSEC);
321 
322  /* Main loop */
323  while (Status == CFE_SUCCESS)
324  {
325  /* Increment the Main task Execution Counter */
326  CFE_ES_IncrementTaskCounter();
327 
328  CFE_ES_PerfLogExit(CFE_EVS_MAIN_PERF_ID);
329 
330  /* Pend on receipt of packet */
331  Status = CFE_SB_RcvMsg(&EVS_MsgPtr,
332  CFE_EVS_GlobalData.EVS_CommandPipe,
333  CFE_SB_PEND_FOREVER);
334 
335  CFE_ES_PerfLogEntry(CFE_EVS_MAIN_PERF_ID);
336 
337  if (Status == CFE_SUCCESS)
338  {
339  /* Process cmd pipe msg */
340  CFE_EVS_ProcessCommandPacket(EVS_MsgPtr);
341  }else{
342  CFE_ES_WriteToSysLog("EVS:Error reading cmd pipe,RC=0x%08X\n",(unsigned int)Status);
343  }/* end if */
344 
345  }/* end while */
346 
347  /* while loop exits only if CFE_SB_RcvMsg returns error */
348  CFE_ES_ExitApp(CFE_ES_RUNSTATUS_CORE_APP_RUNTIME_ERROR);
349 
350 } /* end CFE_EVS_TaskMain */
