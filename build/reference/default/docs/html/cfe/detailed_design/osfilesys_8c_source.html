93 int32 OS_mkfs (char *address, const char *devname, const char * volname, uint32 blocksize,
94  uint32 numblocks)
95 {
96  int i;
97 
98  /*
99  ** Check parameters
100  */
101  if ( devname == NULL || volname == NULL )
102  {
103  return OS_FS_ERR_INVALID_POINTER;
104  }
105 
106  if( strlen(devname) >= OS_FS_DEV_NAME_LEN || strlen(volname) >= OS_FS_VOL_NAME_LEN)
107  {
108  return OS_FS_ERR_PATH_TOO_LONG;
109  }
110 
111  /* find an open entry in the Volume Table */
112  for (i = 0; i < NUM_TABLE_ENTRIES; i++)
113  {
114  if (OS_VolumeTable[i].FreeFlag == TRUE && OS_VolumeTable[i].IsMounted == FALSE
115  && strcmp(OS_VolumeTable[i].DeviceName, devname) == 0)
116  break;
117  }
118 
119  if (i >= NUM_TABLE_ENTRIES)
120  {
121  return OS_FS_ERR_DEVICE_NOT_FREE;
122  }
123 
124  /* make a disk if it is FS based */
125  /*------------------------------- */
126  if (OS_VolumeTable[i].VolumeType == FS_BASED)
127  {
128  /* now enter the info in the table */
129  OS_VolumeTable[i].FreeFlag = FALSE;
130  strcpy(OS_VolumeTable[i].VolumeName, volname);
131  OS_VolumeTable[i].BlockSize = blocksize;
132 
133  }
134  else
135  {
136  return OS_FS_ERR_DRIVE_NOT_CREATED;
137  }
138  return OS_FS_SUCCESS;
139 
140 } /* end OS_mkfs */

153 {
154  int i;
155  int32 ReturnCode;
156 
157  if (devname ==NULL)
158  {
159  ReturnCode = OS_FS_ERR_INVALID_POINTER;
160  }

164  }

511 /*-------------------------------------------------------------------------------------
512  * Name: OS_TranslatePath
513  *
514  * Purpose: Because of the abstraction of the filesystem across OSes, we have to change
515  * the name of the {file, directory, drive} to be what the OS can actually
516  * accept
517 ---------------------------------------------------------------------------------------*/
518 int32 OS_TranslatePath(const char *VirtualPath, char *LocalPath)
519 {
520  char devname [OS_MAX_PATH_LEN];
521  char filename[OS_MAX_PATH_LEN];
522  int NumChars;
523  int DeviceLen;
524  int FilenameLen;
525  int i=0;
526 
527  /*
528  ** Check to see if the path pointers are NULL
529  */
530  if (VirtualPath == NULL)
531  {
532  return OS_FS_ERR_INVALID_POINTER;
533  }
534 
535  if (LocalPath == NULL)
536  {
537  return OS_FS_ERR_INVALID_POINTER;
538  }
539 
540  /*
541  ** Check to see if the path is too long
542  */
543  if (strlen(VirtualPath) >= OS_MAX_PATH_LEN)
544  {
545  return OS_FS_ERR_PATH_TOO_LONG;
546  }
547 
548  /*
549  ** All valid Virtual paths must start with a '/' character
550  */
551  if ( VirtualPath[0] != '/' )
552  {
553  return OS_FS_ERR_PATH_INVALID;
554  }
555 
556  /*
557  ** Fill the file and device name to be sure they do not have garbage
558  */
559  memset((void *)devname,0,OS_MAX_PATH_LEN);
560  memset((void *)filename,0,OS_MAX_PATH_LEN);
561 
562  /*
563  ** We want to find the number of chars to where the second "/" is.
564  ** Since we know the first one is in spot 0, we start looking at 1, and go until
565  ** we find it.
566  */
567  NumChars = 1;
568  while ((NumChars <= strlen(VirtualPath)) && (VirtualPath[NumChars] != '/'))
569  {
570  NumChars++;
571  }
572 
573  /*
574  ** Don't let it overflow to cause a segfault when trying to get the highest level
575  ** directory
576  */
577  if (NumChars > strlen(VirtualPath))
578  {
579  NumChars = strlen(VirtualPath);
580  }
581 
582  /*
583  ** copy over only the part that is the device name
584  */
585  strncpy(devname, VirtualPath, NumChars);
586  devname[NumChars] = '\0'; /* Truncate it with a NULL. */
587  DeviceLen = strlen(devname);
588 
589  /*
590  ** Copy everything after the devname as the path/filename
591  */
592  strncpy(filename, &(VirtualPath[NumChars]), OS_MAX_PATH_LEN);
593  FilenameLen = strlen(filename);
594 
595 #ifdef OS_DEBUG_PRINTF
596  printf("VirtualPath: %s, Length: %d\n",VirtualPath, (int)strlen(VirtualPath));
597  printf("NumChars: %d\n",NumChars);
598  printf("devname: %s\n",devname);
599  printf("filename: %s\n",filename);
600 #endif
601 
602  /*
603  ** look for the dev name we found in the VolumeTable
604  */
605  for (i = 0; i < NUM_TABLE_ENTRIES; i++)
606  {
607  if (OS_VolumeTable[i].FreeFlag == FALSE &&
608  strncmp(OS_VolumeTable[i].MountPoint, devname,NumChars) == 0)
609  {
610  break;
611  }
612  }
613 
614  /*
615  ** Make sure we found a valid drive
616  */
617  if (i >= NUM_TABLE_ENTRIES)
618  {
619  return OS_FS_ERR_PATH_INVALID;
620  }
621 
622  /*
623  ** copy over the physical first part of the drive
624  */
625  strncpy(LocalPath,OS_VolumeTable[i].PhysDevName,OS_MAX_LOCAL_PATH_LEN);
626  NumChars = strlen(LocalPath);
627 
628  /*
629  ** Add the file name
630  */
631  strncat(LocalPath, filename, (OS_MAX_LOCAL_PATH_LEN - NumChars));
632 
633 #ifdef OS_DEBUG_PRINTF
634  printf("Result of TranslatePath = %s\n",LocalPath);
635 #endif
636 
637  return OS_FS_SUCCESS;
638 
639 } /* end OS_TranslatePath */
