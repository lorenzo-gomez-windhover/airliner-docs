74 ** Function: CFE_ES_WriteToERLog
75 **
76 ** Purpose: Create an entry in the ES Exception and Reset Log.
77 **
78 */
79 int32 CFE_ES_WriteToERLog( uint32 EntryType, uint32 ResetType, uint32 ResetSubtype,
80  const char *Description, const uint32 *Context, uint32 ContextSize )
81 {
82  uint32 LogIdx;
83 
84  /*
85  ** Code
86  */
87 
88  /*
89  ** Try to clean up an invalid ER log index variable.
90  */
91  if ( CFE_ES_ResetDataPtr->ERLogIndex >= CFE_ES_ER_LOG_ENTRIES )
92  {
93  CFE_ES_ResetDataPtr->ERLogIndex = 0;
94  }
95  LogIdx = CFE_ES_ResetDataPtr->ERLogIndex;
96 
97  /*
98  ** Now that the Local Index variable is set, increment the index for the next entry.
99  */
100  CFE_ES_ResetDataPtr->ERLogIndex++;
101  if ( CFE_ES_ResetDataPtr->ERLogIndex >= CFE_ES_ER_LOG_ENTRIES )
102  {
103  CFE_ES_ResetDataPtr->ERLogIndex = 0;
104  }
105 
106 
107  /*
108  ** Clear out the log entry we are about to use.
109  */
110  CFE_PSP_MemSet( (void *)&(CFE_ES_ResetDataPtr->ERLog[LogIdx]), 0,
111  sizeof (CFE_ES_ERLog_t));
112 
113  /*
114  ** Fill out the log fields
115  */
116  CFE_ES_ResetDataPtr->ERLog[LogIdx].LogEntryType = EntryType;
117  CFE_ES_ResetDataPtr->ERLog[LogIdx].ResetType = ResetType;
118  CFE_ES_ResetDataPtr->ERLog[LogIdx].ResetSubtype = ResetSubtype;
119  CFE_ES_ResetDataPtr->ERLog[LogIdx].BootSource = CFE_ES_ResetDataPtr->ResetVars.BootSource;
120  CFE_ES_ResetDataPtr->ERLog[LogIdx].ProcessorResetCount =
121  CFE_ES_ResetDataPtr->ResetVars.ProcessorResetCount;
122  CFE_ES_ResetDataPtr->ERLog[LogIdx].MaxProcessorResetCount =
123  CFE_ES_ResetDataPtr->ResetVars.MaxProcessorResetCount;
124 
125  /*
126  ** Copy the ES Reset variables to the log (before they are modified by the log entry).
127  */
128  CFE_PSP_MemCpy((void *)&(CFE_ES_ResetDataPtr->ERLog[LogIdx].DebugVars),
129  (void *)&(CFE_ES_Global.DebugVars),
130  sizeof(CFE_ES_DebugVariables_t ));
131 
132  /*
133  ** Time Stamp the log entry with the system time
134  */
135  CFE_ES_ResetDataPtr->ERLog[LogIdx].TimeCode = CFE_TIME_GetTime();
136 
137  /*
138  ** Copy the Description string to the log.
139  */
140  if ( Description == NULL)
141  {
142  strncpy(CFE_ES_ResetDataPtr->ERLog[LogIdx].Description, "No Description String Given.", 80);
143  }
144  else
145  {
146  strncpy(CFE_ES_ResetDataPtr->ERLog[LogIdx].Description, Description, 80);
147  }
148 
149  /*
150  ** In the case of an exception, copy the processor context data to the log.
151  */
152  if (Context != NULL && ContextSize > 0)
153  {
154  /*
155  ** Copy the processor context data (i.e. register dump). Make sure that
156  ** the passed-in context_size is not greater than the declared size in
157  ** the ER Log log entry.
158  */
159  if ( ContextSize <= CFE_ES_ER_LOG_MAX_CONTEXT_SIZE )
160  {
161  CFE_PSP_MemCpy ( (void *)(CFE_ES_ResetDataPtr->ERLog[LogIdx].Context),
162  (void *)Context,
163  ContextSize);
164  }
165  else
166  {
167  CFE_PSP_MemCpy ( (void *)(CFE_ES_ResetDataPtr->ERLog[LogIdx].Context),
168  (void *)Context,
169  CFE_ES_ER_LOG_MAX_CONTEXT_SIZE);
170  }
171  /*
172  ** Indicate that context is valid.
173  ** Using the original context size (not the truncated size) so it will be
174  ** evident if the context information was truncated.
175  */
176  CFE_ES_ResetDataPtr->ERLog[LogIdx].ContextSize = ContextSize;
177  }
178  else
179  {
180  /*
181  ** Context is not valid
182  */
183  CFE_ES_ResetDataPtr->ERLog[LogIdx].ContextSize = 0;
184  } /* end if */
185 
186  /*
187  ** Increment the number of ER log entries made
188  */
189  CFE_ES_ResetDataPtr->ERLogEntries++;
190 
191  return(CFE_SUCCESS);
192 
193 } /* End of CFE_ES_WriteToERLog() */
