170 int32 CFE_SB_CreatePipe(CFE_SB_PipeId_t *PipeIdPtr, uint16 Depth, const char *PipeName)
171 {
172  uint32 AppId = 0xFFFFFFFF;
173  uint32 TskId = 0;
174  uint32 SysQueueId = 0;
175  int32 Status;
176  CFE_SB_PipeId_t PipeTblIdx;
177  char AppName[OS_MAX_API_NAME] = {'\0'};
178  char FullName[(OS_MAX_API_NAME * 2)];
179 
180  /* take semaphore to prevent a task switch during this call */
181  CFE_SB_LockSharedData(__func__,__LINE__);
182 
183  TskId = OS_TaskGetId();
184 
185  /* set user's pipe id value to 'invalid' for error cases below */
186  if(PipeIdPtr != NULL){
187  *PipeIdPtr = CFE_SB_INVALID_PIPE;
188  }/* end if */
189 
190  /* check input parameters */
191  if((PipeIdPtr == NULL)||(Depth > CFE_SB_MAX_PIPE_DEPTH)||(Depth == 0)){
192  CFE_SB.HKTlmMsg.Payload.CreatePipeErrCnt++;
193  CFE_SB_UnlockSharedData(__func__,__LINE__);
194  CFE_EVS_SendEventWithAppID(CFE_SB_CR_PIPE_BAD_ARG_EID,CFE_EVS_ERROR,CFE_SB.AppId,
195  "CreatePipeErr:Bad Input Arg:app=%s,ptr=0x%lx,depth=%d,maxdepth=%d",
196  CFE_SB_GetAppTskName(TskId,FullName),(unsigned long)PipeIdPtr,Depth,CFE_SB_MAX_PIPE_DEPTH);
197  return CFE_SB_BAD_ARGUMENT;
198  }/*end if*/
199 
200  /* get first available entry in pipe table */
201  PipeTblIdx = CFE_SB_GetAvailPipeIdx();
202 
203  /* if pipe table is full, send event and return error */
204  if(PipeTblIdx == CFE_SB_INVALID_PIPE){
205  CFE_SB_UnlockSharedData(__func__,__LINE__);
206  CFE_EVS_SendEventWithAppID(CFE_SB_MAX_PIPES_MET_EID,CFE_EVS_ERROR,CFE_SB.AppId,
207  "CreatePipeErr:Max Pipes(%d)In Use.app %s",
208  CFE_SB_MAX_PIPES,CFE_SB_GetAppTskName(TskId,FullName));
209  return CFE_SB_MAX_PIPES_MET;
210  }/* end if */
211 
212  /* create the queue */
213  Status = OS_QueueCreate(&SysQueueId,PipeName,Depth,sizeof(CFE_SB_BufferD_t *),0);
214  if (Status != OS_SUCCESS) {
215  CFE_SB_UnlockSharedData(__func__,__LINE__);
216  CFE_EVS_SendEventWithAppID(CFE_SB_CR_PIPE_ERR_EID,CFE_EVS_ERROR,CFE_SB.AppId,
217  "CreatePipeErr:OS_QueueCreate returned %d,app %s",
218  (int)Status,CFE_SB_GetAppTskName(TskId,FullName));
219  return CFE_SB_PIPE_CR_ERR;
220  }/* end if */
221 
222  /* get callers AppId */
223  CFE_ES_GetAppID(&AppId);
224 
225  /* get callers name */
226  CFE_ES_GetAppName(AppName, AppId, OS_MAX_API_NAME);
227 
228  /* Hardcode a NULL terminator, in case rcvd name was too long */
229  AppName[OS_MAX_API_NAME-1]= '\0';
230 
231  /* fill in the pipe table fields */
232  CFE_SB.PipeTbl[PipeTblIdx].InUse = CFE_SB_IN_USE;
233  CFE_SB.PipeTbl[PipeTblIdx].SysQueueId = SysQueueId;
234  CFE_SB.PipeTbl[PipeTblIdx].PipeId = PipeTblIdx;
235  CFE_SB.PipeTbl[PipeTblIdx].QueueDepth = Depth;
236  CFE_SB.PipeTbl[PipeTblIdx].AppId = AppId;
237  CFE_SB.PipeTbl[PipeTblIdx].SendErrors = 0;
238  CFE_SB.PipeTbl[PipeTblIdx].CurrentBuff = NULL;
239  CFE_SB.PipeTbl[PipeTblIdx].ToTrashBuff = NULL;
240  strcpy(&CFE_SB.PipeTbl[PipeTblIdx].AppName[0],&AppName[0]);
241  strncpy(&CFE_SB.PipeTbl[PipeTblIdx].PipeName[0],PipeName,OS_MAX_API_NAME);
242  CFE_SB.PipeTbl[PipeTblIdx].PipeName[OS_MAX_API_NAME-1]='\0';
243 
244  /* Increment the Pipes in use ctr and if it's > the high water mark,*/
245  /* adjust the high water mark */
246  CFE_SB.StatTlmMsg.Payload.PipesInUse++;
247  if(CFE_SB.StatTlmMsg.Payload.PipesInUse > CFE_SB.StatTlmMsg.Payload.PeakPipesInUse){
248  CFE_SB.StatTlmMsg.Payload.PeakPipesInUse = CFE_SB.StatTlmMsg.Payload.PipesInUse;
249  }/* end if */
250 
251  /* Reset the pipe depth parameters in the statistics pkt */
252  if (PipeTblIdx < CFE_SB_TLM_PIPEDEPTHSTATS_SIZE)
253  {
254  CFE_SB.StatTlmMsg.Payload.PipeDepthStats[PipeTblIdx].PipeId = PipeTblIdx;
255  CFE_SB.StatTlmMsg.Payload.PipeDepthStats[PipeTblIdx].Depth = Depth;
256  CFE_SB.StatTlmMsg.Payload.PipeDepthStats[PipeTblIdx].InUse = 0;
257  CFE_SB.StatTlmMsg.Payload.PipeDepthStats[PipeTblIdx].PeakInUse = 0;
258  }
259 
260  /* give the pipe handle to the caller */
261  *PipeIdPtr = PipeTblIdx;
262 
263  CFE_SB_UnlockSharedData(__func__,__LINE__);
264 
265  /* send debug event */
266  CFE_EVS_SendEventWithAppID(CFE_SB_PIPE_ADDED_EID,CFE_EVS_DEBUG,CFE_SB.AppId,
267  "Pipe Created:name %s,id %d,app %s",
268  CFE_SB_GetPipeName(CFE_SB.PipeTbl[PipeTblIdx].PipeId),
269  (int)CFE_SB.PipeTbl[PipeTblIdx].PipeId,
270  CFE_SB_GetAppTskName(TskId,FullName));
271 
272  return CFE_SUCCESS;
273 
274 }/* end CFE_SB_CreatePipe */

330 /******************************************************************************
331 ** Function: CFE_SB_DeletePipeFull()
332 **
333 ** Purpose:
334 ** Will unsubscribe to all routes associated with the given pipe id, then remove
335 ** pipe from the pipe table.
336 **
337 ** NOTE:This function cannot be called directly, it would not be semaphore protected
338 **
339 ** Arguments:
340 ** PipeId - The ID of the pipe to delete.
341 **
342 ** Return:
343 ** CFE_SUCCESS or cFE Error Code
344 */
345 int32 CFE_SB_DeletePipeFull(CFE_SB_PipeId_t PipeId,uint32 AppId)
346 {
347  uint8 PipeTblIdx;
348  int32 RtnFromVal,Stat;
349  uint32 Owner,i;
350  uint32 TskId = 0;
351  CFE_SB_Msg_t *PipeMsgPtr;
352  CFE_SB_DestinationD_t *DestPtr = NULL;
353  char FullName[(OS_MAX_API_NAME * 2)];
354 
355  /* take semaphore to prevent a task switch during this call */
356  CFE_SB_LockSharedData(__func__,__LINE__);
357 
358  /* get TaskId of caller for events */
359  TskId = OS_TaskGetId();
360 
361  /* check input parameter */
362  PipeTblIdx = CFE_SB_GetPipeIdx(PipeId);
363  RtnFromVal = CFE_SB_ValidatePipeId(PipeId);
364  if((RtnFromVal != CFE_SUCCESS)||(PipeTblIdx == CFE_SB_INVALID_PIPE))
365  {
366  CFE_SB.HKTlmMsg.Payload.CreatePipeErrCnt++;
367  CFE_SB_UnlockSharedData(__func__,__LINE__);
368  CFE_EVS_SendEventWithAppID(CFE_SB_DEL_PIPE_ERR1_EID,CFE_EVS_ERROR,CFE_SB.AppId,
369  "Pipe Delete Error:Bad Argument,PipedId %d,Requestor %s,Idx %d,Stat %d",
370  (int)PipeId,CFE_SB_GetAppTskName(TskId,FullName),(int)PipeTblIdx,(int)RtnFromVal);
371  return CFE_SB_BAD_ARGUMENT;
372  }/* end if */
373 
374  Owner = CFE_SB.PipeTbl[PipeTblIdx].AppId;
375 
376  /* check that the given AppId is the owner of the pipe */
377  if(AppId != Owner){
378  CFE_SB.HKTlmMsg.Payload.CreatePipeErrCnt++;
379  CFE_SB_UnlockSharedData(__func__,__LINE__);
380  CFE_EVS_SendEventWithAppID(CFE_SB_DEL_PIPE_ERR2_EID,CFE_EVS_ERROR,CFE_SB.AppId,
381  "Pipe Delete Error:Caller(%s) is not the owner of pipe %d",
382  CFE_SB_GetAppTskName(TskId,FullName),PipeId);
383  return CFE_SB_BAD_ARGUMENT;
384  }/* end if */
385 
386  /* check destination list of every in-use MsgId, for the given pipeid. */
387  /* when found, remove the pipe ID from the destination list via 'unsubscribe' */
388  for(i=0;i<CFE_SB_MAX_MSG_IDS;i++){
389  if(CFE_SB.RoutingTbl[i].MsgId != CFE_SB_INVALID_MSG_ID){
390 
391  DestPtr = CFE_SB.RoutingTbl[i].ListHeadPtr;
392 
393  while(DestPtr != NULL){
394 
395  if(DestPtr -> PipeId == PipeId){
396  /* release the semaphore, unsubscribe will need to take it */
397  CFE_SB_UnlockSharedData(__func__,__LINE__);
398  CFE_SB_UnsubscribeWithAppId(CFE_SB.RoutingTbl[i].MsgId,
399  PipeId,AppId);
400  CFE_SB_LockSharedData(__func__,__LINE__);
401  }/* end if */
402 
403  DestPtr = DestPtr -> Next;
404 
405  }/* end while */
406 
407  }/* end if */
408  }/* end for */
409 
410  if (CFE_SB.PipeTbl[PipeTblIdx].ToTrashBuff != NULL) {
411 
412  /* Decrement the Buffer Use Count and Free buffer if cnt=0) */
413  CFE_SB_DecrBufUseCnt(CFE_SB.PipeTbl[PipeTblIdx].ToTrashBuff);
414  CFE_SB.PipeTbl[PipeTblIdx].ToTrashBuff = NULL;
415 
416  }/* end if */
417 
418 
419  /* remove any messages that might be on the pipe */
420  /* this step will free the memory used to store the message */
421  do{
422  CFE_SB_UnlockSharedData(__func__,__LINE__);
423  Stat = CFE_SB_RcvMsg(&PipeMsgPtr,PipeId,CFE_SB_POLL);
424  CFE_SB_LockSharedData(__func__,__LINE__);
425  }while(Stat == CFE_SUCCESS);
426 
427  /* Delete the underlying OS queue */
428  OS_QueueDelete(CFE_SB.PipeTbl[PipeTblIdx].SysQueueId);
429 
430  /* remove the pipe from the pipe table */
431  CFE_SB.PipeTbl[PipeTblIdx].InUse = CFE_SB_NOT_IN_USE;
432  CFE_SB.PipeTbl[PipeTblIdx].SysQueueId = CFE_SB_UNUSED_QUEUE;
433  CFE_SB.PipeTbl[PipeTblIdx].PipeId = CFE_SB_INVALID_PIPE;
434  CFE_SB.PipeTbl[PipeTblIdx].CurrentBuff = NULL;
435  CFE_PSP_MemSet(&CFE_SB.PipeTbl[PipeTblIdx].PipeName[0],0,OS_MAX_API_NAME);
436 
437  /* zero out the pipe depth stats */
438  if (PipeTblIdx < CFE_SB_TLM_PIPEDEPTHSTATS_SIZE)
439  {
440  CFE_SB.StatTlmMsg.Payload.PipeDepthStats[PipeTblIdx].PipeId = 0;
441  CFE_SB.StatTlmMsg.Payload.PipeDepthStats[PipeTblIdx].Depth = 0;
442  CFE_SB.StatTlmMsg.Payload.PipeDepthStats[PipeTblIdx].InUse = 0;
443  CFE_SB.StatTlmMsg.Payload.PipeDepthStats[PipeTblIdx].PeakInUse = 0;
444  }
445 
446  CFE_SB.StatTlmMsg.Payload.PipesInUse--;
447 
448  CFE_SB_UnlockSharedData(__func__,__LINE__);
449 
450  CFE_EVS_SendEventWithAppID(CFE_SB_PIPE_DELETED_EID,CFE_EVS_DEBUG,CFE_SB.AppId,
451  "Pipe Deleted:id %d,owner %s",PipeId, CFE_SB_GetAppTskName(Owner,FullName));
452 
453  return CFE_SUCCESS;
454 
455 }/* end CFE_SB_DeletePipeFull */

577 /******************************************************************************
578 ** Name: CFE_SB_SubscribeFull
579 **
580 ** Purpose: CFE Internal API used to subscribe to a message. Contains an input
581 ** parameter for all possible subscription choices. This function is
582 ** called by CFE_SB_SubscribeEx, CFE_SB_Subscribe and
583 ** CFE_SB_SubscribeLocal.
584 **
585 ** Assumptions, External Events, and Notes:
586 ** None
587 **
588 ** Date Written:
589 ** 04/25/2005
590 **
591 ** Input Arguments:
592 ** MsgId - The requesting message
593 ** PipeId - The Pipe ID to send the message to
594 ** Quality - Quality of Service (Qos) - priority and reliability
595 ** MsgLim - Max number of messages, with this MsgId, allowed on the
596 ** pipe at any time.
597 ** Scope - Local subscription or broadcasted to peers
598 **
599 ** Output Arguments:
600 ** None
601 **
602 ** Return Values:
603 ** Status
604 **
605 ******************************************************************************/
606 int32 CFE_SB_SubscribeFull(CFE_SB_MsgId_t MsgId,
607  CFE_SB_PipeId_t PipeId,
608  CFE_SB_Qos_t Quality,
609  uint16 MsgLim,
610  uint8 Scope)
611 {
612  uint16 Idx;
613  int32 Stat;
614  uint32 TskId = 0;
615  uint32 AppId = 0xFFFFFFFF;
616  uint8 PipeIdx;
617  CFE_SB_DestinationD_t *DestBlkPtr = NULL;
618  char FullName[(OS_MAX_API_NAME * 2)];
619 
620  /* take semaphore to prevent a task switch during this call */
621  CFE_SB_LockSharedData(__func__,__LINE__);
622 
623  /* get task id for events */
624  TskId = OS_TaskGetId();
625 
626  /* get the callers Application Id */
627  CFE_ES_GetAppID(&AppId);
628 
629  /* check that the pipe has been created */
630  PipeIdx = CFE_SB_GetPipeIdx(PipeId);
631  if(PipeIdx==CFE_SB_INVALID_PIPE){
632  CFE_SB.HKTlmMsg.Payload.SubscribeErrCnt++;
633  CFE_SB_UnlockSharedData(__func__,__LINE__);
634  CFE_EVS_SendEventWithAppID(CFE_SB_SUB_INV_PIPE_EID,CFE_EVS_ERROR,CFE_SB.AppId,
635  "Subscribe Err:Invalid Pipe Id,Msg=0x%x,PipeId=%d,App %s",MsgId,PipeId,
636  CFE_SB_GetAppTskName(TskId,FullName));
637  return CFE_SB_BAD_ARGUMENT;
638  }/* end if */
639 
640  /* check that the requestor is the owner of the pipe */
641  if(CFE_SB.PipeTbl[PipeIdx].AppId != AppId){
642  CFE_SB.HKTlmMsg.Payload.SubscribeErrCnt++;
643  CFE_SB_UnlockSharedData(__func__,__LINE__);
644  CFE_EVS_SendEventWithAppID(CFE_SB_SUB_INV_CALLER_EID,CFE_EVS_ERROR,CFE_SB.AppId,
645  "Subscribe Err:Caller(%s) is not the owner of pipe %d,Msg=0x%x",
646  CFE_SB_GetAppTskName(TskId,FullName),PipeId,MsgId);
647  return CFE_SB_BAD_ARGUMENT;
648  }/* end if */
649 
650  /* check message id and scope */
651  if((CFE_SB_ValidateMsgId(MsgId) != CFE_SUCCESS)||(Scope > 1))
652  {
653  CFE_SB.HKTlmMsg.Payload.SubscribeErrCnt++;
654  CFE_SB_UnlockSharedData(__func__,__LINE__);
655  CFE_EVS_SendEventWithAppID(CFE_SB_SUB_ARG_ERR_EID,CFE_EVS_ERROR,CFE_SB.AppId,
656  "Subscribe Err:Bad Arg,MsgId 0x%x,PipeId %d,app %s,scope %d",
657  MsgId,PipeId,CFE_SB_GetAppTskName(TskId,FullName),Scope);
658  return CFE_SB_BAD_ARGUMENT;
659  }/* end if */
660 
661  /* check for duplicate subscription */
662  if(CFE_SB_DuplicateSubscribeCheck(MsgId,PipeId)==CFE_SB_DUPLICATE){
663  CFE_SB.HKTlmMsg.Payload.DupSubscriptionsCnt++;
664  CFE_SB_UnlockSharedData(__func__,__LINE__);
665  CFE_EVS_SendEventWithAppID(CFE_SB_DUP_SUBSCRIP_EID,CFE_EVS_INFORMATION,CFE_SB.AppId,
666  "Duplicate Subscription,MsgId 0x%x on %s pipe,app %s",
667  MsgId,CFE_SB_GetPipeName(PipeId),CFE_SB_GetAppTskName(TskId,FullName));
668  return CFE_SUCCESS;
669  }/* end if */
670 
671  /*
672  ** If there has been a subscription for this message id earlier,
673  ** get the element number in the routing table.
674  */
675  Idx = CFE_SB_GetRoutingTblIdx(MsgId);
676 
677  /* if first subscription for this message... */
678  if(Idx==CFE_SB_AVAILABLE){
679 
680  /* Get the index to the first available element in the routing table */
681  Idx = CFE_SB_GetAvailRoutingIdx();
682 
683  /* if all routing table elements are used, send event */
684  if(Idx >= CFE_SB_MAX_MSG_IDS){
685  CFE_SB_UnlockSharedData(__func__,__LINE__);
686  CFE_EVS_SendEventWithAppID(CFE_SB_MAX_MSGS_MET_EID,CFE_EVS_ERROR,CFE_SB.AppId,
687  "Subscribe Err:Max Msgs(%d)In Use,MsgId 0x%x,pipe %s,app %s",
688  CFE_SB_MAX_MSG_IDS,MsgId,CFE_SB_GetPipeName(PipeId),CFE_SB_GetAppTskName(TskId,FullName));
689  return CFE_SB_MAX_MSGS_MET;
690  }/* end if */
691 
692  /* Increment the MsgIds in use ctr and if it's > the high water mark,*/
693  /* adjust the high water mark */
694  CFE_SB.StatTlmMsg.Payload.MsgIdsInUse++;
695  if(CFE_SB.StatTlmMsg.Payload.MsgIdsInUse > CFE_SB.StatTlmMsg.Payload.PeakMsgIdsInUse){
696  CFE_SB.StatTlmMsg.Payload.PeakMsgIdsInUse = CFE_SB.StatTlmMsg.Payload.MsgIdsInUse;
697  }/* end if */
698 
699  /* populate the look up table with the routing table index */
700  CFE_SB_SetRoutingTblIdx(MsgId,Idx);
701 
702  /* label the new routing block with the message identifier */
703  CFE_SB.RoutingTbl[Idx].MsgId = MsgId;
704 
705  }/* end if */
706 
707  if(CFE_SB.RoutingTbl[Idx].Destinations >= CFE_SB_MAX_DEST_PER_PKT){
708  CFE_SB_UnlockSharedData(__func__,__LINE__);
709  CFE_EVS_SendEventWithAppID(CFE_SB_MAX_DESTS_MET_EID,CFE_EVS_ERROR,CFE_SB.AppId,
710  "Subscribe Err:Max Dests(%d)In Use For Msg 0x%x,pipe %s,app %s",
711  CFE_SB_MAX_DEST_PER_PKT,MsgId,CFE_SB_GetPipeName(PipeId),
712  CFE_SB_GetAppTskName(TskId,FullName));
713  return CFE_SB_MAX_DESTS_MET;
714  }/* end if */
715 
716  DestBlkPtr = CFE_SB_GetDestinationBlk();
717  if(DestBlkPtr == NULL){
718  CFE_SB_UnlockSharedData(__func__,__LINE__);
719  CFE_EVS_SendEventWithAppID(CFE_SB_DEST_BLK_ERR_EID,CFE_EVS_ERROR,CFE_SB.AppId,
720  "Subscribe Err:Request for Destination Blk failed for Msg 0x%x", (unsigned int)MsgId);
721  return CFE_SB_BUF_ALOC_ERR;
722  }/* end if */
723 
724  /* initialize destination block */
725  DestBlkPtr -> PipeId = PipeId;
726  DestBlkPtr -> MsgId2PipeLim = (uint16)MsgLim;
727  DestBlkPtr -> Active = CFE_SB_ACTIVE;
728  DestBlkPtr -> BuffCount = 0;
729  DestBlkPtr -> DestCnt = 0;
730  DestBlkPtr -> Scope = Scope;
731  DestBlkPtr -> Prev = NULL;
732  DestBlkPtr -> Next = NULL;
733 
734  /* add destination block to head of list */
735  CFE_SB_AddDest(Idx, DestBlkPtr);
736 
737  CFE_SB.RoutingTbl[Idx].Destinations++;
738 
739  CFE_SB.StatTlmMsg.Payload.SubscriptionsInUse++;
740  if(CFE_SB.StatTlmMsg.Payload.SubscriptionsInUse > CFE_SB.StatTlmMsg.Payload.PeakSubscriptionsInUse)
741  {
742  CFE_SB.StatTlmMsg.Payload.PeakSubscriptionsInUse = CFE_SB.StatTlmMsg.Payload.SubscriptionsInUse;
743  }/* end if */
744 
745  if((CFE_SB.SubscriptionReporting == CFE_SB_ENABLE)&&(Scope==CFE_SB_GLOBAL)){
746  CFE_SB.SubRprtMsg.Payload.MsgId = MsgId;
747  CFE_SB.SubRprtMsg.Payload.Pipe = PipeId;
748  CFE_SB.SubRprtMsg.Payload.Qos.Priority = Quality.Priority;
749  CFE_SB.SubRprtMsg.Payload.Qos.Reliability = Quality.Reliability;
750  CFE_SB.SubRprtMsg.Payload.SubType = CFE_SB_SUBSCRIPTION;
751  CFE_SB_UnlockSharedData(__func__,__LINE__);
752  Stat = CFE_SB_SendMsg((CFE_SB_Msg_t *)&CFE_SB.SubRprtMsg);
753  CFE_EVS_SendEventWithAppID(CFE_SB_SUBSCRIPTION_RPT_EID,CFE_EVS_DEBUG,CFE_SB.AppId,
754  "Sending Subscription Report Msg=0x%x,Pipe=%d,Stat=0x%x",
755  (unsigned int)MsgId,(int)PipeId,(unsigned int)Stat);
756  CFE_SB_LockSharedData(__func__,__LINE__);/* to prevent back-to-back unlock */
757  }/* end if */
758 
759  /* release the semaphore */
760  CFE_SB_UnlockSharedData(__func__,__LINE__);
761 
762  CFE_EVS_SendEventWithAppID(CFE_SB_SUBSCRIPTION_RCVD_EID,CFE_EVS_DEBUG,CFE_SB.AppId,
763  "Subscription Rcvd:MsgId 0x%x on %s(%d),app %s",
764  (unsigned int)MsgId,CFE_SB_GetPipeName(PipeId),(int)PipeId,CFE_SB_GetAppTskName(TskId,FullName));
765 
766  return CFE_SUCCESS;
767 
768 }/* end CFE_SB_SubscribeFull */

879 **
880 ** Assumptions, External Events, and Notes:
881 **
882 **
883 ** Notes:This function cannot be called directly,it would not be semaphore protected.
884 ** Also,if more than one subscription is found, this function will remove all
885 ** entries that match.
886 **
887 ** Date Written:
888 ** 04/25/2005
889 **
890 ** Input Arguments:
891 ** MsgId
892 ** PipeId
893 ** Scope
894 ** AppId
895 **
896 ** Output Arguments:
897 ** None
898 **
899 ** Return Values:
900 ** Status
901 **
902 ******************************************************************************/
903 int32 CFE_SB_UnsubscribeFull(CFE_SB_MsgId_t MsgId,CFE_SB_PipeId_t PipeId,
904  uint8 Scope,uint32 AppId)
905 {
906  uint16 Idx;
907  uint32 PipeIdx;
908  uint32 TskId = 0;
909  uint32 MatchFound = FALSE;
910  int32 Stat;
911  CFE_SB_DestinationD_t *DestPtr = NULL;
912  char FullName[(OS_MAX_API_NAME * 2)];
913 
914 
915  /* take semaphore to prevent a task switch during this call */
916  CFE_SB_LockSharedData(__func__,__LINE__);
917 
918  /* get task id for events */
919  TskId = OS_TaskGetId();
920 
921  /* check that the pipe has been created */
922  PipeIdx = CFE_SB_GetPipeIdx(PipeId);
923  if(PipeIdx==CFE_SB_INVALID_PIPE){
924  CFE_SB_UnlockSharedData(__func__,__LINE__);
925  CFE_EVS_SendEventWithAppID(CFE_SB_UNSUB_INV_PIPE_EID,CFE_EVS_ERROR,CFE_SB.AppId,
926  "Unsubscribe Err:Invalid Pipe Id Msg=0x%x,Pipe=%d,app=%s",
927  (unsigned int)MsgId,(int)PipeId,CFE_SB_GetAppTskName(TskId,FullName));
928  return CFE_SB_BAD_ARGUMENT;
929  }/* end if */
930 
931  /* if given 'AppId' is not the owner of the pipe, send error event and return */
932  if(CFE_SB.PipeTbl[PipeIdx].AppId != AppId){
933  CFE_SB_UnlockSharedData(__func__,__LINE__);
934  CFE_EVS_SendEventWithAppID(CFE_SB_UNSUB_INV_CALLER_EID,CFE_EVS_ERROR,CFE_SB.AppId,
935  "Unsubscribe Err:Caller(%s) is not the owner of pipe %d,Msg=0x%x",
936  CFE_SB_GetAppTskName(TskId,FullName),(int)PipeId,MsgId);
937  return CFE_SB_BAD_ARGUMENT;
938  }/* end if */
939 
940  /* check input parameters */
941  if((CFE_SB_ValidateMsgId(MsgId) != CFE_SUCCESS)||
942  (CFE_SB_ValidatePipeId(PipeId) != CFE_SUCCESS)||
943  (Scope > 1))
944  {
945  CFE_SB_UnlockSharedData(__func__,__LINE__);

983

993  CFE_SB.RoutingTbl[Idx].MsgId = CFE_SB_INVALID_MSG_ID;
994  CFE_SB_SetRoutingTblIdx(MsgId,CFE_SB_AVAILABLE);
995 
996  /* Send unsubscribe report only if there are zero requests for this pkt */
997  if((CFE_SB.SubscriptionReporting == CFE_SB_ENABLE)&&
998  (Scope == CFE_SB_GLOBAL))
999  {
1000  CFE_SB.SubRprtMsg.Payload.MsgId = MsgId;
1001  CFE_SB.SubRprtMsg.Payload.Pipe = PipeId;

1006  Stat = CFE_SB_SendMsg((CFE_SB_Msg_t *)&CFE_SB.SubRprtMsg);
1007  CFE_EVS_SendEventWithAppID(CFE_SB_UNSUBSCRIPTION_RPT_EID,CFE_EVS_DEBUG,CFE_SB.AppId,
1008  "Sending Unsubscription Report Msg=0x%x,Pipe=%d,Stat=0x%x",
1009  (unsigned int)MsgId,(int)PipeId,(unsigned int)Stat);
1010  CFE_SB_LockSharedData(__func__,__LINE__);
1011  }/* end if */
1012 
1013  }/* end if */

1019  MsgId,PipeId,CFE_SB_GetAppTskName(TskId,FullName));
1020
