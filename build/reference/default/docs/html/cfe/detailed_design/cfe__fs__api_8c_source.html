73 int32 CFE_FS_ReadHeader(CFE_FS_Header_t *Hdr, int32 FileDes)
74 {
75  int32 Result;
76  int32 EndianCheck = 0x01020304;
77 
78  /*
79  ** Ensure that we are at the start of the file...
80  */
81  Result = OS_lseek(FileDes, 0, OS_SEEK_SET);
82 
83  if (Result == OS_SUCCESS)
84  {
85  /*
86  ** Read header structure into callers buffer...
87  */
88  Result = OS_read(FileDes, Hdr, sizeof(CFE_FS_Header_t));
89 
90  /* Determine if this processor is a little endian processor */
91  if ((*(char *)(&EndianCheck)) == 0x04)
92  {
93  /* If this is a little endian processor, then convert the header data structure from */
94  /* its standard big-endian format into a little endian format to ease user access */
95  CFE_FS_ByteSwapCFEHeader(Hdr);
96  }
97  }
98 
99  return(Result);
100 
101 } /* End of CFE_FS_ReadHeader() */

121 int32 CFE_FS_WriteHeader(int32 FileDes, CFE_FS_Header_t *Hdr)
122 {
123  CFE_TIME_SysTime_t Time;
124  int32 Result;
125  int32 EndianCheck = 0x01020304;
126 
127  /*
128  ** Ensure that we are at the start of the file...
129  */
130  Result = OS_lseek(FileDes, 0, OS_SEEK_SET);
131 
132  if (Result == OS_SUCCESS)
133  {
134  /*
135  ** Fill in the ID fields...
136  */
137  Hdr->SpacecraftID = CFE_PSP_GetSpacecraftId();
138  Hdr->ProcessorID = CFE_PSP_GetProcessorId();
139  CFE_ES_GetAppID((uint32 *)&(Hdr->ApplicationID));
140 
141  /* Fill in length field */
142 
143  Hdr->Length = sizeof(CFE_FS_Header_t);
144 
145  /* put the header, 'cfe1' in hex, in to the content type */
146  Hdr->ContentType = 0x63464531;
147 
148 
149  /*
150  ** Fill in the timestamp fields...
151  */
152  Time = CFE_TIME_GetTime();
153  Hdr->TimeSeconds = Time.Seconds;
154  Hdr->TimeSubSeconds = Time.Subseconds;
155 
156  /*
157  ** Determine if this is a little endian processor
158  */
159  if ((*(char *)(&EndianCheck)) == 0x04)
160  {
161  /* If this is a little endian processor, then convert the header data structure from */
162  /* the native little endian format to the required CFE standard big endian format */
163  CFE_FS_ByteSwapCFEHeader(Hdr);
164  }
165 
166  /*
167  ** Write header structure from callers buffer...
168  */
169  Result = OS_write(FileDes, Hdr, sizeof(CFE_FS_Header_t));
170 
171  /*
172  ** Determine if this is a little endian processor
173  */
174  if ((*(char *)(&EndianCheck)) == 0x04)
175  {
176  /* If this is a little endian processor, then convert the header data structure back */
177  /* from the required CFE standard big endian format to the little endian format */
178  CFE_FS_ByteSwapCFEHeader(Hdr);
179  }
180 
181  }
182 
183  return(Result);
184 
185 } /* End of CFE_FS_WriteHeader() */

188 int32 CFE_FS_SetTimestamp(int32 FileDes, CFE_TIME_SysTime_t NewTimestamp)
189 {
190  int32 Result = OS_FS_SUCCESS;
191  CFE_FS_Header_t TempHdr;
192  int32 EndianCheck = 0x01020304;
193  CFE_TIME_SysTime_t OutTimestamp = NewTimestamp;
194  int32 FileOffset = 0;
195 
196  FileOffset = ((char *)&TempHdr.TimeSeconds - (char *)&TempHdr.ContentType);
197  Result = OS_lseek(FileDes, FileOffset, OS_SEEK_SET);
198 
199  if (Result == FileOffset)
200  {
201  /*
202  ** Determine if this is a little endian processor
203  */
204  if ((*(char *)(&EndianCheck)) == 0x04)
205  {
206  /* If this processor is a little endian processor, then convert the timestamp to a big */
207  /* endian format so that it is compatible with the standard cFE File Header format */
208  CFE_FS_ByteSwapUint32(&OutTimestamp.Seconds);
209  CFE_FS_ByteSwapUint32(&OutTimestamp.Subseconds);
210  }
211 
212  Result = OS_write(FileDes, &OutTimestamp.Seconds, sizeof(OutTimestamp.Seconds));
213 
214  /* On a good write, the value returned will equal the number of bytes written */
215  if (Result == sizeof(OutTimestamp.Seconds))
216  {
217  Result = OS_write(FileDes, &OutTimestamp.Subseconds, sizeof(OutTimestamp.Subseconds));
218 
219  if (Result == sizeof(OutTimestamp.Subseconds))
220  {
221  Result = OS_FS_SUCCESS;
222  }
223  else
224  {
225  CFE_ES_WriteToSysLog("CFE_FS:SetTime-Failed to write Seconds (Status=0x%08X)\n", (unsigned int)Result);
226  }
227  }
228  else
229  {
230  CFE_ES_WriteToSysLog("CFE_FS:SetTime-Failed to write Seconds (Status=0x%08X)\n", (unsigned int)Result);
231  }
232  }
233  else
234  {
235  CFE_ES_WriteToSysLog("CFE_FS:SetTime-Failed to lseek time fields (Status=0x%08X)\n", (unsigned int)Result);
236  }
237 
238  return(Result);
239 } /* End of CFE_FS_SetTimestamp() */

283 /* Function: CFE_FS_ExtractFilenameFromPath */
284 /* */
285 /* This function extracts the filename from a given path/filename */
286 /* combination. */
287 /* */
288 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
289 int32 CFE_FS_ExtractFilenameFromPath(const char *OriginalPath, char *FileNameOnly)
290 {
291  uint32 i,j;
292  int StringLength;
293  int DirMarkIdx;
294  int32 ReturnCode;
295 
296  if ( OriginalPath == NULL || FileNameOnly == NULL )
297  {
298  ReturnCode = CFE_FS_BAD_ARGUMENT;
299  }
300  else
301  {
302 
303  /*
304  ** Get the string length of the original file path
305  */
306  StringLength = strlen(OriginalPath);
307 
308  /*
309  ** Extract the filename from the Path
310  */
311 
312  /*
313  ** Find the last '/' Character
314  */
315  DirMarkIdx = -1;
316  for ( i = 0; i < StringLength; i++ )
317  {
318  if ( OriginalPath[i] == '/' )
319  {
320  DirMarkIdx = i;
321  }
322  }
323 
324  /*
325  ** Verify the filename isn't too long
326  */
327  if ((StringLength - (DirMarkIdx + 1)) < OS_MAX_PATH_LEN)
328  {
329  /*
330  ** Extract the filename portion
331  */
332  if ( DirMarkIdx > 0 )
333  {
334  /*
335  ** Extract the filename portion
336  */
337  j = 0;
338  for ( i = DirMarkIdx + 1; i < StringLength; i++ )
339  {
340  FileNameOnly[j] = OriginalPath[i];
341  j++;
342  }
343  FileNameOnly[j] = '\0';
344 
345  ReturnCode = CFE_SUCCESS;
346  }
347  else
348  {
349  ReturnCode = CFE_FS_INVALID_PATH;
350  }
351  }
352  else
353  {
354  ReturnCode = CFE_FS_FNAME_TOO_LONG;
355  }
356  }
357 
358  return(ReturnCode);
359 }
