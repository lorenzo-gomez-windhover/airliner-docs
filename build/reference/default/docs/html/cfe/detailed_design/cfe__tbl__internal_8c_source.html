88 /******************************************************************************
89 ** Function: CFE_TBL_EarlyInit()
90 **
91 ** Purpose:
92 ** Initialize the Table Services
93 **
94 ** Arguments:
95 **
96 ** Notes:
97 ** This function MUST be called before any TBL API's are called.
98 **
99 ** Return:
100 ** none
101 */
102 int32 CFE_TBL_EarlyInit (void)
103 {
104  uint16 i;
105  uint32 j;
106  int32 Status;
107 
108 
109  /* Initialize the Table Registry */
110  for(i=0; i < CFE_TBL_MAX_NUM_TABLES; i++)
111  {
112  CFE_TBL_InitRegistryRecord(&CFE_TBL_TaskData.Registry[i]);
113  }
114 
115  /* Initialize the Table Access Descriptors */
116  for (i=0; i<CFE_TBL_MAX_NUM_HANDLES; i++)
117  {
118  CFE_TBL_TaskData.Handles[i].AppId = CFE_ES_ERR_APPID;
119  CFE_TBL_TaskData.Handles[i].RegIndex = 0;
120  CFE_TBL_TaskData.Handles[i].PrevLink = CFE_TBL_END_OF_LIST;
121  CFE_TBL_TaskData.Handles[i].NextLink = CFE_TBL_END_OF_LIST;
122  CFE_TBL_TaskData.Handles[i].UsedFlag = FALSE;
123  CFE_TBL_TaskData.Handles[i].LockFlag = FALSE;
124  CFE_TBL_TaskData.Handles[i].Updated = FALSE;
125  CFE_TBL_TaskData.Handles[i].BufferIndex = 0;
126  }
127 
128  /* Initialize the Table Validation Results Records */
129  for (i=0; i<CFE_TBL_MAX_NUM_VALIDATIONS; i++)
130  {
131  CFE_TBL_TaskData.ValidationResults[i].State = CFE_TBL_VALIDATION_FREE;
132  CFE_TBL_TaskData.ValidationResults[i].CrcOfTable = 0;
133  CFE_TBL_TaskData.ValidationResults[i].Result = 0;
134  CFE_TBL_TaskData.ValidationResults[i].ActiveBuffer = FALSE;
135  CFE_TBL_TaskData.ValidationResults[i].TableName[0] = '\0';
136  }
137 
138  /* Initialize the Dump-Only Table Dump Control Blocks */
139  for (i=0; i<CFE_TBL_MAX_SIMULTANEOUS_LOADS; i++)
140  {
141  CFE_TBL_TaskData.DumpControlBlocks[i].State = CFE_TBL_DUMP_FREE;
142  CFE_TBL_TaskData.DumpControlBlocks[i].DumpBufferPtr = NULL;
143  CFE_TBL_TaskData.DumpControlBlocks[i].Size = 0;
144  CFE_TBL_TaskData.DumpControlBlocks[i].TableName[0] = '\0';
145 
146  /* Prevent Shared Buffers from being used until successfully allocated */
147  CFE_TBL_TaskData.LoadBuffs[i].Taken = TRUE;
148  }
149 
150  CFE_TBL_TaskData.ValidationCtr = 0;
151 
152  CFE_TBL_TaskData.HkTlmTblRegIndex = CFE_TBL_NOT_FOUND;
153  CFE_TBL_TaskData.LastTblUpdated = CFE_TBL_NOT_FOUND;
154 
155  /*
156  ** Create table registry access mutex
157  */
158  Status = OS_MutSemCreate(&CFE_TBL_TaskData.RegistryMutex,
159  CFE_TBL_MUT_REG_NAME,
160  CFE_TBL_MUT_REG_VALUE);
161  if(Status != OS_SUCCESS)
162  {
163  CFE_ES_WriteToSysLog("CFE_TBL:Registry mutex creation failed! RC=0x%08x\n",(unsigned int)Status);
164  return Status;
165  }/* end if */
166 
167  /*
168  ** Create working buffer access mutex
169  */
170  Status = OS_MutSemCreate(&CFE_TBL_TaskData.WorkBufMutex,
171  CFE_TBL_MUT_WORK_NAME,
172  CFE_TBL_MUT_WORK_VALUE);
173  if(Status != OS_SUCCESS)
174  {
175  CFE_ES_WriteToSysLog("CFE_TBL:Working buffer mutex creation failed! RC=0x%08x\n",(unsigned int)Status);
176  return Status;
177  }/* end if */
178 
179  /*
180  ** Initialize housekeeping packet (clear user data area)...
181  */
182  CFE_SB_InitMsg(&CFE_TBL_TaskData.HkPacket,
183  CFE_TBL_HK_TLM_MID,
184  sizeof(CFE_TBL_HkPacket_t), TRUE);
185 
186  /*
187  ** Initialize table registry report packet (clear user data area)...
188  */
189  CFE_SB_InitMsg(&CFE_TBL_TaskData.TblRegPacket,
190  CFE_TBL_REG_TLM_MID,
191  sizeof(CFE_TBL_TblRegPacket_t), TRUE);
192 
193  /* Initialize memory partition and allocate shared table buffers. */
194  Status = CFE_ES_PoolCreate(&CFE_TBL_TaskData.Buf.PoolHdl,
195  CFE_TBL_TaskData.Buf.Partition,
196  CFE_TBL_BUF_MEMORY_BYTES);
197 
198  if(Status < 0)
199  {
200  CFE_ES_WriteToSysLog("CFE_TBL:InitBuffers PoolCreate fail Status=0x%X\n", (unsigned int)Status);
201  return Status;
202  }
203  else
204  {
205  /* Initialize each of the shared load buffers */
206  j=0;
207  do
208  {
209  /* Allocate memory for shared load buffers */
210  Status = CFE_ES_GetPoolBuf((uint32 **)&CFE_TBL_TaskData.LoadBuffs[j].BufferPtr,
211  CFE_TBL_TaskData.Buf.PoolHdl,
212  CFE_TBL_MAX_SNGL_TABLE_SIZE);
213 
214  if (Status < CFE_TBL_MAX_SNGL_TABLE_SIZE)
215  {
216  CFE_ES_WriteToSysLog("CFE_TBL:InitBuffers GetPoolBuf Fail Index=%d, Status=0x%X\n", (int)j, (unsigned int)Status);
217  return Status;
218  }
219  else
220  {
221  /* The buffer is successfully created, so allow it to be used */
222  CFE_TBL_TaskData.LoadBuffs[j].Taken = FALSE;
223 
224  CFE_PSP_MemSet(CFE_TBL_TaskData.LoadBuffs[j].DataSource, 0, OS_MAX_PATH_LEN);
225  CFE_TBL_TaskData.LoadBuffs[j].FileCreateTimeSecs = 0;
226  CFE_TBL_TaskData.LoadBuffs[j].FileCreateTimeSubSecs = 0;
227  }
228 
229  j++;
230  } while ((j < CFE_TBL_MAX_SIMULTANEOUS_LOADS) &&
231  (Status >= CFE_TBL_MAX_SNGL_TABLE_SIZE));
232  }
233 
234  /* Try to obtain a previous image of the Critical Table Registry from the Critical Data Store */
235  Status = CFE_ES_RegisterCDSEx(&CFE_TBL_TaskData.CritRegHandle,
236  (sizeof(CFE_TBL_CritRegRec_t)*CFE_TBL_MAX_CRITICAL_TABLES),
237  "CFE_TBL.CritReg", TRUE);
238 
239  /* Assume for the moment that nothing is already in the CDS and zero out the Critical Table Registry */
240  for (i=0; i<CFE_TBL_MAX_CRITICAL_TABLES; i++)
241  {
242  CFE_TBL_TaskData.CritReg[i].CDSHandle = CFE_ES_CDS_BAD_HANDLE;
243  CFE_TBL_TaskData.CritReg[i].FileCreateTimeSecs = 0;
244  CFE_TBL_TaskData.CritReg[i].FileCreateTimeSubSecs = 0;
245  CFE_TBL_TaskData.CritReg[i].LastFileLoaded[0] = '\0';
246  CFE_TBL_TaskData.CritReg[i].TimeOfLastUpdate.Seconds = 0;
247  CFE_TBL_TaskData.CritReg[i].TimeOfLastUpdate.Subseconds = 0;
248  CFE_TBL_TaskData.CritReg[i].TableLoadedOnce = FALSE;
249 
250  }
251 
252  if (Status == CFE_ES_CDS_ALREADY_EXISTS)
253  {
254  /* Try to recover the Critical Table Registry from the CDS */
255  Status = CFE_ES_RestoreFromCDS(CFE_TBL_TaskData.CritReg, CFE_TBL_TaskData.CritRegHandle);
256 
257  if (Status != CFE_SUCCESS)
258  {
259  /* Note if we were unable to recover error free Critical Table Registry from the CDS */
260  CFE_ES_WriteToSysLog("CFE_TBL:EarlyInit-Failed to recover Critical Table Registry (Err=0x%08X)\n", (unsigned int)Status);
261  }
262 
263  /* Whether we recovered the Critical Table Registry or not, we are successful with initialization */
264  Status = CFE_SUCCESS;
265  }
266  else if (Status != CFE_SUCCESS)
267  {
268  /* Not being able to support Critical Tables is not the end of the world */
269  /* Note the problem and move on */
270  CFE_ES_WriteToSysLog("CFE_TBL:EarlyInit-Failed to create Critical Table Registry (Err=0x%08X)\n", (unsigned int)Status);
271 
272  /* Failure to support critical tables is not a good enough reason to exit the cFE on start up */
273  Status = CFE_SUCCESS;
274  }
275  else
276  {
277  /* Save the initial version of the Critical Table Registry in the CDS */
278  Status = CFE_ES_CopyToCDS(CFE_TBL_TaskData.CritRegHandle, CFE_TBL_TaskData.CritReg);
279 
280  if (Status != CFE_SUCCESS)
281  {
282  /* Not being able to support Critical Tables is not the end of the world */
283  /* Note the problem and move on */
284  CFE_ES_WriteToSysLog("CFE_TBL:EarlyInit-Failed to save Critical Table Registry (Err=0x%08X)\n", (unsigned int)Status);
285 
286  /* Failure to support critical tables is not a good enough reason to exit the cFE on start up */
287  Status = CFE_SUCCESS;
288  }
289  }
290 
291  return Status;
292 
293 }/* end CFE_TBL_EarlyInit */

473 /*******************************************************************

475 ** CFE_TBL_RemoveAccessLink
476 **
477 ** NOTE: For complete prolog information, see 'cfe_tbl_internal.h'
478 ********************************************************************/
479 
480 int32 CFE_TBL_RemoveAccessLink(CFE_TBL_Handle_t TblHandle)
481 {
482  int32 Status = CFE_SUCCESS;
483  CFE_TBL_AccessDescriptor_t *AccessDescPtr = &CFE_TBL_TaskData.Handles[TblHandle];
484  CFE_TBL_RegistryRec_t *RegRecPtr = &CFE_TBL_TaskData.Registry[AccessDescPtr->RegIndex];
485 
486  /* Lock Access to the table while we modify the linked list */
487  CFE_TBL_LockRegistry();
488 
489  /* If we are removing the head of the linked list, then point */
490  /* the head pointer to the link after this one */
491  if (AccessDescPtr->PrevLink == CFE_TBL_END_OF_LIST)
492  {
493  RegRecPtr->HeadOfAccessList = AccessDescPtr->NextLink;
494 
495  /* Update the next link, if there is one, to be the new head of the list */
496  if (AccessDescPtr->NextLink != CFE_TBL_END_OF_LIST)
497  {
498  CFE_TBL_TaskData.Handles[AccessDescPtr->NextLink].PrevLink = CFE_TBL_END_OF_LIST;
499  }
500  }
501  else /* Access Descriptor is not the head of the list */
502  {
503  /* Set the next link on the previous link to the next link of the link being removed */
504  CFE_TBL_TaskData.Handles[AccessDescPtr->PrevLink].NextLink = AccessDescPtr->NextLink;
505 
506  /* If this link is not the end of the list, then complete two way linkage */
507  /* by setting the next link's previous link to the previous link of the link being removed */
508  if (AccessDescPtr->NextLink != CFE_TBL_END_OF_LIST)
509  {
510  CFE_TBL_TaskData.Handles[AccessDescPtr->NextLink].PrevLink = AccessDescPtr->PrevLink;
511  }
512  }
513 
514  /* Return the Access Descriptor to the pool */
515  AccessDescPtr->UsedFlag = FALSE;
516 
517  /* If this was the last Access Descriptor for this table, we can free the memory buffers as well */
518  if (RegRecPtr->HeadOfAccessList == CFE_TBL_END_OF_LIST)
519  {
520  /* Only free memory that we have allocated. If the image is User Defined, then don't bother */
521  if (RegRecPtr->UserDefAddr == FALSE)
522  {
523  /* Free memory allocated to buffers */
524  Status = CFE_ES_PutPoolBuf(CFE_TBL_TaskData.Buf.PoolHdl, (uint32 *)RegRecPtr->Buffers[0].BufferPtr);
525  RegRecPtr->Buffers[0].BufferPtr = NULL;
526 
527  if (Status < 0)
528  {
529  CFE_ES_WriteToSysLog("CFE_TBL:RemoveAccessLink-PutPoolBuf[0] Fail Stat=0x%08X, Hndl=0x%08lX, Buf=0x%08lX\n",
530  (unsigned int)Status, (unsigned long)CFE_TBL_TaskData.Buf.PoolHdl, (unsigned long)RegRecPtr->Buffers[0].BufferPtr);
531  }
532 
533  /* If a double buffered table, then free the second buffer as well */
534  if (RegRecPtr->DblBuffered)
535  {
536  Status = CFE_ES_PutPoolBuf(CFE_TBL_TaskData.Buf.PoolHdl, (uint32 *)RegRecPtr->Buffers[1].BufferPtr);
537  RegRecPtr->Buffers[1].BufferPtr = NULL;
538 
539  if (Status < 0)
540  {
541  CFE_ES_WriteToSysLog("CFE_TBL:RemoveAccessLink-PutPoolBuf[1] Fail Stat=0x%08X, Hndl=0x%08lX, Buf=0x%08lX\n",
542  (unsigned int)Status, (unsigned long)CFE_TBL_TaskData.Buf.PoolHdl, (unsigned long)RegRecPtr->Buffers[1].BufferPtr);
543  }
544  }
545  else
546  {
547  /* If a shared buffer has been allocated to the table, then release it as well */
548  if (RegRecPtr->LoadInProgress != CFE_TBL_NO_LOAD_IN_PROGRESS)

832 **
833 ** NOTE: For complete prolog information, see 'cfe_tbl_internal.h'
834 ********************************************************************/
835 
836 int32 CFE_TBL_GetWorkingBuffer(CFE_TBL_LoadBuff_t **WorkingBufferPtr,
837  CFE_TBL_RegistryRec_t *RegRecPtr,
838  boolean CalledByApp)
839 {
840  int32 Status = CFE_SUCCESS;
841  int32 i;
842  int32 InactiveBufferIndex;
843  CFE_TBL_Handle_t AccessIterator;
844 
845  /* Initialize return pointer to NULL */
846  *WorkingBufferPtr = NULL;
847 
848  /* If a load is already in progress, return the previously allocated working buffer */
849  if (RegRecPtr->LoadInProgress != CFE_TBL_NO_LOAD_IN_PROGRESS)
850  {
851  if (RegRecPtr->DblBuffered)
852  {
853  *WorkingBufferPtr = &RegRecPtr->Buffers[(1U-RegRecPtr->ActiveBufferIndex)];
854  }
855  else
856  {
857  *WorkingBufferPtr = &CFE_TBL_TaskData.LoadBuffs[RegRecPtr->LoadInProgress];
858  }
859  }
860  else
861  {
862  /* If the table is uninitialized and the function is called by an application (rather than */
863  /* by the Table Services application), then use the current active buffer as the working buffer. */
864  /* This allows many tasks with many tables to perform the initialization without conflict */
865  /* over the accessibility of the shared working buffers. */
866  if ((RegRecPtr->TableLoadedOnce == FALSE) && (CalledByApp == TRUE))
867  {
868  if (RegRecPtr->DblBuffered)
869  {
870  *WorkingBufferPtr = &RegRecPtr->Buffers[RegRecPtr->ActiveBufferIndex];
871  }
872  else
873  {
874  *WorkingBufferPtr = &RegRecPtr->Buffers[0];
875  }
876  }
877  else
878  {
879  /* If the table is a double buffered table, then check to make sure the */
880  /* inactive buffer has been freed by any Applications that may have been using it */
881  if (RegRecPtr->DblBuffered)
882  {
883  /* Determine the index of the Inactive Buffer Pointer */
884  InactiveBufferIndex = 1 - RegRecPtr->ActiveBufferIndex;
885 
886  /* Scan the access descriptor table to determine if anyone is still using the inactive buffer */
887  AccessIterator = RegRecPtr->HeadOfAccessList;
888  while ((AccessIterator != CFE_TBL_END_OF_LIST) && (Status == CFE_SUCCESS))
889  {
890  if ((CFE_TBL_TaskData.Handles[AccessIterator].BufferIndex == InactiveBufferIndex) &&
891  (CFE_TBL_TaskData.Handles[AccessIterator].LockFlag))
892  {
893  Status = CFE_TBL_ERR_NO_BUFFER_AVAIL;
894 
895  CFE_ES_WriteToSysLog("CFE_TBL:GetWorkingBuffer-Inactive Dbl Buff Locked for '%s' by AppId=%d\n",
896  RegRecPtr->Name, (int)CFE_TBL_TaskData.Handles[AccessIterator].AppId);
897  }
898 
899  /* Move to next access descriptor in linked list */
900  AccessIterator = CFE_TBL_TaskData.Handles[AccessIterator].NextLink;
901  }
902 
903  /* If buffer is free, then return the pointer to it */
904  if (Status == CFE_SUCCESS)
905  {
906  *WorkingBufferPtr = &RegRecPtr->Buffers[InactiveBufferIndex];
907  RegRecPtr->LoadInProgress = InactiveBufferIndex;
908  }
909  }
910  else /* Single Buffered Table */
911  {
912  /* Take Mutex to make sure we are not trying to grab a working buffer that some */
913  /* other application is also trying to grab. */
914  Status = OS_MutSemTake(CFE_TBL_TaskData.WorkBufMutex);
915 
916  /* Make note of any errors but continue and hope for the best */
917  if (Status != OS_SUCCESS)
918  {
919  CFE_ES_WriteToSysLog("CFE_TBL:GetWorkBuf-Internal error taking WorkBuf Mutex (Status=0x%08X)\n",
920  (unsigned int)Status);
921  }
922 
923  /* Determine if there are any common buffers available */
924  i = 0;
925  while ((i < CFE_TBL_MAX_SIMULTANEOUS_LOADS) && (CFE_TBL_TaskData.LoadBuffs[i].Taken == TRUE))
926  {
927  i++;
928  }
929 
930  /* If a free buffer was found, then return the address to the associated shared buffer */
931  if (i < CFE_TBL_MAX_SIMULTANEOUS_LOADS)
932  {
933  CFE_TBL_TaskData.LoadBuffs[i].Taken = TRUE;
934  *WorkingBufferPtr = &CFE_TBL_TaskData.LoadBuffs[i];
935  RegRecPtr->LoadInProgress = i;
936 
937  /* Translate OS_SUCCESS into CFE_SUCCESS */
938  Status = CFE_SUCCESS;
939  }
940  else
941  {
942  Status = CFE_TBL_ERR_NO_BUFFER_AVAIL;
943 
944  CFE_ES_WriteToSysLog("CFE_TBL:GetWorkingBuffer-All shared buffers are locked\n");
945  }
946 
947  /* Allow others to obtain a shared working buffer */
948  OS_MutSemGive(CFE_TBL_TaskData.WorkBufMutex);
949  }
950 
951  if ((*WorkingBufferPtr) != NULL)
952  {
953  /* In case the file contains a partial table load, get the active buffer contents first */
954  CFE_PSP_MemCpy((*WorkingBufferPtr)->BufferPtr,
955  RegRecPtr->Buffers[RegRecPtr->ActiveBufferIndex].BufferPtr,
956  RegRecPtr->Size);
957  }
958  }
959  }
960 
961 
962  return Status;
963 
964 } /* End of CFE_TBL_GetWorkingBuffer() */

967 /*******************************************************************

969 ** CFE_TBL_LoadFromFile
970 **
971 ** NOTE: For complete prolog information, see 'cfe_tbl_internal.h'

973 
974 int32 CFE_TBL_LoadFromFile(CFE_TBL_LoadBuff_t *WorkingBufferPtr,
975  CFE_TBL_RegistryRec_t *RegRecPtr,
976  const char *Filename)
977 {
978  int32 Status = CFE_SUCCESS;
979  CFE_FS_Header_t StdFileHeader;
980  CFE_TBL_File_Hdr_t TblFileHeader;
981  int32 FileDescriptor;
982  size_t FilenameLen = strlen(Filename);
983  uint32 NumBytes;
984  uint8 ExtraByte;
985 
986  if (FilenameLen > (OS_MAX_PATH_LEN-1))
987  {
988  Status = CFE_TBL_ERR_FILENAME_TOO_LONG;
989  }
990  else
991  {
992  /* Try to open the specified table file */
993  FileDescriptor = OS_open(Filename, OS_READ_ONLY, 0);
994 
995  if (FileDescriptor >= 0)
996  {
997  Status = CFE_TBL_ReadHeaders(FileDescriptor, &StdFileHeader, &TblFileHeader, Filename);
998 
999  if (Status == CFE_SUCCESS)
1000  {
1001  /* Verify that the specified file has compatible data for specified table */
1002  if (strcmp(RegRecPtr->Name, TblFileHeader.TableName) == 0)
1003  {
1004  if ((TblFileHeader.Offset + TblFileHeader.NumBytes) > RegRecPtr->Size)
1005  {
1006  Status = CFE_TBL_ERR_FILE_TOO_LARGE;
1007  }
1008  else
1009  {
1010  /* Any Table load that starts beyond the first byte is a "partial load" */
1011  /* But a file that starts with the first byte and ends before filling */
1012  /* the whole table is just considered "short". */
1013  if (TblFileHeader.Offset > 0)
1014  {
1015  Status = CFE_TBL_WARN_PARTIAL_LOAD;
1016  }
1017  else if (TblFileHeader.NumBytes < RegRecPtr->Size)
1018  {
1019  Status = CFE_TBL_WARN_SHORT_FILE;
1020  }
1021 
1022  NumBytes = OS_read(FileDescriptor,
1023  &WorkingBufferPtr->BufferPtr[TblFileHeader.Offset],
1024  TblFileHeader.NumBytes);
1025 
1026  if (NumBytes != TblFileHeader.NumBytes)
1027  {
1028  Status = CFE_TBL_ERR_LOAD_INCOMPLETE;
1029  }
1030 
1031  /* Check to see if the file is too large (ie - more data than header claims) */
1032  NumBytes = OS_read(FileDescriptor, &ExtraByte, 1);
1033 
1034  /* If successfully read another byte, then file must have too much data */
1035  if (NumBytes == 1)
1036  {
1037  Status = CFE_TBL_ERR_FILE_TOO_LARGE;
1038  }
1039 
1040  CFE_PSP_MemSet(WorkingBufferPtr->DataSource, 0, OS_MAX_PATH_LEN);
1041  strncpy(WorkingBufferPtr->DataSource, Filename, OS_MAX_PATH_LEN);
1042 
1043  /* Save file creation time for later storage into Registry */
1044  WorkingBufferPtr->FileCreateTimeSecs = StdFileHeader.TimeSeconds;
1045  WorkingBufferPtr->FileCreateTimeSubSecs = StdFileHeader.TimeSubSeconds;
1046 
1047  /* Compute the CRC on the specified table buffer */
1048  WorkingBufferPtr->Crc = CFE_ES_CalculateCRC(WorkingBufferPtr->BufferPtr,
1049  RegRecPtr->Size,
1050  0,
1051  CFE_ES_DEFAULT_CRC);
1052  }
1053  }
1054  else
1055  {
1056  Status = CFE_TBL_ERR_FILE_FOR_WRONG_TABLE;
1057  }
1058  }
1059 
1060  OS_close(FileDescriptor);
1061  }
1062  else
1063  {
1064  /* Return error code obtained from OS_open */
1065  Status = FileDescriptor;
1066  }
1067  }
1068 
1069  return Status;
1070 }

1160  CFE_TBL_NotifyTblUsersOfUpdate(RegRecPtr);
