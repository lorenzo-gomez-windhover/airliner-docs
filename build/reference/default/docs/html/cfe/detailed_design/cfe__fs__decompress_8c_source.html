192 {
193  uint8 flags; /* compression flags */
194  int8 magic[2]; /* magic header */
195  int8 thisByte; /* temporary holder for current byte in a string read */
196  uint32 stamp; /* time stamp */
197  boolean keep_going;
198 
199  /* read input buffer (check for read error) */
200  magic[0] = (int8)NEXTBYTE();
201  if( State->Error != CFE_SUCCESS ) return CFE_FS_GZIP_READ_ERROR_HEADER;
202 
203  magic[1] = (int8)NEXTBYTE();
204  if ( State->Error != CFE_SUCCESS ) return CFE_FS_GZIP_READ_ERROR_HEADER;
205 
206  if ( memcmp( magic, GZIP_MAGIC, 2 ) == 0 || memcmp( magic, OLD_GZIP_MAGIC, 2 ) == 0 )
207  {
208 
209  NEXTBYTE();
210  if ( State->Error != CFE_SUCCESS ) return CFE_FS_GZIP_READ_ERROR_HEADER;
211 
212  flags = NEXTBYTE();
213  if ( State->Error != CFE_SUCCESS ) return CFE_FS_GZIP_READ_ERROR_HEADER;
214 
215  stamp = (uint32)NEXTBYTE();
216  if( State->Error != CFE_SUCCESS ) return CFE_FS_GZIP_READ_ERROR_HEADER;
217  stamp |= (uint32)NEXTBYTE() << 8;
218  if( State->Error != CFE_SUCCESS ) return CFE_FS_GZIP_READ_ERROR_HEADER;
219  stamp |= (uint32)NEXTBYTE() << 16;
220  if( State->Error != CFE_SUCCESS ) return CFE_FS_GZIP_READ_ERROR_HEADER;
221  stamp |= (uint32)NEXTBYTE() << 24;
222  if( State->Error != CFE_SUCCESS ) return CFE_FS_GZIP_READ_ERROR_HEADER;
223 
224  NEXTBYTE(); /* Ignore extra flags for the moment */
225  if( State->Error != CFE_SUCCESS ) return CFE_FS_GZIP_READ_ERROR_HEADER;
226  NEXTBYTE(); /* Ignore OS type for the moment */
227  if( State->Error != CFE_SUCCESS ) return CFE_FS_GZIP_READ_ERROR_HEADER;
228 
229  if ( (flags & CONTINUATION) != 0 ) {
230  NEXTBYTE();
231  if( State->Error != CFE_SUCCESS ) return CFE_FS_GZIP_READ_ERROR_HEADER;
232  NEXTBYTE();
233  if( State->Error != CFE_SUCCESS ) return CFE_FS_GZIP_READ_ERROR_HEADER;
234  }
235 
236  if ( (flags & EXTRA_FIELD) != 0 ) {
237  uint32 len;
238  len = NEXTBYTE();
239  if( State->Error != CFE_SUCCESS ) return CFE_FS_GZIP_READ_ERROR_HEADER;
240  len |= NEXTBYTE() << 8;
241  if( State->Error != CFE_SUCCESS ) return CFE_FS_GZIP_READ_ERROR_HEADER;
242  while (len--) {
243  NEXTBYTE();
244  if( State->Error != CFE_SUCCESS ) return CFE_FS_GZIP_READ_ERROR_HEADER;
245  }
246  }
247 
248  /* Get original file name if it was truncated */
249  if ( (flags & ORIG_NAME) != 0 ) {
250  keep_going = TRUE;
251  while ( keep_going ) {
252 
253  /* Get the next byte */
254  thisByte = NEXTBYTE();
255 
256  /* Header failure when end of file is reached or a read failure occurs */
257  if ((thisByte == EOF) || (State->Error != CFE_SUCCESS)) return CFE_FS_GZIP_READ_ERROR_HEADER;
258 
259  /* End of string was found */
260  if (thisByte == 0) keep_going = FALSE;
261  }
262  }
263 
264  /* Discard file comment if any */
265  if ( (flags & COMMENT) != 0 ) {
266  keep_going = TRUE;
267  while ( keep_going ) {
268 
269  /* Get the next byte */
270  thisByte = NEXTBYTE();
271 
272  /* Header failure when end of file is reached or a read failure occurs */
273  if ((thisByte == EOF) || (State->Error != CFE_SUCCESS)) return CFE_FS_GZIP_READ_ERROR_HEADER;
274 
275  /* End of string was found */
276  if (thisByte == 0) keep_going = FALSE;
277  }
278  }
279  }
280  else
281  {
282  State->Error = CFE_FS_GZIP_NON_ZIP_FILE;
283  }
284 
285  return State->Error;

358 /* tables to decode that set of codes. Return zero on success, one if the */
359 /* given code set is incomplete (the tables are still built in this case), */
360 /* two if the input is invalid (all zero length codes or an oversubscribed */
361 /* set of lengths), and three if not enough memory. */

363 /* INPUT: */
364 /* uint32 *b; */ /* code lengths in bits (all assumed <= BMAX) */
365 /* uint32 n; */ /* number of codes (assumed <= N_MAX) */
366 /* uint32 s; */ /* number of simple-valued codes (0..s-1) */
367 /* uint16 *d; */ /* list of base values for non-simple codes */
368 /* uint16 *e; */ /* list of extra bits for non-simple codes */

371 int32 FS_gz_huft_build_Reentrant( CFE_FS_Decompress_State_t *State, uint32 * b, uint32 n, uint32 s, uint16 * d, uint16 * e, int32 * m )
372 {
373 
374  uint32 a; /* counter for codes of length k */
375  uint32 c[BMAX + 1]; /* bit length count table */
376  uint32 f; /* i repeats in table every f entries */
377  int32 g; /* maximum code length */
378  int32 h; /* table level */
379  register uint32 i; /* counter, current code */
380  register uint32 j; /* counter */
381  register int32 k; /* number of bits in current code */
382  int32 l; /* bits per table (returned in m) */
383  uint32 lu; /* l unsigned */
384  register uint32 *p; /* pointer into c[], b[], or v[] */
385  register HufTable *q; /* points to current table */
386  HufTable r; /* table entry for structure assignment */
387  HufTable *u[BMAX]; /* table stack */
388  uint32 v[N_MAX]; /* values in order of bit length */
389  register int32 w; /* bits before this table == (l * h) */
390  uint32 x[BMAX + 1]; /* bit offsets, then code stack */
391  uint32 *xp; /* pointer into x */
392  int32 y; /* number of dummy codes added */
393  uint32 z; /* number of entries in current table */
394 
395  uint32 huft_index_q;
396  uint32 huft_index_t = 0;
397  boolean not_first_table = FALSE;
398  huft_index_t = 0;
399 
400  /* Generate counts for each bit length */
401  CFE_PSP_MemSet( (void*)(c), 0, sizeof(c) );
402  r.b=0;
403  r.e=0;
404  r.v.n=0;
405  r.v.t=0;
406  p = b;
407  i = n;
408  do
409  {
410  c[*p]++;
411  p++;
412  } while (--i);
413 
414  if (c[0] == n)
415  { /* null input--all zero length codes */
416  *m = 0;
417  return CFE_SUCCESS;
418  }
419 
420 
421  /* Find minimum and maximum length, bound *m by those */
422  l = *m;
423  for ( j = 1; j <= BMAX; j++ ) if ( c[j] ) break;
424  k = j;
425  lu = (uint32)l; /* minimum code length */
426  if ( lu < j ) { l = j; lu = (uint32)l; }
427  for ( i = BMAX; i; i-- ) if (c[i]) break;
428  g = i; /* maximum code length */
429  if ( lu > i ) { l = i; lu = (uint32)l; }
430  *m = l;
431 
432 
433  /* Adjust last length count to fill out codes, if needed */
434  for ( y = 1 << j; j < i; j++, y <<= 1 )
435  if ( ( y -= c[j] ) < 0 ) return CFE_FS_GZIP_BAD_CODE_BLOCK; /* bad input: more codes than bits */
436  if ( (y -= c[i] ) < 0 ) return CFE_FS_GZIP_BAD_CODE_BLOCK;
437  c[i] += y;
438 
439 
440  /* Generate starting offsets into the value table for each length */
441  x[1] = j = 0;
442  p = c + 1;
443  xp = x + 2;
444  while (--i) { /* note that i == g from above */
445  j += *p++;
446  *xp++ = j;
447  }
448 
449 
450  /* Make a table of values in order of bit lengths */
451  p = b;
452  i = 0;
453  do {
454  if ( (j = *p++) != 0 ) v[x[j]++] = i;
455  } while ( ++i < n );
456 
457 
458  /* Generate the Huffman codes and for each, make the table entries */
459  x[0] = i = 0; /* first Huffman code is zero */
460  p = v; /* grab values in bit order */
461  h = -1; /* no tables yet--level -1 */
462  w = -l; /* bits decoded == (l * h) */
463  u[0] = (HufTable *)NULL; /* just to keep compilers happy */
464  q = (HufTable *)NULL; /* ditto */
465  z = 0; /* ditto */
466 
467 
468  /* go through the bit lengths (k already is bits in shortest code) */
469  for (; k <= g; k++) {
470  a = c[k];
471  while (a--) {
472 
473  /* here i is the Huffman code of length k bits for value *p */
474  /* make tables up to required level */
475  while (k > w + l) {
476  h++;
477  w += l; /* previous table always l bits */
478 
479  /* compute minimum size table less than or equal to l bits */
480  z = g - w;
481  z = ( z > lu ? l : z ); /* upper limit on table size */
482  j = k - w;
483  f = 1<<j;
484  if ( f > a + 1 ) { /* try a k-w bit table */
485 
486  /* too few codes for k-w bit table */
487  f -= a + 1; /* deduct codes from patterns left */
488  xp = c + k;
489  while ( ++j < z ) { /* try smaller tables up to z bits */
490  f <<= 1;
491  if ( f <= *++xp ) break; /* enough codes to use up j bits */
492  f -= *xp; /* else deduct codes from patterns */
493  }
494 
495  }
496  z = 1 << j; /* table entries for j-bit table */
497 
498  /* if space remains in the Huffman table memory array, "allocate" and link in new table */
499  if ( State->hufts + (z + 1) > MAX_HUF_TABLES ) return CFE_FS_GZIP_NO_MEMORY;
500 
501  q = &State->hufTable[State->hufts];
502  huft_index_q = State->hufts;
503  State->hufts += z + 1;
504  if ( State->hufts > State->max_hufts ) State->max_hufts = State->hufts;
505 
506  if ( not_first_table ) State->hufTable[huft_index_t].v.t = huft_index_q + 1;
507  else not_first_table = TRUE;
508 
509  huft_index_t = huft_index_q;
510  State->hufTable[huft_index_t].v.t = 0xffff; /* stand-in for NULL */
511 
512  u[h] = ++q; /* table starts after link */
513  huft_index_q++;
514 
515  /* connect to last table, if there is one */
516  if ( h ) {
517  x[h] = i; /* save pattern for backing up */
518  r.b = (uint8)l; /* bits to dump before this table */
519  r.e = (uint8)(16 + j); /* bits in this table */
520  /* r.t = q; */ /* pointer to this table */
521  r.v.t = huft_index_q; /* pointer to this table */
522  j = i >> (w - l); /* (get around Turbo C bug) */
523  u[h - 1][j] = r; /* connect to last table */
524  }
525 
526  }
527 
528  /* set up table entry in r */
529  r.b = (uint8)(k - w);
530  if (p >= v + n) {
531  r.e = 99; /* out of values--invalid code */
532  } else if ( *p < s ) {
533  r.e = (uint8)( *p < 256 ? 16 : 15 ); /* 256 is end-of-block code */
534  r.v.n = *p; /* simple code is just the value */
535  p++; /* one compiler does not like *p++ */
536  } else {
537  r.e = (uint8)( e[*p - s] ); /* non-simple--look up in lists */
538  r.v.n = d[*p++ - s];
539  }
540 
541  /* fill code-like entries with r */
542  f = 1 << (k - w);
543  j = i >> w;
544 
545  if( (q == (HufTable *)NULL) && (j < z) )
546  {
547  return CFE_FS_GZIP_BAD_CODE_BLOCK;
548  }
549  else
550  {
551  for ( j = i >> w; j < z; j += f ) q[j] = r;
552  }
553 
554  /* backwards increment the k-bit code i */
555  for ( j = 1 << (k - 1); i & j; j >>= 1 ) i ^= j;
556  i ^= j;
557 
558  /* backup over finished tables */
559  while (( h >= 0 ) && ( ( i & ((1 << w) - 1) ) != x[h] )) {
560  h--; /* don't need to update q */
561  w -= l;
562  }
563 
564  }
565 
566  }
567 
568  /* Return true (1) if we were given an incomplete table */
569  if((y != 0 && g != 1))
570  {
571  return CFE_FS_GZIP_BAD_DATA;
572  }
573 
574  return CFE_SUCCESS;
575 
576 }
