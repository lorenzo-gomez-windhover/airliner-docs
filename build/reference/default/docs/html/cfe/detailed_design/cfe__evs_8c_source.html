85 ** Function Prologue
86 **
87 ** Function Name: CFE_EVS_Register
88 **
89 ** Purpose: This routine registers an application for receiving event services.
90 **
91 ** Assumptions and Notes:
92 **
93 */
94 int32 CFE_EVS_Register (void *Filters, uint16 NumEventFilters, uint16 FilterScheme)
95 {
96  uint16 FilterLimit;
97  uint16 i;
98  int32 Status = CFE_SUCCESS;
99  uint32 AppID = CFE_EVS_UNDEF_APPID;
100  CFE_EVS_BinFilter_t *AppFilters;
101  EVS_AppData_t *AppDataPtr;
102 
103  /* Query and verify the caller's AppID */
104  Status = EVS_GetAppID(&AppID);
105 
106  if (Status == CFE_SUCCESS)
107  {
108  /* Cleanup if already registered */
109  if (CFE_EVS_GlobalData.AppData[AppID].RegisterFlag == TRUE)
110  {
111  CFE_PSP_MemSet(&CFE_EVS_GlobalData.AppData[AppID], 0, sizeof(EVS_AppData_t));
112  }
113 
114  /* Verify filter arguments */
115  if (FilterScheme != CFE_EVS_BINARY_FILTER)
116  {
117  Status = CFE_EVS_UNKNOWN_FILTER;
118  }
119  else if ((NumEventFilters != 0) && (Filters == NULL))
120  {
121  Status = CFE_ES_ERR_BUFFER;
122  }
123  else
124  {
125  /* Initialize application event data */
126  AppDataPtr = &CFE_EVS_GlobalData.AppData[AppID];
127  AppDataPtr->RegisterFlag = TRUE;
128  AppDataPtr->ActiveFlag = TRUE;
129  AppDataPtr->EventCount = 0;
130  AppDataPtr->EventTypesActiveFlag = CFE_EVS_DEFAULT_TYPE_FLAG;
131 
132  /* Set limit for number of provided filters */
133  if (NumEventFilters < CFE_EVS_MAX_EVENT_FILTERS)
134  {
135  FilterLimit = NumEventFilters;
136  }
137  else
138  {
139  FilterLimit = CFE_EVS_MAX_EVENT_FILTERS;
140  }
141 
142  if (Filters != NULL)
143  {
144  AppFilters = (CFE_EVS_BinFilter_t *) Filters;
145 
146  /* Copy provided filters */
147  for (i = 0; i < FilterLimit; i++)
148  {
149  AppDataPtr->BinFilters[i].EventID = AppFilters[i].EventID;
150  AppDataPtr->BinFilters[i].Mask = AppFilters[i].Mask;
151  AppDataPtr->BinFilters[i].Count = 0;
152  }
153  }
154 
155  /* Initialize remainder of filters as unused */
156  for (i = FilterLimit; i < CFE_EVS_MAX_EVENT_FILTERS; i++)
157  {
158  AppDataPtr->BinFilters[i].EventID = CFE_EVS_FREE_SLOT;
159  AppDataPtr->BinFilters[i].Mask = 0;
160  AppDataPtr->BinFilters[i].Count = 0;
161  }
162  }
163  }
164 
165  return(Status);
166 
167 } /* End CFE_EVS_Register */

202 /*
203 ** Function Prologue
204 **
205 ** Function Name: CFE_EVS_SendEvent
206 **
207 ** Purpose: This routine sends an event message, with message ID containing the current
208 ** processor/box and task ID as well as the eventID.
209 **
210 ** Assumptions and Notes:
211 */
212 int32 CFE_EVS_SendEvent (uint16 EventID, uint16 EventType, const char *Spec, ... )
213 {
214  CFE_EVS_Packet_t EVS_Packet;
215  int32 ExpandedLength;
216  int32 Status = CFE_SUCCESS;
217  uint32 AppID = CFE_EVS_UNDEF_APPID;
218  CFE_TIME_SysTime_t Time;
219  va_list Ptr;
220 
221  /* Query and verify the caller's AppID */
222  Status = EVS_GetAppID(&AppID);
223 
224  if (Status == CFE_SUCCESS)
225  {
226  if (CFE_EVS_GlobalData.AppData[AppID].RegisterFlag == FALSE)
227  {
228  /* Handler for events from apps not registered with EVS */
229  Status = EVS_NotRegistered(AppID);
230  }
231  else if (EVS_IsFiltered(AppID, EventID, EventType) == FALSE)
232  {
233  /* Initialize EVS event packet */
234  CFE_SB_InitMsg(&EVS_Packet, CFE_EVS_EVENT_MSG_MID, sizeof(CFE_EVS_Packet_t), TRUE);
235  EVS_Packet.Payload.PacketID.EventID = EventID;
236  EVS_Packet.Payload.PacketID.EventType = EventType;
237 
238  /* vsnprintf() returns the total expanded length of the formatted string */
239  /* vsnprintf() copies and zero terminates portion that fits in the buffer */
240  va_start(Ptr, Spec);
241  ExpandedLength = vsnprintf((char *)EVS_Packet.Payload.Message, sizeof(EVS_Packet.Payload.Message), Spec, Ptr);
242  va_end(Ptr);
243 
244  /* Were any characters truncated in the buffer? */
245  if (ExpandedLength >= sizeof(EVS_Packet.Payload.Message))
246  {
247  /* Mark character before zero terminator to indicate truncation */
248  EVS_Packet.Payload.Message[sizeof(EVS_Packet.Payload.Message) - 2] = CFE_EVS_MSG_TRUNCATED;
249  CFE_EVS_GlobalData.EVS_TlmPkt.Payload.MessageTruncCounter++;
250  }
251 
252  /* Get current spacecraft time */
253  Time = CFE_TIME_GetTime();
254 
255  /* Send the event packet */
256  EVS_SendPacket(AppID, Time, &EVS_Packet);
257  }
258  }
259 
260  return(Status);
261 
262 } /* End CFE_EVS_SendEvent */
