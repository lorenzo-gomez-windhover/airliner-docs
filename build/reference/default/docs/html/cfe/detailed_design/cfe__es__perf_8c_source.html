301 /* Function: CFE_ES_PerfLogDump() */
302 /* */
303 /* Purpose: */
304 /* Write performance data to a file */
305 /* */
306 /* Arguments: */
307 /* None */
308 /* */
309 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
310 void CFE_ES_PerfLogDump(void){
311 
312  int32 WriteStat;
313  uint32 i;
314  uint32 FileSize;
315  CFE_FS_Header_t FileHdr;
316 
317  CFE_ES_RegisterChildTask();
318 
319 
320  /* Zero cFE header, then fill in fields */
321  CFE_FS_InitHeader(&FileHdr, CFE_ES_PERF_LOG_DESC, CFE_FS_ES_PERFDATA_SUBTYPE);
322 
323  /* write the cFE header to the file */
324  WriteStat = CFE_FS_WriteHeader( CFE_ES_PerfLogDumpStatus.DataFileDescriptor, &FileHdr);
325  if(WriteStat != sizeof(CFE_FS_Header_t))
326  {
327  CFE_ES_FileWriteByteCntErr(&CFE_ES_PerfLogDumpStatus.DataFileName[0],
328  sizeof(CFE_FS_Header_t),WriteStat);
329 
330  OS_close(CFE_ES_PerfLogDumpStatus.DataFileDescriptor);
331  CFE_ES_ExitChildTask();
332  /* normally ExitChildTask() does not return, but it DOES under UT */
333  return;
334  }/* end if */
335  FileSize = WriteStat;
336 
337  /* write the performance metadata to the file */
338  WriteStat = OS_write(CFE_ES_PerfLogDumpStatus.DataFileDescriptor,&Perf->MetaData,sizeof(CFE_ES_PerfMetaData_t));
339  if(WriteStat != sizeof(CFE_ES_PerfMetaData_t))
340  {
341  CFE_ES_FileWriteByteCntErr(&CFE_ES_PerfLogDumpStatus.DataFileName[0],
342  sizeof(CFE_ES_PerfMetaData_t),WriteStat);
343  OS_close(CFE_ES_PerfLogDumpStatus.DataFileDescriptor);
344  CFE_ES_ExitChildTask();
345  /* normally ExitChildTask() does not return, but it DOES under UT */
346  return;
347  }/* end if */
348  FileSize += WriteStat;
349 
350  CFE_ES_PerfLogDumpStatus.DataToWrite = Perf->MetaData.DataCount;
351 
352  /* write the collected data to the file */
353  for(i=0; i < Perf->MetaData.DataCount; i++){
354  WriteStat = OS_write (CFE_ES_PerfLogDumpStatus.DataFileDescriptor, &Perf->DataBuffer[i], sizeof(CFE_ES_PerfDataEntry_t));
355  if(WriteStat != sizeof(CFE_ES_PerfDataEntry_t))
356  {
357  CFE_ES_FileWriteByteCntErr(&CFE_ES_PerfLogDumpStatus.DataFileName[0],
358  sizeof(CFE_ES_PerfDataEntry_t),WriteStat);
359  OS_close(CFE_ES_PerfLogDumpStatus.DataFileDescriptor);
360  /* Reset the DataToWrite variable, so a new file can be written */
361  CFE_ES_PerfLogDumpStatus.DataToWrite = 0;
362  CFE_ES_ExitChildTask();
363  /* normally ExitChildTask() does not return, but it DOES under UT */
364  return;
365  }/* end if */
366  FileSize += WriteStat;
367  CFE_ES_PerfLogDumpStatus.DataToWrite--;
368  if((i % CFE_ES_PERF_ENTRIES_BTWN_DLYS) == 0){
369  OS_TaskDelay(CFE_ES_PERF_CHILD_MS_DELAY);
370  }/* end if */
371 
372  }/* end for */
373 
374  OS_close(CFE_ES_PerfLogDumpStatus.DataFileDescriptor);
375 
376  CFE_EVS_SendEvent(CFE_ES_PERF_DATAWRITTEN_EID,CFE_EVS_DEBUG,
377  "%s written:Size=%d,EntryCount=%d",
378  &CFE_ES_PerfLogDumpStatus.DataFileName[0],(int)FileSize,
379  (int)Perf->MetaData.DataCount);
380 
381  CFE_ES_ExitChildTask();
382 
383 }/* end CFE_ES_PerfLogDump */

459 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
460 /* Name: CFE_ES_PerfLogAdd */
461 /* */
462 /* Purpose: This function adds a new entry to the data buffer. */
463 /* */
464 /* Assumptions and Notes: */
465 /* */
466 /* This function implements a circular buffer using an array. */
467 /* DataStart points to first stored entry */
468 /* DataEnd points to next available entry */
469 /* if DataStart == DataEnd then the buffer is either empty or full */
470 /* depending on the value of the DataCount */
471 /* */
472 /* Time is stored as 2 32 bit integers, (TimerLower32, TimerUpper32): */
473 /* TimerLower32 is the curent value of the hardware timer register. */
474 /* TimerUpper32 is the number of times the timer has rolled over. */
475 /* */
476 /* Time is stored as a absolute time instead of a relative time between log */
477 /* entries. This will yield better accuracy since storing relative time between */
478 /* entries will accumulate (rounding/sampling) errors over time. It also is */
479 /* faster since the time does not need to be calculated. */
480 /* */
481 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
482 void CFE_ES_PerfLogAdd(uint32 Marker, uint32 EntryExit)
483 {
484  int32 IntFlags;
485 
486  if (Perf->MetaData.State != CFE_ES_PERF_IDLE) {
487 
488  /* if marker is out of range... */
489  if(Marker >= CFE_ES_PERF_MAX_IDS){
490 
491  /* if marker has not been reported previously ... */
492  if(Perf->MetaData.InvalidMarkerReported == FALSE){
493  CFE_ES_WriteToSysLog("ES PERF:Invalid performance marker %d,max is %d\n",
494  (unsigned int)Marker,(CFE_ES_PERF_MAX_IDS - 1));
495  Perf->MetaData.InvalidMarkerReported = TRUE;
496  }/* end if */
497 
498  return;
499 
500  }/* end if */
501 
502 
503  /* is this id filtered */
504  if (CFE_ES_TEST_LONG_MASK(Perf->MetaData.FilterMask, Marker)) {
505 
506  /* disable interrupts to guarentee exclusive access to the data structures */
507  IntFlags = OS_IntLock();
508
