150 /*
151 ** Name:
152 ** CFE_ES_StartApplications
153 **
154 ** Purpose:
155 ** This routine loads/starts cFE applications.
156 **
157 */
158 void CFE_ES_StartApplications(uint32 ResetType, const char *StartFilePath )
159 {
160  char ES_AppLoadBuffer[ES_START_BUFF_SIZE]; /* A buffer of for a line in a file */
161  uint32 BuffLen = 0; /* Length of the current buffer */
162  int32 AppFile = 0;
163  char c;
164  int32 ReadStatus;
165  boolean LineTooLong = FALSE;
166  boolean FileOpened = FALSE;
167 
168 #ifdef _ENHANCED_BUILD_
169  CFE_StaticModuleLoadEntry_t *Entry;
170  CFE_ES_AppStaticModuleApi_t *ApiPtr;
171  uint32 StaticId;
172 
173  /* First run through the static application list */
174  Entry = GLOBAL_CONFIGDATA.CfsModuleList;
175  if (Entry != NULL)
176  {
177  while (Entry->Name != NULL)
178  {
179  ApiPtr = (CFE_ES_AppStaticModuleApi_t *)Entry->Api;
180  if (ApiPtr->EntryType == CFE_ES_STATICENTRYTYPE_APPLICATION)
181  {
182  ReadStatus = CFE_ES_AppCreate(&StaticId, NULL,
183  Entry->Api, Entry->Name, ApiPtr->Priority,
184  ApiPtr->StackSize, CFE_ES_APP_EXCEPTION_PROC_RESTART, 0);
185  }
186  else if (ApiPtr->EntryType == CFE_ES_STATICENTRYTYPE_LIBRARY)
187  {
188  ReadStatus = CFE_ES_LoadLibrary(&StaticId, NULL,
189  Entry->Api, Entry->Name);
190  }
191  ++Entry;
192  }
193  }
194 #endif
195 
196  /*
197  ** Get the ES startup script filename.
198  ** If this is a Processor Reset, try to open the file in the volatile disk first.
199  */
200  if ( ResetType == CFE_PSP_RST_TYPE_PROCESSOR )
201  {
202  /*
203  ** Open the file in the volatile disk.
204  */
205  AppFile = OS_open( CFE_ES_VOLATILE_STARTUP_FILE, O_RDONLY, 0);
206 
207  if ( AppFile >= 0 )
208  {
209  CFE_ES_WriteToSysLog ("ES Startup: Opened ES App Startup file: %s\n",
210  CFE_ES_VOLATILE_STARTUP_FILE);
211  FileOpened = TRUE;
212  }
213  else
214  {
215  CFE_ES_WriteToSysLog ("ES Startup: Cannot Open Volatile Startup file, Trying Nonvolatile.\n");
216  FileOpened = FALSE;
217  }
218 
219  } /* end if */
220 
221  /*
222  ** This if block covers two cases: A Power on reset, and a Processor reset when
223  ** the startup file on the volatile file system could not be opened.
224  */
225  if ( FileOpened == FALSE )
226  {
227  /*
228  ** Try to Open the file passed in to the cFE start.
229  */
230  AppFile = OS_open( (const char *)StartFilePath, O_RDONLY, 0);
231 
232  if ( AppFile >= 0 )
233  {
234  CFE_ES_WriteToSysLog ("ES Startup: Opened ES App Startup file: %s\n",StartFilePath);
235  FileOpened = TRUE;
236  }
237  else
238  {
239  CFE_ES_WriteToSysLog ("ES Startup: Error, Can't Open ES App Startup file: %s EC = 0x%08X\n",
240  StartFilePath, (unsigned int)AppFile );
241  FileOpened = FALSE;
242  }
243 
244  }
245 
246  /*
247  ** If the file is opened in either the Nonvolatile or the Volatile disk, process it.
248  */
249  if ( FileOpened == TRUE)
250  {
251  memset(ES_AppLoadBuffer,0x0,ES_START_BUFF_SIZE);
252  BuffLen = 0;
253 
254  /*
255  ** Parse the lines from the file. If it has an error
256  ** or reaches EOF, then abort the loop.
257  */
258  while(1)
259  {
260  ReadStatus = OS_read(AppFile, &c, 1);
261  if ( ReadStatus == OS_FS_ERROR )
262  {
263  CFE_ES_WriteToSysLog ("ES Startup: Error Reading Startup file. EC = 0x%08X\n",(unsigned int)ReadStatus);
264  break;
265  }
266  else if ( ReadStatus == 0 )
267  {
268  /*
269  ** EOF Reached
270  */
271  break;
272  }
273  else if(c != '!')
274  {
275  if ( c <= ' ')
276  {
277  /*
278  ** Skip all white space in the file
279  */
280  ;
281  }
282  else if ( c == ',' )
283  {
284  /*
285  ** replace the field delimiter with a space
286  ** This is used for the sscanf string parsing
287  */
288  if ( BuffLen < ES_START_BUFF_SIZE )
289  {
290  ES_AppLoadBuffer[BuffLen] = ' ';
291  }
292  else
293  {
294  LineTooLong = TRUE;
295  }
296  BuffLen++;
297  }
298  else if ( c != ';' )
299  {
300  /*
301  ** Regular data gets copied in
302  */
303  if ( BuffLen < ES_START_BUFF_SIZE )
304  {
305  ES_AppLoadBuffer[BuffLen] = c;
306  }
307  else
308  {
309  LineTooLong = TRUE;
310  }
311  BuffLen++;
312  }
313  else
314  {
315  if ( LineTooLong == TRUE )
316  {
317  /*
318  ** The was too big for the buffer
319  */
320  CFE_ES_WriteToSysLog ("ES Startup: ES Startup File Line is too long: %u bytes.\n",(unsigned int)BuffLen);
321  LineTooLong = FALSE;
322  }
323  else
324  {
325  /*
326  ** Send the line to the file parser
327  */
328  CFE_ES_ParseFileEntry(ES_AppLoadBuffer);
329  }
330  memset(ES_AppLoadBuffer,0x0,ES_START_BUFF_SIZE);
331  BuffLen = 0;
332  }
333  }
334  else
335  {
336  /*
337  ** break when EOF character '!' is reached
338  */
339  break;
340  }
341  }
342  /*
343  ** close the file
344  */
345  OS_close(AppFile);
346 
347  }
348 }

350 /*
351 **---------------------------------------------------------------------------------------
352 ** Name: CFE_ES_ParseFileEntry
353 **
354 ** Purpose: This function parses the startup file line for an individual
355 ** cFE application.
356 **---------------------------------------------------------------------------------------
357 */
358 int32 CFE_ES_ParseFileEntry(const char *FileEntry)
359 {
360  char FileName[OS_MAX_PATH_LEN];
361  char AppName[OS_MAX_API_NAME];
362  char EntryPoint[OS_MAX_API_NAME];
363  char EntryType[OS_MAX_API_NAME];
364  int ScanfStatus;
365  unsigned int Priority;
366  unsigned int StackSize;
367  unsigned int Unused;
368  unsigned int ExceptionAction;
369  unsigned int Flags;
370  uint32 ApplicationId;
371  int32 CreateStatus = CFE_ES_ERR_APP_CREATE;
372 
373  /*
374  ** Use sscanf to parse the string.
375  ** When this routine is called, the buffer is already verified to be less than the
376  ** maximum size, and the parameters have been separated by a single space each.
377  ** If the sscanf fails, then it is a malformed file entry.
378  **
379  ** CFS CCB is tracking this issue separately, and we no longer need
380  ** warnings from cppcheck about this call.
381  */
382  /* cppcheck-suppress invalidscanf */
383  ScanfStatus = sscanf(FileEntry,"%s %s %s %s %u %u %x %u %u",
384  EntryType, FileName, EntryPoint, AppName,
385  &Priority, &StackSize, &Unused, &ExceptionAction,
386  &Flags);
387 
388  /*
389  ** Check to see if the correct number of items were parsed
390  */
391  if ( ScanfStatus != 9 )
392  {
393  CFE_ES_WriteToSysLog("ES Startup: Invalid ES Startup file entry: %d\n",ScanfStatus);
394  return (CreateStatus);
395  }
396 
397  if(strncmp(EntryType,"CFE_APP",OS_MAX_API_NAME)==0)
398  {
399  CFE_ES_WriteToSysLog("ES Startup: Loading file: %s, APP: %s\n",
400  FileName, AppName);
401 
402  /*
403  ** Validate Some parameters
404  ** Exception action should be 0 ( Restart App ) or
405  ** 1 ( Processor reset ). If it's non-zero, assume it means
406  ** reset CPU.
407  */
408  if ( ExceptionAction > CFE_ES_APP_EXCEPTION_RESTART_APP )
409  ExceptionAction = CFE_ES_APP_EXCEPTION_PROC_RESTART;
410  /*
411  ** Now create the application
412  */
413  CreateStatus = CFE_ES_AppCreate(&ApplicationId, FileName,
414  EntryPoint, AppName, (uint32) Priority,
415  (uint32) StackSize, (uint32) ExceptionAction,
416  (uint32) Flags );
417  }
418  else if(strncmp(EntryType,"CFE_LIB",OS_MAX_API_NAME)==0)
419  {
420  CFE_ES_WriteToSysLog("ES Startup: Loading shared library: %s\n",FileName);
421 
422  /*
423  ** Now load the library
424  */
425  CreateStatus = CFE_ES_LoadLibrary(&ApplicationId, FileName,
426  EntryPoint, AppName);
427 
428  }
429  else if(strncmp(EntryType,"CFE_DRV",OS_MAX_API_NAME)==0)
430  {
431  CFE_ES_WriteToSysLog("ES Startup: Device Drivers not yet implemented. %s\n",FileName);
432  }
433  else
434  {
435  CFE_ES_WriteToSysLog("ES Startup: Unexpected EntryType %s in startup file.\n",EntryType);
436  }
437 
438  return (CreateStatus);
439 
440 }

442 /*
443 **---------------------------------------------------------------------------------------
444 ** Name: ES_AppCreate
445 **
446 ** Purpose: This function loads and creates a cFE Application.
447 ** This function can be called from the ES startup code when it
448 ** loads the cFE Applications from the disk using the startup script, or it
449 ** can be called when the ES Start Application command is executed.
450 **
451 **---------------------------------------------------------------------------------------
452 */
453 int32 CFE_ES_AppCreate(uint32 *ApplicationIdPtr,
454  const char *FileName,
455  const void *EntryPointData,
456  const char *AppName,
457  uint32 Priority,
458  uint32 StackSize,
459  uint32 ExceptionAction,
460  uint32 Flags)
461 {
462  cpuaddr StartAddr;
463  int32 ReturnCode;
464  uint32 i;
465  boolean AppSlotFound;
466  uint32 TaskId;
467  uint32 ModuleId;
468  char FileNameOnly[OS_MAX_PATH_LEN];
469  char RamDiskPath[OS_MAX_PATH_LEN];
470  size_t StringLength;
471  boolean IsRamDiskFile = FALSE;
472  CFE_ES_AppStaticModuleApi_t *StaticEntryPtr = NULL;
473 
474  /*
475  ** Allocate an ES_AppTable entry
476  */
477  CFE_ES_LockSharedData(__func__,__LINE__);
478  AppSlotFound = FALSE;
479  for ( i = 0; i < CFE_ES_MAX_APPLICATIONS; i++ )
480  {
481  if ( CFE_ES_Global.AppTable[i].RecordUsed == FALSE )
482  {
483  AppSlotFound = TRUE;
484  CFE_ES_Global.AppTable[i].RecordUsed = TRUE; /* Reserve Slot */
485  break;
486  }
487  }
488  CFE_ES_UnlockSharedData(__func__,__LINE__);
489 
490  /*
491  ** If a slot was found, create the application
492  */
493  if ( AppSlotFound == TRUE)
494  {
495  /*
496  * Check if dynamically loading -
497  * if statically linked then FileName is NULL and EntryPointData points to the API structure
498  */
499  if (FileName != NULL)
500  {
501  /*
502  ** Check to see if the code is a Gzip file
503  */
504  if ( CFE_FS_IsGzFile(FileName) == TRUE )
505  {
506  /*
507  ** Build up the destination path in the RAM disk
508  */
509  (void) CFE_SB_MessageStringGet(RamDiskPath,
510  CFE_ES_RAM_DISK_MOUNT_STRING"/",
511  NULL,
512  sizeof(RamDiskPath),
513  sizeof(CFE_ES_RAM_DISK_MOUNT_STRING"/"));
514 
515  /*
516  ** Extract the filename from the path
517  */
518  ReturnCode = CFE_FS_ExtractFilenameFromPath(FileName, FileNameOnly);
519 
520  if ( ReturnCode == CFE_SUCCESS )
521  {
522  if ((strlen(RamDiskPath) + strlen(FileNameOnly)) < OS_MAX_PATH_LEN)
523  {
524 
525  /*
526  ** Cat the Filename to the RamDiskPath
527  */
528  strcat(RamDiskPath, FileNameOnly);
529 
530  /*
531  ** Remove the ".gz" prefix from the filename
532  ** Already Determined that the filename ends in ".gz"
533  */
534  StringLength = strlen(RamDiskPath);
535  RamDiskPath[StringLength - 3] = '\0';
536 
537  /*
538  ** Decompress the file:
539  */
540  ReturnCode = CFE_FS_Decompress( FileName, RamDiskPath);
541 
542  if ( ReturnCode != OS_SUCCESS )
543  {
544  CFE_ES_WriteToSysLog("ES Startup: Unable to decompress Application File: %s\n",FileName);
545 
546  CFE_ES_LockSharedData(__func__,__LINE__);
547  CFE_ES_Global.AppTable[i].RecordUsed = FALSE; /* Release slot */
548  CFE_ES_UnlockSharedData(__func__,__LINE__);
549 
550  return(CFE_ES_ERR_APP_CREATE);
551  }
552  else
553  {
554  /*
555  ** All ready to use unzipped RAM disk file
556  */
557  IsRamDiskFile = TRUE;
558  ReturnCode = OS_ModuleLoad( &ModuleId, AppName, RamDiskPath);
559  }
560  }
561  else
562  {
563  /* Can't include the name string since it could be too long for the message */
564  CFE_ES_WriteToSysLog("ES Startup: Application path plus file name length (%d) exceeds max allowed (%d)\n",
565  (int)(strlen(RamDiskPath) + strlen(FileNameOnly)), OS_MAX_PATH_LEN);
566 
567  CFE_ES_LockSharedData(__func__,__LINE__);
568  CFE_ES_Global.AppTable[i].RecordUsed = FALSE; /* Release slot */
569  CFE_ES_UnlockSharedData(__func__,__LINE__);
570 
571  return(CFE_ES_ERR_APP_CREATE);
572  }
573 
574  }
575  else
576  {
577  CFE_ES_WriteToSysLog("ES Startup: Unable to extract filename from path: %s.\n",FileName);
578  CFE_ES_LockSharedData(__func__,__LINE__);
579  CFE_ES_Global.AppTable[i].RecordUsed = FALSE; /* Release slot */
580  CFE_ES_UnlockSharedData(__func__,__LINE__);
581  return(CFE_ES_ERR_APP_CREATE);
582  }
583 
584  }
585  else
586  {
587  /*
588  ** Load the module directly
589  */
590  ReturnCode = OS_ModuleLoad ( &ModuleId, AppName, FileName );
591  }
592  /*
593  ** If the Load was OK, then lookup the address of the entry point
594  */
595  if ( ReturnCode == OS_SUCCESS )
596  {
597  ReturnCode = OS_SymbolLookup( &StartAddr, (const char*)EntryPointData );
598  if ( ReturnCode != OS_SUCCESS )
599  {
600  CFE_ES_WriteToSysLog("ES Startup: Could not find symbol:%s. EC = 0x%08X\n",
601  (const char*)EntryPointData, (unsigned int)ReturnCode);
602 
603  CFE_ES_LockSharedData(__func__,__LINE__);
604  CFE_ES_Global.AppTable[i].RecordUsed = FALSE; /* Release slot */
605  CFE_ES_UnlockSharedData(__func__,__LINE__);
606 
607  return(CFE_ES_ERR_APP_CREATE);
608  }
609  }
610  else /* load not successful */
611  {
612  CFE_ES_WriteToSysLog("ES Startup: Could not load cFE application file:%s. EC = 0x%08X\n",
613  FileName, (unsigned int)ReturnCode);
614 
615  CFE_ES_LockSharedData(__func__,__LINE__);
616  CFE_ES_Global.AppTable[i].RecordUsed = FALSE; /* Release slot */
617  CFE_ES_UnlockSharedData(__func__,__LINE__);
618 
619  return(CFE_ES_ERR_APP_CREATE);
620  }
621  }
622  else /* Not a dynamic load - app must be statically linked into current exe */
623  {
624  StaticEntryPtr = (CFE_ES_AppStaticModuleApi_t *)EntryPointData;
625  StartAddr = StaticEntryPtr->Ptrs.EntryFuncAddr;
626  ModuleId = 0;
627  }
628 
629  /*
630  ** If the EntryPoint symbol was found, then start creating the App
631  */
632  CFE_ES_LockSharedData(__func__,__LINE__);
633  /*
634  ** Allocate and populate the ES_AppTable entry
635  */
636  CFE_PSP_MemSet ( (void *)&(CFE_ES_Global.AppTable[i]), 0, sizeof(CFE_ES_AppRecord_t));
637  CFE_ES_Global.AppTable[i].RecordUsed = TRUE;
638  CFE_ES_Global.AppTable[i].Type = CFE_ES_APP_TYPE_EXTERNAL;
639 
640  /*
641  ** Fill out the parameters in the AppStartParams sub-structure
642  */
643  strncpy((char *)CFE_ES_Global.AppTable[i].StartParams.Name, AppName, OS_MAX_API_NAME);
644  CFE_ES_Global.AppTable[i].StartParams.Name[OS_MAX_API_NAME - 1] = '\0';
645 
646  if (StaticEntryPtr != NULL)
647  {
648  /* Leave entry point / filename blank if static load entry */
649  CFE_ES_Global.AppTable[i].StartParams.EntryPoint[0] = 0;
650  CFE_ES_Global.AppTable[i].StartParams.FileName[0] = 0;
651  }
652  else
653  {
654  strncpy((char *)CFE_ES_Global.AppTable[i].StartParams.EntryPoint, (const char *)EntryPointData, OS_MAX_API_NAME);

957 
958  if(ReturnCode != CFE_SUCCESS)
959  {
960  CFE_ES_Global.LibTable[i].RecordUsed = FALSE;
961  CFE_ES_UnlockSharedData(__func__,__LINE__);
962 
963  CFE_ES_WriteToSysLog("ES Startup: Load Shared Library Init Error.\n");
964  return(CFE_ES_ERR_LOAD_LIB);
965  }
966  else
967  {
968  CFE_ES_Global.RegisteredLibs++;
969  *LibraryIdPtr = i;
970 
971  CFE_ES_UnlockSharedData(__func__,__LINE__);
972 
973  /*
974  ** Remove the temporary RAM disk file
975  */
976  if ( IsRamDiskFile == TRUE )
977  {
978  ReturnCode = OS_remove(RamDiskPath);
979  }
980  return(CFE_SUCCESS);
981  }
982  }
983  else /* libSlot not found */
984  {
985  CFE_ES_WriteToSysLog("ES Startup: No free library slots available\n");
986  return(CFE_ES_ERR_LOAD_LIB);

1102  ** Kill the app, and dont restart it

1118  case CFE_ES_RUNSTATUS_SYS_DELETE:

1120  ** Kill the app, and dont restart it
1121  */

1123

1126  CFE_EVS_SendEvent(CFE_ES_STOP_INF_EID, CFE_EVS_INFORMATION,

1275  ** Delete the ES Resources

1280  ** Get Main Task ID
1281  */

1300  {
