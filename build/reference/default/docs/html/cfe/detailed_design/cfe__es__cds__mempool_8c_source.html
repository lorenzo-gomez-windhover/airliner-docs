117 /*
118 ** CFE_ES_CreateCDSPool will initialize a pre-allocated memory pool.
119 */
120 int32 CFE_ES_CreateCDSPool(uint32 CDSPoolSize, uint32 StartOffset)
121 {
122  char MutexName[10] = {"CDS_POOL"};
123  uint32 i = 0;
124  uint32 Size = (CDSPoolSize & 0xfffffffc);
125 
126  /* create a semphore to protect this memory pool */
127  OS_MutSemCreate(&(CFE_ES_CDSMemPool.MutexId), MutexName, 0);
128 
129  /* Take the semaphore to ensure the mem pool is not being used during it's creation */
130  OS_MutSemTake(CFE_ES_CDSMemPool.MutexId);
131 
132  CFE_ES_CDSMemPool.Start = StartOffset;
133  CFE_ES_CDSMemPool.End = StartOffset + Size;
134  CFE_ES_CDSMemPool.Size = Size;
135  CFE_ES_CDSMemPool.Current = StartOffset;
136  CFE_ES_CDSMemPool.SizeIndex = -1;
137 
138  CFE_ES_CDSMemPool.CheckErrCntr = 0;
139  CFE_ES_CDSMemPool.RequestCntr = 0;
140 
141  for (i=0; i<CFE_ES_CDS_NUM_BLOCK_SIZES; i++)
142  {
143  CFE_ES_CDSMemPool.SizeDesc[i].Top = 0;
144  CFE_ES_CDSMemPool.SizeDesc[i].NumCreated = 0;
145  CFE_ES_CDSMemPool.SizeDesc[i].MaxSize = CFE_ES_CDSMemPoolDefSize[i];
146  }
147 
148  if (CDSPoolSize < (CFE_ES_CDSMemPool.MinBlockSize + sizeof(CFE_ES_CDSBlockDesc_t)))
149  {
150  /* Must be able make Pool verification, block descriptor and at least one of the smallest blocks */
151  CFE_ES_WriteToSysLog("CFE_ES:CreateCDSPool-Pool size(%u) too small for one CDS Block, need >=%u\n",
152  (unsigned int)CDSPoolSize, (unsigned int)(CFE_ES_CDSMemPool.MinBlockSize + sizeof(CFE_ES_CDSBlockDesc_t)));
153 
154  /* Give and delete semaphore since CDS Pool creation failed */
155  OS_MutSemGive(CFE_ES_CDSMemPool.MutexId);
156  OS_MutSemDelete(CFE_ES_CDSMemPool.MutexId);
157  return(CFE_ES_BAD_ARGUMENT);
158  }
159 
160  OS_MutSemGive(CFE_ES_CDSMemPool.MutexId);
161 
162  return(CFE_SUCCESS);
163 }

166 /*
167 ** Function:
168 ** CFE_ES_RebuildCDSPool
169 **
170 ** Purpose:
171 **
172 */
173 int32 CFE_ES_RebuildCDSPool(uint32 CDSPoolSize, uint32 StartOffset)
174 {
175  char MutexName[10] = {"CDS_POOL"};
176  uint32 i = 0;
177  uint32 Size = (CDSPoolSize & 0xfffffffc);
178  int32 Status = OS_SUCCESS;
179  uint32 Offset = StartOffset;
180  int32 BinIndex = 0;
181 
182  /* create a semphore to protect this memory pool */
183  OS_MutSemCreate(&(CFE_ES_CDSMemPool.MutexId), MutexName, 0);
184 
185  /* Take the semaphore to ensure the mem pool is not being used during it's creation */
186  OS_MutSemTake(CFE_ES_CDSMemPool.MutexId);
187 
188  CFE_ES_CDSMemPool.Start = StartOffset;
189  CFE_ES_CDSMemPool.End = StartOffset + Size;
190  CFE_ES_CDSMemPool.Size = Size;
191  CFE_ES_CDSMemPool.Current = 0;
192  CFE_ES_CDSMemPool.SizeIndex = -1;
193 
194  CFE_ES_CDSMemPool.CheckErrCntr = 0;
195  CFE_ES_CDSMemPool.RequestCntr = 0;
196 
197  for (i=0; i<CFE_ES_CDS_NUM_BLOCK_SIZES; i++)
198  {
199  CFE_ES_CDSMemPool.SizeDesc[i].Top = 0;
200  CFE_ES_CDSMemPool.SizeDesc[i].NumCreated = 0;
201  CFE_ES_CDSMemPool.SizeDesc[i].MaxSize = CFE_ES_CDSMemPoolDefSize[i];
202  }
203 
204  if (CDSPoolSize < (CFE_ES_CDSMemPool.MinBlockSize + sizeof(CFE_ES_CDSBlockDesc_t)))
205  {
206  /* Must be able make Pool verification, block descriptor and at least one of the smallest blocks */
207  CFE_ES_WriteToSysLog("CFE_ES:RebuildCDSPool-Pool size(%u) too small for one CDS Block, need >=%u\n",
208  (unsigned int)CDSPoolSize, (unsigned int)(CFE_ES_CDSMemPool.MinBlockSize + sizeof(CFE_ES_CDSBlockDesc_t)));
209 
210  /* Give and delete semaphore since CDS Pool rebuild failed */
211  OS_MutSemGive(CFE_ES_CDSMemPool.MutexId);
212  OS_MutSemDelete(CFE_ES_CDSMemPool.MutexId);
213  return(CFE_ES_BAD_ARGUMENT);
214  }
215 
216  /* Scan the CDS memory trying to find blocks that were created but are now free */
217  while ((Status == OS_SUCCESS) &&
218  (Offset < (CFE_ES_CDSMemPool.End - sizeof(CFE_ES_CDSBlockDesc_t))) &&
219  (CFE_ES_CDSMemPool.Current == 0))
220  {
221  /* Read the block descriptor for the first block in the memory pool */
222  Status = CFE_PSP_ReadFromCDS(&CFE_ES_CDSBlockDesc, Offset, sizeof(CFE_ES_CDSBlockDesc_t));
223 
224  if (Status == CFE_PSP_SUCCESS)
225  {
226  /* First, determine if the block is being or has been used */
227  if (CFE_ES_CDSBlockDesc.CheckBits == CFE_ES_CDS_CHECK_PATTERN)
228  {
229  /* See if the block is currently being used */
230  if (CFE_ES_CDSBlockDesc.AllocatedFlag != CFE_ES_CDS_BLOCK_USED)
231  {
232  /* If the block is not currently being used, */
233  /* then add it to the appropriate linked list in the memory pool */
234  BinIndex = CFE_ES_CDSGetBinIndex(CFE_ES_CDSBlockDesc.SizeUsed);
235 
236  /* Sanity-check the block descriptor */
237  if (BinIndex >= 0)
238  {
239  CFE_ES_CDSBlockDesc.Next = CFE_ES_CDSMemPool.SizeDesc[BinIndex].Top;
240  CFE_ES_CDSBlockDesc.AllocatedFlag = CFE_ES_CDS_BLOCK_UNUSED;
241  CFE_ES_CDSMemPool.SizeDesc[BinIndex].Top = Offset;
242 
243  /* Store the new CDS Block Descriptor in the CDS */
244  Status = CFE_PSP_WriteToCDS(&CFE_ES_CDSBlockDesc, Offset, sizeof(CFE_ES_CDSBlockDesc_t));
245 
246  if (Status != CFE_PSP_SUCCESS)
247  {
248  CFE_ES_WriteToSysLog("CFE_ES:RebuildCDS-Err writing to CDS (Stat=0x%08x)\n", (unsigned int)Status);
249  Status = CFE_ES_CDS_ACCESS_ERROR;
250  }
251  }
252  else
253  {
254  CFE_ES_CDSMemPool.CheckErrCntr++;
255  CFE_ES_WriteToSysLog("CFE_ES:RebuildCDS-Invalid Block Descriptor \n");
256  Status = CFE_ES_CDS_ACCESS_ERROR;
257  }
258  }
259 
260  /* Skip to the next block of memory */
261  Offset = Offset + CFE_ES_CDSBlockDesc.ActualSize + sizeof(CFE_ES_CDSBlockDesc_t);
262  }
263  else
264  {
265  /* If the block has never been used, then we should save the offset as the current offset */
266  /* which in turn will finish the scan of the CDS memory */
267  CFE_ES_CDSMemPool.Current = Offset;
268  }
269  }
270  else
271  {
272  CFE_ES_WriteToSysLog("CFE_ES:RebuildCDS-Err reading from CDS (Stat=0x%08x)\n", (unsigned int)Status);
273  Status = CFE_ES_CDS_ACCESS_ERROR;
274  }
275  } /* end while */
276 
277  OS_MutSemGive(CFE_ES_CDSMemPool.MutexId);
278 
279  return Status;
280 }

283 /*
284 ** Function:
285 ** CFE_ES_GetCDSBlock
286 **
287 ** Purpose:
288 ** CFE_ES_GetCDSBlock allocates a block from the CDS memory pool.
289 */
290 int32 CFE_ES_GetCDSBlock(CFE_ES_CDSBlockHandle_t *BlockHandle,
291  uint32 BlockSize )
292 {
293  int32 BinIndex;
294  int32 Status;
295 
296  OS_MutSemTake(CFE_ES_CDSMemPool.MutexId);
297 
298  BinIndex = CFE_ES_CDSGetBinIndex(BlockSize);
299  if (BinIndex < 0)
300  {
301  CFE_ES_WriteToSysLog("CFE_ES:GetCDSBlock-err:size(%d) > max(%d).\n", (int)BlockSize, CFE_ES_CDS_MAX_BLOCK_SIZE);
302  OS_MutSemGive(CFE_ES_CDSMemPool.MutexId);
303  return(CFE_ES_ERR_MEM_BLOCK_SIZE);
304  }
305 
306  /*
307  ** Check if any of the requested size are available
308  */
309  if (CFE_ES_CDSMemPool.SizeDesc[BinIndex].Top != 0)
310  {
311  /*
312  ** Get it off the top on the list
313  */
314  Status = CFE_PSP_ReadFromCDS(&CFE_ES_CDSBlockDesc,
315  CFE_ES_CDSMemPool.SizeDesc[BinIndex].Top,
316  sizeof(CFE_ES_CDSBlockDesc_t));
317 
318  if (Status != CFE_PSP_SUCCESS)
319  {
320  CFE_ES_WriteToSysLog("CFE_ES:GetCDSBlock-Err reading from CDS (Stat=0x%08x)\n", (unsigned int)Status);
321  OS_MutSemGive(CFE_ES_CDSMemPool.MutexId);
322  return(CFE_ES_CDS_ACCESS_ERROR);
323  }
324 
325  /* The handle returned is the byte offset of the block in the CDS */
326  *BlockHandle = CFE_ES_CDSMemPool.SizeDesc[BinIndex].Top;
327 
328  /* A local version of the block descriptor is initialized */
329  CFE_ES_CDSBlockDesc.CheckBits = CFE_ES_CDS_CHECK_PATTERN;
330  CFE_ES_CDSBlockDesc.AllocatedFlag = CFE_ES_CDS_BLOCK_USED;
331  CFE_ES_CDSBlockDesc.SizeUsed = BlockSize;
332  CFE_ES_CDSBlockDesc.ActualSize = CFE_ES_CDSMemPool.SizeDesc[BinIndex].MaxSize;
333  CFE_ES_CDSMemPool.SizeDesc[BinIndex].Top = CFE_ES_CDSBlockDesc.Next;
334  CFE_ES_CDSBlockDesc.CRC = 0;
335  CFE_ES_CDSBlockDesc.Next = 0;
336  }
337  else /* Create a new block */
338  {
339  if ( (CFE_ES_CDSMemPool.Current == 0) ||
340  (((uint32)CFE_ES_CDSMemPool.Current +
341  sizeof(CFE_ES_CDSBlockDesc_t) +
342  CFE_ES_CDSMemPool.SizeDesc[BinIndex].MaxSize ) >= CFE_ES_CDSMemPool.End) )
343  {
344  CFE_ES_WriteToSysLog("CFE_ES:GetCDSBlock-err:Request for %d bytes won't fit in remaining memory\n", (int)BlockSize);
345  OS_MutSemGive(CFE_ES_CDSMemPool.MutexId);
346  return(CFE_ES_ERR_MEM_BLOCK_SIZE);
347  }
348 
349  *BlockHandle = (CFE_ES_CDSBlockHandle_t)CFE_ES_CDSMemPool.Current;
350 
351  CFE_ES_CDSMemPool.SizeDesc[BinIndex].NumCreated++;
352  CFE_ES_CDSMemPool.RequestCntr++;
353 
354  /*
355  ** Initialize the buffer descriptor that will be kept in front of the CDS Block
356  */
357  CFE_ES_CDSBlockDesc.CheckBits = CFE_ES_CDS_CHECK_PATTERN;
358  CFE_ES_CDSBlockDesc.AllocatedFlag = CFE_ES_CDS_BLOCK_USED;
359  CFE_ES_CDSBlockDesc.SizeUsed = BlockSize;
360  CFE_ES_CDSBlockDesc.ActualSize = CFE_ES_CDSMemPool.SizeDesc[BinIndex].MaxSize;
361  CFE_ES_CDSBlockDesc.CRC = 0;
362  CFE_ES_CDSBlockDesc.Next = 0;
363 
364  /*
365  ** Adjust pool current pointer to first unallocated byte in CDS
366  */
367  CFE_ES_CDSMemPool.Current = CFE_ES_CDSMemPool.Current
368  + CFE_ES_CDSBlockDesc.ActualSize
369  + sizeof(CFE_ES_CDSBlockDesc_t);
370  }
371 
372  /* Store the new CDS Block Descriptor in the CDS */
373  Status = CFE_PSP_WriteToCDS(&CFE_ES_CDSBlockDesc, *BlockHandle, sizeof(CFE_ES_CDSBlockDesc_t));
374 
375  if (Status != CFE_PSP_SUCCESS)
376  {
377  CFE_ES_WriteToSysLog("CFE_ES:GetCDSBlock-Err writing to CDS (Stat=0x%08x)\n", (unsigned int)Status);
378  OS_MutSemGive(CFE_ES_CDSMemPool.MutexId);
379  return(CFE_ES_CDS_ACCESS_ERROR);
380  }
381 
382  OS_MutSemGive(CFE_ES_CDSMemPool.MutexId);
383 
384  return Status;
385 }

387 /*
388 ** CFE_ES_PutCDSBlock returns a block back to the CDS memory pool.
389 */
390 int32 CFE_ES_PutCDSBlock(CFE_ES_CDSBlockHandle_t BlockHandle)
391 {
392  int32 BinIndex;
393  int32 Status;
394 
395  /* Perform some sanity checks on the BlockHandle */
396  /* First check, is the handle within an acceptable range of CDS offsets */
397  if ((BlockHandle < sizeof(CFE_ES_Global.CDSVars.ValidityField)) ||
398  (BlockHandle > (CFE_ES_CDSMemPool.End - sizeof(CFE_ES_CDSBlockDesc_t) -
399  CFE_ES_CDSMemPool.MinBlockSize - sizeof(CFE_ES_Global.CDSVars.ValidityField))))
400  {
401  CFE_ES_WriteToSysLog("CFE_ES:PutCDSBlock-Invalid Memory Handle.\n");
402  return(CFE_ES_ERR_MEM_HANDLE);
403  }
404 
405  OS_MutSemTake(CFE_ES_CDSMemPool.MutexId);
406 
407  /* Read a copy of the contents of the block descriptor being freed */
408  Status = CFE_PSP_ReadFromCDS(&CFE_ES_CDSBlockDesc, BlockHandle, sizeof(CFE_ES_CDSBlockDesc_t));
409 
410  if (Status != CFE_PSP_SUCCESS)
411  {
412  CFE_ES_WriteToSysLog("CFE_ES:PutCDSBlock-Err reading from CDS (Stat=0x%08x)\n", (unsigned int)Status);
413  OS_MutSemGive(CFE_ES_CDSMemPool.MutexId);
414  return(CFE_ES_CDS_ACCESS_ERROR);
415  }
416 
417  /* Make sure the contents of the Block Descriptor look reasonable */
418  if ((CFE_ES_CDSBlockDesc.CheckBits != CFE_ES_CDS_CHECK_PATTERN) ||
419  (CFE_ES_CDSBlockDesc.AllocatedFlag != CFE_ES_CDS_BLOCK_USED))
420  {
421  CFE_ES_WriteToSysLog("CFE_ES:PutCDSBlock-Invalid Handle or Block Descriptor.\n");
422  OS_MutSemGive(CFE_ES_CDSMemPool.MutexId);
423  return(CFE_ES_ERR_MEM_HANDLE);
424  }
425 
426  BinIndex = CFE_ES_CDSGetBinIndex(CFE_ES_CDSBlockDesc.ActualSize);
427 
428  /* Final sanity check on block descriptor, is the Actual size reasonable */
429  if (BinIndex < 0)
430  {
431  CFE_ES_CDSMemPool.CheckErrCntr++;
432  CFE_ES_WriteToSysLog("CFE_ES:PutCDSBlock-Invalid Block Descriptor\n");
433  OS_MutSemGive(CFE_ES_CDSMemPool.MutexId);
434  return(CFE_ES_ERR_MEM_HANDLE);
435  }
436 
437  CFE_ES_CDSBlockDesc.Next = CFE_ES_CDSMemPool.SizeDesc[BinIndex].Top;
438  CFE_ES_CDSBlockDesc.AllocatedFlag = CFE_ES_CDS_BLOCK_UNUSED;
439  CFE_ES_CDSMemPool.SizeDesc[BinIndex].Top = BlockHandle;
440 
441  /* Store the new CDS Block Descriptor in the CDS */
442  Status = CFE_PSP_WriteToCDS(&CFE_ES_CDSBlockDesc, BlockHandle, sizeof(CFE_ES_CDSBlockDesc_t));
443 
444  if (Status != CFE_PSP_SUCCESS)
445  {
446  CFE_ES_WriteToSysLog("CFE_ES:PutCDSBlock-Err writing to CDS (Stat=0x%08x)\n", (unsigned int)Status);
447  OS_MutSemGive(CFE_ES_CDSMemPool.MutexId);
448  return(CFE_ES_CDS_ACCESS_ERROR);
449  }
450 
451  OS_MutSemGive(CFE_ES_CDSMemPool.MutexId);
452 
453  return Status;
454 }

456 /*
457 ** Function:
458 ** CFE_ES_CDSGetBinIndex
459 **
460 ** Purpose:
461 **
462 */
463 int32 CFE_ES_CDSGetBinIndex(uint32 DesiredSize)
464 {
465  int32 i=0;
466 
467  if (DesiredSize > CFE_ES_CDSMemPool.SizeDesc[0].MaxSize)
468  {
469  return(-1);
470  }
471 
472  /* Look ahead to see if the next bin has a size too small */
473  while ((i < (CFE_ES_CDS_NUM_BLOCK_SIZES-1)) &&
474  (DesiredSize <= CFE_ES_CDSMemPool.SizeDesc[i+1].MaxSize))
475  {
476  i++;
477  }
478 
479  return(i);
480 }

497  (BlockHandle > (CFE_ES_CDSMemPool.End - sizeof(CFE_ES_CDSBlockDesc_t) -
498  CFE_ES_CDSMemPool.MinBlockSize - sizeof(CFE_ES_Global.CDSVars.ValidityField))))
499  {
500  CFE_ES_WriteToSysLog("CFE_ES:CDSBlkWrite-Invalid Memory Handle.\n");
501  return(CFE_ES_ERR_MEM_HANDLE);
502  }
503 
504  OS_MutSemTake(CFE_ES_CDSMemPool.MutexId);
505 
506  /* Get a copy of the block descriptor associated with the specified handle */
507  /* Read the block descriptor for the first block in the memory pool */
508  Status = CFE_PSP_ReadFromCDS(&CFE_ES_CDSBlockDesc, BlockHandle, sizeof(CFE_ES_CDSBlockDesc_t));
509 
510  if (Status == CFE_PSP_SUCCESS)
511  {
512  /* Validate the block to make sure it is still active and not corrupted */
513  if ((CFE_ES_CDSBlockDesc.CheckBits != CFE_ES_CDS_CHECK_PATTERN) ||
514  (CFE_ES_CDSBlockDesc.AllocatedFlag != CFE_ES_CDS_BLOCK_USED))
515  {
516  CFE_ES_WriteToSysLog("CFE_ES:CDSBlkWrite-Invalid Handle or Block Descriptor.\n");
517  OS_MutSemGive(CFE_ES_CDSMemPool.MutexId);
518  return(CFE_ES_ERR_MEM_HANDLE);
519  }
520 
521  BinIndex = CFE_ES_CDSGetBinIndex(CFE_ES_CDSBlockDesc.ActualSize);
522 
523  /* Final sanity check on block descriptor, is the Actual size reasonable */
524  if (BinIndex < 0)
525  {
526  CFE_ES_CDSMemPool.CheckErrCntr++;
527  CFE_ES_WriteToSysLog("CFE_ES:CDSBlkWrite-Invalid Block Descriptor\n");
528  OS_MutSemGive(CFE_ES_CDSMemPool.MutexId);
529  return(CFE_ES_ERR_MEM_HANDLE);
530  }
531 
532  /* Use the size specified when the CDS was created to compute the CRC */
