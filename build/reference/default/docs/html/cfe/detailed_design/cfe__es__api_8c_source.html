193 /*
194 ** Function: CFE_ES_ResetCFE
195 **
196 ** Purpose: Reset the cFE core and all apps.
197 **
198 */
199 int32 CFE_ES_ResetCFE(uint32 ResetType)
200 {
201  int32 ReturnCode;
202 
203  if ( ResetType == CFE_PSP_RST_TYPE_PROCESSOR )
204  {
205  /*
206  ** Increment the processor reset count
207  */
208  CFE_ES_ResetDataPtr->ResetVars.ProcessorResetCount++;
209 
210  /*
211  ** Before doing a Processor reset, check to see
212  ** if the maximum number has been exceeded
213  */
214  if ( CFE_ES_ResetDataPtr->ResetVars.ProcessorResetCount >
215  CFE_ES_ResetDataPtr->ResetVars.MaxProcessorResetCount )
216  {
217  CFE_ES_WriteToSysLog("POWER ON RESET due to max proc resets (Commanded).\n");
218 
219  /*
220  ** Log the reset in the ER Log. The log will be wiped out, but it's good to have
221  ** the entry just in case something fails.
222  */
223  CFE_ES_WriteToERLog(CFE_ES_CORE_LOG_ENTRY, CFE_PSP_RST_TYPE_POWERON,
224  CFE_PSP_RST_SUBTYPE_RESET_COMMAND,
225  "POWER ON RESET due to max proc resets (Commanded).", NULL,0 );
226  /*
227  ** Call the BSP reset routine
228  */
229  CFE_PSP_Restart(CFE_PSP_RST_TYPE_POWERON);
230  }
231  else
232  {
233  CFE_ES_WriteToSysLog("PROCESSOR RESET called from CFE_ES_ResetCFE (Commanded).\n");
234 
235  /*
236  ** Update the reset variables
237  */
238  CFE_ES_ResetDataPtr->ResetVars.ES_CausedReset = TRUE;
239 
240  /*
241  ** Log the reset in the ER Log
242  */
243  CFE_ES_WriteToERLog(CFE_ES_CORE_LOG_ENTRY, CFE_ES_PROCESSOR_RESET,
244  CFE_ES_RESET_COMMAND,
245  "PROCESSOR RESET called from CFE_ES_ResetCFE (Commanded).", NULL,0 );
246  /*
247  ** Call the BSP reset routine
248  */
249  CFE_PSP_Restart(CFE_PSP_RST_TYPE_PROCESSOR);
250 
251  } /* end if */
252 
253  /*
254  ** If the BSP routine is not implemented,
255  ** it will return.
256  */
257  ReturnCode = CFE_ES_NOT_IMPLEMENTED;
258  }
259  else if ( ResetType == CFE_PSP_RST_TYPE_POWERON )
260  {
261  CFE_ES_WriteToSysLog("POWERON RESET called from CFE_ES_ResetCFE (Commanded).\n");
262 
263  /*
264  ** Log the reset in the ER Log. The log will be wiped out, but it's good to have
265  ** the entry just in case something fails.
266  */
267  CFE_ES_WriteToERLog(CFE_ES_CORE_LOG_ENTRY, CFE_PSP_RST_TYPE_POWERON,
268  CFE_PSP_RST_SUBTYPE_RESET_COMMAND,
269  "POWERON RESET called from CFE_ES_ResetCFE (Commanded).", NULL,0 );
270 
271  /*
272  ** Call the BSP reset routine
273  */
274  CFE_PSP_Restart(CFE_PSP_RST_TYPE_POWERON);
275 
276  /*
277  ** If the BSP routine is not implemented,
278  ** it will return.
279  */
280  ReturnCode = CFE_ES_NOT_IMPLEMENTED;
281  }
282  else
283  {
284  CFE_ES_WriteToSysLog("ES ResetCFE: Invalid Reset Type: %d.\n",(int)ResetType);
285  ReturnCode = CFE_ES_BAD_ARGUMENT;
286  }
287 
288  return(ReturnCode);
289 
290 } /* End of CFE_ES_ResetCFE() */

434 void CFE_ES_ExitApp(uint32 ExitStatus)
435 {
436  int32 ReturnCode;
437  uint32 AppID;
438 
439  CFE_ES_LockSharedData(__func__,__LINE__);
440 
441  /*
442  ** Get App ID - Be careful not to lock the shared data before calling GetAppID
443  */
444  ReturnCode = CFE_ES_GetAppIDInternal(&AppID);
445  if ( ReturnCode == CFE_SUCCESS )
446  {
447  /*
448  ** Check to see if the App is an external cFE App.
449  */
450  if ( CFE_ES_Global.AppTable[AppID].Type == CFE_ES_APP_TYPE_CORE )
451  {
452  /*
453  ** A core app should only call this function with one of two ExitStatus codes.
454  */
455  if ( ExitStatus == CFE_ES_RUNSTATUS_CORE_APP_INIT_ERROR )
456  {
457  CFE_ES_WriteToSysLog ("CFE_ES_ExitApp: CORE Application %s Had an Init Error.\n",
458  CFE_ES_Global.AppTable[AppID].StartParams.Name );
459 
460  /*
461  ** Unlock the ES Shared data before calling ResetCFE
462  */
463  CFE_ES_UnlockSharedData(__func__,__LINE__);
464 
465 
466  /*
467  ** Do a Processor Reset the cFE
468  */
469  ReturnCode = CFE_ES_ResetCFE(CFE_PSP_RST_TYPE_PROCESSOR);
470 
471  /*
472  ** Code will not return
473  */
474 
475  }
476  else if ( ExitStatus == CFE_ES_RUNSTATUS_CORE_APP_RUNTIME_ERROR )
477  {
478  CFE_ES_WriteToSysLog ("CFE_ES_ExitApp: CORE Application %s Had a Runtime Error.\n",
479  CFE_ES_Global.AppTable[AppID].StartParams.Name );
480 
481  /*
482  ** Unlock the ES Shared data before killing the main task
483  */
484  CFE_ES_UnlockSharedData(__func__,__LINE__);
485 
486  /*
487  ** Exit this task
488  */
489  OS_TaskExit();
490 
491  /*
492  ** Code will not return, except under unit test
493  */
494  return;
495  }
496  else
497  {
498  CFE_ES_WriteToSysLog ("CFE_ES_ExitApp, Cannot Exit CORE Application %s\n",
499  CFE_ES_Global.AppTable[AppID].StartParams.Name );
500  }
501 
502  }
503  else /* It is an external App */
504  {
505 
506  CFE_ES_WriteToSysLog ("Application %s called CFE_ES_ExitApp\n",
507  CFE_ES_Global.AppTable[AppID].StartParams.Name);

509

513  */
514  CFE_ES_UnlockSharedData(__func__,__LINE__);
515

518  ** It might be better to have a way of suspending the app in the OS
519  */
520  while(1)
521  {
522  OS_TaskDelay(500);
523  }
524 
525  } /* end if */
526 
527  } /* end if ReturnCode == CFE_SUCCESS */
528 
529  CFE_ES_UnlockSharedData(__func__,__LINE__);
530 
531 } /* End of CFE_ES_ExitApp() */

533 /*
534 ** Function: CFE_ES_RunLoop
535 **
536 ** Purpose: Check the Run Status of an Application.
537 **
538 */
539 int32 CFE_ES_RunLoop(uint32 *RunStatus)
540 {
541  int32 ReturnCode;
542  uint32 AppID;
543  uint32 TaskID;
544 
545  CFE_ES_LockSharedData(__func__,__LINE__);
546 
547  /*
548  ** Get App ID
549  */
550  ReturnCode = CFE_ES_GetAppIDInternal(&AppID);
551 
552  if ( ReturnCode == CFE_SUCCESS )
553  {
554 
555  /*
556  ** Get the Task ID for the main task
557  */
558  OS_ConvertToArrayIndex(CFE_ES_Global.AppTable[AppID].TaskInfo.MainTaskId, &TaskID);
559 
560  /*
561  ** Increment the execution counter for the main task
562  */
563  CFE_ES_Global.TaskTable[TaskID].ExecutionCounter++;
564 
565  /*
566  ** Validate RunStatus
567  */
568  if ( *RunStatus == CFE_ES_RUNSTATUS_APP_RUN || *RunStatus == CFE_ES_RUNSTATUS_APP_EXIT || *RunStatus == CFE_ES_RUNSTATUS_APP_ERROR )
569  {
570  /*
571  ** Look up the system control request
572  */
573  if ( *RunStatus == CFE_ES_RUNSTATUS_APP_RUN )
574  {
575  /*
576  ** The first time an App calls this function, change the state from
577  ** INITIALIZING to RUNNING. This is for the startup syncronization code below
578  */
579  if ( CFE_ES_Global.AppTable[AppID].StateRecord.AppState == CFE_ES_APP_STATE_INITIALIZING)
580  {
581  /*
