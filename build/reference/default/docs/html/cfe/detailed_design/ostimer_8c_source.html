70 /*
71 ** The timers use the RT Signals. The system that this code was developed
72 ** and tested on has 32 available RT signals ( SIGRTMIN -> SIGRTMAX ).
73 ** OS_MAX_TIMERS should not be set to more than this number.
74 */
75 #define OS_STARTING_SIGNAL (SIGRTMAX-1)
76 
77 /*
78 ** Since the API is storing the timer values in a 32 bit integer as Microseconds,
79 ** there is a limit to the number of seconds that can be represented.
80 */
81 #define MAX_SECS_IN_USEC 4293

114 /****************************************************************************************
115  INITIALIZATION FUNCTION
116 ****************************************************************************************/
117 int32 OS_TimerAPIInit ( void )
118 {
119  int i;
120  int status;
121  struct timespec clock_resolution;
122  int32 return_code = OS_SUCCESS;
123 
124  /*
125  ** Mark all timers as available
126  */
127  for ( i = 0; i < OS_MAX_TIMERS; i++ )
128  {
129  OS_timer_table[i].free = TRUE;
130  OS_timer_table[i].creator = UNINITIALIZED;
131  strcpy(OS_timer_table[i].name,"");
132 
133  }
134 
135  /*
136  ** get the resolution of the realtime clock
137  */
138  status = clock_getres(CLOCK_REALTIME, &clock_resolution);
139  if ( status < 0 )
140  {
141  return_code = OS_ERROR;
142  }
143  else
144  {
145  /*
146  ** Convert to microseconds
147  */
148  OS_TimespecToUsec(clock_resolution, &os_clock_accuracy);
149 
150  /*
151  ** Create the Timer Table mutex
152  */
153  status = pthread_mutex_init((pthread_mutex_t *) & OS_timer_table_mut,NULL);
154  if ( status < 0 )
155  {
156  return_code = OS_ERROR;
157  }
158  }
159  return(return_code);
160 
161 }

167 /*
168 ** Timer Signal Handler.
169 ** The purpose of this function is to convert the POSIX signal number to the
170 ** OSAL signal number and pass it to the User defined signal handler.
171 */
172 void OS_TimerSignalHandler(int signum)
173 {
174  uint32 timer_id = 0;
175 
176  timer_id = OS_STARTING_SIGNAL - signum;
177 
178  if ( timer_id < OS_MAX_TIMERS )
179  {
180  if ( OS_timer_table[timer_id].free == FALSE )
181  {
182  (OS_timer_table[timer_id].callback_ptr)(timer_id);
183  }
184  }
185 
186 }

188 /******************************************************************************
189  ** Function: OS_UsecToTimespec
190  **
191  ** Purpose: Convert Microseconds to a POSIX timespec structure.
192  **
193  */
194 void OS_UsecToTimespec(uint32 usecs, struct timespec *time_spec)
195 {
196 
197  if ( usecs < 1000000 )
198  {
199  time_spec->tv_nsec = (usecs * 1000);
200  time_spec->tv_sec = 0;
201  }
202  else
203  {
204  time_spec->tv_sec = usecs / 1000000;
205  time_spec->tv_nsec = (usecs % 1000000) * 1000;
206  }
207 
208 }
209 
210 /******************************************************************************
211  ** Function: OS_TimespecToUsec
212  **
213  ** Purpose: Convert a POSIX timespec structure to microseconds
214  **
215  */
216 void OS_TimespecToUsec(struct timespec time_spec, uint32 *usecs)
217 {
218  if ( (*usecs) < 1000000 )
219  {
220  *usecs = time_spec.tv_nsec / 1000;
221  }
222  else
223  {
224  /*
225  ** Need to account for maximum number of seconds that will fit in the
226  ** 32 bit usec integer
227  */
228  if ( time_spec.tv_sec > MAX_SECS_IN_USEC )
229  {
230  time_spec.tv_sec = MAX_SECS_IN_USEC;
231  }
232  *usecs = (time_spec.tv_sec * 1000000 ) + (time_spec.tv_nsec / 1000 );
233  }
234 }

242 /******************************************************************************
243 ** Function: OS_TimerCreate
244 **
245 ** Purpose: Create a new OSAL Timer
246 **
247 ** Arguments:
248 **
249 ** Return:
250 */
251 int32 OS_TimerCreate(uint32 *timer_id, const char *timer_name, uint32 *clock_accuracy, OS_TimerCallback_t callback_ptr)
252 {
253  uint32 possible_tid;
254  int32 i;
255  sigset_t previous;
256  sigset_t mask;
257 
258  int status;
259  struct sigaction sig_act;
260  struct sigevent evp;
261 
262  if ( timer_id == NULL || timer_name == NULL || clock_accuracy == NULL)
263  {
264  return OS_INVALID_POINTER;
265  }
266 
267  /*
268  ** we don't want to allow names too long
269  ** if truncated, two names might be the same
270  */
271  if (strlen(timer_name) > OS_MAX_API_NAME)
272  {
273  return OS_ERR_NAME_TOO_LONG;
274  }
275 
276  /*
277  ** Check Parameters
278  */
279  OS_InterruptSafeLock(&OS_timer_table_mut, &mask, &previous);
280 
281  for(possible_tid = 0; possible_tid < OS_MAX_TIMERS; possible_tid++)
282  {
283  if (OS_timer_table[possible_tid].free == TRUE)
284  break;
285  }
286 
287 
288  if( possible_tid >= OS_MAX_TIMERS || OS_timer_table[possible_tid].free != TRUE)
289  {
290  OS_InterruptSafeUnlock(&OS_timer_table_mut, &previous);
291  return OS_ERR_NO_FREE_IDS;
292  }
293 
294  /*
295  ** Check to see if the name is already taken
296  */
297  for (i = 0; i < OS_MAX_TIMERS; i++)
298  {
299  if ((OS_timer_table[i].free == FALSE) &&
300  strcmp ((char*) timer_name, OS_timer_table[i].name) == 0)
301  {
302  OS_InterruptSafeUnlock(&OS_timer_table_mut, &previous);
303  return OS_ERR_NAME_TAKEN;
304  }
305  }
306 
307  /*
308  ** Verify callback parameter
309  */
310  if (callback_ptr == NULL )
311  {
312  OS_InterruptSafeUnlock(&OS_timer_table_mut, &previous);
313  return OS_TIMER_ERR_INVALID_ARGS;
314  }
315 
316  /*
317  ** Set the possible timer Id to not free so that
318  ** no other task can try to use it
319  */
320  OS_timer_table[possible_tid].free = FALSE;
321  OS_InterruptSafeUnlock(&OS_timer_table_mut, &previous);
322 
323  OS_timer_table[possible_tid].creator = OS_FindCreator();
324  strncpy(OS_timer_table[possible_tid].name, timer_name, OS_MAX_API_NAME);
325  OS_timer_table[possible_tid].start_time = 0;
326  OS_timer_table[possible_tid].interval_time = 0;
327 
328  OS_timer_table[possible_tid].callback_ptr = callback_ptr;
329 
330  /*
331  ** Initialize the sigaction and sigevent structures for the handler.
332  */
333  memset((void *)&sig_act, 0, sizeof(sig_act));
334  sigemptyset(&sig_act.sa_mask);
335  sig_act.sa_handler = OS_TimerSignalHandler;
336  sig_act.sa_flags = SA_RESTART;
337 
338  memset((void *)&evp, 0, sizeof(evp));
339  evp.sigev_notify = SIGEV_SIGNAL;
340  evp.sigev_signo = OS_STARTING_SIGNAL - possible_tid;
341 
342  /*
343  ** Create the timer
344  */
345  status = timer_create(CLOCK_REALTIME, &evp, (timer_t *)&(OS_timer_table[possible_tid].host_timerid));
346  if (status < 0)
347  {
348  OS_timer_table[possible_tid].free = TRUE;
349  return ( OS_TIMER_ERR_UNAVAILABLE);
350  }
351 
352  /*
353  ** Set the signal action for the timer
354  */
355  sigaction(OS_STARTING_SIGNAL - possible_tid, &(sig_act), 0);
356 
357  /*
358  ** Return the clock accuracy to the user
359  */
360  *clock_accuracy = os_clock_accuracy;
361 
362  /*
363  ** Return timer ID
364  */
365  *timer_id = possible_tid;
366 
367  return OS_SUCCESS;
368 }

370 /******************************************************************************
371 ** Function: OS_TimerSet
372 **
373 ** Purpose:
374 **
375 ** Arguments:
376 ** (none)
377 **
378 ** Return:
379 ** (none)
380 */
381 int32 OS_TimerSet(uint32 timer_id, uint32 start_time, uint32 interval_time)
382 {
383  int status;
384  struct itimerspec timeout;
385 
386  /*
387  ** Check to see if the timer_id given is valid
388  */
389  if (timer_id >= OS_MAX_TIMERS || OS_timer_table[timer_id].free == TRUE)
390  {
391  return OS_ERR_INVALID_ID;
392  }
393 
394  /*
395  ** Round up the accuracy of the start time and interval times
396  */
397  if (( start_time > 0 ) && ( start_time < os_clock_accuracy ))
398  {
399  start_time = os_clock_accuracy;
400  }
401 
402  if (( interval_time > 0) && ( interval_time < os_clock_accuracy ))
403  {
404  interval_time = os_clock_accuracy;
405  }
406 
407  /*
408  ** Save the start and interval times
409  */
410  OS_timer_table[timer_id].start_time = start_time;
411  OS_timer_table[timer_id].interval_time = interval_time;
412 
413  /*
414  ** Convert from Microseconds to timespec structures
415  */
416  OS_UsecToTimespec(start_time, &(timeout.it_value));
417  OS_UsecToTimespec(interval_time, &(timeout.it_interval));
418 
419  /*
420  ** Program the real timer
421  */
422  status = timer_settime((timer_t)(OS_timer_table[timer_id].host_timerid),
423  0, /* Flags field can be zero */
424  &timeout, /* struct itimerspec */
425  0); /* Oldvalue */
426  if (status < 0)
427  {
428  return ( OS_TIMER_ERR_INTERNAL);
429  }
430 
431  return OS_SUCCESS;
432 }

471 /***********************************************************************************
472 **
473 ** Name: OS_TimerGetIdByName
474 **
475 ** Purpose: This function tries to find a Timer Id given the name
476 ** The id is returned through timer_id
477 **
478 ** Returns: OS_INVALID_POINTER if timer_id or timer_name are NULL pointers
479 ** OS_ERR_NAME_TOO_LONG if the name given is to long to have been stored
480 ** OS_ERR_NAME_NOT_FOUND if the name was not found in the table
481 ** OS_SUCCESS if success
482 **
483 */
484 int32 OS_TimerGetIdByName (uint32 *timer_id, const char *timer_name)
485 {
486  uint32 i;
487 
488  if (timer_id == NULL || timer_name == NULL)
489  {
490  return OS_INVALID_POINTER;
491  }
492 
493  /*
494  ** a name too long wouldn't have been allowed in the first place
495  ** so we definitely won't find a name too long
496  */
497  if (strlen(timer_name) > OS_MAX_API_NAME)
498  {
499  return OS_ERR_NAME_TOO_LONG;
500  }
501 
502  for (i = 0; i < OS_MAX_TIMERS; i++)
503  {
504  if (OS_timer_table[i].free != TRUE &&
505  (strcmp (OS_timer_table[i].name , (char*) timer_name) == 0))
506  {
507  *timer_id = i;
508  return OS_SUCCESS;
509  }
510  }
511 
512  /*
513  ** The name was not found in the table,
514  ** or it was, and the sem_id isn't valid anymore
515  */
516  return OS_ERR_NAME_NOT_FOUND;
517 
518 }/* end OS_TimerGetIdByName */
