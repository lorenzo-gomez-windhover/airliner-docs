187 /*--------------------------------------------------------------------------------------
188  Name: OS_ModuleLoad
189 
190  Purpose: Loads an object file into the running operating system
191 
192  Parameters:
193 
194  Returns: OS_ERROR if the module cannot be loaded
195  OS_INVALID_POINTER if one of the parameters is NULL
196  OS_ERR_NO_FREE_IDS if the module table is full
197  OS_ERR_NAME_TAKEN if the name is in use
198  OS_SUCCESS if the module is loaded successfuly
199 ---------------------------------------------------------------------------------------*/
200 int32 OS_ModuleLoad ( uint32 *module_id, const char *module_name, const char *filename )
201 {
202  int i;
203  uint32 possible_moduleid;
204  char translated_path[OS_MAX_LOCAL_PATH_LEN] = "";
205  int32 return_code;
206  void *function_lib; /* Handle to shared lib file */
207  const char *dl_error; /* Pointer to error string */
208  sigset_t previous;
209  sigset_t mask;
210 
211  /*
212  ** Check parameters
213  */
214  if (( filename == NULL ) || (module_id == NULL ) || (module_name == NULL))
215  {
216  return(OS_INVALID_POINTER);
217  }
218 
219  OS_InterruptSafeLock(&OS_module_table_mut, &mask, &previous);
220 
221  /*
222  ** Find a free module id
223  */
224  for( possible_moduleid = 0; possible_moduleid < OS_MAX_MODULES; possible_moduleid++)
225  {
226  if (OS_module_table[possible_moduleid].name[0] == '\0')
227  {
228  break;
229  }
230  }
231 
232  /*
233  ** Check to see if the id is out of bounds
234  */
235  if( possible_moduleid >= OS_MAX_MODULES || OS_module_table[possible_moduleid].name[0] != '\0')
236  {
237  OS_InterruptSafeUnlock(&OS_module_table_mut, &previous);
238  return OS_ERR_NO_FREE_IDS;
239  }
240 
241  /*
242  ** Check to see if the module file is already loaded
243  */
244  for (i = 0; i < OS_MAX_MODULES; i++)
245  {
246  if ((OS_module_table[i].name[0] != '\0') &&
247  ( strcmp((char*) module_name, OS_module_table[i].name) == 0))
248  {
249  OS_InterruptSafeUnlock(&OS_module_table_mut, &previous);
250  return OS_ERR_NAME_TAKEN;
251  }
252  }
253 
254  /*
255  ** Set the possible task Id to not free so that
256  ** no other task can try to use it
257  */
258  OS_module_table[possible_moduleid].name[0] = '\n' ;
259  OS_InterruptSafeUnlock(&OS_module_table_mut, &previous);
260 
261  /*
262  ** Translate the filename to the Host System
263  */
264  return_code = OS_TranslatePath((const char *)filename, (char *)translated_path);
265  if ( return_code != OS_SUCCESS )
266  {
267  OS_module_table[possible_moduleid].name[0] = '\0';
268  return(return_code);
269  }
270 
271  /*
272  ** File is ready to load
273  */
274 
275  /*
276  ** Open the loadable bundle .. just opening it loads it into the system.
277  */
278  function_lib = dlopen(translated_path, RTLD_NOW | RTLD_GLOBAL);
279  dl_error = dlerror();
280  if( dl_error )
281  {
282  OS_printf("OSAL: dlopen() failed. errno=%u. %s\n", errno, dl_error);
283  OS_module_table[possible_moduleid].name[0] = '\0';
284  return(OS_ERROR);
285  }
286 
287  /*
288  ** fill out the OS_module_table entry for this new module
289  */
290  OS_module_table[possible_moduleid].entry_point = 0; /* Only for certain targets */
291  OS_module_table[possible_moduleid].host_module_id = (uint32) function_lib;
292  strncpy(OS_module_table[possible_moduleid].filename , filename, OS_MAX_PATH_LEN);
293  strncpy(OS_module_table[possible_moduleid].name , module_name, OS_MAX_API_NAME);
294 
295  /*
296  ** For now, do not store the module address information
297  ** Let the OS_ModuleInfo function fetch that information and return it.
298  */
299  OS_module_table[possible_moduleid].addr.valid = FALSE;
300 
301  /*
302  ** Return the OSAPI Module ID
303  */
304  *module_id = possible_moduleid;
305 
306  return(OS_SUCCESS);
307 
308 }/* end OS_ModuleLoad */
