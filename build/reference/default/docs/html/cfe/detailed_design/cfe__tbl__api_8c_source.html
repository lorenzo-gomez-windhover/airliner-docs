83 /*******************************************************************
84 **
85 ** CFE_TBL_Register() -- Register a table with cFE to obtain
86 ** Table Management Services
87 **
88 ** NOTE: For complete prolog information, see 'cfe_tbl.h'
89 ********************************************************************/
90 int32 CFE_TBL_Register( CFE_TBL_Handle_t *TblHandlePtr,
91  const char *Name,
92  uint32 Size,
93  uint16 TblOptionFlags,
94  CFE_TBL_CallbackFuncPtr_t TblValidationFuncPtr )
95 {
96  CFE_TBL_AccessDescriptor_t *AccessDescPtr = NULL;
97  CFE_TBL_RegistryRec_t *RegRecPtr = NULL;
98  CFE_TBL_LoadBuff_t *WorkingBufferPtr;
99  CFE_TBL_CritRegRec_t *CritRegRecPtr = NULL;
100  int32 Status = CFE_SUCCESS;
101  size_t NameLen = 0;
102  int16 RegIndx = -1;
103  uint32 ThisAppId;
104  char AppName[OS_MAX_API_NAME] = {"UNKNOWN"};
105  char TblName[CFE_TBL_MAX_FULL_NAME_LEN] = {""};
106  CFE_TBL_Handle_t AccessIndex;
107 
108  /* Check to make sure calling application is legit */
109  Status = CFE_TBL_ValidateAppID(&ThisAppId);
110 
111  if (Status == CFE_SUCCESS)
112  {
113  /* Assume we can't make a table and return a bad handle for now */
114  *TblHandlePtr = CFE_TBL_BAD_TABLE_HANDLE;
115 
116  /* Make sure specified table name is not too long or too short */
117  NameLen = strlen(Name);
118  if ((NameLen > CFE_TBL_MAX_NAME_LENGTH) || (NameLen == 0))
119  {
120  Status = CFE_TBL_ERR_INVALID_NAME;
121 
122  /* Perform a buffer overrun safe copy of name for debug log message */
123  strncpy(TblName, Name, CFE_TBL_MAX_NAME_LENGTH);
124  TblName[CFE_TBL_MAX_NAME_LENGTH-1] = '\0';
125  CFE_ES_WriteToSysLog("CFE_TBL:Register-Table Name (%s) is bad length (%d)",TblName,(int)NameLen);
126  }
127  else
128  {
129  /* Modify specified name to be processor specific name */
130  /* of the form "AppName.TableName" */
131  CFE_TBL_FormTableName(TblName, Name, ThisAppId);
132 
133  /* Make sure the specified size is acceptable */
134  /* Single buffered tables are allowed to be up to CFE_TBL_MAX_SNGL_TABLE_SIZE */
135  /* Double buffered tables are allowed to be up to CFE_TBL_MAX_DBL_TABLE_SIZE */
136  if (Size == 0)
137  {
138  Status = CFE_TBL_ERR_INVALID_SIZE;
139 
140  CFE_ES_WriteToSysLog("CFE_TBL:Register-Table %s has size of zero\n", Name);
141  }
142  else if ((Size > CFE_TBL_MAX_SNGL_TABLE_SIZE) &&
143  ((TblOptionFlags & CFE_TBL_OPT_BUFFER_MSK) == CFE_TBL_OPT_SNGL_BUFFER))
144  {
145  Status = CFE_TBL_ERR_INVALID_SIZE;
146 
147  CFE_ES_WriteToSysLog("CFE_TBL:Register-Single Buffered Table '%s' has size %d > %d\n",
148  Name, (int)Size, CFE_TBL_MAX_SNGL_TABLE_SIZE);
149  }
150  else if ((Size > CFE_TBL_MAX_DBL_TABLE_SIZE) &&
151  ((TblOptionFlags & CFE_TBL_OPT_BUFFER_MSK) == CFE_TBL_OPT_DBL_BUFFER))
152  {
153  Status = CFE_TBL_ERR_INVALID_SIZE;
154 
155  CFE_ES_WriteToSysLog("CFE_TBL:Register-Dbl Buffered Table '%s' has size %d > %d\n",
156  Name, (int)Size, CFE_TBL_MAX_DBL_TABLE_SIZE);
157  }
158 
159  /* Verify Table Option settings are legal */
160  /* User defined table addresses are only legal for single buffered, dump-only, non-critical tables */
161  if ((TblOptionFlags & CFE_TBL_OPT_USR_DEF_MSK) == (CFE_TBL_OPT_USR_DEF_ADDR & CFE_TBL_OPT_USR_DEF_MSK))
162  {
163  if (((TblOptionFlags & CFE_TBL_OPT_BUFFER_MSK) == CFE_TBL_OPT_DBL_BUFFER) ||
164  ((TblOptionFlags & CFE_TBL_OPT_LD_DMP_MSK) == CFE_TBL_OPT_LOAD_DUMP) ||
165  ((TblOptionFlags & CFE_TBL_OPT_CRITICAL_MSK) == CFE_TBL_OPT_CRITICAL))
166  {
167  Status = CFE_TBL_ERR_INVALID_OPTIONS;
168 
169  CFE_ES_WriteToSysLog("CFE_TBL:Register-User Def tbl '%s' cannot be dbl buff, load/dump or critical\n",
170  Name);
171  }
172  }
173  else if ((TblOptionFlags & CFE_TBL_OPT_LD_DMP_MSK) == CFE_TBL_OPT_DUMP_ONLY)
174  {
175  /* Dump Only tables cannot be double buffered, nor critical */
176  if (((TblOptionFlags & CFE_TBL_OPT_BUFFER_MSK) == CFE_TBL_OPT_DBL_BUFFER) ||
177  ((TblOptionFlags & CFE_TBL_OPT_CRITICAL_MSK) == CFE_TBL_OPT_CRITICAL))
178  {
179  Status = CFE_TBL_ERR_INVALID_OPTIONS;
180 
181  CFE_ES_WriteToSysLog("CFE_TBL:Register-Dump Only tbl '%s' cannot be double buffered or critical\n",
182  Name);
183  }
184  }
185  }
186  }
187  else /* Application ID was invalid */
188  {
189  CFE_ES_WriteToSysLog("CFE_TBL:Register-Bad AppId(%d)\n", (int)ThisAppId);
190  }
191 
192  /* If input parameters appear acceptable, register the table */
193  if (Status == CFE_SUCCESS)
194  {
195  /* Lock Registry for update. This prevents two applications from */
196  /* trying to register/share tables at the same location at the same time */
197  CFE_TBL_LockRegistry();
198 
199  /* Check for duplicate table name */
200  RegIndx = CFE_TBL_FindTableInRegistry(TblName);
201 
202  /* Check to see if table is already in the registry */
203  if (RegIndx != CFE_TBL_NOT_FOUND)
204  {
205  /* Get pointer to Registry Record Entry to speed up processing */
206  RegRecPtr = &CFE_TBL_TaskData.Registry[RegIndx];
207 
208  /* If this app previously owned the table, then allow them to re-register */
209  if (RegRecPtr->OwnerAppId == ThisAppId)
210  {
211  /* If the new table is the same size as the old, then no need to reallocate memory */
212  if (Size != RegRecPtr->Size)
213  {
214  /* If the new size is different, the old table must deleted */
215  /* but this function can't do that because it is probably shared */
216  /* and is probably still being accessed. Someone else will need */
217  /* to clean up this mess. */
218  Status = CFE_TBL_ERR_DUPLICATE_DIFF_SIZE;
219 
220  CFE_ES_WriteToSysLog("CFE_TBL:Register-Attempt to register existing table ('%s') with different size(%d!=%d)\n",
221  TblName, (int)Size, (int)RegRecPtr->Size);
222  }
223  else
224  {
225  /* Warn calling application that this is a duplicate registration */
226  Status = CFE_TBL_WARN_DUPLICATE;
227 
228  /* Find the existing access descriptor for the table */
229  /* and return the same handle that was returned previously */
230  AccessIndex = RegRecPtr->HeadOfAccessList;
231  while ((AccessIndex != CFE_TBL_END_OF_LIST) && (*TblHandlePtr == CFE_TBL_BAD_TABLE_HANDLE))
232  {
233  if ((CFE_TBL_TaskData.Handles[AccessIndex].UsedFlag == TRUE) &&
234  (CFE_TBL_TaskData.Handles[AccessIndex].AppId == ThisAppId) &&
235  (CFE_TBL_TaskData.Handles[AccessIndex].RegIndex == RegIndx))
236  {
237  *TblHandlePtr = AccessIndex;
238  }
239  else
240  {
241  AccessIndex = CFE_TBL_TaskData.Handles[AccessIndex].NextLink;
242  }
243  }
244  }
245  }
246  else /* Duplicate named table owned by another Application */
247  {
248  Status = CFE_TBL_ERR_DUPLICATE_NOT_OWNED;
249 
250  CFE_ES_WriteToSysLog("CFE_TBL:Register-App(%d) Registering Duplicate Table '%s' owned by App(%d)\n",
251  (int)ThisAppId, TblName, (int)RegRecPtr->OwnerAppId);
252  }
253  }
254  else /* Table not already in registry */
255  {
256  /* Locate empty slot in table registry */
257  RegIndx = CFE_TBL_FindFreeRegistryEntry();
258  }
259 
260  /* Check to make sure we found a free entry in registry */
261  if (RegIndx == CFE_TBL_NOT_FOUND)
262  {
263  Status = CFE_TBL_ERR_REGISTRY_FULL;
264  CFE_ES_WriteToSysLog("CFE_TBL:Register-Registry full\n");
265  }
266 
267  /* If this is a duplicate registration, no other work is required */
268  if (Status != CFE_TBL_WARN_DUPLICATE)
269  {
270  /* Search Access Descriptor Array for free Descriptor */
271  *TblHandlePtr = CFE_TBL_FindFreeHandle();
272 
273  /* Check to make sure there was a handle available */
274  if (*TblHandlePtr == CFE_TBL_END_OF_LIST)
275  {
276  Status = CFE_TBL_ERR_HANDLES_FULL;
277  CFE_ES_WriteToSysLog("CFE_TBL:Register-No more free handles\n");
278  }
279 
280  /* If no errors, then initialize the table registry entry */
281  /* and return the registry index to the caller as the handle */
282  if ((Status & CFE_SEVERITY_BITMASK) != CFE_SEVERITY_ERROR)
283  {
284  /* Get pointer to Registry Record Entry to speed up processing */
285  RegRecPtr = &CFE_TBL_TaskData.Registry[RegIndx];
286 
287  /* Initialize Registry Record to default settings */
288  CFE_TBL_InitRegistryRecord(RegRecPtr);
289 
290  if ((TblOptionFlags & CFE_TBL_OPT_USR_DEF_MSK) != (CFE_TBL_OPT_USR_DEF_ADDR & CFE_TBL_OPT_USR_DEF_MSK))
291  {
292  RegRecPtr->UserDefAddr = FALSE;
293 
294  /* Allocate the memory buffer(s) for the table and inactive table, if necessary */
295  Status = CFE_ES_GetPoolBuf((uint32 **)&RegRecPtr->Buffers[0].BufferPtr,
296  CFE_TBL_TaskData.Buf.PoolHdl,
297  Size);
298  if(Status < 0)
299  {
300  CFE_ES_WriteToSysLog("CFE_TBL:Register-1st Buf Alloc GetPool fail Stat=0x%08X MemPoolHndl=0x%08lX\n",
301  (unsigned int)Status, (unsigned long)CFE_TBL_TaskData.Buf.PoolHdl);
302  }
303  else
304  {
305  /* Zero the memory buffer */
306  Status = CFE_SUCCESS;
307  CFE_PSP_MemSet(RegRecPtr->Buffers[0].BufferPtr, 0x0, Size);
308  }
309  }
310  else
311  {
312  /* Set buffer pointer to NULL for user defined address tables */
313  RegRecPtr->Buffers[0].BufferPtr = NULL;
314  RegRecPtr->UserDefAddr = TRUE;
315  }
316 
317  if (((TblOptionFlags & CFE_TBL_OPT_DBL_BUFFER) == CFE_TBL_OPT_DBL_BUFFER) &&
318  ((Status & CFE_SEVERITY_BITMASK) != CFE_SEVERITY_ERROR))
319  {
320  /* Allocate memory for the dedicated secondary buffer */
321  Status = CFE_ES_GetPoolBuf((uint32 **)&RegRecPtr->Buffers[1].BufferPtr,
322  CFE_TBL_TaskData.Buf.PoolHdl,
323  Size);
324  if(Status < 0)
325  {
326  CFE_ES_WriteToSysLog("CFE_TBL:Register-2nd Buf Alloc GetPool fail Stat=0x%08X MemPoolHndl=0x%08lX\n",
327  (unsigned int)Status, (unsigned long)CFE_TBL_TaskData.Buf.PoolHdl);
328  }
329  else
330  {
331  /* Zero the dedicated secondary buffer */
332  Status = CFE_SUCCESS;
333  CFE_PSP_MemSet(RegRecPtr->Buffers[1].BufferPtr, 0x0, Size);
334  }
335 
336  RegRecPtr->ActiveBufferIndex = 0;
337  RegRecPtr->DblBuffered = TRUE;
338  }
339  else /* Single Buffered Table */
340  {
341  RegRecPtr->DblBuffered = FALSE;
342  RegRecPtr->ActiveBufferIndex = 0;
343  }
344 
345  if ((Status & CFE_SEVERITY_BITMASK) != CFE_SEVERITY_ERROR)
346  {
347  /* Save the size of the table */
348  RegRecPtr->Size = Size;
349 
350  /* Save the Callback function pointer */
351  RegRecPtr->ValidationFuncPtr = TblValidationFuncPtr;
352 
353  /* Save Table Name in Registry */
354  strncpy(RegRecPtr->Name, TblName, CFE_TBL_MAX_FULL_NAME_LEN);
355 
356  /* Set the "Dump Only" flag to value based upon selected option */
357  if ((TblOptionFlags & CFE_TBL_OPT_LD_DMP_MSK) == CFE_TBL_OPT_DUMP_ONLY)
358  {
359  RegRecPtr->DumpOnly = TRUE;
360  }
361  else
362  {
363  RegRecPtr->DumpOnly = FALSE;
364  }
365 
366  /* Initialize the Table Access Descriptor */
367  AccessDescPtr = &CFE_TBL_TaskData.Handles[*TblHandlePtr];
368 
369  AccessDescPtr->AppId = ThisAppId;
370  AccessDescPtr->LockFlag = FALSE;
371  AccessDescPtr->Updated = FALSE;
372 
373  if ((RegRecPtr->DumpOnly) && (!RegRecPtr->UserDefAddr))
374  {
375  /* Dump Only Tables are assumed to be loaded at all times */
376  /* unless the address is specified by the application. In */
377  /* that case, it isn't loaded until the address is specified */
378  RegRecPtr->TableLoadedOnce = TRUE;
379  }
380 
381  AccessDescPtr->RegIndex = RegIndx;
382 
383  AccessDescPtr->PrevLink = CFE_TBL_END_OF_LIST; /* We are the head of the list */
384  AccessDescPtr->NextLink = CFE_TBL_END_OF_LIST; /* We are the end of the list */
385 
386  AccessDescPtr->UsedFlag = TRUE;
387 
388  /* Make sure the Table Registry entry points to First Access Descriptor */
389  RegRecPtr->HeadOfAccessList = *TblHandlePtr;
390 
391  /* If the table is a critical table, allocate space for it in the Critical Data Store */
392  /* OR locate its previous incarnation there and extract its previous contents */
393  if ((TblOptionFlags & CFE_TBL_OPT_CRITICAL_MSK) == CFE_TBL_OPT_CRITICAL)
394  {
395  /* Register a CDS under the table name and determine if the table already exists there */
396  Status = CFE_ES_RegisterCDSEx(&RegRecPtr->CDSHandle, Size, TblName, TRUE);
397 
398  if (Status == CFE_ES_CDS_ALREADY_EXISTS)
399  {
400  Status = CFE_TBL_GetWorkingBuffer(&WorkingBufferPtr, RegRecPtr, TRUE);
401 
402  if (Status != CFE_SUCCESS)
403  {
404  /* Unable to get a working buffer - this error is not really */
405  /* possible at this point during table registration. But we */
406  /* do need to handle the error case because if the function */
407  /* call did fail, WorkingBufferPtr would be a NULL pointer. */
408  CFE_ES_GetAppName(AppName, ThisAppId, OS_MAX_API_NAME);
409  CFE_ES_WriteToSysLog("CFE_TBL:Register-Failed to get work buffer for '%s.%s' (ErrCode=0x%08X)\n",
410  AppName, Name, (unsigned int)Status);
411  }
412  else
413  {
414  /* CDS exists for this table - try to restore the data */
415  Status = CFE_ES_RestoreFromCDS(WorkingBufferPtr->BufferPtr, RegRecPtr->CDSHandle);
416 
417  if (Status != CFE_SUCCESS)
418  {
419  CFE_ES_GetAppName(AppName, ThisAppId, OS_MAX_API_NAME);
420  CFE_ES_WriteToSysLog("CFE_TBL:Register-Failed to recover '%s.%s' from CDS (ErrCode=0x%08X)\n",
421  AppName, Name, (unsigned int)Status);
422  }
423  }
424 
425  if (Status != CFE_SUCCESS)
426  {
427  /* Treat a restore from existing CDS error the same as */
428  /* after a power-on reset (CDS was created but is empty) */
429  Status = CFE_SUCCESS;
430  }
431  else
432  {
433  /* Try to locate the associated information in the Critical Table Registry */
434  CFE_TBL_FindCriticalTblInfo(&CritRegRecPtr, RegRecPtr->CDSHandle);
435 
436  if ((CritRegRecPtr != NULL) && (CritRegRecPtr->TableLoadedOnce))
437  {
438  strncpy(WorkingBufferPtr->DataSource, CritRegRecPtr->LastFileLoaded, OS_MAX_PATH_LEN);
439  WorkingBufferPtr->FileCreateTimeSecs = CritRegRecPtr->FileCreateTimeSecs;
440  WorkingBufferPtr->FileCreateTimeSubSecs = CritRegRecPtr->FileCreateTimeSubSecs;
441  RegRecPtr->TimeOfLastUpdate.Seconds = CritRegRecPtr->TimeOfLastUpdate.Seconds;
442  RegRecPtr->TimeOfLastUpdate.Subseconds = CritRegRecPtr->TimeOfLastUpdate.Subseconds;
443  RegRecPtr->TableLoadedOnce = CritRegRecPtr->TableLoadedOnce;
444 
445  /* Compute the CRC on the specified table buffer */
446  WorkingBufferPtr->Crc = CFE_ES_CalculateCRC(WorkingBufferPtr->BufferPtr,
447  RegRecPtr->Size,
448  0,
449  CFE_ES_DEFAULT_CRC);
450 
451  /* Make sure everyone who sees the table knows that it has been updated */
452  CFE_TBL_NotifyTblUsersOfUpdate(RegRecPtr);
453 
454  /* Make sure the caller realizes the contents have been initialized */
455  Status = CFE_TBL_INFO_RECOVERED_TBL;
456  }
457  else
458  {
459  /* If an error occurred while trying to get the previous contents registry info, */
460  /* Log the error in the System Log and pretend like we created a new CDS */
461  CFE_ES_GetAppName(AppName, ThisAppId, OS_MAX_API_NAME);
462  CFE_ES_WriteToSysLog("CFE_TBL:Register-Failed to recover '%s.%s' info from CDS TblReg\n",
463  AppName, Name);
464  Status = CFE_SUCCESS;
465  }
466  }
467 
468  /* Mark the table as critical for future reference */
469  RegRecPtr->CriticalTable = TRUE;
470  }
471 
472  if (Status == CFE_SUCCESS)
473  {
474  /* Find and initialize a free entry in the Critical Table Registry */
475  CFE_TBL_FindCriticalTblInfo(&CritRegRecPtr, CFE_ES_CDS_BAD_HANDLE);
476 
477  if (CritRegRecPtr != NULL)
478  {
479  CritRegRecPtr->CDSHandle = RegRecPtr->CDSHandle;
480  strncpy(CritRegRecPtr->Name, TblName, CFE_TBL_MAX_FULL_NAME_LEN);
481  CritRegRecPtr->FileCreateTimeSecs = 0;
482  CritRegRecPtr->FileCreateTimeSubSecs = 0;
483  CritRegRecPtr->LastFileLoaded[0] = '\0';
484  CritRegRecPtr->TimeOfLastUpdate.Seconds = 0;
485  CritRegRecPtr->TimeOfLastUpdate.Subseconds = 0;
486  CritRegRecPtr->TableLoadedOnce = FALSE;
487 
488  CFE_ES_CopyToCDS(CFE_TBL_TaskData.CritRegHandle, CFE_TBL_TaskData.CritReg);
489  }
490  else
491  {
492  CFE_ES_WriteToSysLog("CFE_TBL:Register-Failed to find a free Crit Tbl Reg Rec for '%s'\n",
493  RegRecPtr->Name);
494  }
495 
496  /* Mark the table as critical for future reference */
497  RegRecPtr->CriticalTable = TRUE;
498  }
499  else if (Status != CFE_TBL_INFO_RECOVERED_TBL)
500  {
501  CFE_ES_WriteToSysLog("CFE_TBL:Register-Failed to register '%s.%s' as a CDS (ErrCode=0x%08X)\n",
502  AppName, Name, (unsigned int)Status);
503 
504  /* Notify caller that although they asked for it to be critical, it isn't */
505  Status = CFE_TBL_WARN_NOT_CRITICAL;
506  }
507  }
508 
509  /* The last step of the registration process is claiming ownership. */
510  /* By making it the last step, other APIs do not have to lock registry */
511  /* to share the table or get its address because registry entries that */
512  /* are unowned are not checked to see if they match names, etc. */
513  RegRecPtr->OwnerAppId = ThisAppId;
514  }
515  }
516  }
517 
518  /* Unlock Registry for update */
519  CFE_TBL_UnlockRegistry();
520  }
521 
522  /* On Error conditions, notify ground of screw up */
523  if (Status < 0)
524  {
525  /* Make sure the returned handle is invalid when an error occurs */
526  *TblHandlePtr = CFE_TBL_BAD_TABLE_HANDLE;
527 
528  /* Translate AppID of caller into App Name */
529  CFE_ES_GetAppName(AppName, ThisAppId, OS_MAX_API_NAME);
530 
531  CFE_EVS_SendEventWithAppID(CFE_TBL_REGISTER_ERR_EID,
532  CFE_EVS_ERROR,
533  CFE_TBL_TaskData.TableTaskAppId,
534  "%s Failed to Register '%s', Status=0x%08X",
535  AppName, TblName, (unsigned int)Status);
536  }
537 
538  return Status;
539 } /* End of CFE_TBL_Register() */

545 ** application
546 **
547 ** NOTE: For complete prolog information, see 'cfe_tbl.h'
548 ********************************************************************/

550 int32 CFE_TBL_Share( CFE_TBL_Handle_t *TblHandlePtr,
551  const char *TblName )
552 {
553  int32 Status = CFE_SUCCESS;
554  uint32 ThisAppId;
555  int16 RegIndx = CFE_TBL_NOT_FOUND;
556  CFE_TBL_AccessDescriptor_t *AccessDescPtr = NULL;
557  CFE_TBL_RegistryRec_t *RegRecPtr = NULL;
558  char AppName[OS_MAX_API_NAME] = {"UNKNOWN"};
559 
560  /* Get a valid Application ID for calling App */
561  Status = CFE_TBL_ValidateAppID(&ThisAppId);
562 
563  if (Status == CFE_SUCCESS)
564  {
565  /* Lock Registry for update. This prevents two applications from */
566  /* trying to register/share tables at the same location at the same time */
567  CFE_TBL_LockRegistry();
568 
569  RegIndx = CFE_TBL_FindTableInRegistry(TblName);
570 
571  /* If we found the table, then get a new Access Descriptor and initialize it */
572  if (RegIndx != CFE_TBL_NOT_FOUND)
573  {
574  /* Get pointer to Registry Record Entry to speed up processing */
575  RegRecPtr = &CFE_TBL_TaskData.Registry[RegIndx];
576 
577  /* Search Access Descriptor Array for free Descriptor */
578  *TblHandlePtr = CFE_TBL_FindFreeHandle();
579 
580  /* Check to make sure there was a handle available */
581  if (*TblHandlePtr == CFE_TBL_END_OF_LIST)
582  {
583  Status = CFE_TBL_ERR_HANDLES_FULL;
584  CFE_ES_WriteToSysLog("CFE_TBL:Share-No more free handles\n");
585  }
586  else
587  {
588  /* Initialize the Table Access Descriptor */
589  AccessDescPtr = &CFE_TBL_TaskData.Handles[*TblHandlePtr];
590 
591  AccessDescPtr->AppId = ThisAppId;
592  AccessDescPtr->LockFlag = FALSE;
593  AccessDescPtr->Updated = FALSE;
594 
595  /* Check current state of table in order to set Notification flags properly */
596  if (RegRecPtr->TableLoadedOnce)
597  {
598  AccessDescPtr->Updated = TRUE;
599  }
600 
601  AccessDescPtr->RegIndex = RegIndx;
602  AccessDescPtr->UsedFlag = TRUE;
603 
604  AccessDescPtr->PrevLink = CFE_TBL_END_OF_LIST; /* We are the new head of the list */
605  AccessDescPtr->NextLink = RegRecPtr->HeadOfAccessList;
606 
607  /* Make sure the old head of the list now sees this as the head */
608  CFE_TBL_TaskData.Handles[RegRecPtr->HeadOfAccessList].PrevLink = *TblHandlePtr;
609 
610  /* Make sure the Registry Record see this as the head of the list */
611  RegRecPtr->HeadOfAccessList = *TblHandlePtr;
612  }
613  }
614  else /* Table could not be found in registry */
615  {
616  Status = CFE_TBL_ERR_INVALID_NAME;
617 
618  CFE_ES_WriteToSysLog("CFE_TBL:Share-Table '%s' not found in Registry\n", TblName);
619  }
620 
621  CFE_TBL_UnlockRegistry();
622  }
623  else /* Application ID was invalid */
624  {
625  CFE_ES_WriteToSysLog("CFE_TBL:Share-Bad AppId(%d)\n", (int)ThisAppId);
626  }
627 
628  /* On Error conditions, notify ground of screw up */
629  if (Status < 0)
630  {
631  /* Translate AppID of caller into App Name */
632  CFE_ES_GetAppName(AppName, ThisAppId, OS_MAX_API_NAME);
633 
634  CFE_EVS_SendEventWithAppID(CFE_TBL_SHARE_ERR_EID,
635  CFE_EVS_ERROR,
636  CFE_TBL_TaskData.TableTaskAppId,
637  "%s Failed to Share '%s', Status=0x%08X",
638  AppName, TblName, (unsigned int)Status);
639  }
640 
641  return Status;
642 } /* End of CFE_TBL_Share() */

714 /*******************************************************************
715 **
716 ** CFE_TBL_Load() -- Load a specified table with data from the
717 ** specified source
718 **
719 ** NOTE: For complete prolog information, see 'cfe_tbl.h'
720 ********************************************************************/
721 
722 int32 CFE_TBL_Load( CFE_TBL_Handle_t TblHandle,
723  CFE_TBL_SrcEnum_t SrcType,
724  const void *SrcDataPtr )
725 {
726  int32 Status;
727  uint32 ThisAppId;
728  CFE_TBL_LoadBuff_t *WorkingBufferPtr;
729  CFE_TBL_RegistryRec_t *RegRecPtr = NULL;
730  CFE_TBL_AccessDescriptor_t *AccessDescPtr;
731  char AppName[OS_MAX_API_NAME]={"UNKNOWN"};
732  uint16 EventMsgType = CFE_EVS_INFORMATION;
733  boolean FirstTime = FALSE;
734 
735 
736  /* Initialize return pointer to NULL */
737  WorkingBufferPtr = NULL;
738 
739  /* Verify access rights and get a valid Application ID for calling App */
740  Status = CFE_TBL_ValidateAccess(TblHandle, &ThisAppId);
741 
742  if (Status == CFE_SUCCESS)
743  {
744  /* Get pointers to pertinent records in registry and handles */
745  AccessDescPtr = &CFE_TBL_TaskData.Handles[TblHandle];
746  RegRecPtr = &CFE_TBL_TaskData.Registry[AccessDescPtr->RegIndex];
747 
748  /* Check to see if this is a dump only table */
749  if (RegRecPtr->DumpOnly)
750  {
751  if ((RegRecPtr->UserDefAddr) && (!RegRecPtr->TableLoadedOnce))
752  {
753  /* The Application is allowed to call Load once when the address */
754  /* of the dump only table is being defined by the application. */
755  RegRecPtr->Buffers[0].BufferPtr = (void *)SrcDataPtr;
756  RegRecPtr->TableLoadedOnce = TRUE;
757 
758  snprintf(RegRecPtr->Buffers[0].DataSource, sizeof(RegRecPtr->Buffers[0].DataSource),
759  "Addr 0x%08lX", (unsigned long)SrcDataPtr);
760  RegRecPtr->Buffers[0].FileCreateTimeSecs = 0;
761  RegRecPtr->Buffers[0].FileCreateTimeSubSecs = 0;
762 
763  CFE_EVS_SendEventWithAppID(CFE_TBL_LOAD_SUCCESS_INF_EID,
764  CFE_EVS_DEBUG,
765  CFE_TBL_TaskData.TableTaskAppId,
766  "Successfully loaded '%s' from '%s'",
767  RegRecPtr->Name,
768  RegRecPtr->Buffers[0].DataSource);
769  }
770  else
771  {
772  Status = CFE_TBL_ERR_DUMP_ONLY;
773 
774  CFE_ES_WriteToSysLog("CFE_TBL:Load-App(%d) attempted to load Dump Only Tbl '%s'\n",
775  (int)ThisAppId, RegRecPtr->Name);
776  }
777  }
778  else
779  {
780  /* Loads by an Application are not allowed if a table load is already in progress */
781  if (RegRecPtr->LoadInProgress != CFE_TBL_NO_LOAD_IN_PROGRESS)
782  {
783  Status = CFE_TBL_ERR_LOAD_IN_PROGRESS;
784 
785  CFE_ES_WriteToSysLog("CFE_TBL:Load-Tbl Load already in progress for '%s'\n", RegRecPtr->Name);
786  }
787  else
788  {
789  /* Obtain a working buffer (either the table's dedicated buffer or one of the shared buffers) */
790  Status = CFE_TBL_GetWorkingBuffer(&WorkingBufferPtr, RegRecPtr, TRUE);
791 
792  if (Status == CFE_SUCCESS)
793  {
794  /* Perform appropriate update to working buffer */
795  /* Determine whether the load is to occur from a file or from a block of memory */
796  if (SrcType == CFE_TBL_SRC_FILE)
797  {
798  /* Load the data from the file into the specified buffer */
799  Status = CFE_TBL_LoadFromFile(WorkingBufferPtr, RegRecPtr, (const char *)SrcDataPtr);
800 
801  if (Status < 0)
802  {
803  CFE_ES_WriteToSysLog("CFE_TBL:Load-App(%d) Fail to load Tbl '%s' from '%s' (Stat=0x%08X)\n",
804  (int)ThisAppId, RegRecPtr->Name, (const char *)SrcDataPtr, (unsigned int)Status);
805  }
806  else if ((Status == CFE_TBL_WARN_PARTIAL_LOAD) && (!RegRecPtr->TableLoadedOnce))
807  {
808  /* Uninitialized tables cannot be loaded with partial table loads */
809  /* Partial loads can only occur on previously loaded tables. */
810  Status = CFE_TBL_ERR_PARTIAL_LOAD;
811  }
812  }
813  else if (SrcType == CFE_TBL_SRC_ADDRESS)
814  {
815  /* When the source is a block of memory, it is assumed to be a complete load */
816  CFE_PSP_MemCpy(WorkingBufferPtr->BufferPtr,
817  (uint8 *)SrcDataPtr,
818  RegRecPtr->Size);
819 
820  snprintf(WorkingBufferPtr->DataSource, sizeof(WorkingBufferPtr->DataSource), "Addr 0x%08lX", (unsigned long)SrcDataPtr);
821  WorkingBufferPtr->FileCreateTimeSecs = 0;
822  WorkingBufferPtr->FileCreateTimeSubSecs = 0;
823 
824  /* Compute the CRC on the specified table buffer */
825  WorkingBufferPtr->Crc = CFE_ES_CalculateCRC(WorkingBufferPtr->BufferPtr,
826  RegRecPtr->Size,
827  0,
828  CFE_ES_DEFAULT_CRC);
829  }
830  else
831  {
832  Status = CFE_TBL_ERR_ILLEGAL_SRC_TYPE;
833  CFE_ES_WriteToSysLog("CFE_TBL:Load-App(%d) attempt to load from illegal source type=%d\n",
834  (int)ThisAppId, (int)SrcType);
835  }
836 
837  /* If the data was successfully loaded, then validate its contents */
838  if ((Status >= CFE_SUCCESS) && (RegRecPtr->ValidationFuncPtr != NULL))
839  {
840  Status = (RegRecPtr->ValidationFuncPtr)(WorkingBufferPtr->BufferPtr);
841 
842  if (Status > CFE_SUCCESS)
843  {
844  CFE_ES_WriteToSysLog("CFE_TBL:Load-App(%d) Validation func return code invalid (Stat=0x%08X) for '%s'\n",
845  (int)ThisAppId, (unsigned int)Status, RegRecPtr->Name);
846  Status = -1;
847  }
848 
849  if (Status < 0)
850  {
851  CFE_ES_WriteToSysLog("CFE_TBL:Load-App(%d) reports load invalid (Stat=0x%08X) for '%s'\n",
852  (int)ThisAppId, (unsigned int)Status, RegRecPtr->Name);
853 
854  /* Zero out the buffer to remove any bad data */
855  CFE_PSP_MemSet(WorkingBufferPtr->BufferPtr, 0, RegRecPtr->Size);
856  }
857  }
858 
859  /* Perform the table update to complete the load */
860  if (Status >= CFE_SUCCESS)
861  {
862  FirstTime = !RegRecPtr->TableLoadedOnce;
863 
864  /* If this is not the first load, then the data must be moved from the inactive buffer */
865  /* to the active buffer to complete the load. First loads are done directly to the active. */
866  if (!FirstTime)
867  {
868  /* Force the table update */
869  RegRecPtr->LoadPending = TRUE;
870 
871  Status = CFE_TBL_UpdateInternal(TblHandle, RegRecPtr, AccessDescPtr);
872  }
873  else
874  {
875  /* On initial loads, make sure registry is given file/address of data source */
876  strncpy(RegRecPtr->LastFileLoaded,
877  WorkingBufferPtr->DataSource,
878  OS_MAX_PATH_LEN);
879 
880  CFE_TBL_NotifyTblUsersOfUpdate(RegRecPtr);
881 
882  /* If the table is a critical table, update the appropriate CDS with the new data */
883  if (RegRecPtr->CriticalTable == TRUE)
884  {
885  CFE_TBL_UpdateCriticalTblCDS(RegRecPtr);
886  }
887 
888  Status = CFE_SUCCESS;
889  }
890 
891  if (Status != CFE_SUCCESS)
892  {
893  CFE_ES_WriteToSysLog("CFE_TBL:Load-App(%d) fail to update '%s' (Stat=0x%08X)\n",
894  (int)ThisAppId, RegRecPtr->Name, (unsigned int)Status);
895  }
896  else
897  {
898  /* The first time a table is loaded, the event message is DEBUG */
899  /* to help eliminate a flood of events during a startup */
900  if (FirstTime)
901  {
902  EventMsgType = CFE_EVS_DEBUG;
903  }
904 
905  CFE_EVS_SendEventWithAppID(CFE_TBL_LOAD_SUCCESS_INF_EID,
906  EventMsgType,
907  CFE_TBL_TaskData.TableTaskAppId,
908  "Successfully loaded '%s' from '%s'",
909  RegRecPtr->Name,
910  RegRecPtr->LastFileLoaded);
911 
912  /* Save the index of the table for housekeeping telemetry */
913  CFE_TBL_TaskData.LastTblUpdated = AccessDescPtr->RegIndex;
914  }
915  }
916  else
917  {
918  /* The load has had a problem, free the working buffer for another attempt */
919  if ((!RegRecPtr->DblBuffered) && (RegRecPtr->TableLoadedOnce == TRUE))
920  {
921  /* For single buffered tables, freeing entails resetting flag */
922  CFE_TBL_TaskData.LoadBuffs[RegRecPtr->LoadInProgress].Taken = FALSE;
923  }
924 
925  /* For double buffered tables, freeing buffer is simple */
926  RegRecPtr->LoadInProgress = CFE_TBL_NO_LOAD_IN_PROGRESS;
927  }
928  }
929  else
930  {
931  CFE_ES_WriteToSysLog("CFE_TBL:Load-App(%d) Failed to get Working Buffer (Stat=0x%08X)\n",
932  (int)ThisAppId, (unsigned int)Status);
933  }
934  }
935  }
936  }
937  else
938  {
939  CFE_ES_WriteToSysLog("CFE_TBL:Load-App(%d) does not have access to Tbl Handle=%d\n",
940  (int)ThisAppId, (int)TblHandle);
941  }
942 
943  /* On Error conditions, notify ground of screw up */
944  if (Status < 0)
945  {
946  /* Translate AppID of caller into App Name */
947  CFE_ES_GetAppName(AppName, ThisAppId, OS_MAX_API_NAME);
948 
949  if (RegRecPtr == NULL)
950  {
951  CFE_EVS_SendEventWithAppID(CFE_TBL_LOAD_ERR_EID,
952  CFE_EVS_ERROR,
953  CFE_TBL_TaskData.TableTaskAppId,
954  "%s Failed to Load '?', Status=0x%08X",
955  AppName, (unsigned int)Status);
956  }
957  else
958  {
959  if (SrcType == CFE_TBL_SRC_ADDRESS)
960  {
961  CFE_EVS_SendEventWithAppID(CFE_TBL_LOAD_ERR_EID,
962  CFE_EVS_ERROR,
963  CFE_TBL_TaskData.TableTaskAppId,
964  "%s Failed to Load '%s' from Addr 0x%08lX, Status=0x%08X",
965  AppName, RegRecPtr->Name, (unsigned long)SrcDataPtr, (unsigned int)Status);
966  }
967  else if (SrcType == CFE_TBL_SRC_FILE)
968  {
969  CFE_EVS_SendEventWithAppID(CFE_TBL_LOAD_ERR_EID,
970  CFE_EVS_ERROR,
971  CFE_TBL_TaskData.TableTaskAppId,
972  "%s Failed to Load '%s' from '%s', Status=0x%08X",
973  AppName, RegRecPtr->Name, (const char *)SrcDataPtr, (unsigned int)Status);
974  }
975  else
976  {
977  CFE_EVS_SendEventWithAppID(CFE_TBL_LOAD_TYPE_ERR_EID,
978  CFE_EVS_ERROR,
979  CFE_TBL_TaskData.TableTaskAppId,
980  "%s Failed to Load '%s' (Invalid Source Type)",
981  AppName, RegRecPtr->Name);
982  }
983  }
984  }
985 
986  return Status;
987 } /* End of CFE_TBL_Load() */
