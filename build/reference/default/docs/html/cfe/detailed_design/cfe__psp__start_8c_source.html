182 /******************************************************************************
183 ** Function: main()
184 **
185 ** Purpose:
186 ** BSP Application entry point.
187 **
188 ** Arguments:
189 ** (none)
190 **
191 ** Return:
192 ** (none)
193 */
194 void PSP_Main(int argc, char *argv[])
195 {
196  uint32 reset_type;
197  uint32 reset_subtype;
198  struct sigaction sa;
199  struct itimerval timer;
200  int opt = 0;
201  int longIndex = 0;
202  sigset_t mask;
203  pthread_t thread_id;
204  struct sched_param thread_params;
205  int ret;
206  int32 os_timer;
207  CFE_PSP_CommandData_t CommandData;
208 
209 #ifdef CFE_PSP_EEPROM_SUPPORT
210  int32 Status;
211  uint32 eeprom_address;
212  uint32 eeprom_size;
213 #endif
214 
215  /*
216  ** Initialize the CommandData struct
217  */
218  memset(&(CommandData), 0, sizeof(CFE_PSP_CommandData_t));
219 
220  /*
221  ** Process the arguments with getopt_long(), then
222  ** start the cFE
223  */
224  opt = getopt_long( argc, argv, optString, longOpts, &longIndex );
225  while( opt != -1 )
226  {
227  switch( opt )
228  {
229  case 'R':
230  strncpy(CommandData.ResetType, optarg, CFE_PSP_RESET_NAME_LENGTH);
231  if ((strncmp(CommandData.ResetType, "PO", CFE_PSP_RESET_NAME_LENGTH ) != 0 ) &&
232  (strncmp(CommandData.ResetType, "PR", CFE_PSP_RESET_NAME_LENGTH ) != 0 ))
233  {
234  printf("\nERROR: Invalid Reset Type: %s\n\n", CommandData.ResetType);
235  CommandData.GotResetType = 0;
236  CFE_PSP_DisplayUsage(argv[0]);
237  break;
238  }
239  printf("CFE_PSP: Reset Type: %s\n",(char *)optarg);
240  CommandData.GotResetType = 1;
241  break;
242 
243  case 'S':
244  CommandData.SubType = strtol(optarg, NULL, 0 );
245  if ( CommandData.SubType < 1 || CommandData.SubType > 5 )
246  {
247  printf("\nERROR: Invalid Reset SubType: %s\n\n", optarg);
248  CommandData.SubType = 0;
249  CommandData.GotSubType = 0;
250  CFE_PSP_DisplayUsage(argv[0]);
251  break;
252  }
253  printf("CFE_PSP: Reset SubType: %d\n",(int)CommandData.SubType);
254  CommandData.GotSubType = 1;
255  break;
256 
257  case 'N':
258  strncpy(CommandData.CpuName, optarg, CFE_PSP_CPU_NAME_LENGTH );
259  printf("CFE_PSP: CPU Name: %s\n", CommandData.CpuName);
260  CommandData.GotCpuName = 1;
261  break;
262 
263  case 'C':
264  CommandData.CpuId = strtol(optarg, NULL, 0 );
265  printf("CFE_PSP: CPU ID: %d\n",(int)CommandData.CpuId);
266  CommandData.GotCpuId = 1;
267  break;
268 
269  case 'I':
270  CommandData.SpacecraftId = strtol(optarg, NULL, 0 );
271  printf("CFE_PSP: Spacecraft ID: %d\n",(int)CommandData.SpacecraftId);
272  CommandData.GotSpacecraftId = 1;
273  break;
274 
275  case 'h':
276  CFE_PSP_DisplayUsage(argv[0]);
277  break;
278 
279  default:
280  CFE_PSP_DisplayUsage(argv[0]);
281  break;
282  }
283 
284  opt = getopt_long( argc, argv, optString, longOpts, &longIndex );
285  } /* end while */
286 
287  /*
288  ** Set the defaults for values that were not given for the
289  ** optional arguments, and check for arguments that are required.
290  */
291  CFE_PSP_ProcessArgumentDefaults(&CommandData);
292 
293  /*
294  ** Set the reset type
295  */
296  if (strncmp("PR", CommandData.ResetType, 2 ) == 0 )
297  {
298  reset_type = CFE_ES_PROCESSOR_RESET;
299  printf("CFE_PSP: Starting the cFE with a PROCESSOR reset.\n");
300  }
301  else
302  {
303  reset_type = CFE_ES_POWERON_RESET;
304  printf("CFE_PSP: Starting the cFE with a POWER ON reset.\n");
305  }
306 
307  /*
308  ** Assign the Spacecraft ID, CPU ID, and CPU Name
309  */
310  CFE_PSP_GlobalData.CFE_PSP_SpacecraftId = CommandData.SpacecraftId;
311  CFE_PSP_GlobalData.CFE_PSP_CpuId = CommandData.CpuId;
312  strncpy(CFE_PSP_GlobalData.CFE_PSP_CpuName, CommandData.CpuName, CFE_PSP_CPU_NAME_LENGTH);
313 
314  /*
315  ** Set the reset subtype
316  */
317  reset_subtype = CommandData.SubType;
318 
319  /*
320  ** Install sigint_handler as the signal handler for SIGINT.
321  */
322  signal(SIGINT, CFE_PSP_SigintHandler);
323 
324  /*
325  ** Init timer counter
326  */
327  CFE_PSP_GlobalData.TimerCounter = 0;
328 
329  /*
330  ** Install timer_handler as the signal handler for SIGVTALRM.
331  */
332  memset (&sa, 0, sizeof (sa));
333  sa.sa_handler = &CFE_PSP_TimerHandler;
334  sigaction (SIGALRM, &sa, NULL);
335 
336  /*
337  ** Configure the timer to expire after 250ms
338  */
339  timer.it_value.tv_sec = 0;
340  timer.it_value.tv_usec = 250000;
341 
342  /*
343  ** and every 500ms after that.
344  */
345  timer.it_interval.tv_sec = 0;
346  timer.it_interval.tv_usec = 250000;
347 
348  /*
349  ** Initialize the OS API data structures
350  */
351  OS_API_Init();
352 
353  sleep(1);
354 
355  /*
356  ** Initialize the reserved memory
357  */
358  CFE_PSP_InitProcessorReservedMemory(reset_type);
359 
360 
361  /*
362  ** Start the timer
363  */
364  setitimer (ITIMER_REAL, &timer, NULL);
365 
366 #ifdef CFE_PSP_EEPROM_SUPPORT
367  /*
368  ** Create the simulated EEPROM segment by mapping a memory segment to a file.
369  ** Since the file will be saved, the "EEPROM" contents will be preserved.
370  ** Set up 512Kbytes of EEPROM
371  */
372  eeprom_size = 0x80000;
373  Status = CFE_PSP_SetupEEPROM(eeprom_size, &eeprom_address);
374 
375  if ( Status == 0 )
376  {
377  uint32 Dword;
378  /*
379  ** Install the 2nd memory range as the mapped file ( EEPROM )
380  */
381  Status = CFE_PSP_MemRangeSet (1, CFE_PSP_MEM_EEPROM, eeprom_address,
382  eeprom_size, CFE_PSP_MEM_SIZE_DWORD, 0 );
383  printf("CFE_PSP: EEPROM Range (2) created: Start Address = %08X, Size = %08X\n", eeprom_address, eeprom_size);
384 
385  }
386  else
387  {
388  printf("CFE_PSP: Cannot create EEPROM Range from Memory Mapped file.\n");
389  }
390 #endif
391 
392  /*
393  ** Disable Signals to parent thread and therefore all
394  ** child threads create will block all signals
395  ** Note: Timers will not work in the application unless
396  ** threads are spawned in OS_Application_Startup.
397  */
398  sigfillset(&mask);
399  sigdelset(&mask, SIGINT); /* Needed to kill program */
400  sigprocmask(SIG_SETMASK, &mask, NULL);
401 
402 
403  /*
404  ** Raise the priority of the main thread so ES Main completes
405  */
406  thread_id = pthread_self();
407  thread_params.sched_priority = sched_get_priority_max(SCHED_FIFO);
408  ret = pthread_setschedparam(thread_id, SCHED_FIFO, &thread_params);
409  if ( ret != 0 )
410  {
411  printf("Unable to set main thread priority to max\n");
412  }
413 
414  /*
415  ** Call cFE entry point.
416  */
417  CFE_ES_Main(reset_type, reset_subtype, 1, (uint8 *)CFE_ES_NONVOL_STARTUP_FILE);
418 
419  CFE_PSP_DrawBanner();
420 
421  //uint32 clock_accuracy;
422  //OS_TimerCreate(&os_timer, "CFE_PSP_TIMER", &clock_accuracy, CFE_PSP_TimerHandler);
423  //OS_TimerSet(os_timer, 250000, 250000);
424 
425  /*
426  ** Re-enable Signals to current thread so that
427  ** any signals will interrupt in this threads context
428  ** ... this is needed for timers
429  */
430  sigprocmask(SIG_UNBLOCK, &mask, NULL);
431 
432  /*
433  ** Lower the thread priority to before entering the sleep loop
434  */
435  thread_params.sched_priority = sched_get_priority_min(SCHED_FIFO);
436  ret = pthread_setschedparam(thread_id, SCHED_FIFO, &thread_params);
437  if ( ret != 0 )
438  {
439  printf("Unable to set main thread priority to minimum\n");
440  }
441 }
