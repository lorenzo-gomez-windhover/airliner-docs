175 void CFE_ES_TaskMain(void)
176 {
177  int32 Status;
178  uint32 TimeOut = CFE_ES_APP_SCAN_RATE;
179  uint32 AppRunStatus = CFE_ES_RUNSTATUS_APP_RUN;
180 
181 
182  /*
183  ** Performance Time Stamp Entry
184  */
185  CFE_ES_PerfLogEntry(CFE_ES_MAIN_PERF_ID);
186 
187  /*
188  ** Perform task specific initialization.
189  */
190  Status = CFE_ES_TaskInit();
191  if ( Status != CFE_SUCCESS )
192  {
193  /*
194  ** Create a syslog entry
195  */
196  CFE_ES_WriteToSysLog("ES:Application Init Failed,RC=0x%08X\n", (unsigned int)Status);
197 
198 
199  /*
200  ** Allow Core App to Exit
201  */
202  AppRunStatus = CFE_ES_RUNSTATUS_CORE_APP_INIT_ERROR;
203 
204  } /* end if */
205 
206  /*
207  * Wait for other apps to start.
208  * It is important that the core apps are present before this starts receiving
209  * messages from the command pipe, as some of those handlers might depend on
210  * the other core apps.
211  */
212  CFE_ES_WaitForStartupSync(CFE_CORE_MAX_STARTUP_MSEC);
213 
214  /*
215  ** Main process loop
216  */
217  while (AppRunStatus == CFE_ES_RUNSTATUS_APP_RUN)
218  {
219  /*
220  ** Increment the main task execution counter
221  ** This is normally done in the CFE_ES_RunLoop call, but
222  ** currently CFE Child tasks and the cFE core tasks do not
223  ** use the RunLoop call.
224  */
225  CFE_ES_IncrementTaskCounter();
226 
227  /*
228  ** Performance Time Stamp Exit
229  */
230  CFE_ES_PerfLogExit(CFE_ES_MAIN_PERF_ID);
231 
232  /*
233  ** Wait for the next Software Bus message.
234  */
235  Status = CFE_SB_RcvMsg(&CFE_ES_TaskData.MsgPtr,
236  CFE_ES_TaskData.CmdPipe,
237  TimeOut);
238 
239  /*
240  ** Performance Time Stamp Entry
241  */
242  CFE_ES_PerfLogEntry(CFE_ES_MAIN_PERF_ID);
243 
244  /*
245  ** Scan the App table for Application Deletion requests
246  */
247  if ( Status == CFE_SB_TIME_OUT )
248  {
249  CFE_ES_ScanAppTable();
250  }
251  else if (Status == CFE_SUCCESS)
252  {
253  /*
254  ** Process Software Bus message.
255  */
256  CFE_ES_TaskPipe(CFE_ES_TaskData.MsgPtr);
257 
258  /*
259  ** Scan the App Table for changes after processing a command
260  */
261  CFE_ES_ScanAppTable();
262  }
263  else
264  {
265  /*
266  ** SB Error: Write a SysLog Message
267  */
268  CFE_ES_WriteToSysLog("ES:Error reading cmd pipe,RC=0x%08X\n",(unsigned int)Status);
269 
270  /*
271  ** Allow Core App to Exit
272  */
273  AppRunStatus = CFE_ES_RUNSTATUS_CORE_APP_RUNTIME_ERROR;
274 
275  } /* end if */
276 
277  } /* end while */
278 
279  /*
280  ** Performance Time Stamp Exit
281  */
282  CFE_ES_PerfLogExit(CFE_ES_MAIN_PERF_ID);
283 
284  /*
285  ** Exit the application, CFE_ES_ExitApp will not return.
286  */
287  CFE_ES_ExitApp(AppRunStatus);
288 
289 } /* End of CFE_ES_TaskMain() */
