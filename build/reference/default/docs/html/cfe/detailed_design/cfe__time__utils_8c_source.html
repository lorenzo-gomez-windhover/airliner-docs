114 /* CFE_TIME_QueryResetVars() -- query contents of Reset Variables */

116 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
117 
118 void CFE_TIME_QueryResetVars(void)
119 {
120 
121  CFE_TIME_ResetVars_t LocalResetVars;
122  uint32 DefSubsMET;
123  uint32 DefSubsSTCF;
124  int32 status;
125 
126  uint32 resetAreaSize;
127  cpuaddr resetAreaAddr;
128  CFE_ES_ResetData_t *CFE_TIME_ResetDataPtr;
129 
130  /*
131  ** Get the pointer to the Reset area from the BSP
132  */
133  status = CFE_PSP_GetResetArea (&(resetAreaAddr), &(resetAreaSize));
134 
135  if (status != CFE_PSP_SUCCESS)
136  {
137  /* There is something wrong with the Reset Area */
138  CFE_TIME_TaskData.DataStoreStatus = CFE_TIME_RESET_AREA_BAD;
139  }
140 
141  else
142  {
143  CFE_TIME_ResetDataPtr = (CFE_ES_ResetData_t *)resetAreaAddr;
144 
145  /* Get the structure from the Reset Area */
146  LocalResetVars = CFE_TIME_ResetDataPtr -> TimeResetVars;
147 
148  /*
149  ** Verify TIME data signature and clock signal selection...
150  ** (other data fields have no verifiable limits)
151  */
152  if ((LocalResetVars.Signature == CFE_TIME_RESET_SIGNATURE) &&
153  ((LocalResetVars.ClockSignal == CFE_TIME_TONE_PRI) ||
154  (LocalResetVars.ClockSignal == CFE_TIME_TONE_RED)))
155  {
156  /*
157  ** Initialize TIME to valid Reset Area values...
158  */
159  CFE_TIME_TaskData.AtToneMET = LocalResetVars.CurrentMET;
160  CFE_TIME_TaskData.AtToneSTCF = LocalResetVars.CurrentSTCF;
161  CFE_TIME_TaskData.AtToneDelay = LocalResetVars.CurrentDelay;
162  CFE_TIME_TaskData.AtToneLeaps = LocalResetVars.LeapSeconds;
163  CFE_TIME_TaskData.ClockSignal = LocalResetVars.ClockSignal;
164 
165  CFE_TIME_TaskData.DataStoreStatus = CFE_TIME_RESET_AREA_EXISTING;
166  }
167  else
168  {
169  /*
170  ** We got a blank area from the reset variables
171  */
172  CFE_TIME_TaskData.DataStoreStatus = CFE_TIME_RESET_AREA_NEW;
173  }
174 
175  }
176  /*
177  ** Initialize TIME to default values if no valid Reset data...
178  */
179  if (CFE_TIME_TaskData.DataStoreStatus != CFE_TIME_RESET_AREA_EXISTING)
180  {
181  DefSubsMET = CFE_TIME_Micro2SubSecs(CFE_TIME_DEF_MET_SUBS);
182  DefSubsSTCF = CFE_TIME_Micro2SubSecs(CFE_TIME_DEF_STCF_SUBS);
183 
184  CFE_TIME_TaskData.AtToneMET.Seconds = CFE_TIME_DEF_MET_SECS;
185  CFE_TIME_TaskData.AtToneMET.Subseconds = DefSubsMET;
186  CFE_TIME_TaskData.AtToneSTCF.Seconds = CFE_TIME_DEF_STCF_SECS;
187  CFE_TIME_TaskData.AtToneSTCF.Subseconds = DefSubsSTCF;
188  CFE_TIME_TaskData.AtToneLeaps = CFE_TIME_DEF_LEAPS;
189  CFE_TIME_TaskData.ClockSignal = CFE_TIME_TONE_PRI;
190  CFE_TIME_TaskData.AtToneDelay.Seconds = 0;
191  CFE_TIME_TaskData.AtToneDelay.Subseconds = 0;
192  }
193 
194  return;
195 
196 } /* End of CFE_TIME_QueryResetVars() */

206 {

208  uint32 resetAreaSize;
209  cpuaddr resetAreaAddr;
210  CFE_ES_ResetData_t *CFE_TIME_ResetDataPtr;
211  /*
212  ** Update the data only if our Reset Area is valid...
213  */
214  if (CFE_TIME_TaskData.DataStoreStatus != CFE_TIME_RESET_AREA_ERROR)
215  {
216 
217  /* Store all of our critical variables to a ResetVars_t
218  * then copy that to the Reset Area */
219  LocalResetVars.Signature = CFE_TIME_RESET_SIGNATURE;
220 
221  LocalResetVars.CurrentMET = Reference->CurrentMET;
222  LocalResetVars.CurrentSTCF = Reference->AtToneSTCF;
223  LocalResetVars.CurrentDelay = Reference->AtToneDelay;

227 
228  /*
229  ** Get the pointer to the Reset area from the BSP
230  */
231  if (CFE_PSP_GetResetArea (&(resetAreaAddr), &(resetAreaSize)) == CFE_PSP_SUCCESS)
232  {
233  CFE_TIME_ResetDataPtr = (CFE_ES_ResetData_t *)resetAreaAddr;
234  CFE_TIME_ResetDataPtr -> TimeResetVars = LocalResetVars;
235  }
236  }
237 
238  return;

444 uint16 CFE_TIME_GetStateFlags(void)
445 {
446  uint16 StateFlags = 0;
447 
448  /*
449  ** Spacecraft time has been set...
450  */
451  if (CFE_TIME_TaskData.ClockSetState == CFE_TIME_WAS_SET)
452  {
453  StateFlags |= CFE_TIME_FLAG_CLKSET;
454  }
455  /*
456  ** This instance of Time Service is in FLYWHEEL mode...
457  */
458  if (CFE_TIME_TaskData.ClockFlyState == CFE_TIME_IS_FLY)
459  {
460  StateFlags |= CFE_TIME_FLAG_FLYING;
461  }
462  /*
463  ** Clock source set to "internal"...
464  */
465  if (CFE_TIME_TaskData.ClockSource == CFE_TIME_USE_INTERN)
466  {
467  StateFlags |= CFE_TIME_FLAG_SRCINT;
468  }
469  /*
470  ** Clock signal set to "primary"...
471  */
472  if (CFE_TIME_TaskData.ClockSignal == CFE_TIME_TONE_PRI)
473  {
474  StateFlags |= CFE_TIME_FLAG_SIGPRI;
475  }
476  /*
477  ** Time Server is in FLYWHEEL mode...
478  */
479  if (CFE_TIME_TaskData.ServerFlyState == CFE_TIME_IS_FLY)
480  {
481  StateFlags |= CFE_TIME_FLAG_SRVFLY;
482  }
483  /*
484  ** This instance of Time Services commanded into FLYWHEEL...
485  */
486  if (CFE_TIME_TaskData.Forced2Fly)
487  {
488  StateFlags |= CFE_TIME_FLAG_CMDFLY;
489  }
490  /*
491  ** One time STCF adjustment direction...
492  */
493  if (CFE_TIME_TaskData.OneTimeDirection == CFE_TIME_ADD_ADJUST)
494  {
495  StateFlags |= CFE_TIME_FLAG_ADDADJ;
496  }
497  /*
498  ** 1 Hz STCF adjustment direction...
499  */
500  if (CFE_TIME_TaskData.OneHzDirection == CFE_TIME_ADD_ADJUST)
501  {
502  StateFlags |= CFE_TIME_FLAG_ADD1HZ;
503  }
504  /*
505  ** Time Client Latency adjustment direction...
506  */
507  if (CFE_TIME_TaskData.DelayDirection == CFE_TIME_ADD_ADJUST)
508  {
509  StateFlags |= CFE_TIME_FLAG_ADDTCL;
510  }
511  /*
512  ** This instance of Time Service is a "server"...
513  */
514  #if (CFE_TIME_CFG_SERVER == TRUE)
515  StateFlags |= CFE_TIME_FLAG_SERVER;
516  #endif
517 
518  return(StateFlags);
519 
520 } /* End of CFE_TIME_GetStateFlags() */

706 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
707 /* */
708 /* CFE_TIME_GetReference() -- get reference data (time at "tone") */
709 /* */
710 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
711 
712 void CFE_TIME_GetReference(CFE_TIME_Reference_t *Reference)
713 {
714  CFE_TIME_SysTime_t TimeSinceTone;
715  CFE_TIME_SysTime_t CurrentMET;
716  uint32 VersionCount;
717 
718  /*
719  ** VersionCount is incremented when reference data is modified...
720  */
721  do
722  {
723  VersionCount = CFE_TIME_TaskData.VersionCount;
724 
725  Reference->CurrentLatch = CFE_TIME_LatchClock();
726 
727  Reference->AtToneMET = CFE_TIME_TaskData.AtToneMET;
728  Reference->AtToneSTCF = CFE_TIME_TaskData.AtToneSTCF;
729  Reference->AtToneLeaps = CFE_TIME_TaskData.AtToneLeaps;
730  Reference->AtToneDelay = CFE_TIME_TaskData.AtToneDelay;
731  Reference->AtToneLatch = CFE_TIME_TaskData.AtToneLatch;
732 
733  Reference->ClockSetState = CFE_TIME_TaskData.ClockSetState;
734  Reference->ClockFlyState = CFE_TIME_TaskData.ClockFlyState;
735 
736  } while (VersionCount != CFE_TIME_TaskData.VersionCount);
737 
738  /*
739  ** Compute the amount of time "since" the tone...
740  */
741  if (CFE_TIME_Compare(Reference->CurrentLatch, Reference->AtToneLatch) == CFE_TIME_A_LT_B)
742  {
743  /*
744  ** Local clock has rolled over since last tone...
745  */
746  TimeSinceTone = CFE_TIME_Subtract(CFE_TIME_TaskData.MaxLocalClock, Reference->AtToneLatch);
747  TimeSinceTone = CFE_TIME_Add(TimeSinceTone, Reference->CurrentLatch);
748  }
749  else
750  {
751  /*
752  ** Normal case -- local clock is greater than latch at tone...
753  */
754  TimeSinceTone = CFE_TIME_Subtract(Reference->CurrentLatch, Reference->AtToneLatch);
755  }
756 
757  Reference->TimeSinceTone = TimeSinceTone;
758 
759  /*
760  ** Add in the MET at the tone...
761  */
762  CurrentMET = CFE_TIME_Add(TimeSinceTone, Reference->AtToneMET);
763 
764 
765  /*
766  ** Synchronize "this" time client to the time server...
767  */
768  #if (CFE_TIME_CFG_CLIENT == TRUE)
769  if (CFE_TIME_TaskData.DelayDirection == CFE_TIME_ADD_ADJUST)
770  {
771  CurrentMET = CFE_TIME_Add(CurrentMET, Reference->AtToneDelay);
772  }
773  else
774  {
775  CurrentMET = CFE_TIME_Subtract(CurrentMET, Reference->AtToneDelay);
776  }
777  #endif
778 
779  Reference->CurrentMET = CurrentMET;
780 
781  return;
782 
783 } /* End of CFE_TIME_GetReference() */
