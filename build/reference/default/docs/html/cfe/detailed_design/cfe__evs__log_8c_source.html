91 /*
92 ** Function Prologue
93 **
94 ** Function Name: EVS_AddLog
95 **
96 ** Purpose: This routine adds an event packet to the internal event log.
97 **
98 ** Assumptions and Notes:
99 **
100 */
101 void EVS_AddLog (CFE_EVS_Packet_t *EVS_PktPtr)
102 {
103 
104  if (CFE_EVS_GlobalData.EVS_TlmPkt.Payload.LogEnabled == TRUE)
105  {
106  /* Serialize access to event log control variables */
107  OS_MutSemTake(CFE_EVS_GlobalData.EVS_SharedDataMutexID);
108 
109  if ((CFE_EVS_GlobalData.EVS_LogPtr->LogFullFlag == TRUE) &&
110  (CFE_EVS_GlobalData.EVS_LogPtr->LogMode == CFE_EVS_LOG_DISCARD))
111  {
112  /* If log is full and in discard mode, just count the event */
113  CFE_EVS_GlobalData.EVS_LogPtr->LogOverflowCounter++;
114  }
115  else
116  {
117  if (CFE_EVS_GlobalData.EVS_LogPtr->LogFullFlag == TRUE)
118  {
119  /* If log is full and in wrap mode, count it and store it */
120  CFE_EVS_GlobalData.EVS_LogPtr->LogOverflowCounter++;
121  }
122 
123  /* Copy the event data to the next available entry in the log */
124  CFE_PSP_MemCpy(&CFE_EVS_GlobalData.EVS_LogPtr->LogEntry[CFE_EVS_GlobalData.EVS_LogPtr->Next],
125  EVS_PktPtr, sizeof(CFE_EVS_Packet_t));
126 
127  CFE_EVS_GlobalData.EVS_LogPtr->Next++;
128 
129  if (CFE_EVS_GlobalData.EVS_LogPtr->Next >= CFE_EVS_LOG_MAX)
130  {
131  /* This is important, even if we are in discard mode */
132  CFE_EVS_GlobalData.EVS_LogPtr->Next = 0;
133  }
134 
135  /* Log count cannot exceed the number of entries in the log */
136  if (CFE_EVS_GlobalData.EVS_LogPtr->LogCount < CFE_EVS_LOG_MAX)
137  {
138  CFE_EVS_GlobalData.EVS_LogPtr->LogCount++;
139 
140  if (CFE_EVS_GlobalData.EVS_LogPtr->LogCount == CFE_EVS_LOG_MAX)
141  {
142  /* The full flag and log count are somewhat redundant */
143  CFE_EVS_GlobalData.EVS_LogPtr->LogFullFlag = TRUE;
144  }
145  }
146  }
147 
148  OS_MutSemGive(CFE_EVS_GlobalData.EVS_SharedDataMutexID);
149  }
150 
151  return;
152 
153 } /* End EVS_AddLog */

159 ** Function Name: EVS_ClearLog
160 **
161 ** Purpose: This routine clears the contents of the internal event log.
162 **
163 ** Assumptions and Notes:
164 **
165 */
166 void EVS_ClearLog ( void )
167 {
168 
169  /* Serialize access to event log control variables */
170  OS_MutSemTake(CFE_EVS_GlobalData.EVS_SharedDataMutexID);
171 
172  /* Clears everything but LogMode (overwrite vs discard) */
173  CFE_EVS_GlobalData.EVS_LogPtr->Next = 0;
174  CFE_EVS_GlobalData.EVS_LogPtr->LogCount = 0;
175  CFE_EVS_GlobalData.EVS_LogPtr->LogFullFlag = FALSE;
176  CFE_EVS_GlobalData.EVS_LogPtr->LogOverflowCounter = 0;
177 
178  CFE_PSP_MemSet(CFE_EVS_GlobalData.EVS_LogPtr->LogEntry, 0,
179  CFE_EVS_LOG_MAX * sizeof(CFE_EVS_Packet_t));
180 
181  OS_MutSemGive(CFE_EVS_GlobalData.EVS_SharedDataMutexID);
182 
183  return;
184 
185 } /* End EVS_ClearLog */

188 /*
189 ** Function Prologue
190 **
191 ** Function Name: CFE_EVS_WriteLogFileCmd
192 **
193 ** Purpose: This routine writes the contents of the internal event log to a file
194 **
195 ** Assumptions and Notes:
196 **
197 */
198 boolean CFE_EVS_WriteLogFileCmd (CFE_SB_MsgPayloadPtr_t Payload)
199 {
200  const CFE_EVS_LogFileCmd_Payload_t *CmdPtr = (const CFE_EVS_LogFileCmd_Payload_t *)Payload;
201  boolean Result = FALSE;
202  int32 LogIndex;
203  int32 BytesWritten;
204  int32 LogFileHandle;
205  uint32 i;
206  CFE_FS_Header_t LogFileHdr;
207  char LogFilename[OS_MAX_PATH_LEN];
208 
209 
210  /* Serialize access to event log control variables */
211  OS_MutSemTake(CFE_EVS_GlobalData.EVS_SharedDataMutexID);
212 
213  /* Copy the commanded filename into local buffer to ensure size limitation and to allow for modification */
214  CFE_SB_MessageStringGet(LogFilename, (const char *)CmdPtr->LogFilename, CFE_EVS_DEFAULT_LOG_FILE,
215  OS_MAX_PATH_LEN, sizeof(CmdPtr->LogFilename));
216 
217  /* Create the log file */
218  LogFileHandle = OS_creat(LogFilename, OS_WRITE_ONLY);
219 
220  if (LogFileHandle < OS_FS_SUCCESS)
221  {
222  EVS_SendEvent(CFE_EVS_ERR_CRLOGFILE_EID, CFE_EVS_ERROR,
223  "Write Log File Command Error: OS_creat = 0x%08X, filename = %s",
224  LogFileHandle, LogFilename);
225  }
226  else
227  {
228  /* Initialize cFE file header for an event log file */
229  CFE_FS_InitHeader(&LogFileHdr, "cFE EVS Log File", CFE_FS_EVS_EVENTLOG_SUBTYPE);
230 
231  /* Write the file header to the log file */
232  BytesWritten = CFE_FS_WriteHeader(LogFileHandle, &LogFileHdr);
233 
234  if (BytesWritten == sizeof(CFE_FS_Header_t))
235  {
236  /* Is the log full? -- Doesn't matter if wrap mode is enabled */
237  if (CFE_EVS_GlobalData.EVS_LogPtr->LogCount == CFE_EVS_LOG_MAX)
238  {
239  /* Start with log entry that will be overwritten next (oldest) */
240  LogIndex = CFE_EVS_GlobalData.EVS_LogPtr->Next;
241  }
242  else
243  {
244  /* Start with the first entry in the log (oldest) */
245  LogIndex = 0;
246  }
247 
248  /* Write all the "in-use" event log entries to the file */
249  for (i = 0; i < CFE_EVS_GlobalData.EVS_LogPtr->LogCount; i++)
250  {
251  BytesWritten = OS_write(LogFileHandle,
252  &CFE_EVS_GlobalData.EVS_LogPtr->LogEntry[LogIndex],
253  sizeof(CFE_EVS_Packet_t));
254 
255  if (BytesWritten == sizeof(CFE_EVS_Packet_t))
256  {
257  LogIndex++;
258 
259  if (LogIndex >= CFE_EVS_LOG_MAX)
260  {
261  LogIndex = 0;
262  }
263  }
264  else
265  {
266  EVS_SendEvent(CFE_EVS_ERR_WRLOGFILE_EID, CFE_EVS_ERROR,
267  "Write Log File Command Error: OS_write = 0x%08X, filename = %s",
268  (unsigned int)BytesWritten, LogFilename);
269  break;
270  }
271  }
272 
273  /* Process command handler success result */
274  if (i == CFE_EVS_GlobalData.EVS_LogPtr->LogCount)
275  {
276  EVS_SendEvent(CFE_EVS_WRLOG_EID, CFE_EVS_DEBUG,
277  "Write Log File Command: %d event log entries written to %s",
278  (int)CFE_EVS_GlobalData.EVS_LogPtr->LogCount, LogFilename);
279  Result = TRUE;
280  }
281  }
282 
283  OS_close(LogFileHandle);
284  }
285 
286  OS_MutSemGive(CFE_EVS_GlobalData.EVS_SharedDataMutexID);
287 
288  return(Result);
289 
290 } /* End CFE_EVS_WriteLogFileCmd */

293 /*
294 ** Function Prologue
295 **
296 ** Function Name: CFE_EVS_SetLoggingModeCmd
297 **
298 ** Purpose: This routine sets the internal event log mode.
299 **
300 ** Assumptions and Notes:
301 **
302 */
303 boolean CFE_EVS_SetLoggingModeCmd (CFE_SB_MsgPayloadPtr_t Payload)
304 {
305  CFE_EVS_ModeCmd_Payload_t *CmdPtr = (CFE_EVS_ModeCmd_Payload_t *)Payload;
306  boolean Status = TRUE;
307 
308  if ((CmdPtr->Mode == CFE_EVS_LOG_OVERWRITE) || (CmdPtr->Mode == CFE_EVS_LOG_DISCARD))
309  {
310  /* Serialize access to event log control variables */
311  OS_MutSemTake(CFE_EVS_GlobalData.EVS_SharedDataMutexID);
312  CFE_EVS_GlobalData.EVS_LogPtr->LogMode = CmdPtr->Mode;
313  OS_MutSemGive(CFE_EVS_GlobalData.EVS_SharedDataMutexID);
314 
315  EVS_SendEvent(CFE_EVS_LOGMODE_EID, CFE_EVS_DEBUG,
316  "Set Log Mode Command: Log Mode = %d", (int)CmdPtr->Mode);
317  }
318  else
319  {
320  Status = FALSE;
321  EVS_SendEvent(CFE_EVS_ERR_LOGMODE_EID, CFE_EVS_ERROR,
322  "Set Log Mode Command Error: Log Mode = %d", (int)CmdPtr->Mode);
323  }
324 
325  return Status;
