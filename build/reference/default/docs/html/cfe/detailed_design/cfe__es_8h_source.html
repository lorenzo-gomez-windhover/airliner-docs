437 /*
438 ** \brief Compatibility wrapper for older versions of OSAL
439 **
440 ** \par Description
441 ** In future versions of OSAL the task/object ID values might not be zero based
442 ** If that is the case then the OSAL must also define a function to convert back
443 ** to zero-based numbers such that the value can be used as an array index.
444 **
445 ** When using an existing/older version of OSAL, this inline function is defined
446 ** to mimic this call for backward compatibility. It just passes through the same
447 ** value without modification.
448 **
449 ** \param[in] ObjectId The object ID from OSAL
450 ** \param[out] ArrayIndex A zero-based value suitable for use as an array index
451 ** \returns OS_SUCCESS (a real version might return an error code).
452 */
453 static inline int32 OS_ConvertToArrayIndex(uint32 ObjectId, uint32 *ArrayIndex)
454 {
455  *ArrayIndex = ObjectId;
456  return OS_SUCCESS;
457 }
