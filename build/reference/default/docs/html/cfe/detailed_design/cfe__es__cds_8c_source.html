188 int32 CFE_ES_CDS_EarlyInit(void)
189 {
190  uint32 MinRequiredSize = 0;
191  int32 Status = CFE_SUCCESS;
192 
193  /* Compute the minimum size required for the CDS with the current configuration of the cFE */
194  MinRequiredSize = (sizeof(CFE_ES_Global.CDSVars.ValidityField) * 2) + /* Minimum size for validity fields */
195  (CFE_ES_CDS_MAX_NUM_ENTRIES * sizeof(CFE_ES_CDS_RegRec_t)) + /* Minimum size for CDS Registry contents */
196  CFE_ES_CDSReqdMinSize(CFE_ES_CDS_MAX_NUM_ENTRIES); /* Max # of Min Sized Blocks */
197 
198  /* Get CDS size from OS BSP */
199  Status = CFE_PSP_GetCDSSize(&CFE_ES_Global.CDSVars.CDSSize);
200 
201  /* If the size was obtained successfully and meets the minimum requirements, then check its contents */
202  if (Status == CFE_PSP_SUCCESS)
203  {
204  /* Always truncate the size to the nearest 4 byte boundary */
205  CFE_ES_Global.CDSVars.CDSSize &= 0xfffffffc;
206 
207  if (CFE_ES_Global.CDSVars.CDSSize >= MinRequiredSize)
208  {
209  Status = CFE_ES_ValidateCDS();
210 
211  /* If the CDS is accessible but invalid, then create a new one */
212  if (Status == CFE_ES_CDS_INVALID)
213  {
214  Status = CFE_ES_InitializeCDS(CFE_ES_Global.CDSVars.CDSSize);
215  }
216  else if (Status == CFE_SUCCESS)
217  {
218  /* If a valid CDS was found, rebuild the memory pool */
219  Status = CFE_ES_RebuildCDS();
220 
221  /* If rebuilding is not possible, then create a new one from scratch */
222  if (Status == CFE_ES_CDS_INVALID)
223  {
224  Status = CFE_ES_InitializeCDS(CFE_ES_Global.CDSVars.CDSSize);
225  }
226  }
227  else /* Unrecoverable error while reading the CDS */
228  {
229  CFE_ES_WriteToSysLog("CFE_CDS:EarlyInit-Read error validating CDS (Err=0x%08X)\n", (unsigned int)Status);
230  CFE_ES_Global.CDSVars.MemPoolSize = 0;
231  return Status;
232  }
233 
234  if (Status == CFE_SUCCESS)
235  {
236  /* Create CDS registry access mutex */
237  OS_MutSemCreate(&CFE_ES_Global.CDSVars.RegistryMutex,
238  CFE_ES_CDS_MUT_REG_NAME,
239  CFE_ES_CDS_MUT_REG_VALUE);
240  }
241  }
242  else /* Size < MinRequiredSize */
243  {
244  CFE_ES_WriteToSysLog("CFE_CDS:EarlyInit-CDS Size (%d) less than required (%d)\n",
245  (int)CFE_ES_Global.CDSVars.CDSSize, (int)MinRequiredSize);
246  CFE_ES_Global.CDSVars.MemPoolSize = 0;
247  return Status;
248  }
249  }
250  else /* Error getting the size of the CDS from the BSP */
251  {
252  CFE_ES_WriteToSysLog("CFE_CDS:EarlyInit-Unable to obtain CDS Size from BSP (Err=0x%08X)\n", (unsigned int)Status);
253  CFE_ES_Global.CDSVars.MemPoolSize = 0;
254  return Status;
255  }
256 
257 
258  return(CFE_SUCCESS);
259 
260 } /* End of CFE_ES_CDS_EarlyInit() */

269 int32 CFE_ES_RegisterCDSEx(CFE_ES_CDSHandle_t *HandlePtr, int32 BlockSize, const char *Name, boolean CriticalTbl)
270 {
271  int32 Status = CFE_SUCCESS;
272  uint32 RegIndx;
273  CFE_ES_CDS_RegRec_t *RegRecPtr = NULL;
274 
275  /* Lock Registry for update. This prevents two applications from */
276  /* trying to register CDSs at the same location at the same time */
277  CFE_ES_LockCDSRegistry();
278 
279  /* Check for duplicate CDS name */
280  RegIndx = CFE_ES_FindCDSInRegistry(Name);
281 
282  /* Check to see if CDS is already in the registry */
283  if (RegIndx != CFE_ES_CDS_NOT_FOUND)
284  {
285  /* Get pointer to Registry Record Entry to speed up processing */
286  RegRecPtr = &CFE_ES_Global.CDSVars.Registry[RegIndx];
287 
288  /* If the new CDS is the same size as the old, then no need to reallocate memory */
289  if (BlockSize != RegRecPtr->Size)
290  {
291  /* If the new size is different, the old CDS must be deleted first */
292  /* Free the old memory */
293  Status = CFE_ES_PutCDSBlock(RegRecPtr->MemHandle);
294  }
295  else
296  {
297  /* Warn calling application that this is a duplicate registration */
298  Status = CFE_ES_CDS_ALREADY_EXISTS;
299 
300  /* Return the index into the registry as the handle to the CDS */
301  *HandlePtr = RegIndx;
302  }
303  }
304  else /* CDS not already in registry */
305  {
306  /* Locate empty slot in table registry */
307  RegIndx = CFE_ES_FindFreeCDSRegistryEntry();
308 
309  /* Check to make sure we found a free entry in registry */
310  if (RegIndx == CFE_ES_CDS_NOT_FOUND)
311  {
312  Status = CFE_ES_CDS_REGISTRY_FULL;
313  CFE_ES_WriteToSysLog("CFE_CDS:Register-Registry full\n");
314  }
315  else
316  {
317  RegRecPtr = &CFE_ES_Global.CDSVars.Registry[RegIndx];
318  }
319  }
320 
321  /* If no errors, then initialize the CDS registry entry */
322  if (Status == CFE_SUCCESS)
323  {
324  /* Allocate the memory buffer for the CDS */
325  Status = CFE_ES_GetCDSBlock((CFE_ES_CDSBlockHandle_t *)&RegRecPtr->MemHandle, BlockSize);
326 
327  if (Status == CFE_SUCCESS)
328  {
329  RegRecPtr->Taken = TRUE;
330 
331  /* Save the size of the CDS */
332  RegRecPtr->Size = BlockSize;
333 
334  /* Save flag indicating whether it is a Critical Table or not */
335  RegRecPtr->Table = CriticalTbl;
336 
337  /* Save CDS Name in Registry */
338  strncpy(RegRecPtr->Name, Name, CFE_ES_CDS_MAX_FULL_NAME_LEN);
339 
340  /* Return the index into the registry as the handle to the CDS */
341  *HandlePtr = RegIndx;
342 
343  /* If we succeeded at creating a CDS, save updated registry in the CDS */
344  Status = CFE_ES_UpdateCDSRegistry();
345 
346  if (Status != CFE_SUCCESS)
347  {
348  CFE_ES_WriteToSysLog("CFE_CDS:RegCDS-Failed to update CDS Registry (Stat=0x%08X)\n", (unsigned int)Status);
349  }
350  }
351  }
352 
353  /* Unlock Registry for update */
354  CFE_ES_UnlockCDSRegistry();
355 
356  return (Status);
357 
358 } /* End of CFE_ES_RegisterCDSEx() */

367 int32 CFE_ES_ValidateCDS(void)
368 {
369  /* Assume the CDS is invalid */
370  int32 Status;
371 
372  /* Perform 2 checks to validate the CDS Memory Pool */
373  /* First, determine if the first validity check field is correct */
374  Status = CFE_PSP_ReadFromCDS(&CFE_ES_Global.CDSVars.ValidityField, 0, sizeof(CFE_ES_Global.CDSVars.ValidityField));
375 
376  if (Status == CFE_PSP_SUCCESS)
377  {
378  if (strncmp(CFE_ES_Global.CDSVars.ValidityField, "_CDSBeg_", sizeof(CFE_ES_Global.CDSVars.ValidityField)) == 0)
379  {
380  Status = CFE_PSP_ReadFromCDS(&CFE_ES_Global.CDSVars.ValidityField,
381  (CFE_ES_Global.CDSVars.CDSSize-sizeof(CFE_ES_Global.CDSVars.ValidityField)),
382  sizeof(CFE_ES_Global.CDSVars.ValidityField));
383 
384  if (Status == CFE_PSP_SUCCESS)
385  {
386  if (strncmp(CFE_ES_Global.CDSVars.ValidityField, "_CDSEnd_", sizeof(CFE_ES_Global.CDSVars.ValidityField)) == 0)
387  {
388  Status = CFE_SUCCESS;
389  }
390  else /* Validity Field failed */
391  {
392  Status = CFE_ES_CDS_INVALID;
393  }
394  }
395  else /* BSP reported an error reading from CDS */
396  {
397  CFE_ES_WriteToSysLog("CFE_CDS:Validate-2nd ReadFromCDS Failed. Status=0x%X\n", (unsigned int)Status);
398  }
399  }
400  else /* Validity Field failed */
401  {
402  Status = CFE_ES_CDS_INVALID;
403  }
404  }
405  else /* BSP reported an error reading from CDS */
406  {
407  CFE_ES_WriteToSysLog("CFE_CDS:Validate-1st ReadFromCDS Failed. Status=0x%X\n", (unsigned int)Status);
408  }
409 
410  return Status;
411 } /* End of CFE_ES_ValidateCDS() */

420 int32 CFE_ES_InitializeCDS(uint32 CDSSize)
421 {
422  int32 Status = CFE_SUCCESS;
423  uint32 MemBlock[32];
424  uint32 NumWritten=0;
425  uint32 Uint32Zero=0;
426 
427  /* Clear the CDS to ensure everything is gone */
428  /* Create a block of zeros to write to the CDS */
429  CFE_PSP_MemSet(MemBlock, 0, sizeof(MemBlock));
430 
431  /* While there is space to write another block of zeros, then do so */
432  while (((NumWritten + sizeof(MemBlock)) < CDSSize) && (Status == OS_SUCCESS))
433  {
434  Status = CFE_PSP_WriteToCDS(MemBlock, NumWritten, sizeof(MemBlock));
435 
436  if (Status == CFE_PSP_SUCCESS)
437  {
438  NumWritten += sizeof(MemBlock);
439  }
440  }
441 
442  /* While there is space to write a uint32 of zeros, then do so */
443  if ((Status == CFE_PSP_SUCCESS) && (NumWritten < CDSSize))
444  {
445  while (((NumWritten + sizeof(uint32)) < CDSSize) && (Status == CFE_PSP_SUCCESS))
446  {
447  Status = CFE_PSP_WriteToCDS(&Uint32Zero, NumWritten, sizeof(uint32));
448 
449  if (Status == CFE_PSP_SUCCESS)
450  {
451  NumWritten += sizeof(uint32);
452  }
453  }
454  }
455 
456  if (Status != CFE_PSP_SUCCESS)
457  {
458  CFE_ES_WriteToSysLog("CFE_CDS:Init-Clear CDS failed @ Offset=%d Status=0x%08X\n", (int)NumWritten, (unsigned int)Status);
459  }
460  else
461  {
462  /* Initialize the Validity Check strings */
463  Status = CFE_PSP_WriteToCDS("_CDSBeg_", 0, sizeof(CFE_ES_Global.CDSVars.ValidityField));
464  if (Status == CFE_PSP_SUCCESS)
465  {
466  Status = CFE_PSP_WriteToCDS("_CDSEnd_",
467  (CDSSize-sizeof(CFE_ES_Global.CDSVars.ValidityField)),
468  sizeof(CFE_ES_Global.CDSVars.ValidityField));
469  if (Status != CFE_PSP_SUCCESS)
470  {
471  CFE_ES_WriteToSysLog("CFE_CDS:Init-'_CDSEnd_' write failed. Status=0x%08X\n", (unsigned int)Status);
472  }
473  else
474  {
475  /* Determine the amount of CDS memory is available for the CDS Memory Pool */
476  /* Total CDS Size minus the validity fields and the space for the registry and the registry size */
477  if (CDSSize < (CDS_POOL_OFFSET + sizeof(CFE_ES_Global.CDSVars.ValidityField)))
478  {
479  /* Invalid - passing zero will cause the CFE_ES_CreateCDSPool() to fail */
480  CFE_ES_Global.CDSVars.MemPoolSize = 0;
481  }
482  else
483  {
484  CFE_ES_Global.CDSVars.MemPoolSize = CDSSize - CDS_POOL_OFFSET - sizeof(CFE_ES_Global.CDSVars.ValidityField);
485  }
486 
487  /* Initialize the variables for managing the CDS Memory Pool */
488  Status = CFE_ES_CreateCDSPool(CFE_ES_Global.CDSVars.MemPoolSize, CDS_POOL_OFFSET);
489 
490  if (Status == CFE_SUCCESS)
491  {
492  Status = CFE_ES_InitCDSRegistry();
493  }
494  }
495  }
496  else /* BSP reported an error writing to CDS */
497  {
498  CFE_ES_WriteToSysLog("CFE_CDS:Init-'_CDSBeg_' write failed. Status=0x%08X\n", (unsigned int)Status);
499  }
500  }
501 
502  return Status;
503 } /* End of CFE_ES_InitializeCDS() */

736 int32 CFE_ES_RebuildCDS(void)
737 {

739  int32 PoolOffset;
740 
741  /* First, determine if the CDS registry stored in the CDS is smaller or equal */
742  /* in size to the CDS registry we are currently configured for */
743  /* Copy the number of registry entries to the CDS */
744  Status = CFE_PSP_ReadFromCDS(&CFE_ES_Global.CDSVars.MaxNumRegEntries,
745  CDS_REG_SIZE_OFFSET,
746  sizeof(CFE_ES_Global.CDSVars.MaxNumRegEntries));
747 
748  if ((Status == CFE_PSP_SUCCESS) &&
749  (CFE_ES_Global.CDSVars.MaxNumRegEntries <= CFE_ES_CDS_MAX_NUM_ENTRIES))
750  {
751  Status = CFE_PSP_ReadFromCDS(&CFE_ES_Global.CDSVars.Registry,
752  CDS_REG_OFFSET,
753  (CFE_ES_Global.CDSVars.MaxNumRegEntries * sizeof(CFE_ES_CDS_RegRec_t)));
754 
755  if (Status == CFE_PSP_SUCCESS)
756  {
757  /* Calculate the starting offset of the memory pool */
758  PoolOffset = (CDS_REG_OFFSET + (CFE_ES_Global.CDSVars.MaxNumRegEntries * sizeof(CFE_ES_CDS_RegRec_t)) + 3) & 0xfffffffc;;
759 
760  /* Calculate the size of the memory pool */
761  CFE_ES_Global.CDSVars.MemPoolSize = CFE_ES_Global.CDSVars.CDSSize - PoolOffset - sizeof(CFE_ES_Global.CDSVars.ValidityField);
762 
763  /* Scan the memory pool and identify the created but currently unused memory blocks */
764  Status = CFE_ES_RebuildCDSPool(CFE_ES_Global.CDSVars.MemPoolSize, PoolOffset);
765  }
766  else
767  {
768  /* Registry in CDS is unreadable */
769  Status = CFE_ES_CDS_INVALID;
770  CFE_ES_WriteToSysLog("CFE_CDS:Rebuild-Registry in CDS is unreadable\n");
771  }
772  }
773  else
774  {
775  /* Registry in CDS is too large to recover */
776  Status = CFE_ES_CDS_INVALID;
777  CFE_ES_WriteToSysLog("CFE_CDS:Rebuild-Registry in CDS too large to recover\n");
778  }
779 
780  return Status;
781 }

790 
791 int32 CFE_ES_DeleteCDS(const char *CDSName, boolean CalledByTblServices)
792 {
793  int32 Status;
794  int32 RegIndx;
795  CFE_ES_CDS_RegRec_t *RegRecPtr = NULL;
796  char OwnerName[OS_MAX_API_NAME];
797  uint32 AppId;
798  uint32 i;
799 
800  /* Lock Registry for update. This prevents two applications from */
801  /* trying to change the CDS registry at the same time */
802  CFE_ES_LockCDSRegistry();
803 
804  /* Find CDS name in registry */
805  RegIndx = CFE_ES_FindCDSInRegistry(CDSName);
806 
807  /* Check to see if CDS is already in the registry */
808  if (RegIndx != CFE_ES_CDS_NOT_FOUND)
809  {
810  /* Get pointer to Registry Record Entry to speed up processing */
811  RegRecPtr = &CFE_ES_Global.CDSVars.Registry[RegIndx];
812 
813  /* Critical tables are not allowed to be deleted via an ES Command. */
814  /* They must be deleted by a Table Services Command */
815  if (RegRecPtr->Table != CalledByTblServices)
816  {
817  Status = CFE_ES_CDS_WRONG_TYPE_ERR;
818  }
819  else
820  {
821  /* Check to see if the owning application is still active */
822  /* First, extract the owning application name */
823  i=0;
824  while ((i < (OS_MAX_API_NAME-1) && (RegRecPtr->Name[i] != '.')))
825  {
826  OwnerName[i] = RegRecPtr->Name[i];
827  i++;
828  }
829 
830  /* Null terminate the application name */
831  OwnerName[i] = '\0';
832 
833  /* Check to see if the Application Name is in the Registered Apps list */
834  Status = CFE_ES_GetAppIDByName(&AppId, OwnerName);
835 
836  /* If we can't find the name, then things are good */
837  if (Status == CFE_ES_ERR_APPNAME)
838  {
839  /* Free the registry entry and the CDS memory block associated with it */
840  Status = CFE_ES_PutCDSBlock(RegRecPtr->MemHandle);
841 
842  /* Report any errors incurred while freeing the CDS Memory Block */
843  if (Status < 0)
844  {
845  CFE_ES_WriteToSysLog("CFE_ES:DeleteCDS-Failed to free CDS Mem Block (Handle=0x%08lX)(Stat=0x%08X)\n",
846  (unsigned long)RegRecPtr->MemHandle, (unsigned int)Status);
847  }
848  else
849  {
850  /* Remove entry from the CDS Registry */
851  RegRecPtr->Taken = FALSE;
852 
853  Status = CFE_ES_UpdateCDSRegistry();
854 
855  if (Status != CFE_SUCCESS)
856  {
857  CFE_ES_WriteToSysLog("CFE_ES:DeleteCDS-Failed to update CDS Registry (Stat=0x%08X)\n", (unsigned int)Status);
858  }
859  }
860  }
861  else
862  {
863  Status = CFE_ES_CDS_OWNER_ACTIVE_ERR;
864  }
865  }
866  }
867  else /* Error - CDS not in registry */
868  {
869  Status = CFE_ES_CDS_NOT_FOUND_ERR;
870  }
871 
872  /* Unlock Registry for future updates */
873  CFE_ES_UnlockCDSRegistry();
874 
875  return Status;
876 } /* End of CFE_ES_DeleteCDS() */
