109 /******************************************************************************
110 ** Function: CFE_SB_TaskMain()
111 **
112 ** Purpose:
113 ** Main loop for Software Bus task, used to process SB commands.
114 **
115 ** Arguments:
116 ** none
117 **
118 ** Return:
119 ** none
120 */
121 void CFE_SB_TaskMain(void)
122 {
123  int32 Status;
124 
125  CFE_ES_PerfLogEntry(CFE_SB_MAIN_PERF_ID);
126 
127  Status = CFE_SB_AppInit();
128 
129  if(Status != CFE_SUCCESS)
130  {
131  CFE_ES_WriteToSysLog("SB:Application Init Failed,RC=0x%08X\n", (unsigned int)Status);
132  CFE_ES_PerfLogExit(CFE_SB_MAIN_PERF_ID);
133  /* Note: CFE_ES_ExitApp will not return */
134  CFE_ES_ExitApp(CFE_ES_RUNSTATUS_CORE_APP_INIT_ERROR);
135  }/* end if */
136 
137  /*
138  * Wait for other apps to start.
139  * It is important that the core apps are present before this starts receiving
140  * messages from the command pipe, as some of those handlers might depend on
141  * the other core apps.
142  */
143  CFE_ES_WaitForStartupSync(CFE_CORE_MAX_STARTUP_MSEC);
144 
145  /* Main loop */
146  while (Status == CFE_SUCCESS)
147  {
148  /* Increment the Main task Execution Counter */
149  CFE_ES_IncrementTaskCounter();
150 
151  CFE_ES_PerfLogExit(CFE_SB_MAIN_PERF_ID);
152 
153  /* Pend on receipt of packet */
154  Status = CFE_SB_RcvMsg(&CFE_SB.CmdPipePktPtr,
155  CFE_SB.CmdPipe,
156  CFE_SB_PEND_FOREVER);
157 
158  CFE_ES_PerfLogEntry(CFE_SB_MAIN_PERF_ID);
159 
160  if(Status == CFE_SUCCESS)
161  {
162  /* Process cmd pipe msg */
163  CFE_SB_ProcessCmdPipePkt();
164  }else{
165  CFE_ES_WriteToSysLog("SB:Error reading cmd pipe,RC=0x%08X\n",(unsigned int)Status);
166  }/* end if */
167 
168  }/* end while */
169 
170  /* while loop exits only if CFE_SB_RcvMsg returns error */
171  CFE_ES_ExitApp(CFE_ES_RUNSTATUS_CORE_APP_RUNTIME_ERROR);
172 
173 }/* end CFE_SB_TaskMain */

372  switch(CFE_SB_GetMsgId(CFE_SB.CmdPipePktPtr)){

375  /* Note: Command counter not incremented for this command */
376  CFE_SB_SendHKTlmPkt();

378

380  switch (CFE_SB_GetCmdCode(CFE_SB.CmdPipePktPtr)) {
381  case CFE_SB_NOOP_CC:

385  CFE_SB.HKTlmMsg.Payload.CommandCnt++;
386  break;
387

389  /* Note: Command counter not incremented for this command */

393  break;
394 
395  case CFE_SB_SEND_SB_STATS_CC:
396  CFE_SB_SendStats();
397  break;
398

400  CFE_SB_ProcessSendRtgInfoCmd(Payload);
401  break;
402 
403  case CFE_SB_ENABLE_ROUTE_CC:
