94 void CFE_TBL_TaskMain(void)
95 {
96  int32 Status;
97 
98  CFE_ES_PerfLogEntry(CFE_TBL_MAIN_PERF_ID);
99 
100  Status = CFE_TBL_TaskInit();
101 
102  if(Status != CFE_SUCCESS)
103  {
104  CFE_ES_WriteToSysLog("TBL:Application Init Failed,RC=0x%08X\n", (unsigned int)Status);
105  CFE_ES_PerfLogExit(CFE_TBL_MAIN_PERF_ID);
106  /* Note: CFE_ES_ExitApp will not return */
107  CFE_ES_ExitApp(CFE_ES_RUNSTATUS_CORE_APP_INIT_ERROR);
108  }/* end if */
109 
110  /*
111  * Wait for other apps to start.
112  * It is important that the core apps are present before this starts receiving
113  * messages from the command pipe, as some of those handlers might depend on
114  * the other core apps.
115  */
116  CFE_ES_WaitForStartupSync(CFE_CORE_MAX_STARTUP_MSEC);
117 
118  /* Main loop */
119  while (Status == CFE_SUCCESS)
120  {
121  /* Increment the Main task Execution Counter */
122  CFE_ES_IncrementTaskCounter();
123 
124  CFE_ES_PerfLogExit(CFE_TBL_MAIN_PERF_ID);
125 
126  /* Pend on receipt of packet */
127  Status = CFE_SB_RcvMsg( &CFE_TBL_TaskData.MsgPtr,
128  CFE_TBL_TaskData.CmdPipe,
129  CFE_SB_PEND_FOREVER);
130 
131  CFE_ES_PerfLogEntry(CFE_TBL_MAIN_PERF_ID);
132 
133  if (Status == CFE_SUCCESS)
134  {
135  /* Process cmd pipe msg */
136  CFE_TBL_TaskPipe(CFE_TBL_TaskData.MsgPtr);
137  }else{
138  CFE_ES_WriteToSysLog("TBL:Error reading cmd pipe,RC=0x%08X\n",(unsigned int)Status);
139  }/* end if */
140 
141  }/* end while */
142 
143  /* while loop exits only if CFE_SB_RcvMsg returns error */
144  CFE_ES_ExitApp(CFE_ES_RUNSTATUS_CORE_APP_RUNTIME_ERROR);
145 
146 } /* end CFE_TBL_TaskMain() */

151 int32 CFE_TBL_TaskInit(void)
152 {
153  int32 Status = CFE_SUCCESS;
154 
155  /*
156  ** Register Table Services with ES
157  */
158  Status = CFE_ES_RegisterApp();
159 
160  if(Status != CFE_SUCCESS)
161  {
162  CFE_ES_WriteToSysLog("TBL:Call to CFE_ES_RegisterApp Failed:RC=0x%08X\n",(unsigned int)Status);
163  return Status;
164  }/* end if */
165 
166  /*
167  ** Initialize global Table Services data
168  */
169  CFE_TBL_InitData();
170 
171  /*
172  ** Register event filter table
173  */
174  Status = CFE_EVS_Register(NULL, 0, 0);
175 
176  if(Status != CFE_SUCCESS)
177  {
178  CFE_ES_WriteToSysLog("TBL:Call to CFE_EVS_Register Failed:RC=0x%08X\n",(unsigned int)Status);
179  return Status;
180  }/* end if */
181 
182  /*
183  ** Create Software Bus message pipe
184  */
185  Status = CFE_SB_CreatePipe(&CFE_TBL_TaskData.CmdPipe,
186  CFE_TBL_TaskData.PipeDepth,
187  CFE_TBL_TaskData.PipeName);
188  if(Status != CFE_SUCCESS)
189  {
190  CFE_ES_WriteToSysLog("TBL:Error creating cmd pipe:RC=0x%08X\n",(unsigned int)Status);
191  return Status;
192  }/* end if */
193 
194  /*
195  ** Subscribe to Housekeeping request commands
196  */
197  Status = CFE_SB_Subscribe(CFE_TBL_SEND_HK_MID, CFE_TBL_TaskData.CmdPipe);
198 
199  if(Status != CFE_SUCCESS)
200  {
201  CFE_ES_WriteToSysLog("TBL:Error subscribing to HK Request:RC=0x%08X\n",(unsigned int)Status);
202  return Status;
203  }/* end if */
204 
205  /*
206  ** Subscribe to Table task ground command packets
207  */
208  Status = CFE_SB_Subscribe(CFE_TBL_CMD_MID, CFE_TBL_TaskData.CmdPipe);
209 
210  if(Status != CFE_SUCCESS)
211  {
212  CFE_ES_WriteToSysLog("TBL:Error subscribing to gnd cmds:RC=0x%08X\n",(unsigned int)Status);
213  return Status;
214  }/* end if */
215 
216  /*
217  ** Task startup event message
218  */
219  Status = CFE_EVS_SendEvent(CFE_TBL_INIT_INF_EID, CFE_EVS_INFORMATION, "cFE TBL Initialized. cFE Version %d.%d.%d.%d",
220  CFE_MAJOR_VERSION,CFE_MINOR_VERSION,CFE_REVISION,CFE_MISSION_REV);
221 
222  if(Status != CFE_SUCCESS)
223  {
224  CFE_ES_WriteToSysLog("TBL:Error sending init event:RC=0x%08X\n",(unsigned int)Status);
225  return Status;
226  }/* end if */
227 
228  return CFE_SUCCESS;
229 
230 } /* End of CFE_TBL_TaskInit() */

266 void CFE_TBL_TaskPipe(CFE_SB_Msg_t *MessagePtr)
267 {
268  CFE_SB_MsgId_t MessageID = CFE_SB_GetMsgId(MessagePtr);
269  uint16 CommandCode = CFE_SB_GetCmdCode(MessagePtr);
270  int16 CmdIndx;
271  uint32 ActualLength;
272  CFE_TBL_CmdProcRet_t CmdStatus = CFE_TBL_INC_ERR_CTR; /* Assume a failed command */
273 
274  /* Search the Command Handler Table for a matching message */
275  CmdIndx = CFE_TBL_SearchCmdHndlrTbl(MessageID, CommandCode);
276 
277  /* Check to see if a matching command was found */
278  if (CmdIndx >= 0)
279  {
280  /* Verify Message Length before processing */
281  ActualLength = CFE_SB_GetTotalMsgLength(MessagePtr);
282  if (ActualLength == CFE_TBL_CmdHandlerTbl[CmdIndx].ExpectedLength)
283  {
284  /* All checks have passed, call the appropriate message handler */
285  CmdStatus = (CFE_TBL_CmdHandlerTbl[CmdIndx].MsgProcFuncPtr)(&MessagePtr->Byte[CFE_SB_CMD_HDR_SIZE]);
286  }
287  else /* Bad Message Length */
288  {
289  CFE_EVS_SendEvent( CFE_TBL_LEN_ERR_EID, CFE_EVS_ERROR,
290  "Invalid msg length -- ID = 0x%04X, CC = %d, Len = %d (!= %d)",
291  (unsigned int)MessageID, (int)CommandCode, (int)ActualLength,
292  (int)CFE_TBL_CmdHandlerTbl[CmdIndx].ExpectedLength );
293  }
294 
295  /* Only update command counters when message has a command code */
296  if (CFE_TBL_CmdHandlerTbl[CmdIndx].MsgTypes == CFE_TBL_CMD_MSGTYPE)
297  {
298  if (CmdStatus == CFE_TBL_INC_CMD_CTR)
299  {
300  CFE_TBL_TaskData.CmdCounter++;
301  }
302  else if (CmdStatus == CFE_TBL_INC_ERR_CTR)
303  {
304  CFE_TBL_TaskData.ErrCounter++;
305  }
306  }
307  }
308  else
309  {
310  /* Determine whether event message should be */
311  /* "Bad Command Code" or "Bad Message ID" */
312  if (CmdIndx == CFE_TBL_BAD_CMD_CODE)
313  {
314  CFE_EVS_SendEvent(CFE_TBL_CC1_ERR_EID, CFE_EVS_ERROR,
315  "Invalid command code -- ID = 0x%04X, CC = %d",
316  (unsigned int)MessageID, (int)CommandCode);
317 
318  /* Update the command error counter */
319  CFE_TBL_TaskData.ErrCounter++;
320  }
321  else /* CmdIndx == CFE_TBL_BAD_MSG_ID */
322  {
323  CFE_EVS_SendEvent(CFE_TBL_MID_ERR_EID, CFE_EVS_ERROR,
324  "Invalid message ID -- ID = 0x%04X",
325  (unsigned int)MessageID);
326  /*
327  ** Note: we only increment the command error counter when
328  ** processing messages with command codes
329  */
330  }
331  }
332 
333  return;
334 
335 } /* End of CFE_TBL_TaskPipe() */

340 int16 CFE_TBL_SearchCmdHndlrTbl( CFE_SB_MsgId_t MessageID, uint16 CommandCode )
341 {
342  int16 TblIndx = CFE_TBL_BAD_CMD_CODE;
343  boolean FoundMsg = FALSE;
344  boolean FoundMatch = FALSE;
345 
346  do
347  {
348  /* Point to next entry in Command Handler Table */
349  TblIndx++;
350 
351  /* Check to see if we found a matching Message ID */
352  if ((CFE_TBL_CmdHandlerTbl[TblIndx].MsgId == MessageID) &&
353  (CFE_TBL_CmdHandlerTbl[TblIndx].MsgTypes != CFE_TBL_TERM_MSGTYPE))
354  {
355  /* Flag any found message IDs so that if there is an error, */
356  /* we can determine if it was a bad message ID or bad command code */
357  FoundMsg = TRUE;
358 
359  /* If entry in the Command Handler Table is a command entry, */
360  /* then check for a matching command code */
361  if (CFE_TBL_CmdHandlerTbl[TblIndx].MsgTypes == CFE_TBL_CMD_MSGTYPE)
362  {
363  if (CFE_TBL_CmdHandlerTbl[TblIndx].CmdCode == CommandCode)
364  {
365  /* Found matching message ID and Command Code */
366  FoundMatch = TRUE;
367  }
368  }
369  else /* Message is not a command message with specific command code */
370  {
371  /* Automatically assume a match when legit */
372  /* Message ID is all that is required */
373  FoundMatch = TRUE;
374  }
375  }
376  } while ((!FoundMatch) && (CFE_TBL_CmdHandlerTbl[TblIndx].MsgTypes != CFE_TBL_TERM_MSGTYPE));
377 
378  /* If we failed to find a match, return a negative index */
379  if (!FoundMatch)
380  {
381  /* Determine if the message ID was bad or the command code */
382  if (FoundMsg)
383  {
384  /* A matching message ID was found, so the command code must be bad */
385  TblIndx = CFE_TBL_BAD_CMD_CODE;
386  }
387  else /* No matching message ID was found */
388  {
389  TblIndx = CFE_TBL_BAD_MSG_ID;
390  }
391  }
392 
393  return TblIndx;
394 } /* End of CFE_TBL_SearchCmdHndlrTbl() */
