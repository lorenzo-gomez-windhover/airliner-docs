84 /**********************************************************************
85 ** Structure definitions for CCSDS headers. All items in the structure
86 ** must be aligned on 16-bit words. Bitfields must be avoided since
87 ** some compilers (such as gcc) force them into 32-bit alignment.

90 /*----- CCSDS packet primary header. -----*/
91 
92 typedef struct {
93 
94  uint8 StreamId[2]; /* packet identifier word (stream ID) */
95  /* bits shift ------------ description ---------------- */
96  /* 0x07FF 0 : application ID */
97  /* 0x0800 11 : secondary header: 0 = absent, 1 = present */
98  /* 0x1000 12 : packet type: 0 = TLM, 1 = CMD */
99  /* 0xE000 13 : CCSDS version, always set to 0 */
100 
101  uint8 Sequence[2]; /* packet sequence word */
102  /* bits shift ------------ description ---------------- */
103  /* 0x3FFF 0 : sequence count */
104  /* 0xC000 14 : segmentation flags: 3 = complete packet */
105 
106  uint8 Length[2]; /* packet length word */
107  /* bits shift ------------ description ---------------- */
108  /* 0xFFFF 0 : (total packet length) - 7 */
109 
110 } CCSDS_PriHdr_t;

186 /**********************************************************************
187 ** Macros for reading and writing bit fields in a 16-bit integer.
188 ** These are used to implement the read and write macros below.
189 **********************************************************************/
190 
191 /* Read bits specified by 'mask' from 'word' and shift down by 'shift'. */
192 #define CCSDS_RD_BITS(word,mask,shift) \
193  (((word) & mask) >> shift)
194 
195 /* Shift 'value' up by 'shift' and write to those bits in 'word' that
196 ** are specified by 'mask'. Other bits in 'word' are unchanged. */
197 #define CCSDS_WR_BITS(word,mask,shift,value) \
198  ((word) = (uint16)(((word) & ~mask) | (((value) & (mask >> shift)) << shift)))

201 /**********************************************************************
202 ** Macros for reading and writing the fields in a CCSDS header. All
203 ** of the macros are used in a similar way:
204 **
205 ** CCSDS_RD_xxx(header) -- Read field xxx from header.
206 ** CCSDS_WR_xxx(header,value) -- Write value to field xxx of header.
207 **
208 ** Note that 'header' is a reference to the actual header structure,
209 ** not to a pointer to the structure. If using a pointer, one must
210 ** refer to the structure as *pointer.
211 **
212 ** The CCSDS_WR_xxx macros may refer to 'header' more than once; thus
213 ** the expression for 'header' must NOT contain any side effects.
214 **********************************************************************/
