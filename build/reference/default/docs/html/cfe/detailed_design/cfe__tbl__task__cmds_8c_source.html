80 /*******************************************************************
81 **
82 ** CFE_TBL_HousekeepingCmd() -- Process Housekeeping Request Message
83 **
84 ** NOTE: For complete prolog information, see 'cfe_tbl_task_cmds.h'
85 ********************************************************************/
86 
87 CFE_TBL_CmdProcRet_t CFE_TBL_HousekeepingCmd( const CFE_SB_MsgPayloadPtr_t Payload )
88 {
89  int32 Status;
90  uint32 i;
91  CFE_TBL_DumpControl_t *DumpCtrlPtr;
92  CFE_TIME_SysTime_t DumpTime;
93  int32 FileDescriptor;
94 
95  /*
96  ** Collect housekeeping data from Table Services
97  */
98  CFE_TBL_GetHkData();
99 
100  /*
101  ** Send housekeeping telemetry packet
102  */
103  CFE_SB_TimeStampMsg((CFE_SB_Msg_t *) &CFE_TBL_TaskData.HkPacket);
104  Status = CFE_SB_SendMsg((CFE_SB_Msg_t *) &CFE_TBL_TaskData.HkPacket);
105 
106  if (Status != CFE_SUCCESS)
107  {
108  CFE_EVS_SendEvent(CFE_TBL_FAIL_HK_SEND_ERR_EID,
109  CFE_EVS_ERROR,
110  "Unable to send Hk Packet (Status=0x%08X)",
111  (unsigned int)Status);
112  }
113 
114  /* If a table's registry entry has been requested for telemetry, then pack it and send it */
115  if (CFE_TBL_TaskData.HkTlmTblRegIndex != CFE_TBL_NOT_FOUND)
116  {
117  CFE_TBL_GetTblRegData();
118 
119  /*
120  ** Send Table Registry Info Packet
121  */
122  CFE_SB_TimeStampMsg((CFE_SB_Msg_t *) &CFE_TBL_TaskData.TblRegPacket);
123  CFE_SB_SendMsg((CFE_SB_Msg_t *) &CFE_TBL_TaskData.TblRegPacket);
124 
125  /* Once the data has been sent, clear the index so that we don't send it again and again */
126  CFE_TBL_TaskData.HkTlmTblRegIndex = CFE_TBL_NOT_FOUND;
127  }
128 
129  /* Check to see if there are any dump-only table dumps pending */
130  for (i=0; i < CFE_TBL_MAX_SIMULTANEOUS_LOADS; i++)
131  {
132  if (CFE_TBL_TaskData.DumpControlBlocks[i].State == CFE_TBL_DUMP_PERFORMED)
133  {
134  DumpCtrlPtr = &CFE_TBL_TaskData.DumpControlBlocks[i];
135  Status = CFE_TBL_DumpToFile(DumpCtrlPtr->DumpBufferPtr->DataSource,
136  DumpCtrlPtr->TableName,
137  DumpCtrlPtr->DumpBufferPtr->BufferPtr,
138  DumpCtrlPtr->Size);
139 
140  /* If dump file was successfully written, update the file header so that the timestamp */
141  /* is the time of the actual capturing of the data, NOT the time when it was written to the file */
142  if (Status == CFE_TBL_INC_CMD_CTR)
143  {
144  DumpTime.Seconds = DumpCtrlPtr->DumpBufferPtr->FileCreateTimeSecs;
145  DumpTime.Subseconds = DumpCtrlPtr->DumpBufferPtr->FileCreateTimeSubSecs;
146 
147  FileDescriptor = OS_open(DumpCtrlPtr->DumpBufferPtr->DataSource, OS_READ_WRITE, 0);
148 
149  if (FileDescriptor >= 0)
150  {
151  Status = CFE_FS_SetTimestamp(FileDescriptor, DumpTime);
152 
153  if (Status != OS_FS_SUCCESS)
154  {
155  CFE_ES_WriteToSysLog("CFE_TBL:HkCmd-Unable to update timestamp in dump file '%s'\n",
156  DumpCtrlPtr->DumpBufferPtr->DataSource);
157  }
158 
159  OS_close(FileDescriptor);
160  }
161  else
162  {
163  CFE_ES_WriteToSysLog("CFE_TBL:HkCmd-Unable to open dump file '%s' to update timestamp\n",
164  DumpCtrlPtr->DumpBufferPtr->DataSource);
165  }
166  }
167 
168  /* Free the shared working buffer */
169  CFE_TBL_TaskData.LoadBuffs[DumpCtrlPtr->RegRecPtr->LoadInProgress].Taken = FALSE;
170  DumpCtrlPtr->RegRecPtr->LoadInProgress = CFE_TBL_NO_LOAD_IN_PROGRESS;
171 
172  /* Free the Dump Control Block for later use */
173  DumpCtrlPtr->State = CFE_TBL_DUMP_FREE;
174  }
175  }
176 
177  return CFE_TBL_DONT_INC_CTR;
178 
179 } /* End of CFE_TBL_HousekeepingCmd() */

182 /*******************************************************************
183 **
184 ** CFE_TBL_GetHkData() -- Collect data and store it into the Housekeeping Message
185 **
186 ** NOTE: For complete prolog information, see prototype above
187 ********************************************************************/
188 
189 void CFE_TBL_GetHkData(void)
190 {
191  uint32 i;
192  uint16 Count;
193  CFE_TBL_ValidationResult_t *ValPtr = NULL;
194 
195  /* Copy command counter data */
196  CFE_TBL_TaskData.HkPacket.Payload.CmdCounter = CFE_TBL_TaskData.CmdCounter;
197  CFE_TBL_TaskData.HkPacket.Payload.ErrCounter = CFE_TBL_TaskData.ErrCounter;
198  CFE_TBL_TaskData.HkPacket.Payload.FailedValCtr = CFE_TBL_TaskData.FailedValCtr;
199  CFE_TBL_TaskData.HkPacket.Payload.NumLoadPending = 0;
200  CFE_SB_SET_MEMADDR(CFE_TBL_TaskData.HkPacket.Payload.MemPoolHandle, CFE_TBL_TaskData.Buf.PoolHdl);
201 
202  /* Determine the number of tables currently registered and Number of Load Pending Tables */
203  Count = 0;
204  for (i=0; i<CFE_TBL_MAX_NUM_TABLES; i++)
205  {
206  if (CFE_TBL_TaskData.Registry[i].OwnerAppId != CFE_TBL_NOT_OWNED)
207  {
208  Count++;
209 
210  if (CFE_TBL_TaskData.Registry[i].LoadPending)
211  {
212  CFE_TBL_TaskData.HkPacket.Payload.NumLoadPending++;
213  }
214  }
215  }
216  CFE_TBL_TaskData.HkPacket.Payload.NumTables = Count;
217 
218  /* Determine the number of free shared buffers */
219  CFE_TBL_TaskData.HkPacket.Payload.NumFreeSharedBufs = CFE_TBL_MAX_SIMULTANEOUS_LOADS;
220  for (i=0; i<CFE_TBL_MAX_SIMULTANEOUS_LOADS; i++)
221  {
222  if (CFE_TBL_TaskData.LoadBuffs[i].Taken)
223  {
224  CFE_TBL_TaskData.HkPacket.Payload.NumFreeSharedBufs--;
225  }
226  }
227 
228  /* Locate a completed, but unreported, validation request */
229  i=0;
230  while ((i < CFE_TBL_MAX_NUM_VALIDATIONS) && (ValPtr == NULL))
231  {
232  if (CFE_TBL_TaskData.ValidationResults[i].State == CFE_TBL_VALIDATION_PERFORMED)
233  {
234  ValPtr = &CFE_TBL_TaskData.ValidationResults[i];
235  }
236  else
237  {
238  i++;
239  }
240  }
241 
242  if (ValPtr != NULL)
243  {
244  CFE_TBL_TaskData.HkPacket.Payload.LastValCrc = ValPtr->CrcOfTable;
245  CFE_TBL_TaskData.HkPacket.Payload.LastValStatus = ValPtr->Result;
246  CFE_TBL_TaskData.HkPacket.Payload.ActiveBuffer = ValPtr->ActiveBuffer;
247 
248  /* Keep track of the number of failed and successful validations */
249  if (ValPtr->Result == CFE_SUCCESS)
250  {
251  CFE_TBL_TaskData.SuccessValCtr++;
252  }
253  else
254  {
255  CFE_TBL_TaskData.FailedValCtr++;
256  }
257 
258  CFE_SB_MessageStringSet(CFE_TBL_TaskData.HkPacket.Payload.LastValTableName, ValPtr->TableName,
259  sizeof(CFE_TBL_TaskData.HkPacket.Payload.LastValTableName), sizeof(ValPtr->TableName));
260  CFE_TBL_TaskData.ValidationCtr++;
261 
262  /* Free the Validation Response Block for next time */
263  ValPtr->Result = 0;
264  ValPtr->CrcOfTable = 0;
265  ValPtr->TableName[0] = '\0';
266  ValPtr->ActiveBuffer = FALSE;
267  ValPtr->State = CFE_TBL_VALIDATION_FREE;
268  }
269 
270  CFE_TBL_TaskData.HkPacket.Payload.ValidationCtr = CFE_TBL_TaskData.ValidationCtr;
271  CFE_TBL_TaskData.HkPacket.Payload.SuccessValCtr = CFE_TBL_TaskData.SuccessValCtr;
272  CFE_TBL_TaskData.HkPacket.Payload.FailedValCtr = CFE_TBL_TaskData.FailedValCtr;
273  CFE_TBL_TaskData.HkPacket.Payload.NumValRequests = CFE_TBL_TaskData.NumValRequests;
274 
275  /* Validate the index of the last table updated before using it */
276  if ((CFE_TBL_TaskData.LastTblUpdated >= 0) &&
277  (CFE_TBL_TaskData.LastTblUpdated < CFE_TBL_MAX_NUM_TABLES))
278  {
279  /* Check to make sure the Registry Entry is still valid */
280  if (CFE_TBL_TaskData.Registry[CFE_TBL_TaskData.LastTblUpdated].OwnerAppId != CFE_TBL_NOT_OWNED)
281  {
282  /* Get the time at the last table update */
283  CFE_TBL_TaskData.HkPacket.Payload.LastUpdateTime =
284  CFE_TBL_TaskData.Registry[CFE_TBL_TaskData.LastTblUpdated].TimeOfLastUpdate;
285 
286  /* Get the table name used for the last table update */
287  CFE_SB_MessageStringSet(CFE_TBL_TaskData.HkPacket.Payload.LastUpdatedTbl,
288  CFE_TBL_TaskData.Registry[CFE_TBL_TaskData.LastTblUpdated].Name,
289  sizeof(CFE_TBL_TaskData.HkPacket.Payload.LastUpdatedTbl),
290  sizeof(CFE_TBL_TaskData.Registry[CFE_TBL_TaskData.LastTblUpdated].Name));
291  }
292  }
293 }

301 
302 void CFE_TBL_GetTblRegData(void)
303 {
304  CFE_TBL_RegistryRec_t *RegRecPtr;
305 
306  RegRecPtr = &CFE_TBL_TaskData.Registry[CFE_TBL_TaskData.HkTlmTblRegIndex];
307 
308  CFE_TBL_TaskData.TblRegPacket.Payload.Size = RegRecPtr->Size;
309  CFE_SB_SET_MEMADDR(CFE_TBL_TaskData.TblRegPacket.Payload.ActiveBufferAddr,
310  RegRecPtr->Buffers[RegRecPtr->ActiveBufferIndex].BufferPtr);
311 
312  if (RegRecPtr->DblBuffered)
313  {
314  /* For a double buffered table, the inactive is the other allocated buffer */
315  CFE_SB_SET_MEMADDR(CFE_TBL_TaskData.TblRegPacket.Payload.InactiveBufferAddr,
316  RegRecPtr->Buffers[(1U-RegRecPtr->ActiveBufferIndex)].BufferPtr);
317  }
318  else
319  {
320  /* Check to see if an inactive buffer has currently been allocated to the single buffered table */
321  if (RegRecPtr->LoadInProgress != CFE_TBL_NO_LOAD_IN_PROGRESS)
322  {
323  CFE_SB_SET_MEMADDR(CFE_TBL_TaskData.TblRegPacket.Payload.InactiveBufferAddr,
324  CFE_TBL_TaskData.LoadBuffs[RegRecPtr->LoadInProgress].BufferPtr);
325  }
326  else
327  {
328  CFE_TBL_TaskData.TblRegPacket.Payload.InactiveBufferAddr = 0;
329  }
330  }
331 
332  CFE_SB_SET_MEMADDR(CFE_TBL_TaskData.TblRegPacket.Payload.ValidationFuncPtr, RegRecPtr->ValidationFuncPtr);
333  CFE_TBL_TaskData.TblRegPacket.Payload.TimeOfLastUpdate = RegRecPtr->TimeOfLastUpdate;
334  CFE_TBL_TaskData.TblRegPacket.Payload.TableLoadedOnce = RegRecPtr->TableLoadedOnce;
335  CFE_TBL_TaskData.TblRegPacket.Payload.LoadPending = RegRecPtr->LoadPending;
336  CFE_TBL_TaskData.TblRegPacket.Payload.DumpOnly = RegRecPtr->DumpOnly;
337  CFE_TBL_TaskData.TblRegPacket.Payload.DblBuffered = RegRecPtr->DblBuffered;
338  CFE_TBL_TaskData.TblRegPacket.Payload.FileCreateTimeSecs = RegRecPtr->Buffers[RegRecPtr->ActiveBufferIndex].FileCreateTimeSecs;
339  CFE_TBL_TaskData.TblRegPacket.Payload.FileCreateTimeSubSecs = RegRecPtr->Buffers[RegRecPtr->ActiveBufferIndex].FileCreateTimeSubSecs;
340  CFE_TBL_TaskData.TblRegPacket.Payload.Crc = RegRecPtr->Buffers[RegRecPtr->ActiveBufferIndex].Crc;
341  CFE_TBL_TaskData.TblRegPacket.Payload.Critical = RegRecPtr->CriticalTable;
342 
343  CFE_SB_MessageStringSet(CFE_TBL_TaskData.TblRegPacket.Payload.Name, RegRecPtr->Name,
344  sizeof(CFE_TBL_TaskData.TblRegPacket.Payload.Name), sizeof(RegRecPtr->Name));
345  CFE_SB_MessageStringSet(CFE_TBL_TaskData.TblRegPacket.Payload.LastFileLoaded, RegRecPtr->LastFileLoaded,
346  sizeof(CFE_TBL_TaskData.TblRegPacket.Payload.LastFileLoaded), sizeof(RegRecPtr->LastFileLoaded));
347  CFE_ES_GetAppName(CFE_TBL_TaskData.TblRegPacket.Payload.OwnerAppName, RegRecPtr->OwnerAppId,
348  sizeof(CFE_TBL_TaskData.TblRegPacket.Payload.OwnerAppName));
349 }

395 /*******************************************************************
396 **
397 ** CFE_TBL_LoadCmd() -- Process Load Table File to Buffer Command Message
398 **
399 ** NOTE: For complete prolog information, see 'cfe_tbl_task_cmds.h'
400 ********************************************************************/
401 
402 CFE_TBL_CmdProcRet_t CFE_TBL_LoadCmd( const CFE_SB_MsgPayloadPtr_t Payload )
403 {
404  CFE_TBL_CmdProcRet_t ReturnCode = CFE_TBL_INC_ERR_CTR; /* Assume failure */
405  const CFE_TBL_LoadCmd_Payload_t *CmdPtr = (const CFE_TBL_LoadCmd_Payload_t *) Payload;
406  CFE_FS_Header_t StdFileHeader;
407  CFE_TBL_File_Hdr_t TblFileHeader;
408  int32 FileDescriptor;
409  int32 Status;
410  CFE_TBL_RegistryRec_t *RegRecPtr;
411  CFE_TBL_LoadBuff_t *WorkingBufferPtr;
412  char LoadFilename[OS_MAX_PATH_LEN];
413  uint8 ExtraByte;
414 
415  /* Make sure all strings are null terminated before attempting to process them */
416  CFE_SB_MessageStringGet(LoadFilename, (char *)CmdPtr->LoadFilename, NULL,
417  OS_MAX_PATH_LEN, sizeof(CmdPtr->LoadFilename));
418 
419  /* Try to open the specified table file */
420  FileDescriptor = OS_open(LoadFilename, OS_READ_ONLY, 0);
421 
422  if (FileDescriptor >= 0)
423  {
424  Status = CFE_TBL_ReadHeaders(FileDescriptor, &StdFileHeader, &TblFileHeader, &LoadFilename[0]);
425 
426  if (Status == CFE_SUCCESS)
427  {
428  /* Locate specified table in registry */
429  Status = CFE_TBL_FindTableInRegistry(TblFileHeader.TableName);
430 
431  if (Status == CFE_TBL_NOT_FOUND)
432  {
433  CFE_EVS_SendEvent(CFE_TBL_NO_SUCH_TABLE_ERR_EID,
434  CFE_EVS_ERROR,
435  "Unable to locate '%s' in Table Registry",
436  TblFileHeader.TableName);
437  }
438  else
439  {
440  /* Translate the registry index into a record pointer */
441  RegRecPtr = &CFE_TBL_TaskData.Registry[Status];
442 
443  if (RegRecPtr->DumpOnly)
444  {
445  CFE_EVS_SendEvent(CFE_TBL_LOADING_A_DUMP_ONLY_ERR_EID,
446  CFE_EVS_ERROR,
447  "Attempted to load DUMP-ONLY table '%s' from '%s'",
448  TblFileHeader.TableName, LoadFilename);
449  }
450  else if (RegRecPtr->LoadPending)
451  {
452  CFE_EVS_SendEvent(CFE_TBL_LOADING_PENDING_ERR_EID,
453  CFE_EVS_ERROR,
454  "Attempted to load table '%s' while previous load is still pending",
455  TblFileHeader.TableName);
456  }
457  else
458  {
459  /* Make sure of the following: */
460  /* 1) If table has not been loaded previously, then make sure the current */
461  /* load starts with the first byte */
462  /* 2) The number of bytes to load is greater than zero */
463  /* 3) The offset plus the number of bytes does not exceed the table size */
464  if (((RegRecPtr->TableLoadedOnce) || (TblFileHeader.Offset == 0)) &&
465  (TblFileHeader.NumBytes > 0) &&
466  ((TblFileHeader.NumBytes + TblFileHeader.Offset) <= RegRecPtr->Size))
467  {
468  /* Get a working buffer, either a free one or one allocated with previous load command */
469  Status = CFE_TBL_GetWorkingBuffer(&WorkingBufferPtr, RegRecPtr, FALSE);
470 
471  if (Status == CFE_SUCCESS)
472  {
473  /* Copy data from file into working buffer */
474  Status = OS_read(FileDescriptor,
475  &WorkingBufferPtr->BufferPtr[TblFileHeader.Offset],
476  TblFileHeader.NumBytes);
477 
478  /* Make sure the appropriate number of bytes were read */
479  if (Status == (int32)TblFileHeader.NumBytes)
480  {
481  /* Check to ensure the file does not have any extra data at the end */
482  Status = OS_read(FileDescriptor, &ExtraByte, 1);
483 
484  /* If another byte was successfully read, then file contains more data than header claims */
485  if (Status == 1)
486  {
487  CFE_EVS_SendEvent(CFE_TBL_FILE_TOO_BIG_ERR_EID,
488  CFE_EVS_ERROR,
489  "File '%s' has more data than Tbl Hdr indicates (%d)",
490  LoadFilename,
491  (int)TblFileHeader.NumBytes);
492  }
493  else /* If error reading file or zero bytes read, assume it was the perfect size */
494  {
495  CFE_EVS_SendEvent(CFE_TBL_FILE_LOADED_INF_EID,
496  CFE_EVS_INFORMATION,
497  "Successful load of '%s' into '%s' working buffer",
498  LoadFilename,
499  TblFileHeader.TableName);
500 
501  /* Save file information statistics for later use in registry */
502  CFE_PSP_MemCpy(WorkingBufferPtr->DataSource, LoadFilename, OS_MAX_PATH_LEN);
503 
504  /* Save file creation time for later storage into Registry */
505  WorkingBufferPtr->FileCreateTimeSecs = StdFileHeader.TimeSeconds;
506  WorkingBufferPtr->FileCreateTimeSubSecs = StdFileHeader.TimeSubSeconds;
507 
508  /* Compute the CRC on the specified table buffer */
509  WorkingBufferPtr->Crc = CFE_ES_CalculateCRC(WorkingBufferPtr->BufferPtr,
510  RegRecPtr->Size,
511  0,
512  CFE_ES_DEFAULT_CRC);
513 
514  /* Initialize validation flag with TRUE if no Validation Function is required to be called */
515  WorkingBufferPtr->Validated = (RegRecPtr->ValidationFuncPtr == NULL);
516 
517  /* Save file information statistics for housekeeping telemetry */
518  strncpy(CFE_TBL_TaskData.HkPacket.Payload.LastFileLoaded, LoadFilename,
519  sizeof(CFE_TBL_TaskData.HkPacket.Payload.LastFileLoaded));

530  /* is considered a fatal error during a load process */
531  CFE_EVS_SendEvent(CFE_TBL_FILE_INCOMPLETE_ERR_EID,
532  CFE_EVS_ERROR,
533  "Incomplete load of '%s' into '%s' working buffer",

535  TblFileHeader.TableName);

541  CFE_EVS_ERROR,

550  (unsigned int)Status);
551  }
552  }
553  else
554  {
555  if ((TblFileHeader.NumBytes + TblFileHeader.Offset) > RegRecPtr->Size)
556  {
557  CFE_EVS_SendEvent(CFE_TBL_LOAD_EXCEEDS_SIZE_ERR_EID,
558  CFE_EVS_ERROR,

585  }

626  CFE_TBL_MAX_FULL_NAME_LEN, sizeof(CmdPtr->TableName));
627

629  RegIndex = CFE_TBL_FindTableInRegistry(TableName);
630

632  {
633  /* Obtain a pointer to registry information about specified table */
634  RegRecPtr = &CFE_TBL_TaskData.Registry[RegIndex];
635 
636  /* Determine what data is to be dumped */
637  if (CmdPtr->ActiveTblFlag == CFE_TBL_ACTIVE_BUFFER)
638  {
639  DumpDataAddr = RegRecPtr->Buffers[RegRecPtr->ActiveBufferIndex].BufferPtr;
640  }
641  else if (CmdPtr->ActiveTblFlag == CFE_TBL_INACTIVE_BUFFER)/* Dumping Inactive Buffer */
642  {
643  /* If this is a double buffered table, locating the inactive buffer is trivial */
644  if (RegRecPtr->DblBuffered)
645  {
646  DumpDataAddr = RegRecPtr->Buffers[(1U-RegRecPtr->ActiveBufferIndex)].BufferPtr;
647  }
648  else
649  {
650  /* For single buffered tables, the index to the inactive buffer is kept in 'LoadInProgress' */
651  /* Unless this is a table whose address was defined by the owning Application. */
652  if ((RegRecPtr->LoadInProgress != CFE_TBL_NO_LOAD_IN_PROGRESS) && (!RegRecPtr->UserDefAddr))
653  {
654  DumpDataAddr = CFE_TBL_TaskData.LoadBuffs[RegRecPtr->LoadInProgress].BufferPtr;
655  }
656  else
657  {
658  CFE_EVS_SendEvent(CFE_TBL_NO_INACTIVE_BUFFER_ERR_EID,
659  CFE_EVS_ERROR,
660  "No Inactive Buffer for Table '%s' present",
661  TableName);
662  }
663  }
664  }
665  else
666  {
667  CFE_EVS_SendEvent(CFE_TBL_ILLEGAL_BUFF_PARAM_ERR_EID,
668  CFE_EVS_ERROR,
669  "Cmd for Table '%s' had illegal buffer parameter (0x%08X)",
670  TableName, CmdPtr->ActiveTblFlag);
671  }
672 
673  /* If we have located the data to be dumped, then proceed with creating the file and dumping the data */
674  if (DumpDataAddr != NULL)
675  {
676  /* If this is not a dump only table, then we can perform the dump immediately */
677  if (!RegRecPtr->DumpOnly)
678  {
679  ReturnCode = CFE_TBL_DumpToFile(DumpFilename, TableName, DumpDataAddr, RegRecPtr->Size);
680  }
681  else /* Dump Only tables need to synchronize their dumps with the owner's execution */
682  {
683  /* Make sure a dump is not already in progress */
684  if (RegRecPtr->DumpControlIndex == CFE_TBL_NO_DUMP_PENDING)
685  {
686  /* Find a free Dump Control Block */
687  DumpIndex = 0;
688  while ((DumpIndex < CFE_TBL_MAX_SIMULTANEOUS_LOADS) &&
689  (CFE_TBL_TaskData.DumpControlBlocks[DumpIndex].State != CFE_TBL_DUMP_FREE))
690  {
691  DumpIndex++;
692  }
693 
694  if (DumpIndex < CFE_TBL_MAX_SIMULTANEOUS_LOADS)
695  {
696  /* Allocate a shared memory buffer for storing the data to be dumped */
697  Status = CFE_TBL_GetWorkingBuffer(&WorkingBufferPtr, RegRecPtr, FALSE);
698 
699  if (Status == CFE_SUCCESS)
700  {
701  DumpCtrlPtr = &CFE_TBL_TaskData.DumpControlBlocks[DumpIndex];
702  DumpCtrlPtr->State = CFE_TBL_DUMP_PENDING;
703  DumpCtrlPtr->RegRecPtr = RegRecPtr;
704 
705  /* Save the name of the desired dump filename, table name and size for later */
706  DumpCtrlPtr->DumpBufferPtr = WorkingBufferPtr;
707  CFE_PSP_MemCpy(DumpCtrlPtr->DumpBufferPtr->DataSource, DumpFilename, OS_MAX_PATH_LEN);
708  CFE_PSP_MemCpy(DumpCtrlPtr->TableName, TableName, CFE_TBL_MAX_FULL_NAME_LEN);
709  DumpCtrlPtr->Size = RegRecPtr->Size;
710 
711  /* Notify the owning application that a dump is pending */
712  RegRecPtr->DumpControlIndex = DumpIndex;
713 
714  /* If application requested notification by message, then do so */
715  CFE_TBL_SendNotificationMsg(RegRecPtr);
716 
717  /* Consider the command completed successfully */
718  ReturnCode = CFE_TBL_INC_CMD_CTR;
719  }
720  else
721  {
722  CFE_EVS_SendEvent(CFE_TBL_NO_WORK_BUFFERS_ERR_EID,
723  CFE_EVS_ERROR,
724  "No working buffers available for table '%s'",
725  TableName);
726  }
727  }
728  else
729  {
730  CFE_EVS_SendEvent(CFE_TBL_TOO_MANY_DUMPS_ERR_EID,
731  CFE_EVS_ERROR,
732  "Too many Dump Only Table Dumps have been requested");
733  }
734  }
735  else
736  {
737  CFE_EVS_SendEvent(CFE_TBL_DUMP_PENDING_ERR_EID,
738  CFE_EVS_ERROR,
739  "A dump for '%s' is already pending",
740  TableName);
741  }
742  }
743  }
744  }
745  else /* Table could not be found in Registry */
746  {
747  CFE_EVS_SendEvent(CFE_TBL_NO_SUCH_TABLE_ERR_EID,
748  CFE_EVS_ERROR,
749  "Unable to locate '%s' in Table Registry",
750  TableName);
751  }
752 
753  return ReturnCode;
754

758 /*******************************************************************
759 **
760 ** CFE_TBL_DumpToFile() -- Write table data to a file
761 **
762 ** NOTE: For complete prolog information, see prototype above
763 ********************************************************************/
764 
765 CFE_TBL_CmdProcRet_t CFE_TBL_DumpToFile( const char *DumpFilename, const char *TableName, void *DumpDataAddr, uint32 TblSizeInBytes)
766 {
767  CFE_TBL_CmdProcRet_t ReturnCode = CFE_TBL_INC_ERR_CTR; /* Assume failure */
768  boolean FileExistedPrev = FALSE;
769  CFE_FS_Header_t StdFileHeader;
770  CFE_TBL_File_Hdr_t TblFileHeader;
771  int32 FileDescriptor;
772  int32 Status;
773  int32 EndianCheck = 0x01020304;
774 
775  /* Clear Header of any garbage before copying content */
776  CFE_PSP_MemSet(&TblFileHeader, 0, sizeof(CFE_TBL_File_Hdr_t));
777 
778  /* Check to see if the dump file already exists */
779  FileDescriptor = OS_open(DumpFilename, OS_READ_ONLY, 0);
780 
781  if (FileDescriptor >= 0)
782  {
783  FileExistedPrev = TRUE;
784 
785  OS_close(FileDescriptor);
786  }
787 
788  /* Create a new dump file, overwriting anything that may have existed previously */
789  FileDescriptor = OS_creat(DumpFilename, OS_WRITE_ONLY);
790 
791  if (FileDescriptor >= OS_FS_SUCCESS)
792  {
793  /* Initialize the standard cFE File Header for the Dump File */
794  CFE_FS_InitHeader(&StdFileHeader, "Table Dump Image", CFE_FS_TBL_IMG_SUBTYPE);
795 
796  /* Output the Standard cFE File Header to the Dump File */
797  Status = CFE_FS_WriteHeader(FileDescriptor, &StdFileHeader);
798 
799  if (Status == sizeof(CFE_FS_Header_t))
800  {
801  /* Initialize the Table Image Header for the Dump File */
802  strncpy(TblFileHeader.TableName, TableName, sizeof(TblFileHeader.TableName));
803  TblFileHeader.Offset = 0;
804  TblFileHeader.NumBytes = TblSizeInBytes;
805  TblFileHeader.Reserved = 0;
806 
807  /* Determine if this is a little endian processor */
808  if ((*(char *)&EndianCheck) == 0x04)
809  {
810  /* If this is a little endian processor, then byte swap the header to a big endian format */
811  /* to maintain the cFE Header standards */
812  /* NOTE: FOR THE REMAINDER OF THIS FUNCTION, THE CONTENTS OF THE HEADER IS UNREADABLE BY */
813  /* THIS PROCESSOR! THE DATA WOULD NEED TO BE SWAPPED BACK BEFORE READING. */
814  CFE_TBL_ByteSwapTblHeader(&TblFileHeader);
815  }
816 
817  /* Output the Table Image Header to the Dump File */
818  Status = OS_write(FileDescriptor, &TblFileHeader, sizeof(CFE_TBL_File_Hdr_t));
819 
820  /* Make sure the header was output completely */
821  if (Status == sizeof(CFE_TBL_File_Hdr_t))
822  {
823  /* Output the requested data to the dump file */
824  /* Output the active table image data to the dump file */
825  Status = OS_write(FileDescriptor,
826  DumpDataAddr,
827  TblSizeInBytes);
828 
829  if (Status == (int32)TblSizeInBytes)
830  {
831  if (FileExistedPrev)
832  {
833  CFE_EVS_SendEvent(CFE_TBL_OVERWRITE_DUMP_INF_EID,
834  CFE_EVS_INFORMATION,
835  "Successfully overwrote '%s' with Table '%s'",
836  DumpFilename, TableName);
837  }
838  else
839  {
840  CFE_EVS_SendEvent(CFE_TBL_WRITE_DUMP_INF_EID,
841  CFE_EVS_INFORMATION,
842  "Successfully dumped Table '%s' to '%s'",
843  TableName, DumpFilename);
844  }
845 
846  /* Save file information statistics for housekeeping telemetry */
847  strncpy(CFE_TBL_TaskData.HkPacket.Payload.LastFileDumped, DumpFilename,
848  sizeof(CFE_TBL_TaskData.HkPacket.Payload.LastFileDumped));
849 
850  /* Increment Successful Command Counter */
851  ReturnCode = CFE_TBL_INC_CMD_CTR;
852  }
853  else
854  {
855  CFE_EVS_SendEvent(CFE_TBL_WRITE_TBL_IMG_ERR_EID,
856  CFE_EVS_ERROR,
857  "Error writing Tbl image to '%s', Status=0x%08X",
858  DumpFilename, (unsigned int)Status);
859  }
860  }
861  else
862  {
863  CFE_EVS_SendEvent(CFE_TBL_WRITE_TBL_HDR_ERR_EID,
864  CFE_EVS_ERROR,
865  "Error writing Tbl image File Header to '%s', Status=0x%08X",
866  DumpFilename, (unsigned int)Status);
867  }
868  }
869  else
870  {
871  CFE_EVS_SendEvent(CFE_TBL_WRITE_CFE_HDR_ERR_EID,
872  CFE_EVS_ERROR,
873  "Error writing cFE File Header to '%s', Status=0x%08X",
874  DumpFilename, (unsigned int)Status);
875  }
876 
877  /* We are done outputting data to the dump file. Close it. */
878  OS_close(FileDescriptor);
879  }
880  else
881  {
882  CFE_EVS_SendEvent(CFE_TBL_CREATING_DUMP_FILE_ERR_EID,
883  CFE_EVS_ERROR,
884  "Error creating dump file '%s', Status=0x%08X",
885  DumpFilename, (unsigned int)FileDescriptor);
886  }
887 
888  return ReturnCode;
889 }

1011  }

1016  /* as data to be sent to the ground in telemetry. */

1018 
1019  CFE_EVS_SendEvent(CFE_TBL_ASSUMED_VALID_INF_EID,
1020  CFE_EVS_INFORMATION,

1023  }

1032  "Too many Table Validations have been requested");
1033  }
1034  }
1035  }

1038  CFE_EVS_SendEvent(CFE_TBL_NO_SUCH_TABLE_ERR_EID,
