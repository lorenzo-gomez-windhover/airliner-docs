174 void CFE_TIME_TaskMain(void)
175 {
176  int32 Status;
177 
178  CFE_ES_PerfLogEntry(CFE_TIME_MAIN_PERF_ID);
179 
180  Status = CFE_TIME_TaskInit();
181 
182  if(Status != CFE_SUCCESS)
183  {
184  CFE_ES_WriteToSysLog("TIME:Application Init Failed,RC=0x%08X\n", (unsigned int)Status);
185  CFE_ES_PerfLogExit(CFE_TIME_MAIN_PERF_ID);
186  /* Note: CFE_ES_ExitApp will not return */
187  CFE_ES_ExitApp(CFE_ES_RUNSTATUS_CORE_APP_INIT_ERROR);
188  }/* end if */
189 
190  /*
191  * Wait for other apps to start.
192  * It is important that the core apps are present before this starts receiving
193  * messages from the command pipe, as some of those handlers might depend on
194  * the other core apps.
195  */
196  CFE_ES_WaitForStartupSync(CFE_CORE_MAX_STARTUP_MSEC);
197 
198  /* Main loop */
199  while (Status == CFE_SUCCESS)
200  {
201 
202  /* Increment the Main task Execution Counter */
203  CFE_ES_IncrementTaskCounter();
204 
205  CFE_ES_PerfLogExit(CFE_TIME_MAIN_PERF_ID);
206 
207  /* Pend on receipt of packet */
208  Status = CFE_SB_RcvMsg(&CFE_TIME_TaskData.MsgPtr,
209  CFE_TIME_TaskData.CmdPipe,
210  CFE_SB_PEND_FOREVER);
211 
212  CFE_ES_PerfLogEntry(CFE_TIME_MAIN_PERF_ID);
213 
214  if (Status == CFE_SUCCESS)
215  {
216  /* Process cmd pipe msg */
217  CFE_TIME_TaskPipe(CFE_TIME_TaskData.MsgPtr);
218  }else{
219  CFE_ES_WriteToSysLog("TIME:Error reading cmd pipe,RC=0x%08X\n",(unsigned int)Status);
220  }/* end if */
221 
222  }/* end while */
223 
224  /* while loop exits only if CFE_SB_RcvMsg returns error */
225  CFE_ES_ExitApp(CFE_ES_RUNSTATUS_CORE_APP_RUNTIME_ERROR);
226 
227 } /* end CFE_TIME_TaskMain */
