115 /*--------------------------------------------------------------------------------------
116  Name: OS_creat
117 
118  Purpose: creates a file specified by const char *path, with read/write
119  permissions by access. The file is also automatically opened by the
120  create call.
121 
122  Returns: OS_FS_ERR_INVALID_POINTER if path is NULL
123  OS_FS_ERR_PATH_TOO_LONG if path exceeds the maximum number of chars
124  OS_FS_ERR_PATH_INVALID if path cannot be parsed
125  OS_FS_ERR_NAME_TOO_LONG if the name of the file is too long
126  OS_FS_ERROR if permissions are unknown or OS call fails
127  OS_FS_ERR_NO_FREE_FDS if there are no free file descripors left
128  OS_FS_SUCCESS if success
129 
130 ---------------------------------------------------------------------------------------*/
131 
132 int32 OS_creat (const char *path, int32 access)
133 {
134  int status;
135  char local_path[OS_MAX_LOCAL_PATH_LEN];
136  int perm;
137  mode_t mode;
138  uint32 PossibleFD;
139  sigset_t previous;
140  sigset_t mask;
141 
142  /*
143  ** Check to see if the path pointer is NULL
144  */
145  if (path == NULL)
146  {
147  return OS_FS_ERR_INVALID_POINTER;
148  }
149 
150  /*
151  ** Check to see if the path is too long
152  */
153  if (strlen(path) >= OS_MAX_PATH_LEN)
154  {
155  return OS_FS_ERR_PATH_TOO_LONG;
156  }
157 
158  /*
159  ** check if the name of the file is too long
160  */
161  if (OS_check_name_length(path) != OS_FS_SUCCESS)
162  {
163  return OS_FS_ERR_NAME_TOO_LONG;
164  }
165 
166  /*
167  ** Check for a valid access mode
168  ** For creating a file, OS_READ_ONLY does not make sense
169  */
170  switch(access)
171  {
172  case OS_WRITE_ONLY:
173  perm = O_WRONLY;
174  break;
175  case OS_READ_WRITE:
176  perm = O_RDWR;
177  break;
178  default:
179  return OS_FS_ERROR;
180  }
181 
182  /*
183  ** Translate the path
184  */
185  if ( OS_TranslatePath(path, (char *)local_path) != OS_FS_SUCCESS )
186  {
187  return OS_FS_ERR_PATH_INVALID;
188  }
189 
190  OS_InterruptSafeLock(&OS_FDTableMutex, &mask, &previous);
191 
192  for ( PossibleFD = 0; PossibleFD < OS_MAX_NUM_OPEN_FILES; PossibleFD++)
193  {
194  if( OS_FDTable[PossibleFD].IsValid == FALSE)
195  {
196  break;
197  }
198  }
199 
200  if (PossibleFD >= OS_MAX_NUM_OPEN_FILES)
201  {
202  OS_InterruptSafeUnlock(&OS_FDTableMutex, &previous);
203  return OS_FS_ERR_NO_FREE_FDS;
204  }
205 
206 
207  /* Mark the table entry as valid so no other
208  * task can take that ID */
209  OS_FDTable[PossibleFD].IsValid = TRUE;
210 
211  mode = S_IRUSR | S_IWUSR | S_IRGRP | S_IWGRP | S_IROTH | S_IWOTH;
212 
213  status = open(local_path, perm | O_CREAT | O_TRUNC, mode);
214 
215  if (status != ERROR)
216  {
217  /* fill in the table before returning */
218  OS_FDTable[PossibleFD].OSfd = status;
219  strncpy(OS_FDTable[PossibleFD].Path, path, OS_MAX_PATH_LEN);
220  OS_FDTable[PossibleFD].User = OS_FindCreator();
221  OS_InterruptSafeUnlock(&OS_FDTableMutex, &previous);
222  return PossibleFD;
223  }
224  else
225  {
226  /* Operation failed, so reset to false */
227  OS_FDTable[PossibleFD].IsValid = FALSE;
228  OS_InterruptSafeUnlock(&OS_FDTableMutex, &previous);
229  return OS_FS_ERROR;
230  }
231 
232 } /* end OS_creat */

353 /*--------------------------------------------------------------------------------------
354  Name: OS_close
355 
356  Purpose: Closes a file.
357 
358  Returns: OS_FS_ERROR if file descriptor could not be closed
359  OS_FS_ERR_INVALID_FD if the file descriptor passed in is invalid
360  OS_FS_SUCCESS if success
361 ---------------------------------------------------------------------------------------*/
362 
363 int32 OS_close (int32 filedes)
364 {
365  int status;
366  sigset_t previous;
367  sigset_t mask;
368 
369  /* Make sure the file descriptor is legit before using it */
370  if (filedes < 0 || filedes >= OS_MAX_NUM_OPEN_FILES || OS_FDTable[filedes].IsValid == FALSE)
371  {
372  return OS_FS_ERR_INVALID_FD;
373  }
374  else
375  {
376  OS_InterruptSafeLock(&OS_FDTableMutex, &mask, &previous);
377 
378  /*
379  ** call close, and check for an interrupted system call
380  */
381  do
382  {
383  status = close ((int) OS_FDTable[filedes].OSfd);
384  } while ( status == -1 && errno == EINTR );
385 
386  if (status == ERROR)
387  {
388  /*
389  ** Remove the file from the OSAL list
390  ** to free up that slot
391  */
392  /* fill in the table before returning */
393  OS_FDTable[filedes].OSfd = -1;
394  strcpy(OS_FDTable[filedes].Path, "\0");
395  OS_FDTable[filedes].User = 0;
396  OS_FDTable[filedes].IsValid = FALSE;
397  OS_InterruptSafeUnlock(&OS_FDTableMutex, &previous);
398 
399  return OS_FS_ERROR;
400  }
401  else
402  {
403  /* fill in the table before returning */
404  OS_FDTable[filedes].OSfd = -1;
405  strcpy(OS_FDTable[filedes].Path, "\0");
406  OS_FDTable[filedes].User = 0;
407  OS_FDTable[filedes].IsValid = FALSE;
408  OS_InterruptSafeUnlock(&OS_FDTableMutex, &previous);
409 
410  return OS_FS_SUCCESS;
411  }
412 
413  }
414 
415 }/* end OS_close */
