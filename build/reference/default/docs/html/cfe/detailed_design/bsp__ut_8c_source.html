48 void UT_BSP_ParseCommandLine(int argc, char *argv[])
49 {
50  uint8 UserShift;
51  int opt;
52 
53  UserShift = UTASSERT_CASETYPE_NONE;
54  while ((opt = getopt(argc, argv, "v:qd")) != -1)
55  {
56  switch (opt) {
57  case 'd':
58  UserShift = UTASSERT_CASETYPE_DEBUG;
59  break;
60  case 'q':
61  UserShift = UTASSERT_CASETYPE_FAILURE;
62  break;
63  case 'v':
64  UserShift = atoi(optarg);
65  break;
66  default: /* '?' */
67  fprintf(stderr, "Usage: %s [-v verbosity] [-d] [-q]\n",
68  argv[0]);
69  exit(EXIT_FAILURE);
70  }
71  if (UserShift > 0 && UserShift < UTASSERT_CASETYPE_MAX)
72  {
73  CurrVerbosity = (2 << UserShift) - 1;
74  }
75  }

109 void UT_BSP_DoText(uint8 MessageType, const char *OutputMessage)
110 {
111  const char *Prefix;
112 
113  if ((CurrVerbosity >> MessageType) & 1)
114  {
115  switch(MessageType)
116  {
117  case UTASSERT_CASETYPE_ABORT:
118  Prefix = "ABORT";
119  break;
120  case UTASSERT_CASETYPE_FAILURE:
121  Prefix = "FAIL";
122  break;
123  case UTASSERT_CASETYPE_MIR:
124  Prefix = "MIR";
125  break;
126  case UTASSERT_CASETYPE_TSF:
127  Prefix = "TSF";
128  break;
129  case UTASSERT_CASETYPE_TTF:
130  Prefix = "TTF";
131  break;
132  case UTASSERT_CASETYPE_NA:
133  Prefix = "N/A";
134  break;
135  case UTASSERT_CASETYPE_BEGIN:
136  printf("\n"); /* add a bit of extra whitespace between tests */
137  Prefix = "BEGIN";
138  break;
139  case UTASSERT_CASETYPE_END:
140  Prefix = "END";
141  break;
142  case UTASSERT_CASETYPE_PASS:
143  Prefix = "PASS";
144  break;
145  case UTASSERT_CASETYPE_INFO:
146  Prefix = "INFO";
147  break;
148  case UTASSERT_CASETYPE_DEBUG:
149  Prefix = "DEBUG";
150  break;
151  default:
152  Prefix = "OTHER";
153  break;
154  }
155  printf("[%5s] %s\n",Prefix,OutputMessage);
156  }
157 
158  /*
159  * If any ABORT (major failure) message is thrown,
160  * then actually call abort() to stop the test and dump a core
161  */
162  if (MessageType == UTASSERT_CASETYPE_ABORT)
163  {
164  abort();
165  }
166 }

213 void UT_BSP_EndTest(const UtAssert_TestCounter_t *TestCounters)
214 {
215  int status = 0;
216 
217  /*
218  * Only output a "summary" if there is more than one test Segment.
219  * Otherwise it is a duplicate of the report already given.
220  */
221  if (TestCounters->TestSegmentCount > 1)
222  {
223  UT_BSP_DoTestSegmentReport("SUMMARY", TestCounters);
224  }
225 
226  printf("COMPLETE: %u tests Segment(s) executed\n\n", (unsigned int)TestCounters->TestSegmentCount);
227 
228  /*
229  * The Linux UT BSP allows at least a 7 bit status code to be returned to the OS (i.e. the exit status
230  * of the process). This is useful to report pass/fail. Because we have multiple bits, we can make
231  * descriptive exit status codes to indicate what went wrong. Anything nonzero represents failure.
232  *
233  * Consider Failures as well as "TSF" (setup failures) to be grounds for returning nonzero (bad) status.
234  * Also the lack of ANY test cases should produce a bad status.
235  *
236  * "MIR" results should not produce a bad status -- these may have worked fine, we do not know.
237  *
238  * Likewise "N/A" tests are simply not applicable, so we just ignore them.
239  */
240 
241  if (TestCounters->TotalTestCases == 0)
242  {
243  status |= 0x01;
244  }
245 
246  if (TestCounters->CaseCount[UTASSERT_CASETYPE_FAILURE] > 0)
247  {
248  status |= 0x02;
249  }
250 
251  if (TestCounters->CaseCount[UTASSERT_CASETYPE_TSF] > 0)
252  {
253  status |= 0x04;
254  }
255 
256  exit(status);
257 }
