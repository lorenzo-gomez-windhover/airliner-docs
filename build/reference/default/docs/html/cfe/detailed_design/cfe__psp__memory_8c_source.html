117 int32 CFE_PSP_InitCDS(uint32 RestartType )
118 {
119  int32 return_code;
120  key_t key;
121 
122  /*
123  ** Make the Shared memory key
124  */
125  if ((key = ftok(CFE_PSP_CDS_KEY_FILE, 'R')) == -1)
126  {
127  OS_printf("CFE_PSP: Cannot Create CDS Shared memory key!\n");
128  exit(-1);
129  }
130 
131  /*
132  ** connect to (and possibly create) the segment:
133  */
134  if ((CDSShmId = shmget(key, CFE_PSP_CDS_SIZE, 0644 | IPC_CREAT)) == -1)
135  {
136  OS_printf("CFE_PSP: Cannot shmget CDS Shared memory Segment! CFE_PSP_CDS_SIZE=%u errno=%i\n", CFE_PSP_CDS_SIZE, errno);
137  exit(-1);
138  }
139 
140  /*
141  ** attach to the segment to get a pointer to it:
142  */
143  CFE_PSP_CDSPtr = shmat(CDSShmId, (void *)0, 0);
144  if (CFE_PSP_CDSPtr == (uint8 *)(-1))
145  {
146  OS_printf("CFE_PSP: Cannot shmat to CDS Shared memory Segment!\n");
147  exit(-1);
148  }
149 
150  if ( RestartType == CFE_ES_POWERON_RESET )
151  {
152  OS_printf("CFE_PSP: Clearing out CFE CDS Shared memory segment.\n");
153  memset(CFE_PSP_CDSPtr, 0, CFE_PSP_CDS_SIZE);
154  }
155 
156  return_code = CFE_PSP_SUCCESS;
157  return(return_code);
158 
159 }

638 /******************************************************************************
639 ** Function: CFE_PSP_InitProcessorReservedMemory
640 **
641 ** Purpose:
642 ** This function performs the top level reserved memory initialization.
643 **
644 ** Arguments:
645 ** (none)
646 **
647 ** Return:
648 ** (none)
649 */
650 int32 CFE_PSP_InitProcessorReservedMemory( uint32 RestartType )
651 {
652  int32 return_code;
653  int tempFd;
654 
655  /*
656  ** Create the key files for the shared memory segments
657  ** The files are not needed, so they are closed right away.
658  */
659  tempFd = open(CFE_PSP_CDS_KEY_FILE, O_RDONLY | O_CREAT, S_IRWXU );
660  close(tempFd);
661  tempFd = open(CFE_PSP_RESET_KEY_FILE, O_RDONLY | O_CREAT, S_IRWXU );
662  close(tempFd);
663  tempFd = open(CFE_PSP_RESERVED_KEY_FILE, O_RDONLY | O_CREAT, S_IRWXU );
664  close(tempFd);
665 
666  if ( RestartType == CFE_ES_PROCESSOR_RESET )
667  {
668  return_code = CFE_PSP_InitCDS( CFE_ES_PROCESSOR_RESET );
669  return_code = CFE_PSP_InitResetArea( CFE_ES_PROCESSOR_RESET );
670  return_code = CFE_PSP_InitVolatileDiskMem( CFE_ES_PROCESSOR_RESET );
671  return_code = CFE_PSP_InitUserReservedArea( CFE_ES_PROCESSOR_RESET );
672  }
673  else
674  {
675  return_code = CFE_PSP_InitCDS( CFE_ES_POWERON_RESET );
676  return_code = CFE_PSP_InitResetArea( CFE_ES_POWERON_RESET );
677  return_code = CFE_PSP_InitVolatileDiskMem( CFE_ES_POWERON_RESET );
678  return_code = CFE_PSP_InitUserReservedArea( CFE_ES_POWERON_RESET );
679  }
680 
681  return(return_code);
682 }
