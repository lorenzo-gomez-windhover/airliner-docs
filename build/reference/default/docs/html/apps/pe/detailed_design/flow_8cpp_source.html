60 int32 PE::flowMeasure(math::Vector2F &y)
61 {
62  int32 Status = CFE_SUCCESS;
63  float d = 0.0f;
64  float flow_x_rad = 0.0f;
65  float flow_y_rad = 0.0f;
66  float dt_flow = 0.0f;
67  float gyro_x_rad = 0.0f;
68  float gyro_y_rad = 0.0f;
69  math::Vector3F delta_b;
70  math::Vector3F delta_n;
71 
72  y.Zero();
73  delta_b.Zero();
74  delta_n.Zero();
75 
76  /* Check for sane pitch/roll */
77  if (m_Euler[0] > 0.5f || m_Euler[1] > 0.5f) {
78  Status = -1;
79  goto flowMeasure_Exit_Tag;
80  }
81 
82  /* Check if AGL valid */
83  if (m_DistFault) {
84  Status = -1;
85  goto flowMeasure_Exit_Tag;
86  }
87 
88  /* Check for AGL height */
89  if (m_AglLowPass.m_State < ConfigTblPtr->FLOW_MIN_AGL) {
90  Status = -1;
91  goto flowMeasure_Exit_Tag;
92  }
93 
94  /* Check if terrain estimate valid */
95  if (!m_TzEstValid) {
96  Status = -1;
97  goto flowMeasure_Exit_Tag;
98  }
99 
100  /* Check reported quality */
101  if (m_OpticalFlowMsg.Quality < ConfigTblPtr->FLOW_QUALITY_MIN) {
102  Status = -1;
103  goto flowMeasure_Exit_Tag;
104  }
105 
106  /* Calculate range to center of image for flow */
107  d = m_AglLowPass.m_State * cosf(m_Euler[0]) * cosf(m_Euler[1]);
108 
109  /* Optical flow in x, y axis */
110  /* Note: Consider making flow scale a state of the kalman filter */
111  flow_x_rad = m_OpticalFlowMsg.PixelFlowXIntegral * ConfigTblPtr->FLOW_SCALE;
112  flow_y_rad = m_OpticalFlowMsg.PixelFlowYIntegral * ConfigTblPtr->FLOW_SCALE;
113  dt_flow = m_OpticalFlowMsg.IntegrationTimespan / 1.0e6f;
114 
115  if (dt_flow > 0.5f || dt_flow < 1.0e-6f) {
116  Status = -1;
117  goto flowMeasure_Exit_Tag;
118  }
119 
120  /* Angular rotation in x, y axis */
121  gyro_x_rad = m_FlowGyroXHighPass.Update(m_OpticalFlowMsg.GyroXRateIntegral, m_TimeLastFlow - m_Timestamp, FLOW_GYRO_HP_CUTOFF);
122  gyro_y_rad = m_FlowGyroYHighPass.Update(m_OpticalFlowMsg.GyroYRateIntegral, m_TimeLastFlow - m_Timestamp, FLOW_GYRO_HP_CUTOFF);
123 
124  /* Compute velocities in body frame using ground distance */
125  /* Note: Integral rates in the optical_flow msg are RH rotations about body axes */
126  delta_b[0] = +(flow_y_rad - gyro_y_rad) * d;
127  delta_b[1] = -(flow_x_rad - gyro_x_rad) * d;
128  delta_b[2] = 0.0f;
129 
130  /* Rotation of flow from body to nav frame */
131  delta_n = m_RotationMat * delta_b;
132 
133  /* Measurement */
134  y[Y_flow_vx] = delta_n[0] / dt_flow;
135  y[Y_flow_vy] = delta_n[1] / dt_flow;
136 
137  m_TimeLastFlow = m_Timestamp;
138  m_FlowQStats.update(float(m_OpticalFlowMsg.Quality));
139 
140 flowMeasure_Exit_Tag:
141  return Status;
142 }

145 void PE::flowCorrect()
146 {
147  CFE_ES_PerfLogEntry(PE_SENSOR_FLOW_PERF_ID);
148  const float h_min = 2.0f;
149  const float h_max = 8.0f;
150  const float v_min = 0.5f;
151  const float v_max = 1.0f;
152  float h = 0.0f;
153  float v = 0.0f;
154  float flow_vxy_stddev = 0.0f;
155  float rotrate_sq = 0.0f;
156  float rot_sq = 0.0f;
157 
158  /* Polynomial noise model, found using least squares fit
159  ** h, h**2, v, v*h, v*h**2 */
160  const float p[5] = {0.04005232f, -0.00656446f, -0.26265873f, 0.13686658f, -0.00397357f};
161 
162  if (flowMeasure(m_Flow.y) != CFE_SUCCESS)
163  {
164  goto end_of_function;
165  }
166 
167  /* Measured altitude, negative down dir */
168  m_Flow.C[Y_flow_vx][X_vx] = 1.0f;
169  m_Flow.C[Y_flow_vy][X_vy] = 1.0f;
170 
171  /* Prevent extrapolation past end of polynomial fit by bounding independent variables */
172  h = m_AglLowPass.m_State;
173  v = sqrtf(m_Flow.y * m_Flow.y);
174 
175  if (h > h_max) {
176  h = h_max;
177  }
178 
179  if (h < h_min) {
180  h = h_min;
181  }
182 
183  if (v > v_max) {
184  v = v_max;
185  }
186 
187  if (v < v_min) {
188  v = v_min;
189  }
190 
191  /* Compute polynomial value */
192  flow_vxy_stddev = p[0] * h + p[1] * h * h + p[2] * v + p[3] * v * h + p[4] * v * h * h; // TODO: Adding a constant to this may help performance
193 
194  rotrate_sq = m_VehicleAttitudeMsg.RollSpeed * m_VehicleAttitudeMsg.RollSpeed
195  + m_VehicleAttitudeMsg.PitchSpeed * m_VehicleAttitudeMsg.PitchSpeed
196  + m_VehicleAttitudeMsg.YawSpeed * m_VehicleAttitudeMsg.YawSpeed;
197 
198  rot_sq = m_Euler[0] * m_Euler[0] + m_Euler[1] * m_Euler[1];
199 
200  m_Flow.R[Y_flow_vx][Y_flow_vx] = (flow_vxy_stddev * flow_vxy_stddev) +
201  (ConfigTblPtr->FLOW_R * ConfigTblPtr->FLOW_R * rot_sq) +
202  (ConfigTblPtr->FLOW_RR * ConfigTblPtr->FLOW_RR * rotrate_sq);
203  m_Flow.R[Y_flow_vy][Y_flow_vy] = m_Flow.R[Y_flow_vx][Y_flow_vx];
204 
205  /* Vector2F - (2x10 * Vector10F) */
206  m_Flow.r = m_Flow.y - (m_Flow.C * m_StateVec);
207 
208  /* Residual */
209  /* ((2x10 * 10x10) * 10x2) + 2x2) */
210  m_Flow.S_I = m_Flow.C * m_StateCov * m_Flow.C.Transpose() + m_Flow.R;
211  m_Flow.S_I = m_Flow.S_I.Inversed();
212 
213  /* Fault detection 1x2 * 2x2 * 2F */
214  m_Flow.beta = (m_Flow.r.Transpose() * (m_Flow.S_I * m_Flow.r));
215 
216  if (m_Flow.beta > BETA_TABLE[n_y_flow])
217  {
218  if (!m_FlowFault)
219  {
220  if(Initialized())
221  {
222  (void) CFE_EVS_SendEvent(PE_FLOW_FAULT_ERR_EID, CFE_EVS_ERROR,
223  "Flow fault, beta %5.2f", m_Flow.beta);
224  }
225  m_FlowFault = TRUE;
226  }
227  goto end_of_function;
228  }
229  else
230  {
231  if (m_FlowFault)
232  {
233  m_FlowFault = FALSE;
234  (void) CFE_EVS_SendEvent(PE_FLOW_OK_INF_EID, CFE_EVS_INFORMATION,
235  "Flow OK, beta %5.2f", m_Flow.beta);
236 
237  m_FlowInitialized = TRUE;
238  }
239  }
240 
241  /* Kalman filter correction */
242  if (!m_FlowFault)
243  {
244  /* 10x10 * 10x2 * 2x2 */
245  m_Flow.K = (m_StateCov * m_Flow.C.Transpose()) * m_Flow.S_I;
246 
247  /* 10x2 * 2x2 */
248  m_Flow.dx = m_Flow.K * m_Flow.r;
249 
250  /* 10F + 10F*/
251  m_StateVec = m_StateVec + m_Flow.dx;
252 
253  /* 10x10 - 10x2 * 2x10 * 10x10 */
254  m_StateCov = m_StateCov - m_Flow.K * m_Flow.C * m_StateCov;
255  }
256 
257 end_of_function:
258  CFE_ES_PerfLogExit(PE_SENSOR_FLOW_PERF_ID);
259 }

262 void PE::flowCheckTimeout()
263 {
264  uint64 Timestamp = 0;
265 
266  if (m_Timestamp > m_TimeLastFlow)
267  {
268  Timestamp = m_Timestamp - m_TimeLastFlow;
269  }
270  else if (m_Timestamp < m_TimeLastFlow)
271  {
272  Timestamp = m_TimeLastFlow - m_Timestamp;
273  }
274 
275  if (Timestamp > FLOW_TIMEOUT)
276  {
277  if (!m_FlowTimeout)
278  {
279  m_FlowTimeout = TRUE;
280  m_FlowQStats.reset();
281  (void) CFE_EVS_SendEvent(PE_FLOW_TIMEOUT_ERR_EID, CFE_EVS_ERROR,
282  "Flow timeout: %llu us", Timestamp);
283  }
284  }
