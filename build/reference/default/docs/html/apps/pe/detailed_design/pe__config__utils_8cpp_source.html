48 /* Initialize Config Table */
49 /* */
50 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
51 int32 PE::InitConfigTbl()
52 {
53  int32 iStatus=0;
54 
55  /* Register Config table */
56  iStatus = CFE_TBL_Register(&ConfigTblHdl,
57  PE_CONFIG_TABLENAME,
58  (sizeof(PE_ConfigTbl_t)),
59  CFE_TBL_OPT_DEFAULT,
60  PE::ValidateConfigTbl);
61  if (iStatus != CFE_SUCCESS)
62  {
63  /* Note, a critical table could return another nominal code. If this table is
64  * made critical this logic would have to change. */
65  (void) CFE_EVS_SendEvent(PE_CFGTBL_REG_ERR_EID, CFE_EVS_ERROR,
66  "Failed to register config table (0x%08lX)",
67  iStatus);
68  goto PE_InitConfigTbl_Exit_Tag;
69  }
70 
71  /* Load Config table file */
72  iStatus = CFE_TBL_Load(ConfigTblHdl,
73  CFE_TBL_SRC_FILE,
74  PE_CONFIG_TABLE_FILENAME);
75  if (iStatus != CFE_SUCCESS)
76  {
77  /* Note, CFE_SUCCESS is for a successful full table load. If a partial table
78  load is desired then this logic would have to change. */
79  (void) CFE_EVS_SendEvent(PE_CFGTBL_LOAD_ERR_EID, CFE_EVS_ERROR,
80  "Failed to load Config Table (0x%08lX)",
81  iStatus);
82  goto PE_InitConfigTbl_Exit_Tag;
83  }
84 
85  iStatus = AcquireConfigPointers();
86 
87 PE_InitConfigTbl_Exit_Tag:
88  return iStatus;
89 }

390 /* Acquire Config Pointers */
391 /* */
392 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
393 int32 PE::AcquireConfigPointers(void)
394 {
395  int32 iStatus = CFE_SUCCESS;
396 
397  /*
398  ** Release the table
399  */
400  (void) CFE_TBL_ReleaseAddress(ConfigTblHdl);
401 
402  /*
403  ** Manage the table
404  */
405  iStatus = CFE_TBL_Manage(ConfigTblHdl);
406  if ((iStatus != CFE_SUCCESS) && (iStatus != CFE_TBL_INFO_UPDATED))
407  {
408  (void) CFE_EVS_SendEvent(PE_CFGTBL_MANAGE_ERR_EID, CFE_EVS_ERROR,
409  "Failed to manage PE Config table (0x%08lX)",
410  iStatus);
411  goto PE_AcquireConfigPointers_Exit_Tag;
412  }
413 
414  /*
415  ** Get a pointer to the table
416  */
417  iStatus = CFE_TBL_GetAddress((void**)&ConfigTblPtr, ConfigTblHdl);
418  if (iStatus == CFE_TBL_INFO_UPDATED)
419  {
420  m_ParamsUpdated = true;
421  UpdateLocalParams();
422  iStatus = CFE_SUCCESS;
423  }
424  else if(iStatus != CFE_SUCCESS)
425  {
426  ConfigTblPtr = 0;
427  (void) CFE_EVS_SendEvent(PE_CFGTBL_GETADDR_ERR_EID, CFE_EVS_ERROR,
428  "Failed to get Config table's address (0x%08lX)",
429  iStatus);
430  }
431 
432 PE_AcquireConfigPointers_Exit_Tag:
433  return iStatus;
434 }
