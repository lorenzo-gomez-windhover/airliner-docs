82 void PE::landCorrect()
83 {
84  CFE_ES_PerfLogEntry(PE_SENSOR_LAND_PERF_ID);
85 
86  if (landMeasure(m_Land.y) != CFE_SUCCESS)
87  {
88  goto end_of_function;
89  }
90 
91  /* measurement matrix */
92  m_Land.C[Y_land_vx][X_vx] = 1.0f;
93  m_Land.C[Y_land_vy][X_vy] = 1.0f;
94  m_Land.C[Y_land_agl][X_z] = -1.0f; // measured altitude, negative down dir.
95  m_Land.C[Y_land_agl][X_tz] = 1.0f; // measured altitude, negative down dir.
96 
97  /* use parameter covariance */
98  m_Land.R[Y_land_vx][Y_land_vx] = ConfigTblPtr->LAND_VXY_STDDEV * ConfigTblPtr->LAND_VXY_STDDEV;
99  m_Land.R[Y_land_vy][Y_land_vy] = ConfigTblPtr->LAND_VXY_STDDEV * ConfigTblPtr->LAND_VXY_STDDEV;
100  m_Land.R[Y_land_agl][Y_land_agl] = ConfigTblPtr->LAND_Z_STDDEV * ConfigTblPtr->LAND_Z_STDDEV;
101 
102  /* residual */
103  m_Land.S_I = (m_Land.C * (m_StateCov * m_Land.C.Transpose())) + m_Land.R;
104  m_Land.S_I = m_Land.S_I.Inversed();
105 
106  m_Land.r = m_Land.y - m_Land.C * m_StateVec;
107 
108  /* fault detection */
109  m_Land.beta = (m_Land.r.Transpose() * (m_Land.S_I * m_Land.r));
110 
111  /* artifically increase beta threshhold to prevent fault during landing */
112  m_Land.beta_thresh = 1e2f;
113 
114  if (m_Land.beta / BETA_TABLE[n_y_land] > m_Land.beta_thresh)
115  {
116  if (!m_LandFault)
117  {
118  m_LandFault = TRUE;
119  if(Initialized())
120  {
121  (void) CFE_EVS_SendEvent(PE_LAND_FAULT_ERR_EID, CFE_EVS_ERROR,
122  "Land detector fault, beta %5.2f", double(m_Land.beta));
123  }
124  }
125  }
126  else
127  {
128  if (m_LandFault)
129  {
130  m_LandFault = FALSE;
131  (void) CFE_EVS_SendEvent(PE_LAND_OK_INF_EID, CFE_EVS_ERROR,
132  "Land detector OK");
133  m_LandInitialized = TRUE;
134  }
135  }
136 
137  /* kalman filter correction always for land detector */
138  m_Land.K = (m_StateCov * m_Land.C.Transpose()) * m_Land.S_I;
139  m_Land.dx = m_Land.K * m_Land.r;
140  m_StateVec = m_StateVec + m_Land.dx;
141  m_StateCov = m_StateCov - m_Land.K * m_Land.C * m_StateCov;
142 end_of_function:
143 
144  CFE_ES_PerfLogExit(PE_SENSOR_LAND_PERF_ID);
145 }
