68 int32 PE::distMeasure(math::Vector1F &y)
69 {
70  int32 Status = CFE_SUCCESS;
71  float d = m_DistanceSensor.CurrentDistance;
72  float eps = 0.05f; // 5 cm
73  float min_dist = m_DistanceSensor.MinDistance + eps;
74  float max_dist = m_DistanceSensor.MaxDistance - eps;
75 
76  // prevent driver from setting min dist below eps
77  if (min_dist < eps)
78  {
79  min_dist = eps;
80  }
81 
82  // check for bad data
83  if (d > max_dist || d < min_dist)
84  {
85  Status = -1;
86  goto distMeasure_Exit_Tag;
87  }
88 
89  // Check for horizontal speed
90 
91 
92  /* Measure */
93  y.Zero();
94  m_DistStats.update(d);
95  y[0] = (d + ConfigTblPtr->DIST_OFF_Z) * cosf(m_Euler[0]) * cosf(m_Euler[1]);
96  m_TimeLastDist = m_DistanceSensor.Timestamp;
97 
98 distMeasure_Exit_Tag:
99  return Status;
100 }

103 void PE::distCorrect()
104 {
105  CFE_ES_PerfLogEntry(PE_SENSOR_DIST_PERF_ID);
106  float cov = 0.0f;
107 
108  if (distMeasure(m_Dist.y) != CFE_SUCCESS)
109  {
110  goto end_of_function;
111  }
112 
113  /* subtract dist origin alt */
114  //m_Dist.y[0] -= m_DistAltOrigin;
115 
116  /* measured altitude, negative down dir */
117  m_Dist.C[Y_dist_z][X_z] = -1.0f;
118  m_Dist.C[Y_dist_z][X_tz] = 1.0f;
119 
120  cov = m_DistanceSensor.Covariance;
121  if (cov < 1.0e-3f)
122  {
123  m_Dist.R[0][0] = ConfigTblPtr->DIST_STDDEV * ConfigTblPtr->DIST_STDDEV;
124  }
125  else
126  {
127  m_Dist.R[0][0] = cov;
128  }
129 
130  /* residual */
131  /* ((1x10 * 10x10) * 10x1) + 1x1) */
132  m_Dist.S_I = m_Dist.C * m_StateCov * m_Dist.C.Transpose() + m_Dist.R;
133 
134  /* Take the inverse of a 1x1 matrix (reciprical of the single entry) */
135  m_Dist.S_I[0][0] = 1 / m_Dist.S_I[0][0];
136 
137  /* Vector1F - (1x10 * Vector10F) */
138  m_Dist.r = m_Dist.y - (m_Dist.C * m_StateVec);
139 
140  /* fault detection 1F * 1x1 * 1F */
141  m_Dist.beta = m_Dist.r[0] * m_Dist.S_I[0][0] * m_Dist.r[0];
142 
143  if (m_Dist.beta > DIST_BETA_MAX)
144  {
145  if (!m_DistFault)
146  {
147  if(Initialized())
148  {
149  (void) CFE_EVS_SendEvent(PE_DIST_FAULT_ERR_EID, CFE_EVS_ERROR,
150  "Dist fault, d %5.2f m r %5.2f m, beta %5.2f", m_DistanceSensor.CurrentDistance, m_Dist.r[0], m_Dist.beta);
151  }
152  m_DistFault = TRUE;
153  }
154  goto end_of_function;
155  }
156  else
157  {
158  if (m_DistFault)
159  {
160  m_DistFault = FALSE;
161  (void) CFE_EVS_SendEvent(PE_DIST_OK_INF_EID, CFE_EVS_INFORMATION,
162  "Dist OK r %5.2f m, beta %5.2f", m_Dist.r[0], m_Dist.beta);
163 
164  m_DistInitialized = TRUE;
165  }
166  }
167 
168  /* kalman filter correction */
169 
170  /* 10x10 * 10x1 * 1x1 */
171  m_Dist.K = (m_StateCov * m_Dist.C.Transpose()) * m_Dist.S_I;
172 
173  /* 10x1 * 1x1 */
174  m_Dist.temp = m_Dist.K * m_Dist.r;
175 
176  m_Dist.dx = m_Dist.temp.ToVector();
177 
178  /* 10F + 10F*/
179  m_StateVec = m_StateVec + m_Dist.dx;
180 
181  /* 10x10 - 10x1 * 1x10 * 10x10 */
182  m_StateCov = m_StateCov - m_Dist.K * m_Dist.C * m_StateCov;
183 end_of_function:
184 
185  CFE_ES_PerfLogExit(PE_SENSOR_DIST_PERF_ID);
186 }
