193  iStatus = CFE_SB_CreatePipe(&SchPipeId,

196  if (iStatus == CFE_SUCCESS)

198  iStatus = CFE_SB_SubscribeEx(PE_WAKEUP_MID, SchPipeId, CFE_SB_Default_Qos, PE_WAKEUP_MID_MAX_MSG_COUNT);

592 void PE::updateStateSpaceParams()
593 {
594  /* Input noise covariance matrix */
595  m_InputCov.Zero();
596  m_InputCov[U_ax][U_ax] = ConfigTblPtr->ACCEL_XY_STDDEV * ConfigTblPtr->ACCEL_XY_STDDEV;
597  m_InputCov[U_ay][U_ay] = ConfigTblPtr->ACCEL_XY_STDDEV * ConfigTblPtr->ACCEL_XY_STDDEV;
598  m_InputCov[U_az][U_az] = ConfigTblPtr->ACCEL_Z_STDDEV * ConfigTblPtr->ACCEL_Z_STDDEV;
599

601  m_NoiseCov.Zero();
602  float pn_p_sq = ConfigTblPtr->PN_P_NOISE_DENSITY * ConfigTblPtr->PN_P_NOISE_DENSITY;
603  float pn_v_sq = ConfigTblPtr->PN_V_NOISE_DENSITY * ConfigTblPtr->PN_V_NOISE_DENSITY;
604  m_NoiseCov[X_x][X_x] = pn_p_sq;
605  m_NoiseCov[X_y][X_y] = pn_p_sq;
606  m_NoiseCov[X_z][X_z] = pn_p_sq;
607  m_NoiseCov[X_vx][X_vx] = pn_v_sq;
608  m_NoiseCov[X_vy][X_vy] = pn_v_sq;
609  m_NoiseCov[X_vz][X_vz] = pn_v_sq;
610 
611  /* Technically the noise is in the body frame,
612  * but the components are all the same, so
613  * ignoring for now */
614  float pn_b_sq = ConfigTblPtr->PN_B_NOISE_DENSITY * ConfigTblPtr->PN_B_NOISE_DENSITY;
615  m_NoiseCov[X_bx][X_bx] = pn_b_sq;
616  m_NoiseCov[X_by][X_by] = pn_b_sq;
617  m_NoiseCov[X_bz][X_bz] = pn_b_sq;
618 
619  /* Terrain random walk noise ((m/s)/sqrt(hz)), scales with velocity */
620  float pn_t_noise_density = ConfigTblPtr->PN_T_NOISE_DENSITY +
621  (ConfigTblPtr->T_MAX_GRADE / 100.0f) * sqrtf((m_StateVec[X_vx] *
622  m_StateVec[X_vx]) + (m_StateVec[X_vy] * m_StateVec[X_vy]));
623  m_NoiseCov[X_tz][X_tz] = pn_t_noise_density * pn_t_noise_density;
624 }

763 osalbool PE::ProcessDataPipe()
764 {
765  int32 iStatus = CFE_SUCCESS;
766  CFE_SB_Msg_t* MsgPtr=NULL;
767  CFE_SB_MsgId_t MsgId;
768 
769  /* Process command messages until the pipe is empty */
770  while (1)
771  {
772  iStatus = CFE_SB_RcvMsg(&MsgPtr, DataPipeId, CFE_SB_POLL);
773  if(iStatus == CFE_SUCCESS)
774  {
775  MsgId = CFE_SB_GetMsgId(MsgPtr);
776  switch (MsgId)
777  {
778  case PX4_VEHICLE_GPS_POSITION_MID:
779  {
780  HkTlm.VehicleGpsPositionMsgCount++;
781  CFE_PSP_MemCpy(&m_VehicleGpsPositionMsg, MsgPtr, sizeof(m_VehicleGpsPositionMsg));
782 
783  /* Check if fusing distance sensor */
784  if(TRUE == m_GpsFuse)
785  {
786  if(m_GpsTimeout)
787  {
788  gpsInit();
789  }
790  else
791  {
792  gpsCorrect();
793  }
794  }
795  break;
796  }
797 
798  case PX4_VEHICLE_STATUS_MID:
799  {
800  HkTlm.VehicleStatusMsgCount++;
801  CFE_PSP_MemCpy(&m_VehicleStatusMsg, MsgPtr, sizeof(m_VehicleStatusMsg));
802  break;
803  }
804 
805  case PX4_VEHICLE_LAND_DETECTED_MID:
806  {
807  HkTlm.VehicleLandDetectedMsgCount++;
808  CFE_PSP_MemCpy(&m_VehicleLandDetectedMsg, MsgPtr, sizeof(m_VehicleLandDetectedMsg));
809 
810  /* Check if fusing land */
811  if(TRUE == m_LandFuse)
812  {
813  if(landed())
814  {
815  /* Throttle rate */
816  if((m_Timestamp - m_TimeLastLand) > 1.0e6f / LAND_RATE)
817  {
818  if(m_LandTimeout)
819  {
820  landInit();
821  }
822  else
823  {
824  landCorrect();
825  }
826  }
827  }
828  }
829  break;
830  }
831 
832  case PX4_ACTUATOR_ARMED_MID:
833  {
834  HkTlm.ActuatorArmedMsgCount++;
835  CFE_PSP_MemCpy(&m_ActuatorArmedMsg, MsgPtr, sizeof(m_ActuatorArmedMsg));
836  break;
837  }
838 
839  case PX4_VEHICLE_ATTITUDE_MID:
840  {
841  HkTlm.VehicleAttitudeMsgCount++;
842  CFE_PSP_MemCpy(&m_VehicleAttitudeMsg, MsgPtr, sizeof(m_VehicleAttitudeMsg));
843  break;
844  }
845 
846  case PX4_SENSOR_COMBINED_MID:
847  {
848  HkTlm.SensorCombinedMsgCount++;
849  /* Don't prevent memcpy - we fuse accel every cycle */
850  CFE_PSP_MemCpy(&m_SensorCombinedMsg, MsgPtr, sizeof(m_SensorCombinedMsg));
851 
852  /* Check if fusing baro */
853  if(TRUE == m_BaroFuse)
854  {
855  /* If baro is valid */
856  if(m_SensorCombinedMsg.BaroTimestampRelative != PX4_RELATIVE_TIMESTAMP_INVALID)
857  {
858  /* Get the baro timestamp from the sensor combined timestamp
859  * (which is the gyro timestamp) plus the baro relative timestamp.
860  * Baro relative is the difference between the gyro and baro
861  * when received by the sensors application. If baro is fresh.
862  */
863  if((m_SensorCombinedMsg.Timestamp +
864  m_SensorCombinedMsg.BaroTimestampRelative)
865  != m_TimeLastBaro)
866  {
867  if(m_BaroTimeout)
868  {
869  baroInit();
870  }
871  else
872  {
873  baroCorrect();
874  }
875  /* Save the last valid timestamp */
876  m_TimeLastBaro = m_SensorCombinedMsg.Timestamp + m_SensorCombinedMsg.BaroTimestampRelative;
877  }
878  }
879  }
880  break;
881  }
882 
883  case PX4_VEHICLE_ATTITUDE_SETPOINT_MID:
884  {
885  HkTlm.VehicleAttitudeSetpointMsgCount++;
886  CFE_PSP_MemCpy(&m_VehicleAttitudeSetpointMsg, MsgPtr, sizeof(m_VehicleAttitudeSetpointMsg));
887  break;
888  }
889 
890  case PX4_DISTANCE_SENSOR_MID:
891  {
892  HkTlm.DistanceSensorMsgCount++;
893  CFE_PSP_MemCpy(&m_DistanceSensor, MsgPtr, sizeof(m_DistanceSensor));
894 
895  /* Check that distance sensor is expected type for this platform */
896  if(m_DistanceSensor.Type == DIST_SENSOR_TYPE)
897  {
898  /* Check if fusing distance sensor */
899  if(TRUE == m_DistFuse)
900  {
901  /* Don't integrate while landed */
902  if(!landed())
903  {
904  /* Throttle rate */
905  if((m_Timestamp - m_TimeLastLand) > 1.0e6f / DIST_RATE)
906  {
907  if(m_DistTimeout)
908  {
909  distInit();
910  }
911  else
912  {
913  distCorrect();
914  }
915  }
916  }
917  }
918  }
919  break;
920  }
921 
922  case PX4_OPTICAL_FLOW_MID:
923  {
924  HkTlm.OpticalFlowMsgCount++;
925  CFE_PSP_MemCpy(&m_OpticalFlowMsg, MsgPtr, sizeof(m_OpticalFlowMsg));
926 
927  /* Check if fusing distance sensor */
928  if(TRUE == m_FlowFuse)
929  {
930  if(m_FlowTimeout)
931  {
932  flowInit();
933  }
934  else
935  {
936  flowCorrect();
937  }
938  }
939  break;
940  }
941 
942  default:
943  {
944  (void) CFE_EVS_SendEvent(PE_MSGID_ERR_EID, CFE_EVS_ERROR,
945  "Recvd invalid DATA msgId (0x%04X)", (unsigned short)MsgId);
946  break;
947  }
948  }
949  }
950  else if (iStatus == CFE_SB_NO_MESSAGE)
951  {
952  break;
953  }
954  else
955  {
956  (void) CFE_EVS_SendEvent(PE_RCVMSG_ERR_EID, CFE_EVS_ERROR,
957  "DATA pipe read error (0x%08lX)", iStatus);
958  break;
959  }
960  }
961

963 }

1782 void PE::Predict(float dt)
1783 {
1784  /* Get acceleration */
1785  m_Predict.q[0] = m_VehicleAttitudeMsg.Q[0];
1786  m_Predict.q[1] = m_VehicleAttitudeMsg.Q[1];
1787  m_Predict.q[2] = m_VehicleAttitudeMsg.Q[2];
1788  m_Predict.q[3] = m_VehicleAttitudeMsg.Q[3];
1789 
1790  m_RotationMat = math::Dcm(m_Predict.q);
1791  m_Euler = math::Euler(m_RotationMat);
1792  m_Predict.a[0] = m_SensorCombinedMsg.Acc[0];
1793  m_Predict.a[1] = m_SensorCombinedMsg.Acc[1];
1794  m_Predict.a[2] = m_SensorCombinedMsg.Acc[2];
1795 
1796  /* Note: bias is removed in dynamics function */
1797  m_InputVec = m_RotationMat * m_Predict.a;
1798  m_InputVec[U_az] += PE_ONE_G; // add g
1799 
1800  /* Update state space based on new states */
1801  updateStateSpace();
1802 
1803  /* Continuous time Kalman filter prediction. Integrate runge kutta 4th order.
1804  * https://en.wikipedia.org/wiki/Runge%E2%80%93Kutta_methods */
1805  m_Predict.k1 = dynamics(m_StateVec, m_InputVec);
1806  m_Predict.k2 = dynamics(m_StateVec + m_Predict.k1 * dt / 2, m_InputVec);
1807  m_Predict.k3 = dynamics(m_StateVec + m_Predict.k2 * dt / 2, m_InputVec);
1808  m_Predict.k4 = dynamics(m_StateVec + m_Predict.k3 * dt, m_InputVec);
1809  m_Predict.dx = (m_Predict.k1 + m_Predict.k2 * 2 + m_Predict.k3 * 2 +
1810  m_Predict.k4) * (dt / 6);
1811  /* Don't integrate position if no valid xy data */
1812  if (!m_XyEstValid)
1813  {
1814  m_Predict.dx[X_x] = 0;
1815  m_Predict.dx[X_vx] = 0;
1816  m_Predict.dx[X_y] = 0;
1817  m_Predict.dx[X_vy] = 0;
1818  }
1819 
1820  /* Don't integrate z if no valid z data */
1821  if (!m_ZEstValid)
1822  {
1823  m_Predict.dx[X_z] = 0;
1824  }
1825 
1826  /* Don't integrate tz if no valid tz data */
1827  if (!m_TzEstValid)
1828  {
1829  m_Predict.dx[X_tz] = 0;
1830  }
1831 
1832  /* Saturate bias */
1833  m_Predict.bx = m_Predict.dx[X_bx] + m_StateVec[X_bx];
1834  m_Predict.by = m_Predict.dx[X_by] + m_StateVec[X_by];
1835  m_Predict.bz = m_Predict.dx[X_bz] + m_StateVec[X_bz];
1836 
1837  if (::abs(m_Predict.bx) > BIAS_MAX)
1838  {
1839  m_Predict.bx = BIAS_MAX * m_Predict.bx / ::abs(m_Predict.bx);
1840  m_Predict.dx[X_bx] = m_Predict.bx - m_StateVec[X_bx];
1841  }
1842 
1843  if (::abs(m_Predict.by) > BIAS_MAX)
1844  {
1845  m_Predict.by = BIAS_MAX * m_Predict.by / ::abs(m_Predict.by);
1846  m_Predict.dx[X_by] = m_Predict.by - m_StateVec[X_by];
1847  }
1848 
1849  if (::abs(m_Predict.bz) > BIAS_MAX)
1850  {
1851  m_Predict.bz = BIAS_MAX * m_Predict.bz / ::abs(m_Predict.bz);
1852  m_Predict.dx[X_bz] = m_Predict.bz - m_StateVec[X_bz];
1853  }
1854 
1855  /* Propagate */
1856  m_StateVec += m_Predict.dx;
1857  m_Predict.dP = (m_DynamicsMat * m_StateCov + m_StateCov *
1858  m_DynamicsMat.Transpose() + m_InputMat * m_InputCov *
1859  m_InputMat.Transpose() + m_NoiseCov) * dt;
1860 
1861  /* Covariance propagation logic */
1862  for (int i = 0; i < n_x; i++)
1863  {
1864  if (m_StateCov[i][i] > P_MAX)
1865  {
1866  /* If diagonal element greater than max stop propagating */
1867  m_Predict.dP[i][i] = 0;
1868 
1869  for (int j = 0; j < n_x; j++)
1870  {
1871  m_Predict.dP[i][j] = 0;
1872  m_Predict.dP[j][i] = 0;
1873  }
1874  }
1875  }
1876 
1877  /* Update state */
1878  m_StateCov = m_StateCov + m_Predict.dP;
1879  m_XLowPass.Update(m_StateVec, dt, LOW_PASS_CUTOFF);
1880  m_AglLowPass.Update(m_StateVec[X_tz] - m_StateVec[X_z], dt, LOW_PASS_CUTOFF);
1881 }
