36 void PE::gpsInit()
37 {
38  /* check for good gps signal */
39  uint8 nSat = m_VehicleGpsPositionMsg.SatellitesUsed;
40  float eph = m_VehicleGpsPositionMsg.EpH;
41  float epv = m_VehicleGpsPositionMsg.EpV;
42  uint8 fix_type = m_VehicleGpsPositionMsg.FixType;
43  math::Vector6F y;
44 
45  if (nSat < 6 ||
46  eph > ConfigTblPtr->GPS_EPH_MAX ||
47  epv > ConfigTblPtr->GPS_EPV_MAX ||
48  fix_type < 3)
49  {
50  m_GpsStats.reset();
51  goto gpsInit_Exit_Tag;
52  }
53 
54  /* measure */
55  if (gpsMeasure(y) != CFE_SUCCESS)
56  {
57  m_GpsStats.reset();
58  goto gpsInit_Exit_Tag;
59  }
60 
61  /* if finished */
62  if (m_GpsStats.getCount() > REQ_GPS_INIT_COUNT)
63  {
64  /* get mean gps values */
65  double gpsLat = m_GpsStats.getMean()[0];
66  double gpsLon = m_GpsStats.getMean()[1];
67  float gpsAlt = m_GpsStats.getMean()[2];
68 
69  m_GpsTimeout = FALSE;
70  m_GpsStats.reset();
71 
72  if (!m_ReceivedGps)
73  {
74  /* this is the first time we have received gps */
75  m_ReceivedGps = TRUE;
76 
77  /* note we subtract X_z which is in down directon so it is
78  * an addition
79  **/
80  m_GpsAltOrigin = gpsAlt + m_StateVec[X_z];
81 
82  /* find lat, lon of current origin by subtracting x and y
83  * if not using vision position since vision will
84  * have it's own origin, not necessarily where vehicle starts
85  **/
86  if (!m_MapRef.init_done)
87  {
88  double gpsLatOrigin = 0;
89  double gpsLonOrigin = 0;
90  /* reproject at current coordinates */
91  map_projection_init(&m_MapRef, gpsLat, gpsLon, m_Timestamp);
92  /* find origin */
93  map_projection_reproject(&m_MapRef, -m_StateVec[X_x], -m_StateVec[X_y], &gpsLatOrigin, &gpsLonOrigin);
94  /* reinit origin */
95  map_projection_init(&m_MapRef, gpsLatOrigin, gpsLonOrigin, m_Timestamp);
96 
97  /* always override alt origin on first GPS to fix
98  * possible baro offset in global altitude at init
99  **/
100  m_AltOrigin = m_GpsAltOrigin;
101  m_AltOriginInitialized = TRUE;
102 
103  (void) CFE_EVS_SendEvent(PE_GPS_OK_INF_EID, CFE_EVS_INFORMATION,
104  "GPS initialized origin. Lat: %6.2f Lon: %6.2f Alt: %5.1f m",
105  gpsLatOrigin, gpsLonOrigin, double(m_GpsAltOrigin));
106  }
107 
108  (void) CFE_EVS_SendEvent(PE_GPS_OK_INF_EID, CFE_EVS_INFORMATION,
109  "GPS initialized. Lat: %6.2f Lon: %6.2f Alt: %5.1f m",
110  gpsLat, gpsLon, double(gpsAlt));
111  }
112  }
113 
114 gpsInit_Exit_Tag:
115  return;
116 }

119 int PE::gpsMeasure(math::Vector6F &y)
120 {
121  /* gps measurement */
122  y.Zero();
123  y[0] = m_VehicleGpsPositionMsg.Lat * 1e-7;
124  y[1] = m_VehicleGpsPositionMsg.Lon * 1e-7;
125  y[2] = m_VehicleGpsPositionMsg.Alt * 1e-3;
126  y[3] = m_VehicleGpsPositionMsg.Vel_n_m_s;
127  y[4] = m_VehicleGpsPositionMsg.Vel_e_m_s;
128  y[5] = m_VehicleGpsPositionMsg.Vel_d_m_s;
129 
130  /* increament sums for mean */
131  m_GpsStats.update(y);
132  m_TimeLastGps = m_Timestamp;
133  return CFE_SUCCESS;
134 }

137 void PE::gpsCorrect()
138 {
139  CFE_ES_PerfLogEntry(PE_SENSOR_GPS_PERF_ID);
140  int i = 0;
141 
142  /* measure */
143  if (gpsMeasure(m_GPS.y_global) != CFE_SUCCESS)
144  {
145  goto end_of_function;
146  }
147 
148  /* gps measurement in local frame */
149  m_GPS.lat = m_GPS.y_global[0];
150  m_GPS.lon = m_GPS.y_global[1];
151  m_GPS.alt = m_GPS.y_global[2];
152  m_GPS.px = 0;
153  m_GPS.py = 0;
154 
155  m_GPS.pz = -(m_GPS.alt - m_GpsAltOrigin);
156  map_projection_project(&m_MapRef, m_GPS.lat, m_GPS.lon, &m_GPS.px, &m_GPS.py);
157 
158  m_GPS.y[0] = m_GPS.px;
159  m_GPS.y[1] = m_GPS.py;
160  m_GPS.y[2] = m_GPS.pz;
161  m_GPS.y[3] = m_GPS.y_global[3];
162  m_GPS.y[4] = m_GPS.y_global[4];
163  m_GPS.y[5] = m_GPS.y_global[5];
164 
165  /* gps measurement matrix, measures position and velocity */
166  m_GPS.C[Y_gps_x][X_x] = 1;
167  m_GPS.C[Y_gps_y][X_y] = 1;
168  m_GPS.C[Y_gps_z][X_z] = 1;
169  m_GPS.C[Y_gps_vx][X_vx] = 1;
170  m_GPS.C[Y_gps_vy][X_vy] = 1;
171  m_GPS.C[Y_gps_vz][X_vz] = 1;
172 
173  /* default to parameter, use gps cov if provided */
174  m_GPS.var_xy = ConfigTblPtr->GPS_XY_STDDEV * ConfigTblPtr->GPS_XY_STDDEV;
175  m_GPS.var_z = ConfigTblPtr->GPS_Z_STDDEV * ConfigTblPtr->GPS_Z_STDDEV;
176  m_GPS.var_vxy = ConfigTblPtr->GPS_VXY_STDDEV * ConfigTblPtr->GPS_VXY_STDDEV;
177  m_GPS.var_vz = ConfigTblPtr->GPS_VZ_STDDEV * ConfigTblPtr->GPS_VZ_STDDEV;
178 
179  /* if field is not below minimum, set it to the value provided */
180  if(m_VehicleGpsPositionMsg.EpH > ConfigTblPtr->GPS_XY_STDDEV)
181  {
182  m_GPS.var_xy = m_VehicleGpsPositionMsg.EpH * m_VehicleGpsPositionMsg.EpH;
183  }
184  if (m_VehicleGpsPositionMsg.EpV > ConfigTblPtr->GPS_Z_STDDEV)
185  {
186  m_GPS.var_z = m_VehicleGpsPositionMsg.EpV * m_VehicleGpsPositionMsg.EpV;
187  }
188 
189  m_GPS.gps_s_stddev = m_VehicleGpsPositionMsg.SVariance;
190  if(m_GPS.gps_s_stddev > ConfigTblPtr->GPS_VXY_STDDEV)
191  {
192  m_GPS.var_vxy = m_GPS.gps_s_stddev * m_GPS.gps_s_stddev;
193  }
194 
195  if(m_GPS.gps_s_stddev > ConfigTblPtr->GPS_VZ_STDDEV)
196  {
197  m_GPS.var_vz = m_GPS.gps_s_stddev * m_GPS.gps_s_stddev;
198  }
199 
200  m_GPS.R[0][0] = m_GPS.var_xy;
201  m_GPS.R[1][1] = m_GPS.var_xy;
202  m_GPS.R[2][2] = m_GPS.var_z;
203  m_GPS.R[3][3] = m_GPS.var_vxy;
204  m_GPS.R[4][4] = m_GPS.var_vxy;
205  m_GPS.R[5][5] = m_GPS.var_vz;
206 
207  /* get delayed x */
208  m_GPS.i_hist = 0;
209 
210  if(getDelayPeriods(ConfigTblPtr->GPS_DELAY, &m_GPS.i_hist) < 0)
211  {
212  goto end_of_function;
213  }
214 
215  m_GPS.x0 = m_XDelay.Get(m_GPS.i_hist);
216 
217  /* residual */
218  m_GPS.r = m_GPS.y - m_GPS.C * m_GPS.x0;
219 
220  m_GPS.S_I = m_GPS.C * m_StateCov * m_GPS.C.Transpose() + m_GPS.R;
221  m_GPS.S_I = m_GPS.S_I.Inversed();
222 
223  /* fault detection */
224  /* 6x6 * 6x1 */
225  m_GPS.rTranspose[0][0] = m_GPS.r[0];
226  m_GPS.rTranspose[0][1] = m_GPS.r[1];
227  m_GPS.rTranspose[0][2] = m_GPS.r[2];
228  m_GPS.rTranspose[0][3] = m_GPS.r[3];
229  m_GPS.rTranspose[0][4] = m_GPS.r[4];
230  m_GPS.rTranspose[0][5] = m_GPS.r[5];
231 
232  m_GPS.beta = (m_GPS.rTranspose * (m_GPS.S_I * m_GPS.r))[0][0];
233  /* artifically increase beta threshhold to prevent fault during
234  * landing
235  **/
236  m_GPS.beta_thresh = 1e2f;
237 
238  if (m_GPS.beta / BETA_TABLE[n_y_gps] > m_GPS.beta_thresh)
239  {
240  if (!m_GpsFault)
241  {
242  if(Initialized())
243  {
244  (void) CFE_EVS_SendEvent(PE_GPS_FAULT_ERR_EID, CFE_EVS_ERROR,
245  "GPS fault, %3g %3g %3g %3g %3g %3g",
246  double(m_GPS.r[0]*m_GPS.r[0] / m_GPS.S_I[0][0]),
247  double(m_GPS.r[1]*m_GPS.r[1] / m_GPS.S_I[1][1]),
248  double(m_GPS.r[2]*m_GPS.r[2] / m_GPS.S_I[2][2]),
249  double(m_GPS.r[3]*m_GPS.r[3] / m_GPS.S_I[3][3]),
250  double(m_GPS.r[4]*m_GPS.r[4] / m_GPS.S_I[4][4]),
251  double(m_GPS.r[5]*m_GPS.r[5] / m_GPS.S_I[5][5]));
252  }
253  m_GpsFault = TRUE;
254  }
255  }
256  else
257  {
258  if (m_GpsFault)
259  {
260  m_GpsFault = FALSE;
261  (void) CFE_EVS_SendEvent(PE_GPS_OK_INF_EID, CFE_EVS_INFORMATION,
262  "GPS OK");
263 
264  m_GpsInitialized = TRUE;
265  }
266  }
267 
268  /* kalman filter correction always for GPS */
269 
270  /* 10x10 * 6x10' (10x6) * 6x6 */
271  m_GPS.K = (m_StateCov * m_GPS.C.Transpose()) * m_GPS.S_I;
272  m_GPS.dx = m_GPS.K * m_GPS.r;
273 
274  /* 10F * 10F */
275  m_StateVec = m_StateVec + m_GPS.dx;
276 
277  /* 10x10 - (10x6 * 6x10 * 10x10)*/
278  m_StateCov = m_StateCov - m_GPS.K * m_GPS.C * m_StateCov;
279 end_of_function:
280 
281  CFE_ES_PerfLogExit(PE_SENSOR_GPS_PERF_ID);
282 }
