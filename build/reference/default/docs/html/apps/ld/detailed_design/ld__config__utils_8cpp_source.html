48 /* */
49 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
50 int32 LD::InitConfigTbl()
51 {
52  int32 iStatus = 0;
53 
54  /* Register Config table */
55  iStatus = CFE_TBL_Register(&ConfigTblHdl,
56  LD_CONFIG_TABLENAME,
57  (sizeof(LD_ConfigTbl_t)),
58  CFE_TBL_OPT_DEFAULT,
59  LD::ValidateConfigTbl);
60  if (iStatus != CFE_SUCCESS)
61  {
62  /* Note, a critical table could return another nominal code. If this table is
63  * made critical this logic would have to change. */
64  (void) CFE_EVS_SendEvent(LD_CFGTBL_REG_ERR_EID, CFE_EVS_ERROR,
65  "Failed to register config table (0x%08lX)",
66  iStatus);
67  goto LD_InitConfigTbl_Exit_Tag;
68  }
69 
70  /* Load Config table file */
71  iStatus = CFE_TBL_Load(ConfigTblHdl,
72  CFE_TBL_SRC_FILE,
73  LD_CONFIG_TABLE_FILENAME);
74  if (iStatus != CFE_SUCCESS)
75  {
76  /* Note, CFE_SUCCESS is for a successful full table load. If a partial table
77  load is desired then this logic would have to change. */
78  (void) CFE_EVS_SendEvent(LD_CFGTBL_LOAD_ERR_EID, CFE_EVS_ERROR,
79  "Failed to load Config Table (0x%08lX)",
80  iStatus);
81  goto LD_InitConfigTbl_Exit_Tag;
82  }
83 
84  iStatus = AcquireConfigPointers();
85 
86 LD_InitConfigTbl_Exit_Tag:
87  return (iStatus);
88 }

187 /* Acquire Config Pointers */
188 /* */
189 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
190 int32 LD::AcquireConfigPointers(void)
191 {
192  int32 iStatus = CFE_SUCCESS;
193 
194  /*
195  ** Release the table
196  */
197  (void) CFE_TBL_ReleaseAddress(ConfigTblHdl);
198 
199  /*
200  ** Manage the table
201  */
202  iStatus = CFE_TBL_Manage(ConfigTblHdl);
203  if ((iStatus != CFE_SUCCESS) && (iStatus != CFE_TBL_INFO_UPDATED))
204  {
205  (void) CFE_EVS_SendEvent(LD_CFGTBL_MANAGE_ERR_EID, CFE_EVS_ERROR,
206  "Failed to manage LD Config table (0x%08lX)",
207  iStatus);
208  goto LD_AcquireConfigPointers_Exit_Tag;
209  }
210 
211  /*
212  ** Get a pointer to the table
213  */
214  iStatus = CFE_TBL_GetAddress((void**)&ConfigTblPtr, ConfigTblHdl);
215  if (iStatus == CFE_TBL_INFO_UPDATED)
216  {
217  iStatus = CFE_SUCCESS;
218  }
219  else if(iStatus != CFE_SUCCESS)
220  {
221  ConfigTblPtr = 0;
222  (void) CFE_EVS_SendEvent(LD_CFGTBL_GETADDR_ERR_EID, CFE_EVS_ERROR,
223  "Failed to get Config table's address (0x%08lX)",
224  iStatus);
225  }
226 
227 LD_AcquireConfigPointers_Exit_Tag:
228  return (iStatus);
229 }
