272 int32 SIM::RcvSchPipeMsg(int32 iBlocking)
273 {
274  int32 iStatus=CFE_SUCCESS;
275  CFE_SB_Msg_t* MsgPtr=NULL;
276  CFE_SB_MsgId_t MsgId;
277 
278  /* Stop Performance Log entry */
279  CFE_ES_PerfLogExit(SIM_MAIN_TASK_PERF_ID);
280 
281  /* Wait for WakeUp messages from scheduler */
282  iStatus = CFE_SB_RcvMsg(&MsgPtr, SchPipeId, iBlocking);
283 
284  /* Start Performance Log entry */
285  CFE_ES_PerfLogEntry(SIM_MAIN_TASK_PERF_ID);
286 
287  if (iStatus == CFE_SUCCESS)
288  {
289  MsgId = CFE_SB_GetMsgId(MsgPtr);
290  switch (MsgId)
291  {
292  case SIM_WAKEUP_MID:
293  /* TODO: Do something here. */
294  break;
295 
296  case SIM_SEND_HK_MID:
297  ProcessCmdPipe();
298  ReportHousekeeping();
299  break;
300 
301  default:
302  (void) CFE_EVS_SendEvent(SIM_MSGID_ERR_EID, CFE_EVS_ERROR,
303  "Recvd invalid SCH msgId (0x%04X)", MsgId);
304  }
305  }
306  else if (iStatus == CFE_SB_NO_MESSAGE)
307  {
308  /* TODO: If there's no incoming message, you can do something here, or
309  * nothing. Note, this section is dead code only if the iBlocking arg
310  * is CFE_SB_PEND_FOREVER. */
311  iStatus = CFE_SUCCESS;
312  }
313  else if (iStatus == CFE_SB_TIME_OUT)
314  {
315  /* TODO: If there's no incoming message within a specified time (via the
316  * iBlocking arg, you can do something here, or nothing.
317  * Note, this section is dead code only if the iBlocking arg
318  * is CFE_SB_PEND_FOREVER. */
319  iStatus = CFE_SUCCESS;
320  }
321  else
322  {
323  (void) CFE_EVS_SendEvent(SIM_RCVMSG_ERR_EID, CFE_EVS_ERROR,
324  "SCH pipe read error (0x%08lX).", iStatus);
325  }
326 
327  return iStatus;
328 }

337 void SIM::ProcessCmdPipe()
338 {
339  int32 iStatus = CFE_SUCCESS;
340  CFE_SB_Msg_t* CmdMsgPtr=NULL;
341  CFE_SB_MsgId_t CmdMsgId;
342 
343  /* Process command messages until the pipe is empty */
344  while (1)
345  {
346  iStatus = CFE_SB_RcvMsg(&CmdMsgPtr, CmdPipeId, CFE_SB_POLL);
347  if(iStatus == CFE_SUCCESS)
348  {
349  CmdMsgId = CFE_SB_GetMsgId(CmdMsgPtr);
350  switch (CmdMsgId)
351  {
352  case SIM_CMD_MID:
353  ProcessAppCmds(CmdMsgPtr);
354  break;
355 
356  default:
357  /* Bump the command error counter for an unknown command.
358  * (This should only occur if it was subscribed to with this
359  * pipe, but not handled in this switch-case.) */
360  HkTlm.usCmdErrCnt++;
361  (void) CFE_EVS_SendEvent(SIM_MSGID_ERR_EID, CFE_EVS_ERROR,
362  "Recvd invalid CMD msgId (0x%04X)", (unsigned short)CmdMsgId);
363  break;
364  }
365  }
366  else if (iStatus == CFE_SB_NO_MESSAGE)
367  {
368  break;
369  }
370  else
371  {
372  (void) CFE_EVS_SendEvent(SIM_RCVMSG_ERR_EID, CFE_EVS_ERROR,
373  "CMD pipe read error (0x%08lX)", iStatus);
374  break;
375  }
376  }
377 }

415  }

488 void SIM::AppMain()
489 {
490  /* Register the application with Executive Services */
491  uiRunStatus = CFE_ES_APP_RUN;
492 
493  int32 iStatus = CFE_ES_RegisterApp();
494  if (iStatus != CFE_SUCCESS)
495  {
496  (void) CFE_ES_WriteToSysLog("SIM - Failed to register the app (0x%08lX)\n", iStatus);
497  }
498 
499  /* Start Performance Log entry */
500  CFE_ES_PerfLogEntry(SIM_MAIN_TASK_PERF_ID);
501 
502  /* Perform application initializations */
503  if (iStatus == CFE_SUCCESS)
504  {
505  iStatus = InitApp();
506  }
507 
508  if (iStatus == CFE_SUCCESS)
509  {
510  /* Do not perform performance monitoring on startup sync */
511  CFE_ES_PerfLogExit(SIM_MAIN_TASK_PERF_ID);
512  CFE_ES_WaitForStartupSync(SIM_STARTUP_TIMEOUT_MSEC);
513  CFE_ES_PerfLogEntry(SIM_MAIN_TASK_PERF_ID);
514  }
515  else
516  {
517  uiRunStatus = CFE_ES_APP_ERROR;
518  }
519 
520  /* Application main loop */
521  while (CFE_ES_RunLoop(&uiRunStatus) == TRUE)
522  {
523  RcvSchPipeMsg(SIM_SCH_PIPE_PEND_TIME);
524 
525  iStatus = AcquireConfigPointers();
526  if(iStatus != CFE_SUCCESS)
527  {
528  /* We apparently tried to load a new table but failed. Terminate the application. */
529  uiRunStatus = CFE_ES_APP_ERROR;
530  }
531  }
532 
533  /* Stop Performance Log entry */
534  CFE_ES_PerfLogExit(SIM_MAIN_TASK_PERF_ID);
535 
536  /* Exit the application */
537  CFE_ES_ExitApp(uiRunStatus);
538 }
