114 /* Operates from the current location of the file poiner to EOF */
115 /* */
116 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
117 int32 CFS_ComputeCRCFromFile(uint32 FileHandle,
118  uint32 *CrcPtr,
119  uint32 TypeCRC)
120 {
121  int32 ByteCntr;
122  int32 OS_Status = OS_SUCCESS;
123  uint32 TempCrc = 0;
124 static uint8 DataArray[FILE_CRC_BUFFER_SIZE];
125 
126  do
127  {
128  /*
129  ** Read in some data
130  */
131  ByteCntr = OS_read(FileHandle, DataArray, sizeof(DataArray));
132 
133  /*
134  ** If we didn't hit end of file on the last read...
135  */
136  if (ByteCntr > 0)
137  {
138  /*
139  ** Calculate the CRC based upon the previous CRC calculation
140  */
141  TempCrc = CFE_ES_CalculateCRC(DataArray, ByteCntr, TempCrc, TypeCRC);
142  }
143 
144  }
145  while(ByteCntr > 0);
146 
147  /*
148  ** Check if we broke out of the loop because of an error return
149  ** from the OS_read call
150  */
151  if (ByteCntr < 0)
152  {
153  OS_Status = ByteCntr;
154  }
155  else
156  {
157  *CrcPtr = TempCrc;
158  }
159 
160  return(OS_Status);
161 
162 } /* End CFS_ComputeCRCFromFile */

327 /* */
328 /* Apply common filter algorithm to Software Bus packet */
329 /* */
330 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
331 boolean CFS_IsPacketFiltered(CFE_SB_MsgPtr_t MessagePtr,
332  uint16 FilterType,
333  uint16 Algorithm_N,
334  uint16 Algorithm_X,
335  uint16 Algorithm_O)
336 {
337  /*
338  ** Algorithm_N = the filter will pass this many packets
339  ** Algorithm_X = out of every group of this many packets
340  ** Algorithm_O = starting at this offset within the group
341  */
342  boolean PacketIsFiltered = FALSE;
343  CFE_TIME_SysTime_t PacketTime;
344  uint16 PacketValue;
345  uint16 Seconds;
346  uint16 Subsecs;
347 
348  /*
349  ** Verify input values (all errors = packet is filtered)...
350  */
351  if (Algorithm_X == 0)
352  {
353  /*
354  ** Group size of zero will result in divide by zero...
355  */
356  PacketIsFiltered = TRUE;
357  }
358  else if (Algorithm_N == 0)
359  {
360  /*
361  ** Pass count of zero will result in zero packets...
362  */
363  PacketIsFiltered = TRUE;
364  }
365  else if (Algorithm_N > Algorithm_X)
366  {
367  /*
368  ** Pass count cannot exceed group size...
369  */
370  PacketIsFiltered = TRUE;
371  }
372  else if (Algorithm_O >= Algorithm_X)
373  {
374  /*
375  ** Group offset must be less than group size...
376  */
377  PacketIsFiltered = TRUE;
378  }
379  else if ((FilterType != CFS_PKT_TIME_BASED_FILTER_TYPE) &&
380  (FilterType != CFS_PKT_SEQUENCE_BASED_FILTER_TYPE))
381  {
382  /*
383  ** Invalid - unknown filter type...
384  */
385  PacketIsFiltered = TRUE;
386  }
387  else
388  {
389  if (FilterType == CFS_PKT_SEQUENCE_BASED_FILTER_TYPE)
390  {
391  /*
392  ** Create packet filter value from packet sequence count...
393  */
394  PacketValue = CCSDS_RD_SEQ(MessagePtr->Hdr);
395  }
396  else
397  {
398  /*
399  ** Create packet filter value from packet timestamp...
400  */
401  PacketTime = CFE_SB_GetMsgTime(MessagePtr);
402 
403  /*
404  ** Get the least significant 11 bits of timestamp seconds...
405  */
406  Seconds = (uint16) PacketTime.Seconds;
407  Seconds = Seconds & CFS_11_LSB_SECONDS_MASK;
408 
409  /*
410  ** Get the most significant 4 bits of timestamp subsecs...
411  */
412  Subsecs = (uint16) (PacketTime.Subseconds >> CFS_16_MSB_SUBSECS_SHIFT);
413  Subsecs = Subsecs & CFS_4_MSB_SUBSECS_MASK;
414 
415  /*
416  ** Shift seconds and subsecs to allow merge...
417  */
418  Seconds = Seconds << CFS_11_LSB_SECONDS_SHIFT;
419  Subsecs = Subsecs >> CFS_4_MSB_SUBSECS_SHIFT;
420 
421  /*
422  ** Merge seconds and subsecs to create packet filter value...
423  */
424  PacketValue = Seconds | Subsecs;
425  }
426 
427  /*
428  ** Apply the filter algorithm (common for both filter types)...
429  */
430  if (PacketValue < Algorithm_O)
431  {
432  /*
433  ** Value is less than offset of passed range...
434  */
435  PacketIsFiltered = TRUE;
436  }
437  else if (((PacketValue - Algorithm_O) % Algorithm_X) < Algorithm_N)
438  {
439  /*
440  ** This packet was passed by the filter algorithm...
441  */
442  PacketIsFiltered = FALSE;
443  }
444  else
445  {
446  /*
447  ** This packet was filtered by the filter algorithm...
448  */
449  PacketIsFiltered = TRUE;
450  }
451  }
452 
453  return(PacketIsFiltered);
454 
455 } /* End of CFS_IsPacketFiltered() */

464 boolean CFS_VerifyString(char *Buffer, int32 BufferSize,
465  boolean StringIsRequired, boolean NeedCharTest)
466 {
467  boolean FoundTerminator;
468  boolean CharsAreGood;
469  boolean Result = FALSE;
470  int32 i;
471 
472  if (Buffer[0] == CFS_STRING_TERMINATOR)
473  {
474  /*
475  ** Buffer is empty...
476  */
477  if (StringIsRequired)
478  {
479  Result = FALSE;
480  }
481  else
482  {
483  Result = TRUE;
484  }
485  }
486  else
487  {
488  /*
489  ** Buffer is not empty...
490  */
491  FoundTerminator = FALSE;
492  CharsAreGood = FALSE;
493 
494  for (i = 0; i < BufferSize; i++)
495  {
496  /*
497  ** Find string terminator...
498  */
499  if (Buffer[i] == CFS_STRING_TERMINATOR)
500  {
501  FoundTerminator = TRUE;
502 
503  if (NeedCharTest)
504  {
505  /*
506  ** Verify that chars are legal for filenames...
507  */
508  CharsAreGood = CFS_IsValidFilename(Buffer, i);
509  }
510  else
511  {
512  /*
513  ** All chars are OK for descriptive text...
514  */
515  CharsAreGood = TRUE;
516  }
517  break;
518  }
519  }
520 
521  /*
522  ** Verify that string has valid chars and a terminator...
523  */
524  Result = ((FoundTerminator) && (CharsAreGood));
525  }
526 
527  return(Result);
528 
529 } /* End of CFS_VerifyString() */
