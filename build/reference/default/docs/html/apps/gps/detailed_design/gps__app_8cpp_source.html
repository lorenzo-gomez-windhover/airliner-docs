83 /* Initialize event tables. */
84 /* */
85 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
86 int32 GPS::InitEvent()
87 {
88  int32 iStatus = CFE_SUCCESS;
89  int32 ind = 0;
90  int32 customEventCount = 0;
91 
92  CFE_EVS_BinFilter_t EventTbl[CFE_EVS_MAX_EVENT_FILTERS];
93 
94  /* Initialize the event filter table.
95  * Note: 0 is the CFE_EVS_NO_FILTER mask and event 0 is reserved (not used) */
96  memset(EventTbl, 0x00, sizeof(EventTbl));
97 
98  /* TODO: Choose the events you want to filter. CFE_EVS_MAX_EVENT_FILTERS
99  * limits the number of filters per app. An explicit CFE_EVS_NO_FILTER
100  * (the default) has been provided as an example. */
101  EventTbl[ ind].EventID = GPS_RESERVED_EID;
102  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
103  EventTbl[ ind].EventID = GPS_READ_ERR_EID;
104  EventTbl[ind++].Mask = CFE_EVS_FIRST_16_STOP;
105 
106  /* Add custom events to the filter table */
107  customEventCount = GPS_Custom_Init_EventFilters(ind, EventTbl);
108 
109  if(-1 == customEventCount)
110  {
111  iStatus = CFE_EVS_APP_FILTER_OVERLOAD;
112  (void) CFE_ES_WriteToSysLog("Failed to init custom event filters (0x%08X)\n", (unsigned int)iStatus);
113  goto end_of_function;
114  }
115 
116  /* Register the table with CFE */
117  iStatus = CFE_EVS_Register(EventTbl, (ind + customEventCount), CFE_EVS_BINARY_FILTER);
118  if (iStatus != CFE_SUCCESS)
119  {
120  (void) CFE_ES_WriteToSysLog("MPU9250 - Failed to register with EVS (0x%08lX)\n", iStatus);
121  }
122 
123 end_of_function:
124 
125  return (iStatus);
126 }

240  int8 hasEvents = 0;
241 
242  iStatus = InitEvent();
243  if (iStatus != CFE_SUCCESS)
244  {
245  (void) CFE_ES_WriteToSysLog("GPS - Failed to init events (0x%08lX)\n", iStatus);
246  goto GPS_InitApp_Exit_Tag;
247  }
248  else
249  {
250  hasEvents = 1;
251  }
252 
253  iStatus = InitPipe();
254  if (iStatus != CFE_SUCCESS)
255  {
256  goto GPS_InitApp_Exit_Tag;
257  }
258 
259  InitData();
260 
261  iStatus = InitConfigTbl();
262  if (iStatus != CFE_SUCCESS)
263  {
264  goto GPS_InitApp_Exit_Tag;
265  }
266 
267  returnBool = GPS_Custom_Init();
268  if (FALSE == returnBool)
269  {
270  iStatus = -1;
271  goto GPS_InitApp_Exit_Tag;
272  }
273 
274  HkTlm.State = GPS_INITIALIZED;
275 
276  /* Register the cleanup callback */
277  iStatus = OS_TaskInstallDeleteHandler(&GPS_CleanupCallback);
278  if (iStatus != CFE_SUCCESS)
279  {
280  CFE_EVS_SendEvent(GPS_INIT_ERR_EID, CFE_EVS_ERROR,
281  "Failed to init register cleanup callback (0x%08X)",
282  (unsigned int)iStatus);
283  goto GPS_InitApp_Exit_Tag;
284  }
285 
286 GPS_InitApp_Exit_Tag:
287  if (iStatus == CFE_SUCCESS)
288  {
289  (void) CFE_EVS_SendEvent(GPS_INIT_INF_EID, CFE_EVS_INFORMATION,
290  "Initialized. Version %d.%d.%d.%d",
291  GPS_MAJOR_VERSION,
292  GPS_MINOR_VERSION,
293  GPS_REVISION,
294  GPS_MISSION_REV);
295  }
296  else
297  {
298  if (hasEvents == 1)
299  {
300  (void) CFE_ES_WriteToSysLog("GPS - Application failed to initialize\n");
301  }
302  }
303 
304  return (iStatus);
305 }

313 int32 GPS::RcvSchPipeMsg(int32 iBlocking)
314 {
315  int32 iStatus = CFE_SUCCESS;
316  CFE_SB_Msg_t* MsgPtr = NULL;
317  boolean returnBool = FALSE;
318  CFE_SB_MsgId_t MsgId;
319 
320  /* Stop Performance Log entry */
321  CFE_ES_PerfLogExit(GPS_MAIN_TASK_PERF_ID);
322 
323  /* Wait for WakeUp messages from scheduler */
324  iStatus = CFE_SB_RcvMsg(&MsgPtr, SchPipeId, iBlocking);
325 
326  /* Start Performance Log entry */
327  CFE_ES_PerfLogEntry(GPS_MAIN_TASK_PERF_ID);
328 
329  if (iStatus == CFE_SUCCESS)
330  {
331  MsgId = CFE_SB_GetMsgId(MsgPtr);
332  switch (MsgId)
333  {
334  case GPS_READ_SENSOR_MID:
335  {
336  //returnBool = GPS_Custom_Measure_PositionMsg(&VehicleGps);
337  //if(TRUE == returnBool)
338  //{
339  //SendVehicleGps();
340  //}
341  //returnBool = GPS_Custom_Measure_SatInfoMsg(&SatelliteInfo);
342  //if(TRUE == returnBool)
343  //{
344  //SendSatelliteInfo();
345  //}
346  break;
347  }
348  case GPS_SEND_HK_MID:
349  {
350  ProcessCmdPipe();
351  // TODO: Need a mutex for this in the current event driven publish.
352  memcpy(&HkTlm.VehicleGpsMsg, &VehicleGps, sizeof(VehicleGps));
353  ReportHousekeeping();
354  break;
355  }
356  //case PX4_GPS_INJECT_DATA_MID:
357  //{
358  //memcpy(&CVT.GpsInjectData, MsgPtr, sizeof(CVT.GpsInjectData));
359  //break;
360  //}
361  default:
362  {
363  (void) CFE_EVS_SendEvent(GPS_MSGID_ERR_EID, CFE_EVS_ERROR,
364  "Recvd invalid SCH msgId (0x%04X)", MsgId);
365  }
366  }
367  }
368  else if (iStatus == CFE_SB_NO_MESSAGE)
369  {
370  /* If there's no incoming message, do nothing here.
371  * Note, this section is dead code only if the iBlocking arg
372  * is CFE_SB_PEND_FOREVER. */
373  iStatus = CFE_SUCCESS;
374  }
375  else if (iStatus == CFE_SB_TIME_OUT)
376  {
377  /* If there's no incoming message within a specified time (via the
378  * iBlocking arg, do nothing here.
379  * Note, this section is dead code only if the iBlocking arg
380  * is CFE_SB_PEND_FOREVER. */
381  iStatus = CFE_SUCCESS;
382  }
383  else
384  {
385  (void) CFE_EVS_SendEvent(GPS_RCVMSG_ERR_EID, CFE_EVS_ERROR,
386  "SCH pipe read error (0x%08lX).", iStatus);
387  }
388 
389  return (iStatus);
390 }

393 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
394 /* */
395 /* Process Incoming Commands */
396 /* */
397 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
398 void GPS::ProcessCmdPipe()
399 {
400  int32 iStatus = CFE_SUCCESS;
401  CFE_SB_Msg_t* CmdMsgPtr = NULL;
402  CFE_SB_MsgId_t CmdMsgId;
403 
404  /* Process command messages until the pipe is empty */
405  while (1)
406  {
407  iStatus = CFE_SB_RcvMsg(&CmdMsgPtr, CmdPipeId, CFE_SB_POLL);
408  if(iStatus == CFE_SUCCESS)
409  {
410  CmdMsgId = CFE_SB_GetMsgId(CmdMsgPtr);
411  switch (CmdMsgId)
412  {
413  case GPS_CMD_MID:
414  {
415  ProcessAppCmds(CmdMsgPtr);
416  break;
417  }
418  default:
419  {
420  /* Bump the command error counter for an unknown command.
421  * (This should only occur if it was subscribed to with this
422  * pipe, but not handled in this switch-case.) */
423  HkTlm.usCmdErrCnt++;
424  (void) CFE_EVS_SendEvent(GPS_MSGID_ERR_EID, CFE_EVS_ERROR,
425  "Recvd invalid CMD msgId (0x%04X)", (unsigned short)CmdMsgId);
426  break;
427  }
428  }
429  }
430  else if (iStatus == CFE_SB_NO_MESSAGE)
431  {
432  break;
433  }
434  else
435  {
436  (void) CFE_EVS_SendEvent(GPS_RCVMSG_ERR_EID, CFE_EVS_ERROR,
437  "CMD pipe read error (0x%08lX)", iStatus);
438  break;
439  }
440  }
441  return;
442 }

446 /* */
447 /* Process GPS Commands */
448 /* */
449 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
450 void GPS::ProcessAppCmds(CFE_SB_Msg_t* MsgPtr)
451 {
452  uint32 uiCmdCode = 0;
453 
454  if (MsgPtr != NULL)
455  {
456  uiCmdCode = CFE_SB_GetCmdCode(MsgPtr);
457  switch (uiCmdCode)
458  {
459  case GPS_NOOP_CC:
460  {
461  HkTlm.usCmdCnt++;
462  (void) CFE_EVS_SendEvent(GPS_CMD_NOOP_EID, CFE_EVS_INFORMATION,
463  "Recvd NOOP. Version %d.%d.%d.%d",
464  GPS_MAJOR_VERSION,
465  GPS_MINOR_VERSION,
466  GPS_REVISION,
467  GPS_MISSION_REV);
468  break;
469  }
470  case GPS_RESET_CC:
471  {
472  HkTlm.usCmdCnt = 0;
473  HkTlm.usCmdErrCnt = 0;
474  break;
475  }
476  default:
477  {
478  HkTlm.usCmdErrCnt++;
479  (void) CFE_EVS_SendEvent(GPS_CC_ERR_EID, CFE_EVS_ERROR,
480  "Recvd invalid command code (%u)", (unsigned int)uiCmdCode);
481  break;
482  }
483  }
484  }
485  return;
486 }

571 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
572 /* */
573 /* GPS Application C++ style main entry point. */
574 /* */
575 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
576 void GPS::AppMain()
577 {
578  /* Register the application with Executive Services */
579  uiRunStatus = CFE_ES_APP_RUN;
580 
581  int32 iStatus = CFE_ES_RegisterApp();
582  if (iStatus != CFE_SUCCESS)
583  {
584  (void) CFE_ES_WriteToSysLog("GPS - Failed to register the app (0x%08lX)\n", iStatus);
585  }
586 
587  /* Start Performance Log entry */
588  CFE_ES_PerfLogEntry(GPS_MAIN_TASK_PERF_ID);
589 
590  /* Perform application initializations */
591  if (iStatus == CFE_SUCCESS)
592  {
593  iStatus = InitApp();
594  }
595 
596  if (iStatus == CFE_SUCCESS)
597  {
598  /* Do not perform performance monitoring on startup sync */
599  CFE_ES_PerfLogExit(GPS_MAIN_TASK_PERF_ID);
600  CFE_ES_WaitForStartupSync(GPS_STARTUP_TIMEOUT_MSEC);
601  CFE_ES_PerfLogEntry(GPS_MAIN_TASK_PERF_ID);
602  }
603  else
604  {
605  uiRunStatus = CFE_ES_APP_ERROR;
606  }
607 
608  /* Application main loop */
609  while (CFE_ES_RunLoop(&uiRunStatus) == TRUE)
610  {
611  RcvSchPipeMsg(GPS_SCH_PIPE_PEND_TIME);
612 
613  iStatus = AcquireConfigPointers();
614  if(iStatus != CFE_SUCCESS)
615  {
616  /* We apparently tried to load a new table but failed. Terminate the application. */
617  uiRunStatus = CFE_ES_APP_ERROR;
618  }
619  }
620 
621  /* Stop Performance Log entry */
622  CFE_ES_PerfLogExit(GPS_MAIN_TASK_PERF_ID);
