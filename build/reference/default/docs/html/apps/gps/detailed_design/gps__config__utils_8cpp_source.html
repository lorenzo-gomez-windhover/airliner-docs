56  iStatus = CFE_TBL_Register(&ConfigTblHdl,
57  GPS_CONFIG_TABLENAME,

59  CFE_TBL_OPT_DEFAULT,
60  GPS::ValidateConfigTbl);
61  if (iStatus != CFE_SUCCESS)
62  {
63  /* Note, a critical table could return another nominal code. If this table is
64  * made critical this logic would have to change. */
65  (void) CFE_EVS_SendEvent(GPS_CFGTBL_REG_ERR_EID, CFE_EVS_ERROR,
66  "Failed to register config table (0x%08lX)",
67  iStatus);
68  goto GPS_InitConfigTbl_Exit_Tag;
69  }
70 
71  /* Load Config table file */
72  iStatus = CFE_TBL_Load(ConfigTblHdl,
73  CFE_TBL_SRC_FILE,
74  GPS_CONFIG_TABLE_FILENAME);
75  if (iStatus != CFE_SUCCESS)
76  {
77  /* Note, CFE_SUCCESS is for a successful full table load. If a partial table
78  load is desired then this logic would have to change. */
79  (void) CFE_EVS_SendEvent(GPS_CFGTBL_LOAD_ERR_EID, CFE_EVS_ERROR,
80  "Failed to load Config Table (0x%08lX)",
81  iStatus);
82  goto GPS_InitConfigTbl_Exit_Tag;
83  }
84 
85  iStatus = AcquireConfigPointers();
86 
87 GPS_InitConfigTbl_Exit_Tag:

114 int32 GPS::AcquireConfigPointers(void)
115 {
116  int32 iStatus = CFE_SUCCESS;
117 
118  /*
119  ** Release the table
120  */
121  /* TODO: This return value can indicate success, error, or that the info has been
122  * updated. We ignore this return value in favor of checking CFE_TBL_Manage(), but
123  * be sure this is the behavior you want. */
124  (void) CFE_TBL_ReleaseAddress(ConfigTblHdl);
125 
126  /*
127  ** Manage the table
128  */
129  iStatus = CFE_TBL_Manage(ConfigTblHdl);
130  if ((iStatus != CFE_SUCCESS) && (iStatus != CFE_TBL_INFO_UPDATED))
131  {
132  (void) CFE_EVS_SendEvent(GPS_CFGTBL_MANAGE_ERR_EID, CFE_EVS_ERROR,
133  "Failed to manage GPS Config table (0x%08lX)",
134  iStatus);
135  goto GPS_AcquireConfigPointers_Exit_Tag;
136  }
137 
138  /*
139  ** Get a pointer to the table
140  */
141  iStatus = CFE_TBL_GetAddress((void**)&ConfigTblPtr, ConfigTblHdl);
142  if (iStatus == CFE_TBL_INFO_UPDATED)
143  {
144  iStatus = CFE_SUCCESS;
145  }
146  else if(iStatus != CFE_SUCCESS)
147  {
148  ConfigTblPtr = 0;
149  (void) CFE_EVS_SendEvent(GPS_CFGTBL_GETADDR_ERR_EID, CFE_EVS_ERROR,
150  "Failed to get Config table's address (0x%08lX)",
151  iStatus);
152  }
153 
154 GPS_AcquireConfigPointers_Exit_Tag:
155  return iStatus;
156 }
