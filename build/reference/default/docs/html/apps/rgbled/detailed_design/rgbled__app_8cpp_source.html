74 /* Initialize event tables. */
75 /* */
76 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
77 int32 RGBLED::InitEvent()
78 {
79  int32 iStatus = CFE_SUCCESS;
80  int32 ind = 0;
81 
82  CFE_EVS_BinFilter_t EventTbl[CFE_EVS_MAX_EVENT_FILTERS];
83 
84  /* Initialize the event filter table.
85  * Note: 0 is the CFE_EVS_NO_FILTER mask and event 0 is reserved (not used) */
86  CFE_PSP_MemSet(EventTbl, 0x00, sizeof(EventTbl));
87 
88  EventTbl[ ind].EventID = RGBLED_RESERVED_EID;
89  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
90  EventTbl[ ind].EventID = RGBLED_INIT_INF_EID;
91  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
92  EventTbl[ ind].EventID = RGBLED_CMD_NOOP_EID;
93  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
94  EventTbl[ ind].EventID = RGBLED_SUBSCRIBE_ERR_EID;
95  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
96  EventTbl[ ind].EventID = RGBLED_PIPE_INIT_ERR_EID;
97  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
98  EventTbl[ ind].EventID = RGBLED_RCVMSG_ERR_EID;
99  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
100  EventTbl[ ind].EventID = RGBLED_MSGID_ERR_EID;
101  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
102  EventTbl[ ind].EventID = RGBLED_CC_ERR_EID;
103  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
104  EventTbl[ ind].EventID = RGBLED_MSGLEN_ERR_EID;
105  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
106  EventTbl[ ind].EventID = RGBLED_CMD_ERR_EID;
107  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
108  EventTbl[ ind].EventID = RGBLED_INIT_ERR_EID;
109  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
110 
111  /* Register the table with CFE */
112  iStatus = CFE_EVS_Register(EventTbl, ind, CFE_EVS_BINARY_FILTER);
113  if (iStatus != CFE_SUCCESS)
114  {
115  (void) CFE_ES_WriteToSysLog("RGBLED - Failed to register with EVS (0x%08lX)\n", iStatus);
116  }
117 
118  return (iStatus);
119 }

124 /* Initialize Message Pipes */
125 /* */
126 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
127 int32 RGBLED::InitPipe()
128 {
129  int32 iStatus = CFE_SUCCESS;
130 
131  /* Init schedule pipe and subscribe to wakeup messages */
132  iStatus = CFE_SB_CreatePipe(&SchPipeId,
133  RGBLED_SCH_PIPE_DEPTH,
134  RGBLED_SCH_PIPE_NAME);
135  if (iStatus == CFE_SUCCESS)
136  {
137  iStatus = CFE_SB_SubscribeEx(RGBLED_WAKEUP_MID, SchPipeId, CFE_SB_Default_Qos, RGBLED_WAKEUP_MID_MAX_MSG_COUNT);
138  if (iStatus != CFE_SUCCESS)
139  {
140  (void) CFE_EVS_SendEvent(RGBLED_SUBSCRIBE_ERR_EID, CFE_EVS_ERROR,
141  "Sch Pipe failed to subscribe to RGBLED_WAKEUP_MID. (0x%08lX)",
142  iStatus);
143  goto RGBLED_InitPipe_Exit_Tag;
144  }
145 
146  iStatus = CFE_SB_SubscribeEx(RGBLED_SEND_HK_MID, SchPipeId, CFE_SB_Default_Qos, RGBLED_SEND_HK_MID_MAX_MSG_COUNT);
147  if (iStatus != CFE_SUCCESS)
148  {
149  (void) CFE_EVS_SendEvent(RGBLED_SUBSCRIBE_ERR_EID, CFE_EVS_ERROR,
150  "CMD Pipe failed to subscribe to RGBLED_SEND_HK_MID. (0x%08X)",
151  (unsigned int)iStatus);
152  goto RGBLED_InitPipe_Exit_Tag;
153  }
154  iStatus = CFE_SB_SubscribeEx(PX4_LED_CONTROL_MID, SchPipeId, CFE_SB_Default_Qos, 1);
155  if (iStatus != CFE_SUCCESS)
156  {
157  (void) CFE_EVS_SendEvent(RGBLED_SUBSCRIBE_ERR_EID, CFE_EVS_ERROR,
158  "CMD Pipe failed to subscribe to PX4_LED_CONTROL_MID. (0x%08lX)",
159  iStatus);
160  goto RGBLED_InitPipe_Exit_Tag;
161  }
162  }
163  else
164  {
165  (void) CFE_EVS_SendEvent(RGBLED_PIPE_INIT_ERR_EID, CFE_EVS_ERROR,
166  "Failed to create SCH pipe (0x%08lX)",
167  iStatus);
168  goto RGBLED_InitPipe_Exit_Tag;
169  }
170 
171  /* Init command pipe and subscribe to command messages */
172  iStatus = CFE_SB_CreatePipe(&CmdPipeId,
173  RGBLED_CMD_PIPE_DEPTH,
174  RGBLED_CMD_PIPE_NAME);
175  if (iStatus == CFE_SUCCESS)
176  {
177  /* Subscribe to command messages */
178  iStatus = CFE_SB_Subscribe(RGBLED_CMD_MID, CmdPipeId);
179 
180  if (iStatus != CFE_SUCCESS)
181  {
182  (void) CFE_EVS_SendEvent(RGBLED_SUBSCRIBE_ERR_EID, CFE_EVS_ERROR,
183  "CMD Pipe failed to subscribe to RGBLED_CMD_MID. (0x%08lX)",
184  iStatus);
185  goto RGBLED_InitPipe_Exit_Tag;
186  }
187  }
188  else
189  {
190  (void) CFE_EVS_SendEvent(RGBLED_PIPE_INIT_ERR_EID, CFE_EVS_ERROR,
191  "Failed to create CMD pipe (0x%08lX)",
192  iStatus);
193  goto RGBLED_InitPipe_Exit_Tag;
194  }
195 
196 RGBLED_InitPipe_Exit_Tag:
197  return (iStatus);
198 }

216 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
217 /* */
218 /* RGBLED initialization */
219 /* */
220 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
221 int32 RGBLED::InitApp()
222 {
223  int32 iStatus = CFE_SUCCESS;
224  boolean returnBool = FALSE;
225  int8 hasEvents = 0;
226 
227  iStatus = InitEvent();
228  if (iStatus != CFE_SUCCESS)
229  {
230  (void) CFE_ES_WriteToSysLog("RGBLED - Failed to init events (0x%08lX)\n", iStatus);
231  goto RGBLED_InitApp_Exit_Tag;
232  }
233  else
234  {
235  hasEvents = 1;
236  }
237 
238  iStatus = InitPipe();
239  if (iStatus != CFE_SUCCESS)
240  {
241  goto RGBLED_InitApp_Exit_Tag;
242  }
243 
244  InitData();
245 
246  returnBool = RGBLED_Custom_Init();
247  if (FALSE == returnBool)
248  {
249  iStatus = -1;
250  goto RGBLED_InitApp_Exit_Tag;
251  }
252  HkTlm.State = RGBLED_INITIALIZED;
253 
254  /* Register the cleanup callback */
255  iStatus = OS_TaskInstallDeleteHandler(&RGBLED_CleanupCallback);
256  if (iStatus != CFE_SUCCESS)
257  {
258  (void) CFE_EVS_SendEvent(RGBLED_INIT_ERR_EID, CFE_EVS_ERROR,
259  "Failed to init register cleanup callback (0x%08X)",
260  (unsigned int)iStatus);
261  goto RGBLED_InitApp_Exit_Tag;
262  }
263 
264 RGBLED_InitApp_Exit_Tag:
265  if (iStatus == CFE_SUCCESS)
266  {
267  (void) CFE_EVS_SendEvent(RGBLED_INIT_INF_EID, CFE_EVS_INFORMATION,
268  "Initialized. Version %d.%d.%d.%d",
269  RGBLED_MAJOR_VERSION,
270  RGBLED_MINOR_VERSION,
271  RGBLED_REVISION,
272  RGBLED_MISSION_REV);
273  }
274  else
275  {
276  if (hasEvents == 1)
277  {
278  (void) CFE_ES_WriteToSysLog("RGBLED - Application failed to initialize\n");
279  }
280  }
281 
282  return (iStatus);
283 }

288 /* Receive and Process Messages */
289 /* */
290 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
291 int32 RGBLED::RcvSchPipeMsg(int32 iBlocking)
292 {
293  int32 iStatus = CFE_SUCCESS;
294  CFE_SB_Msg_t* MsgPtr = NULL;
295  CFE_SB_MsgId_t MsgId;
296  static uint8 lastColor = 0;
297 
298  /* Stop Performance Log entry */
299  CFE_ES_PerfLogExit(RGBLED_MAIN_TASK_PERF_ID);
300 
301  /* Wait for WakeUp messages from scheduler */
302  iStatus = CFE_SB_RcvMsg(&MsgPtr, SchPipeId, iBlocking);
303 
304  /* Start Performance Log entry */
305  CFE_ES_PerfLogEntry(RGBLED_MAIN_TASK_PERF_ID);
306 
307  if (iStatus == CFE_SUCCESS)
308  {
309  MsgId = CFE_SB_GetMsgId(MsgPtr);
310  switch (MsgId)
311  {
312  case RGBLED_WAKEUP_MID:
313  {
314  ProcessCmdPipe();
315 
316  /* If not in self test mode and the last color is not the
317  * current value update */
318  if(HkTlm.State != RGBLED_SELFTEST &&
319  lastColor != CVT.RGBLEDControl.Color)
320  {
321  HkTlm.State = RGBLED_ON;
322  lastColor = CVT.RGBLEDControl.Color;
323  switch (CVT.RGBLEDControl.Color)
324  {
325  case RGBLED_COLOR_RED:
326  {
327  RGBLED_Custom_SetColor(255, 0, 0);
328  HkTlm.Color = RGBLED_COLOR_RED;
329  break;
330  }
331 
332  case RGBLED_COLOR_GREEN:
333  {
334  RGBLED_Custom_SetColor(0, 255, 0);
335  HkTlm.Color = RGBLED_COLOR_GREEN;
336  break;
337  }
338 
339  case RGBLED_COLOR_BLUE:
340  {
341  RGBLED_Custom_SetColor(0, 0, 255);
342  HkTlm.Color = RGBLED_COLOR_BLUE;
343  break;
344  }
345 
346  case RGBLED_COLOR_AMBER:
347  {
348  RGBLED_Custom_SetColor(255, 255, 0);
349  HkTlm.Color = RGBLED_COLOR_AMBER;
350  break;
351  }
352 
353  case RGBLED_COLOR_YELLOW:
354  {
355  RGBLED_Custom_SetColor(255, 255, 0);
356  HkTlm.Color = RGBLED_COLOR_YELLOW;
357  break;
358  }
359 
360  case RGBLED_COLOR_PURPLE:
361  {
362  RGBLED_Custom_SetColor(255, 0, 255);
363  HkTlm.Color = RGBLED_COLOR_PURPLE;
364  break;
365  }
366 
367  case RGBLED_COLOR_CYAN:
368  {
369  RGBLED_Custom_SetColor(0, 255, 255);
370  HkTlm.Color = RGBLED_COLOR_CYAN;
371  break;
372  }
373 
374  case RGBLED_COLOR_WHITE:
375  {
376  RGBLED_Custom_SetColor(255, 255, 255);
377  HkTlm.Color = RGBLED_COLOR_WHITE;
378  break;
379  }
380 
381  default: /* COLOR_OFF */
382  {
383  RGBLED_Custom_SetColor(0, 0, 0);
384  HkTlm.Color = RGBLED_COLOR_OFF;
385  HkTlm.State = RGBLED_INITIALIZED;
386  break;
387  }
388  }
389  }
390 
391  break;
392  }
393  case RGBLED_SEND_HK_MID:
394  {
395  ReportHousekeeping();
396  break;
397  }
398  case PX4_LED_CONTROL_MID:
399  {
400  CFE_PSP_MemCpy(&CVT.RGBLEDControl, MsgPtr, sizeof(CVT.RGBLEDControl));
401  break;
402  }
403  default:
404  {
405  (void) CFE_EVS_SendEvent(RGBLED_MSGID_ERR_EID, CFE_EVS_ERROR,
406  "Recvd invalid SCH msgId (0x%04X)", MsgId);
407  break;
408  }
409  }
410  }
411  else if (iStatus == CFE_SB_NO_MESSAGE)
412  {
413  /* If there's no incoming message, do nothing here.
414  * Note, this section is dead code only if the iBlocking arg
415  * is CFE_SB_PEND_FOREVER. */
416  iStatus = CFE_SUCCESS;
417  }
418  else if (iStatus == CFE_SB_TIME_OUT)
419  {
420  /* If there's no incoming message within a specified time (via the
421  * iBlocking arg, do nothing here.
422  * Note, this section is dead code only if the iBlocking arg
423  * is CFE_SB_PEND_FOREVER. */
424  iStatus = CFE_SUCCESS;
425  }
426  else
427  {
428  (void) CFE_EVS_SendEvent(RGBLED_RCVMSG_ERR_EID, CFE_EVS_ERROR,
429  "SCH pipe read error (0x%08lX).", iStatus);
430  }
431 
432  return (iStatus);
433 }

436 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
437 /* */
438 /* Process Incoming Commands */
439 /* */
440 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
441 void RGBLED::ProcessCmdPipe()
442 {
443  int32 iStatus = CFE_SUCCESS;
444  CFE_SB_Msg_t* CmdMsgPtr = NULL;
445  CFE_SB_MsgId_t CmdMsgId;
446 
447  /* Process command messages until the pipe is empty */
448  while (1)
449  {
450  iStatus = CFE_SB_RcvMsg(&CmdMsgPtr, CmdPipeId, CFE_SB_POLL);
451  if(iStatus == CFE_SUCCESS)
452  {
453  CmdMsgId = CFE_SB_GetMsgId(CmdMsgPtr);
454  switch (CmdMsgId)
455  {
456  case RGBLED_CMD_MID:
457  {
458  ProcessAppCmds(CmdMsgPtr);
459  break;
460  }
461  default:
462  {
463  /* Bump the command error counter for an unknown command.
464  * (This should only occur if it was subscribed to with this
465  * pipe, but not handled in this switch-case.) */
466  HkTlm.usCmdErrCnt++;
467  (void) CFE_EVS_SendEvent(RGBLED_MSGID_ERR_EID, CFE_EVS_ERROR,
468  "Recvd invalid CMD msgId (0x%04X)", (unsigned short)CmdMsgId);
469  break;
470  }
471  }
472  }
473  else if (iStatus == CFE_SB_NO_MESSAGE)
474  {
475  break;
476  }
477  else
478  {
479  (void) CFE_EVS_SendEvent(RGBLED_RCVMSG_ERR_EID, CFE_EVS_ERROR,
480  "CMD pipe read error (0x%08lX)", iStatus);
481  break;
482  }
483  }
484  return;
485 }

488 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
489 /* */
490 /* Process RGBLED Commands */
491 /* */
492 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
493 void RGBLED::ProcessAppCmds(CFE_SB_Msg_t* MsgPtr)
494 {
495  uint32 uiCmdCode = 0;
496 
497  if (MsgPtr != NULL)
498  {
499  /* All three commands currently have no arguments */
500  uint16 ExpectedLength = sizeof(RGBLED_NoArgCmd_t);
501  /* Length verification is then the same for all three commands */
502  if (VerifyCmdLength(MsgPtr, ExpectedLength))
503  {
504  uiCmdCode = CFE_SB_GetCmdCode(MsgPtr);
505  switch (uiCmdCode)
506  {
507  case RGBLED_NOOP_CC:
508  {
509  HkTlm.usCmdCnt++;
510  (void) CFE_EVS_SendEvent(RGBLED_CMD_NOOP_EID, CFE_EVS_INFORMATION,
511  "Recvd NOOP. Version %d.%d.%d.%d",
512  RGBLED_MAJOR_VERSION,
513  RGBLED_MINOR_VERSION,
514  RGBLED_REVISION,
515  RGBLED_MISSION_REV);
516  break;
517  }
518  case RGBLED_RESET_CC:
519  {
520  HkTlm.usCmdCnt = 0;
521  HkTlm.usCmdErrCnt = 0;
522  break;
523  }
524  case RGBLED_SELFTEST_CC:
525  {
526  if(HkTlm.State != RGBLED_SELFTEST)
527  {
528  HkTlm.usCmdCnt++;
529  previousState = HkTlm.State;
530  HkTlm.State = RGBLED_SELFTEST;
531  RGBLED_Custom_SelfTest();
532  }
533  else
534  {
535  HkTlm.usCmdErrCnt++;
536  CFE_EVS_SendEvent(RGBLED_CMD_ERR_EID, CFE_EVS_ERROR,
537  "RGBLED is already running self-test");
538  }
539  break;
540  }
541  default:
542  {
543  HkTlm.usCmdErrCnt++;
544  (void) CFE_EVS_SendEvent(RGBLED_CC_ERR_EID, CFE_EVS_ERROR,
545  "Recvd invalid command code (%u)", (unsigned int)uiCmdCode);
546  break;
547  }
548  }
549  }
550  }
551  return;
552 }

566 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
567 /* */
568 /* Verify Command Length */
569 /* */
570 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
571 boolean RGBLED::VerifyCmdLength(CFE_SB_Msg_t* MsgPtr,
572  uint16 usExpectedLen)
573 {
574  boolean bResult = TRUE;
575  uint16 usMsgLen = 0;
576 
577  if (MsgPtr != NULL)
578  {
579  usMsgLen = CFE_SB_GetTotalMsgLength(MsgPtr);
580 
581  if (usExpectedLen != usMsgLen)
582  {
583  bResult = FALSE;
584  CFE_SB_MsgId_t MsgId = CFE_SB_GetMsgId(MsgPtr);
585  uint16 usCmdCode = CFE_SB_GetCmdCode(MsgPtr);
586 
587  (void) CFE_EVS_SendEvent(RGBLED_MSGLEN_ERR_EID, CFE_EVS_ERROR,
588  "Rcvd invalid msgLen: msgId=0x%08X, cmdCode=%d, "
589  "msgLen=%d, expectedLen=%d",
590  MsgId, usCmdCode, usMsgLen, usExpectedLen);
591  HkTlm.usCmdErrCnt++;
592  }
593  }
594 
595  return (bResult);
596 }
