307 /* Receive and Process Messages */
308 /* */
309 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
310 int32 BAT::RcvSchPipeMsg(int32 iBlocking)
311 {
312  int32 iStatus=CFE_SUCCESS;
313  CFE_SB_Msg_t* MsgPtr=NULL;
314  CFE_SB_MsgId_t MsgId;
315 
316  /* Stop Performance Log entry */
317  CFE_ES_PerfLogExit(BAT_MAIN_TASK_PERF_ID);
318 
319  /* Wait for WakeUp messages from scheduler */
320  iStatus = CFE_SB_RcvMsg(&MsgPtr, SchPipeId, iBlocking);
321 
322  /* Start Performance Log entry */
323  CFE_ES_PerfLogEntry(BAT_MAIN_TASK_PERF_ID);
324 
325  if (iStatus == CFE_SUCCESS)
326  {
327  MsgId = CFE_SB_GetMsgId(MsgPtr);
328  switch (MsgId)
329  {
330  case BAT_WAKEUP_MID:
331  {
332  HkTlm.WakeupCount++;
333  ProcessDataPipe();
334  PublishBatteryStatus();
335  break;
336  }
337 
338  case BAT_SEND_HK_MID:
339  {
340  ProcessCmdPipe();
341  ReportHousekeeping();
342  break;
343  }
344 
345  default:
346  {
347  (void) CFE_EVS_SendEvent(BAT_MSGID_ERR_EID, CFE_EVS_ERROR,
348  "Recvd invalid SCH msgId (0x%04X)", MsgId);
349  break;
350  }
351  }
352  }
353  else if (iStatus == CFE_SB_NO_MESSAGE)
354  {
355  iStatus = CFE_SUCCESS;
356  }
357  else if (iStatus == CFE_SB_TIME_OUT)
358  {
359  iStatus = CFE_SUCCESS;
360  }
361  else
362  {
363  (void) CFE_EVS_SendEvent(BAT_RCVMSG_ERR_EID, CFE_EVS_ERROR,
364  "SCH pipe read error (0x%08lX).", iStatus);
365  }
366 
367  return iStatus;
368 }

375 void BAT::ProcessCmdPipe()
376 {
377  int32 iStatus = CFE_SUCCESS;
378  CFE_SB_Msg_t* CmdMsgPtr=NULL;
379  CFE_SB_MsgId_t CmdMsgId;
380 
381  /* Process command messages until the pipe is empty */
382  while (1)
383  {
384  iStatus = CFE_SB_RcvMsg(&CmdMsgPtr, CmdPipeId, CFE_SB_POLL);
385  if(iStatus == CFE_SUCCESS)
386  {
387  CmdMsgId = CFE_SB_GetMsgId(CmdMsgPtr);
388  switch (CmdMsgId)
389  {
390  case BAT_CMD_MID:
391  ProcessAppCmds(CmdMsgPtr);
392  break;
393 
394  default:
395  /* Bump the command error counter for an unknown command.
396  * (This should only occur if it was subscribed to with this
397  * pipe, but not handled in this switch-case.) */
398  HkTlm.usCmdErrCnt++;
399  (void) CFE_EVS_SendEvent(BAT_MSGID_ERR_EID, CFE_EVS_ERROR,
400  "Recvd invalid CMD msgId (0x%04X)", (unsigned short)CmdMsgId);
401  break;
402  }
403  }
404  else if (iStatus == CFE_SB_NO_MESSAGE)
405  {
406  break;
407  }
408  else
409  {
410  (void) CFE_EVS_SendEvent(BAT_RCVMSG_ERR_EID, CFE_EVS_ERROR,
411  "CMD pipe read error (0x%08lX)", iStatus);
412  break;
413  }
414  }
415 }

428

454  "Recvd invalid DATA msgId (0x%04X)", (unsigned short)MsgId);
455  break;
456  }
457  }
458  else if (iStatus == CFE_SB_NO_MESSAGE)
459  {
460  break;
461  }
462  else
463  {
464  (void) CFE_EVS_SendEvent(BAT_RCVMSG_ERR_EID, CFE_EVS_ERROR,

477 void BAT::ProcessAppCmds(CFE_SB_Msg_t* MsgPtr)
478 {
479  uint32 uiCmdCode=0;
480 
481  if (MsgPtr != NULL)
482  {
483  uiCmdCode = CFE_SB_GetCmdCode(MsgPtr);
484  switch (uiCmdCode)
485  {
486  case BAT_NOOP_CC:
487  {
488  HkTlm.usCmdCnt++;
489  (void) CFE_EVS_SendEvent(BAT_CMD_NOOP_EID,
490  CFE_EVS_INFORMATION,
491  "Recvd NOOP. Version %d.%d.%d.%d",
492  BAT_MAJOR_VERSION,
493  BAT_MINOR_VERSION,
494  BAT_REVISION,
495  BAT_MISSION_REV);
496  break;
497  }
498  case BAT_RESET_CC:
499  {
500  HkTlm.usCmdCnt = 0;
501  HkTlm.usCmdErrCnt = 0;
502  HkTlm.ActuatorControls0MsgCount = 0;
503  HkTlm.ActuatorArmedMsgCount = 0;
504  break;
505  }
506  default:
507  {
508  HkTlm.usCmdErrCnt++;
509  (void) CFE_EVS_SendEvent(BAT_CC_ERR_EID, CFE_EVS_ERROR,
510  "Recvd invalid command code (%u)", (unsigned int)uiCmdCode);
511  break;
512  }
513  }
514  }
515 }

818 float BAT::GetDischarged(float Current)
819 {
820  CFE_TIME_SysTime_t zeroTime;
821  CFE_TIME_SysTime_t msgTime;
822  zeroTime.Seconds = 0;
823  zeroTime.Subseconds = 0;
824 
825  if(Current < 0.0f)
826  {
827  /* Not a valid measurement. Because the measurement was invalid we
828  * need to stop integration and re-init8ialize with the next valid
829  * measurement. */
830  m_SampleTime.Seconds = 0;
831  m_SampleTime.Subseconds = 0;
832  }
833 
834  /* Check to see if we have read a valid measurement yet. If not, ignore
835  * the first measurement since we don't know dT yet. */
836  if(CFE_TIME_Compare(m_SampleTime, zeroTime) != CFE_TIME_EQUAL)
837  {
838  CFE_TIME_SysTime_t currentTime = CFE_TIME_GetMET();
839  CFE_TIME_SysTime_t dSysTime = CFE_TIME_Subtract(currentTime, m_SampleTime);
840  float dt = dSysTime.Seconds + (CFE_TIME_Sub2MicroSecs(dSysTime.Subseconds)* 0.000001);
841 
842  /* Integrate the current to calculate the new discharge, but calculate
843  * it using mAh units. */
844  m_Discharged += m_Discharged + (Current * dt / 1000.0 / 3600.0f);
845 
846  /* Finally, set the message time to the current time. */
847  m_SampleTime.Seconds = currentTime.Seconds;
848  m_SampleTime.Subseconds = currentTime.Subseconds;
849  }
850 
851  return m_Discharged;
852 }

855 float BAT::GetRemaining(float Voltage, float Current, float ThrottleNormalized, osalbool Armed)
856 {
857  float remaining = 0.0f;
858  float batVEmptyDynamic = 0.0f;
859  float voltageRange = 0.0f;
860  float rVoltage = 0.0f;
861  float rVoltageFilt = 0.0f;
862  float rCap = 0.0f;
863  float rCapFilt = 0.0f;
864 
865  if(ConfigTblPtr->RInternal > 0.0f)
866  {
867  batVEmptyDynamic = ConfigTblPtr->VEmpty - (Current * ConfigTblPtr->RInternal);
868  }
869  else
870  {
871  /* Assume 10% voltage drop of the full drop range with motors idle */
872  float thr = (Armed) ? ((fabsf(ThrottleNormalized) + 0.1f) / 1.1f) : 0.0f;
873 
874  batVEmptyDynamic = ConfigTblPtr->VEmpty - (ConfigTblPtr->VLoadDrop * thr);
875  }
876 
877  /* The range from full to empty is the same for batteries under load and
878  * without load, since the voltage drop applies to both the full and
879  * empty state. */
880  voltageRange = ConfigTblPtr->VFull - ConfigTblPtr->VEmpty;
881 
882  /* Remaining battery capacity based on voltage. */
883  rVoltage = (Voltage - (ConfigTblPtr->NumCells * batVEmptyDynamic))
884  / (ConfigTblPtr->NumCells * voltageRange);
885  rVoltageFilt = m_RemainingVoltage * 0.99f + rVoltage * 0.01f;
886  m_RemainingVoltage = rVoltageFilt;
887 
888  /* Remaining battery capacity based on used current integrated time. */
889  rCap = 1.0f - m_Discharged / ConfigTblPtr->Capacity;
890  rCapFilt = m_RemainingCapacity * 0.99f + rCap * 0.01f;
891 
892  m_RemainingCapacity = rCapFilt;
893 
894  if(ConfigTblPtr->Capacity > 0.0f)
895  {
896  remaining = fminf(m_RemainingVoltage, m_RemainingCapacity);
897  }
898  else
899  {
900  remaining = m_RemainingVoltage;
901  }
902 
903  return remaining;
904 }

926 float BAT::GetScale(void)
927 {
928  float scale = 0.0;
929  float voltageRange = ConfigTblPtr->VFull - ConfigTblPtr->VEmpty;
930  float batV = ConfigTblPtr->VEmpty + (voltageRange * m_RemainingVoltage);
931 
932  /* Reusing capacity calculation to get single cell voltage before drop. */
933  scale = ConfigTblPtr->VFull / batV;
934 
935  if(scale > 1.3f)
936  {
937  /* Allow at most, 30% compensation */
938  scale = 1.3f;
939  }
940  else if(scale < 1.0f)
941  {
942  /* Shouldn't ever be more than the power at full battery. */
943  scale = 1.0f;
944  }
945 
946  return scale;
