48 /* Initialize Config Table */
49 /* */
50 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
51 int32 BAT::InitConfigTbl()
52 {
53  int32 iStatus=0;
54 
55  /* Register Config table */
56  iStatus = CFE_TBL_Register(&ConfigTblHdl,
57  BAT_CONFIG_TABLENAME,
58  (sizeof(BAT_ConfigTbl_t)),
59  CFE_TBL_OPT_DEFAULT,
60  BAT::ValidateConfigTbl);
61  if (iStatus != CFE_SUCCESS)
62  {
63  /* Note, a critical table could return another nominal code. If this table is
64  * made critical this logic would have to change. */
65  (void) CFE_EVS_SendEvent(BAT_CFGTBL_REG_ERR_EID, CFE_EVS_ERROR,
66  "Failed to register config table (0x%08lX)",
67  iStatus);
68  goto BAT_InitConfigTbl_Exit_Tag;
69  }
70 
71  /* Load Config table file */
72  iStatus = CFE_TBL_Load(ConfigTblHdl,
73  CFE_TBL_SRC_FILE,
74  BAT_CONFIG_TABLE_FILENAME);
75  if (iStatus != CFE_SUCCESS)
76  {
77  /* Note, CFE_SUCCESS is for a successful full table load. If a partial table
78  load is desired then this logic would have to change. */
79  (void) CFE_EVS_SendEvent(BAT_CFGTBL_LOAD_ERR_EID, CFE_EVS_ERROR,
80  "Failed to load Config Table (0x%08lX)",
81  iStatus);
82  goto BAT_InitConfigTbl_Exit_Tag;
83  }
84 
85  iStatus = AcquireConfigPointers();
86 
87 BAT_InitConfigTbl_Exit_Tag:
88  return iStatus;
89 }

97 int32 BAT::ValidateConfigTbl(void* ConfigTblPtr)
98 {
99  int32 iStatus = CFE_SUCCESS;
100  BAT_ConfigTbl_t* BAT_ConfigTblPtr = (BAT_ConfigTbl_t*)(ConfigTblPtr);
101 
102  if (BAT_ConfigTblPtr->VEmpty < BAT_V_EMPTY_MIN ||
103  BAT_ConfigTblPtr->VEmpty > BAT_V_EMPTY_MAX)
104  {
105  iStatus = -1;
106  goto BAT_ValidateConfigTbl_Exit_Tag;
107  }
108 
109  if (BAT_ConfigTblPtr->VFull < BAT_V_CHARGED_MIN ||
110  BAT_ConfigTblPtr->VFull > BAT_V_CHARGED_MAX)
111  {
112  iStatus = -1;
113  goto BAT_ValidateConfigTbl_Exit_Tag;
114  }
115 
116  if (BAT_ConfigTblPtr->NumCells < BAT_N_CELLS_MIN ||
117  BAT_ConfigTblPtr->NumCells > BAT_N_CELLS_MAX)
118  {
119  iStatus = -1;
120  goto BAT_ValidateConfigTbl_Exit_Tag;
121  }
122 
123  if (BAT_ConfigTblPtr->Capacity < BAT_CAPACITY_MIN ||
124  BAT_ConfigTblPtr->Capacity > BAT_CAPACITY_MAX)
125  {
126  iStatus = -1;
127  goto BAT_ValidateConfigTbl_Exit_Tag;
128  }
129 
130  if (BAT_ConfigTblPtr->LowThreshold < BAT_LOW_THR_MIN ||
131  BAT_ConfigTblPtr->LowThreshold > BAT_LOW_THR_MAX)
132  {
133  iStatus = -1;
134  goto BAT_ValidateConfigTbl_Exit_Tag;
135  }
136 
137  if (BAT_ConfigTblPtr->CriticalThreshold < BAT_CRIT_THR_MIN ||
138  BAT_ConfigTblPtr->CriticalThreshold > BAT_CRIT_THR_MAX)
139  {
140  iStatus = -1;
141  goto BAT_ValidateConfigTbl_Exit_Tag;
142  }
143 
144  if (BAT_ConfigTblPtr->EmergencyThreshold < BAT_EMERGEN_THR_MIN ||

172 /* Acquire Config Pointers */
173 /* */
174 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
175 int32 BAT::AcquireConfigPointers(void)
176 {
177  int32 iStatus = CFE_SUCCESS;
178 
179  OS_MutSemTake(BatteryMutex);
180 
181  /*
182  ** Release the table
183  */
184  (void) CFE_TBL_ReleaseAddress(ConfigTblHdl);
185 
186  /*
187  ** Manage the table
188  */
189  iStatus = CFE_TBL_Manage(ConfigTblHdl);
190  if ((iStatus != CFE_SUCCESS) && (iStatus != CFE_TBL_INFO_UPDATED))
191  {
192  (void) CFE_EVS_SendEvent(BAT_CFGTBL_MANAGE_ERR_EID, CFE_EVS_ERROR,
193  "Failed to manage BAT Config table (0x%08lX)",
194  iStatus);
195  goto BAT_AcquireConfigPointers_Exit_Tag;
196  }
197 
198  /*
199  ** Get a pointer to the table
200  */
201  iStatus = CFE_TBL_GetAddress((void**)&ConfigTblPtr, ConfigTblHdl);
202  if (iStatus == CFE_TBL_INFO_UPDATED)
203  {
204  iStatus = CFE_SUCCESS;
205  }
206  else if(iStatus != CFE_SUCCESS)
207  {
208  ConfigTblPtr = 0;
209  (void) CFE_EVS_SendEvent(BAT_CFGTBL_GETADDR_ERR_EID, CFE_EVS_ERROR,
210  "Failed to get Config table's address (0x%08lX)",
211  iStatus);
212  }
213 
214 BAT_AcquireConfigPointers_Exit_Tag:
215  OS_MutSemGive(BatteryMutex);
216  return iStatus;
217 }
