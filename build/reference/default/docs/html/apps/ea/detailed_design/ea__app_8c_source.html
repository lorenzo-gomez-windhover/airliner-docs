155 /* Initialize Message Pipes */
156 /* */
157 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
158 int32 EA_InitPipe()
159 {
160  int32 iStatus = CFE_SUCCESS;
161 
162  /* Init schedule pipe and subscribe to wakeup messages */
163  iStatus = CFE_SB_CreatePipe(&EA_AppData.SchPipeId,
164  EA_SCH_PIPE_DEPTH,
165  EA_SCH_PIPE_NAME);
166  if (iStatus == CFE_SUCCESS)
167  {
168  iStatus = CFE_SB_SubscribeEx(EA_WAKEUP_MID, EA_AppData.SchPipeId, CFE_SB_Default_Qos, EA_SCH_PIPE_WAKEUP_RESERVED);
169 
170  if (iStatus != CFE_SUCCESS)
171  {
172  (void) CFE_EVS_SendEvent(EA_INIT_ERR_EID, CFE_EVS_ERROR,
173  "Sch Pipe failed to subscribe to EA_WAKEUP_MID. (0x%08X)",
174  (unsigned int)iStatus);
175  goto EA_InitPipe_Exit_Tag;
176  }
177 
178  iStatus = CFE_SB_SubscribeEx(EA_SEND_HK_MID, EA_AppData.SchPipeId, CFE_SB_Default_Qos, EA_SCH_PIPE_SEND_HK_RESERVED);
179 
180  if (iStatus != CFE_SUCCESS)
181  {
182  (void) CFE_EVS_SendEvent(EA_INIT_ERR_EID, CFE_EVS_ERROR,
183  "CMD Pipe failed to subscribe to EA_SEND_HK_MID. (0x%08X)",
184  (unsigned int)iStatus);
185  goto EA_InitPipe_Exit_Tag;
186  }
187 
188  }
189  else
190  {
191  (void) CFE_EVS_SendEvent(EA_INIT_ERR_EID, CFE_EVS_ERROR,
192  "Failed to create SCH pipe (0x%08X)",
193  (unsigned int)iStatus);
194  goto EA_InitPipe_Exit_Tag;
195  }
196 
197  /* Init command pipe and subscribe to command messages */
198  iStatus = CFE_SB_CreatePipe(&EA_AppData.CmdPipeId,
199  EA_CMD_PIPE_DEPTH,
200  EA_CMD_PIPE_NAME);
201  if (iStatus == CFE_SUCCESS)
202  {
203  /* Subscribe to command messages */
204  iStatus = CFE_SB_Subscribe(EA_CMD_MID, EA_AppData.CmdPipeId);
205 
206  if (iStatus != CFE_SUCCESS)
207  {
208  (void) CFE_EVS_SendEvent(EA_INIT_ERR_EID, CFE_EVS_ERROR,
209  "CMD Pipe failed to subscribe to EA_CMD_MID. (0x%08X)",
210  (unsigned int)iStatus);
211  goto EA_InitPipe_Exit_Tag;
212  }
213  }
214  else
215  {
216  (void) CFE_EVS_SendEvent(EA_INIT_ERR_EID, CFE_EVS_ERROR,
217  "Failed to create CMD pipe (0x%08X)",
218  (unsigned int)iStatus);
219  goto EA_InitPipe_Exit_Tag;
220  }
221 
222 EA_InitPipe_Exit_Tag:
223  return (iStatus);
224 }

247 /* EA initialization */
248 /* */
249 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
250 int32 EA_InitApp()
251 {
252  int32 iStatus = CFE_SUCCESS;
253  int8 hasEvents = 0;
254 
255  iStatus = EA_InitEvent();
256  if (iStatus != CFE_SUCCESS)
257  {
258  (void) CFE_ES_WriteToSysLog("EA - Failed to init events (0x%08X)\n", (unsigned int)iStatus);
259  goto EA_InitApp_Exit_Tag;
260  }
261  else
262  {
263  hasEvents = 1;
264  }
265 
266  iStatus = EA_InitPipe();
267  if (iStatus != CFE_SUCCESS)
268  {
269  (void) CFE_EVS_SendEvent(EA_INIT_ERR_EID, CFE_EVS_ERROR,
270  "Failed to init pipes (0x%08X)",
271  (unsigned int)iStatus);
272  goto EA_InitApp_Exit_Tag;
273  }
274 
275  iStatus = EA_InitData();
276  if (iStatus != CFE_SUCCESS)
277  {
278  (void) CFE_EVS_SendEvent(EA_INIT_ERR_EID, CFE_EVS_ERROR,
279  "Failed to init data (0x%08X)",
280  (unsigned int)iStatus);
281  goto EA_InitApp_Exit_Tag;
282  }
283 
284 EA_InitApp_Exit_Tag:
285  if (iStatus == CFE_SUCCESS)
286  {
287  (void) CFE_EVS_SendEvent(EA_INIT_INF_EID, CFE_EVS_INFORMATION,
288  "Initialized. Version %d.%d.%d.%d",
289  EA_MAJOR_VERSION,
290  EA_MINOR_VERSION,
291  EA_REVISION,
292  EA_MISSION_REV);
293  }
294  else
295  {
296  if (hasEvents == 1)
297  {
298  (void) CFE_EVS_SendEvent(EA_INIT_ERR_EID, CFE_EVS_ERROR, "Application failed to initialize");
299  }
300  else
301  {
302  (void) CFE_ES_WriteToSysLog("EA - Application failed to initialize\n");
303  }
304  }
305 
306  return (iStatus);
307 }

310 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
311 /* */
312 /* Receive and Process Messages */
313 /* */
314 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
315 int32 EA_RcvMsg(int32 iBlocking)
316 {
317  int32 iStatus=CFE_SUCCESS;
318  CFE_SB_Msg_t* MsgPtr=NULL;
319  CFE_SB_MsgId_t MsgId;
320 
321  /* Stop Performance Log entry */
322  CFE_ES_PerfLogExit(EA_MAIN_TASK_PERF_ID);
323 
324  /* Wait for WakeUp messages from scheduler */
325  iStatus = CFE_SB_RcvMsg(&MsgPtr, EA_AppData.SchPipeId, iBlocking);
326 
327  /* Start Performance Log entry */
328  CFE_ES_PerfLogEntry(EA_MAIN_TASK_PERF_ID);
329 
330  if (iStatus == CFE_SUCCESS)
331  {
332  MsgId = CFE_SB_GetMsgId(MsgPtr);
333  switch (MsgId)
334  {
335  case EA_WAKEUP_MID:
336  EA_ProcessNewCmds();
337  break;
338 
339  case EA_SEND_HK_MID:
340  EA_ReportHousekeeping();
341  break;
342 
343  default:
344  (void) CFE_EVS_SendEvent(EA_MSGID_ERR_EID, CFE_EVS_ERROR,
345  "Recvd invalid SCH msgId (0x%04X)", (unsigned short)MsgId);
346  }
347  }
348  else if (iStatus == CFE_SB_NO_MESSAGE)
349  {
350  /* If there's no incoming message, you can do something here, or
351  * nothing. Note, this section is dead code only if the iBlocking arg
352  * is CFE_SB_PEND_FOREVER. */
353  iStatus = CFE_SUCCESS;
354  }
355  else if (iStatus == CFE_SB_TIME_OUT)
356  {
357  /* If there's no incoming message within a specified time (via the
358  * iBlocking arg, you can do something here, or nothing.
359  * Note, this section is dead code only if the iBlocking arg
360  * is CFE_SB_PEND_FOREVER. */
361  iStatus = CFE_SUCCESS;
362  }
363  else
364  {
365  /* This is an example of exiting on an error (either CFE_SB_BAD_ARGUMENT, or
366  * CFE_SB_PIPE_RD_ERROR).
367  */
368  (void) CFE_EVS_SendEvent(EA_PIPE_ERR_EID, CFE_EVS_ERROR,
369  "SB pipe read error (0x%08X), app will exit", (unsigned int)iStatus);
370  EA_AppData.uiRunStatus= CFE_ES_APP_ERROR;
371  }
372 
373  return (iStatus);
374 }

376 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
377 /* */
378 /* Process Incoming Commands */
379 /* */
380 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
381 void EA_ProcessNewCmds()
382 {
383  int iStatus = CFE_SUCCESS;
384  CFE_SB_Msg_t* CmdMsgPtr = NULL;
385  CFE_SB_MsgId_t CmdMsgId;
386 
387  /* Process command messages till the pipe is empty */
388  while (1)
389  {
390  iStatus = CFE_SB_RcvMsg(&CmdMsgPtr, EA_AppData.CmdPipeId, CFE_SB_POLL);
391  if(iStatus == CFE_SUCCESS)
392  {
393  CmdMsgId = CFE_SB_GetMsgId(CmdMsgPtr);
394  switch (CmdMsgId)
395  {
396  case EA_CMD_MID:
397  EA_ProcessNewAppCmds(CmdMsgPtr);
398  break;
399 
400  default:
401  /* Bump the command error counter for an unknown command.
402  * (This should only occur if it was subscribed to with this
403  * pipe, but not handled in this switch-case.) */
404  EA_AppData.HkTlm.usCmdErrCnt++;
405  (void) CFE_EVS_SendEvent(EA_MSGID_ERR_EID, CFE_EVS_ERROR,
406  "Recvd invalid CMD msgId (0x%04X)", (unsigned short)CmdMsgId);
407  break;
408  }
409  }
410  else if (iStatus == CFE_SB_NO_MESSAGE)
411  {
412  break;
413  }
414  else
415  {
416  (void) CFE_EVS_SendEvent(EA_PIPE_ERR_EID, CFE_EVS_ERROR,
417  "CMD pipe read error (0x%08X)", (unsigned int)iStatus);
418  EA_AppData.uiRunStatus = CFE_ES_APP_ERROR;
419  break;
420  }
421  }
422 }

425 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
426 /* */
427 /* Process EA Commands */
428 /* */
429 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
430 void EA_ProcessNewAppCmds(CFE_SB_Msg_t* MsgPtr)
431 {
432  uint32 uiCmdCode=0;
433  if (MsgPtr != NULL)
434  {
435  uiCmdCode = CFE_SB_GetCmdCode(MsgPtr);
436  switch (uiCmdCode)
437  {
438  case EA_NOOP_CC:
439  EA_AppData.HkTlm.usCmdCnt++;
440  (void) CFE_EVS_SendEvent(EA_CMD_NOOP_EID, CFE_EVS_INFORMATION,
441  "Recvd NOOP cmd (%u), Version %d.%d.%d.%d",
442  (unsigned int)uiCmdCode,
443  EA_MAJOR_VERSION,
444  EA_MINOR_VERSION,
445  EA_REVISION,
446  EA_MISSION_REV);
447  break;
448 
449  case EA_RESET_CC:
450  EA_AppData.HkTlm.usCmdCnt = 0;
451  EA_AppData.HkTlm.usCmdErrCnt = 0;
452  CFE_PSP_MemSet(EA_AppData.HkTlm.ActiveApp, '\0', EA_MAX_PATH_LEN);
453  EA_AppData.HkTlm.ActiveAppUtil = 0;
454  EA_AppData.HkTlm.ActiveAppPID = 0;
455  CFE_PSP_MemSet(EA_AppData.HkTlm.LastAppRun, '\0', EA_MAX_PATH_LEN);
456  EA_AppData.HkTlm.LastAppStatus = 0;
457  EA_AppData.ProcData.p_time = 0;
458  EA_AppData.ProcData.total_time = 0;
459  EA_AppData.ChildAppTaskInUse = FALSE;
460 
461  (void) CFE_EVS_SendEvent(EA_CMD_RESET_EID, CFE_EVS_INFORMATION,
462  "Recvd RESET cmd (%u)", (unsigned int)uiCmdCode);
463  break;
464 
465  case EA_START_APP_CC:
466  (void) CFE_EVS_SendEvent(EA_CMD_INF_EID, CFE_EVS_INFORMATION,
467  "Recvd Start App cmd (%u)", (unsigned int)uiCmdCode);
468  EA_StartApp(MsgPtr);
469  break;
470 
471  case EA_TERM_APP_CC:
472  (void) CFE_EVS_SendEvent(EA_CMD_INF_EID, CFE_EVS_INFORMATION,
473  "Recvd Terminate App cmd (%u)", (unsigned int)uiCmdCode);
474  EA_TermApp(MsgPtr);
475  break;
476 
477  case EA_PERFMON_CC:
478  (void) CFE_EVS_SendEvent(EA_CMD_INF_EID, CFE_EVS_INFORMATION,
479  "Recvd Perfmon cmd (%u)", (unsigned int)uiCmdCode);
480  EA_Perfmon();
481  break;
482 
483  default:
484  EA_AppData.HkTlm.usCmdErrCnt++;
485  (void) CFE_EVS_SendEvent(EA_CC_ERR_EID, CFE_EVS_ERROR,
486  "Recvd invalid cmd code (%u)", (unsigned int)uiCmdCode);
487  break;
488  }
489  }
490 }

492 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
493 /* */
494 /* EA Start App */
495 /* */
496 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
497 int32 EA_StartApp(CFE_SB_Msg_t* MsgPtr)
498 {
499  uint16 ExpectedLength = sizeof(EA_StartCmd_t);
500  uint32 ChildAppTaskID = 0;
501  int32 Status = CFE_ES_APP_ERROR;
502  EA_StartCmd_t *CmdPtr = 0;
503  os_fstat_t FileStats;
504 
505  /* Verify command packet length... */
506  if (EA_VerifyCmdLength(MsgPtr, ExpectedLength))
507  {
508  if (EA_AppData.ChildAppTaskInUse == FALSE)
509  {
510  CmdPtr = ((EA_StartCmd_t *) MsgPtr);
511 
512  /*
513  ** NUL terminate the very end of the filename string as a
514  ** safety measure
515  */
516  CmdPtr->Cmd[EA_MAX_PATH_LEN - 1] = '\0';
517  CmdPtr->Args[EA_MAX_PATH_LEN - 1] = '\0';
518 
519  /*
520  ** Check if specified interpreter exists
521  */
522  if(access(CmdPtr->Cmd, F_OK) != -1)
523  {
524  /*
525  ** Update ChildData with validated data
526  */
527  (void) strncpy(EA_AppData.ChildData.Cmd, CmdPtr->Cmd, EA_MAX_PATH_LEN);
528  (void) strncpy(EA_AppData.ChildData.Args, CmdPtr->Args, EA_MAX_PATH_LEN);
529 
530  Status = CFE_ES_CreateChildTask(&ChildAppTaskID,
531  EA_START_APP_TASK_NAME,
532  EA_StartAppCustom,
533  NULL,
534  CFE_ES_DEFAULT_STACK_SIZE,
535  EA_CHILD_TASK_PRIORITY,
536  EA_CHILD_TASK_FLAGS);
537  if (Status == CFE_SUCCESS)
538  {
539  (void) CFE_EVS_SendEvent (EA_CHILD_TASK_START_EID, CFE_EVS_DEBUG, "Created child task for app start");
540  EA_AppData.ChildAppTaskInUse = TRUE;
541  EA_AppData.ChildAppTaskID = ChildAppTaskID;
542  }
543  else /* Child task creation failed */
544  {
545  (void) CFE_EVS_SendEvent (EA_CHILD_TASK_START_ERR_EID,
546  CFE_EVS_ERROR,
547  "Create child tasked failed (%lu). Unable to start external application",
548  Status);
549 
550  EA_AppData.HkTlm.usCmdErrCnt++;
551  EA_AppData.ChildAppTaskInUse = FALSE;
552  CFE_PSP_MemSet(EA_AppData.ChildData.Cmd, '\0', EA_MAX_PATH_LEN);
553  CFE_PSP_MemSet(EA_AppData.ChildData.Args, '\0', EA_MAX_PATH_LEN);
554  }
555  }
556  else
557  {
558  EA_AppData.HkTlm.usCmdErrCnt++;
559  (void) CFE_EVS_SendEvent(EA_APP_ERR_EID, CFE_EVS_ERROR,
560  "Specified app does not exist");
561  }
562  }
563  else
564  {
565  /* Send event that we can't start another task right now */
566  (void) CFE_EVS_SendEvent (EA_CHILD_TASK_START_ERR_EID,
567  CFE_EVS_ERROR,
568  "Create child tasked failed. A child task is in use");
569 
570  EA_AppData.HkTlm.usCmdErrCnt++;
571  }
572  }
573  return (Status);
574 }

581 void EA_TermApp(CFE_SB_Msg_t* MsgPtr)

583  uint16 ExpectedLength = sizeof(EA_NoArgCmd_t);
584 
585  /*
586  ** Verify command packet length

654  "msgLen=%d, expectedLen=%d",

677  }

680  CFE_ES_PerfLogEntry(EA_MAIN_TASK_PERF_ID);
681

683  if (iStatus == CFE_SUCCESS)
684  {
685  iStatus = EA_InitApp();
686  }

689  {
690  /* Do not perform performance monitoring on startup sync */
691  CFE_ES_PerfLogExit(EA_MAIN_TASK_PERF_ID);
692  CFE_ES_WaitForStartupSync(EA_STARTUP_TIMEOUT_MSEC);
693  CFE_ES_PerfLogEntry(EA_MAIN_TASK_PERF_ID);
694  }
695  else
696  {
697  EA_AppData.uiRunStatus = CFE_ES_APP_ERROR;
698  }

702  {
703  EA_RcvMsg(EA_SCH_PIPE_PEND_TIME);

707  EA_Perfmon();

711  CFE_ES_PerfLogExit(EA_MAIN_TASK_PERF_ID);
