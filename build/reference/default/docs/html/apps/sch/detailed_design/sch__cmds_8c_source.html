59 /*******************************************************************
60 **
61 ** SCH_AppPipe
62 **
63 ** NOTE: For complete prolog information, see 'sch_cmds.h'
64 ********************************************************************/
65 
66 int32 SCH_AppPipe(CFE_SB_MsgPtr_t MessagePtr)
67 {
68  int32 Result = CFE_SUCCESS;
69  CFE_SB_MsgId_t MessageID = 0;
70  uint16 CommandCode = 0;
71 
72  MessageID = CFE_SB_GetMsgId(MessagePtr);
73  switch (MessageID)
74  {
75  /*
76  ** Housekeeping telemetry request
77  */
78  case SCH_SEND_HK_MID:
79  Result = SCH_HousekeepingCmd(MessagePtr);
80  break;
81 
82  /*
83  ** SCH ground commands
84  */
85  case SCH_CMD_MID:
86 
87  CommandCode = CFE_SB_GetCmdCode(MessagePtr);
88  switch (CommandCode)
89  {
90  case SCH_NOOP_CC:
91  SCH_NoopCmd(MessagePtr);
92  break;
93 
94  case SCH_RESET_CC:
95  SCH_ResetCmd(MessagePtr);
96  break;
97 
98  case SCH_ENABLE_CC:
99  SCH_EnableCmd(MessagePtr);
100  break;
101 
102  case SCH_DISABLE_CC:
103  SCH_DisableCmd(MessagePtr);
104  break;
105 
106  case SCH_ENABLE_GROUP_CC:
107  SCH_EnableGroupCmd(MessagePtr);
108  break;
109 
110  case SCH_DISABLE_GROUP_CC:
111  SCH_DisableGroupCmd(MessagePtr);
112  break;
113 
114  case SCH_ENABLE_SYNC_CC:
115  SCH_EnableSyncCmd(MessagePtr);
116  break;
117 
118  case SCH_SEND_DIAG_TLM_CC:
119  SCH_SendDiagTlmCmd(MessagePtr);
120  break;
121 
122  /*
123  ** SCH ground commands with unknown command codes...
124  */
125  default:
126  CFE_EVS_SendEvent(SCH_CC_ERR_EID, CFE_EVS_ERROR,
127  "Invalid command code: ID = 0x%04X, CC = %d",
128  MessageID, CommandCode);
129 
130  SCH_AppData.ErrCounter++;
131  break;
132  }
133  break;
134 
135  /*
136  ** Unknown message ID's
137  */
138  default:
139  CFE_EVS_SendEvent(SCH_MD_ERR_EID, CFE_EVS_ERROR,
140  "Msg with Invalid message ID Rcvd -- ID = 0x%04X",
141  MessageID);
142  break;
143  }
144 
145  return(Result);
146 
147 } /* End of SCH_AppPipe() */

308 void SCH_EnableCmd(CFE_SB_MsgPtr_t MessagePtr)
309 {
310  boolean GoodCommand = FALSE;
311  SCH_EntryCmd_t *EnableCmd = NULL;
312  uint16 SlotNumber = 0;
313  uint16 EntryNumber = 0;
314  uint16 TableIndex = 0;
315 
316  /*
317  ** Extract contents of command
318  */
319  EnableCmd = (SCH_EntryCmd_t *) MessagePtr;
320 
321  if(SCH_VerifyCmdLength(MessagePtr, sizeof(SCH_EntryCmd_t)) == SCH_SUCCESS)
322  {
323  SlotNumber = EnableCmd->SlotNumber;
324  EntryNumber = EnableCmd->EntryNumber;
325  TableIndex = (SlotNumber * SCH_ENTRIES_PER_SLOT) + EntryNumber;
326 
327  if ((SlotNumber >= SCH_TOTAL_SLOTS) || (EntryNumber >= SCH_ENTRIES_PER_SLOT))
328  {
329  /*
330  ** Invalid command packet argument
331  */
332  CFE_EVS_SendEvent(SCH_ENABLE_CMD_ARG_ERR_EID, CFE_EVS_ERROR,
333  "ENABLE cmd: invalid argument, slot=%d (<%d), entry=%d (<%d)",
334  SlotNumber, SCH_TOTAL_SLOTS,
335  EntryNumber, SCH_ENTRIES_PER_SLOT);
336  }
337  else if ((SCH_AppData.ScheduleTable[TableIndex].EnableState != SCH_ENABLED) &&
338  (SCH_AppData.ScheduleTable[TableIndex].EnableState != SCH_DISABLED))
339  {
340  /*
341  ** Invalid schedule table enable state (unused or corrupt)
342  */
343  CFE_EVS_SendEvent(SCH_ENABLE_CMD_ENTRY_ERR_EID, CFE_EVS_ERROR,
344  "ENABLE command: invalid state = %d, slot = %d, entry = %d",
345  SCH_AppData.ScheduleTable[TableIndex].EnableState,
346  SlotNumber, EntryNumber);
347  }
348  else
349  {
350  /*
351  ** Success
352  */
353  GoodCommand = TRUE;
354 
355  SCH_AppData.ScheduleTable[TableIndex].EnableState = SCH_ENABLED;
356  CFE_TBL_Modified(SCH_AppData.ScheduleTableHandle);
357 
358  CFE_EVS_SendEvent(SCH_ENABLE_CMD_EID, CFE_EVS_DEBUG,
359  "ENABLE command: slot = %d, entry = %d",
360  SlotNumber, EntryNumber );
361  }
362  }
363 
364  SCH_PostCommandResult(GoodCommand);
365 
366  return;
367 
368 } /* End of SCH_EnableCmd() */

448 void SCH_EnableGroupCmd(CFE_SB_MsgPtr_t MessagePtr)
449 {
450  boolean GoodCommand = FALSE;
451  uint32 TblGroupNumber = 0;
452  uint32 TblMultiGroup = 0;
453  int32 LoopCount = 0;
454  int32 MatchCount = 0;
455  SCH_GroupCmd_t *EnableCmd = NULL;
456  SCH_ScheduleEntry_t *TableEntry = NULL;
457  uint32 CmdGroupNumber = 0;
458  uint32 CmdMultiGroup = 0;
459 
460  /*
461  ** Extract command parameters
462  */
463  EnableCmd = (SCH_GroupCmd_t *) MessagePtr;
464 
465  if(SCH_VerifyCmdLength(MessagePtr, sizeof(SCH_GroupCmd_t)) == SCH_SUCCESS)
466  {
467  TableEntry = &SCH_AppData.ScheduleTable[0];
468  CmdGroupNumber = EnableCmd->GroupData & SCH_GROUP_NUMBER_BIT_MASK;
469  CmdMultiGroup = EnableCmd->GroupData & SCH_MULTI_GROUP_BIT_MASK;
470 
471  if ((CmdGroupNumber == SCH_UNUSED) && (CmdMultiGroup == SCH_UNUSED))
472  {
473  /*
474  ** No groups selected
475  */
476  CFE_EVS_SendEvent(SCH_ENA_GRP_CMD_ERR_EID, CFE_EVS_ERROR,
477  "ENABLE GROUP command: invalid argument, no groups selected");
478  }
479  else
480  {
481  /*
482  ** Search entire schedule table for group members
483  */
484  for (LoopCount = 0; LoopCount < SCH_TABLE_ENTRIES; LoopCount++)
485  {
486  /*
487  ** Skip unused table entries
488  */
489  if (TableEntry->GroupData != SCH_UNUSED)
490  {
491  TblGroupNumber = TableEntry->GroupData & SCH_GROUP_NUMBER_BIT_MASK;
492  TblMultiGroup = TableEntry->GroupData & SCH_MULTI_GROUP_BIT_MASK;
493 
494  /*
495  ** Look for matching table entries
496  */
497  if (((CmdGroupNumber != SCH_UNUSED) && (CmdGroupNumber == TblGroupNumber)) ||
498  ((CmdMultiGroup & TblMultiGroup) != SCH_UNUSED))
499  {
500  MatchCount++;
501  TableEntry->EnableState = SCH_ENABLED;
502  }
503  }
504 
505  TableEntry++;
506  }
507 
508  if (MatchCount > 0)
509  {
510  CFE_TBL_Modified(SCH_AppData.ScheduleTableHandle);
511  CFE_EVS_SendEvent(SCH_ENA_GRP_CMD_EID, CFE_EVS_DEBUG,
512  "ENABLE GROUP command: match count = %d",
513  (int)MatchCount);
514  GoodCommand = TRUE;
515  }
516  else
517  {
518  CFE_EVS_SendEvent(SCH_ENA_GRP_NOT_FOUND_ERR_EID, CFE_EVS_ERROR,
519  "ENABLE GROUP command: Neither Group %d nor Multi-Group 0x%06X found",
520  (int)(CmdGroupNumber>>24), (unsigned int)CmdMultiGroup);
521  }
522  }
523  }
524 
525  SCH_PostCommandResult(GoodCommand);
526 
527  return;
528 
529 } /* End of SCH_EnableGroupCmd() */

655 /*******************************************************************
656 **
657 ** SCH_SendDiagTlmCmd
658 **
659 ** NOTE: For complete prolog information, see above
660 ********************************************************************/
661 
662 void SCH_SendDiagTlmCmd(CFE_SB_MsgPtr_t MessagePtr)
663 {
664  boolean GoodCommand = FALSE;
665  uint32 TblIndex = 0;
666  uint32 WordIndex = 0;
667  uint32 BitIndex = 0;
668  SCH_ScheduleEntry_t *TableEntry = NULL;
669 
670  if(SCH_VerifyCmdLength(MessagePtr, sizeof(SCH_NoArgsCmd_t)) == SCH_SUCCESS)
671  {
672  GoodCommand = TRUE;
673 
674  /* Zero out the previous entry states */
675  CFE_PSP_MemSet(&SCH_AppData.DiagPacket.EntryStates[0], 0x0, SCH_NUM_STATUS_BYTES_REQD);
676 
677  for (TblIndex=0; TblIndex<SCH_TABLE_ENTRIES; TblIndex++)
678  {
679  TableEntry = &SCH_AppData.ScheduleTable[TblIndex];
680  WordIndex = TblIndex/8; /* 8 states can fit in each word */
681  BitIndex = (7-(TblIndex%8))*2; /* Determine bit pair, MSBs contain lowest index */
682 
683  if (TableEntry->EnableState == SCH_ENABLED)
684  {
685  SCH_AppData.DiagPacket.EntryStates[WordIndex] |= (1 << BitIndex);
686  SCH_AppData.DiagPacket.MsgIDs[TblIndex] =
687  CFE_SB_GetMsgId((CFE_SB_MsgPtr_t)&SCH_AppData.MessageTable[SCH_AppData.ScheduleTable[TblIndex].MessageIndex]);
688  }
689  else if (TableEntry->EnableState == SCH_DISABLED)
690  {
691  SCH_AppData.DiagPacket.EntryStates[WordIndex] |= (2 << BitIndex);
692  SCH_AppData.DiagPacket.MsgIDs[TblIndex] =
693  CFE_SB_GetMsgId((CFE_SB_MsgPtr_t)&SCH_AppData.MessageTable[SCH_AppData.ScheduleTable[TblIndex].MessageIndex]);
694  }
695  else
696  {
697  SCH_AppData.DiagPacket.MsgIDs[TblIndex] = 0x0000;
698  }
699  }
700  /*
701  ** Timestamp and send diagnostic packet
702  */
703  CFE_SB_TimeStampMsg((CFE_SB_Msg_t *) &SCH_AppData.DiagPacket);
704  CFE_SB_SendMsg((CFE_SB_Msg_t *) &SCH_AppData.DiagPacket);
705 
706  CFE_EVS_SendEvent(SCH_SEND_DIAG_CMD_EID, CFE_EVS_DEBUG,
707  "Transmitting Diagnostic Message");
708  }
709 
710  SCH_PostCommandResult(GoodCommand);
711 
712  return;
713 
714 } /* End of SCH_SendDiagTlmCmd() */
