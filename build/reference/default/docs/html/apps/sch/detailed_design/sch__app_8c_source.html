129 ** SCH_AppMain

131 ** NOTE: For complete prolog information, see 'sch_app.h'

133 
134 void SCH_AppMain(void)
135 {
136  int32 Status = CFE_SUCCESS;
137  uint32 RunStatus = CFE_ES_APP_RUN;
138 
139  /*
140  ** Performance Log (start time counter)
141  */
142  CFE_ES_PerfLogEntry(SCH_APPMAIN_PERF_ID);
143 
144  /*
145  ** Register application
146  */
147  Status = CFE_ES_RegisterApp();
148 
149  /*
150  ** Perform application specific initialization
151  */
152  if (Status == CFE_SUCCESS)
153  {
154  Status = SCH_AppInit();
155  }
156 
157  /* If no errors were detected during initialization, then wait for everyone to start */
158  if (Status == CFE_SUCCESS)
159  {
160  Status = SCH_CustomLateInit();
161  if (Status != CFE_SUCCESS)
162  {
163  CFE_EVS_SendEvent(SCH_MAJOR_FRAME_SUB_ERR_EID, CFE_EVS_ERROR,
164  "Error initializing Timers (RC=0x%08X)",
165  (unsigned int)Status);
166  }
167  }
168 
169  /*
170  ** Check for start-up error
171  */
172  if (Status != CFE_SUCCESS)
173  {
174  /*
175  ** Set request to terminate main loop
176  */
177  RunStatus = CFE_ES_APP_ERROR;
178  }
179 
180  /*
181  ** Main process loop
182  */
183  while (CFE_ES_RunLoop(&RunStatus))
184  {
185  /*
186  ** Performance Log (stop time counter)
187  */
188  CFE_ES_PerfLogExit(SCH_APPMAIN_PERF_ID);
189 
190  /*
191  ** Wait for the next slot (Major or Minor Frame)
192  */
193  Status = OS_BinSemTake(SCH_AppData.TimeSemaphore);
194 
195  /*
196  ** Performance Log (start time counter)
197  */
198  CFE_ES_PerfLogEntry(SCH_APPMAIN_PERF_ID);
199 
200  /*
201  ** Report if during the previous frame the major has determined to be unstable
202  */
203  if (SCH_AppData.IgnoreMajorFrame == TRUE)
204  {
205  if (SCH_AppData.IgnoreMajorFrameMsgSent == FALSE)
206  {
207  CFE_EVS_SendEvent(SCH_NOISY_MAJOR_FRAME_ERR_EID, CFE_EVS_ERROR,
208  "Major Frame Sync too noisy (Slot %d). Disabling synchronization.",
209  SCH_AppData.MinorFramesSinceTone);
210  SCH_AppData.IgnoreMajorFrameMsgSent = TRUE;
211  }
212  }
213  else
214  {
215  SCH_AppData.IgnoreMajorFrameMsgSent = FALSE;
216  }
217 
218  /*
219  ** Process schedule table activities
220  */
221 #if SCH_LIB_PRESENCE == 1
222  if ((Status == OS_SUCCESS) &&
223  (SCH_GetProcessingState() == TRUE))
224  {
225  Status = SCH_ProcessScheduleTable();
226  }
227 #else
228  if (Status == OS_SUCCESS)
229  {
230  Status = SCH_ProcessScheduleTable();
231  }
232 #endif
233 
234  /*
235  ** Note: If there were some reason to exit the task
236  ** normally (without error) then we would set
237  ** RunStatus = CFE_ES_APP_EXIT
238  */
239  if (Status != SCH_SUCCESS)
240  {
241  /*
242  ** Set request to terminate main loop
243  */
244  RunStatus = CFE_ES_APP_ERROR;
245  }
246 
247  SCH_CheckDeadlines();
248 #ifdef SCH_RTM_SUPPORTED
249  OS_RtmEndFrame();
250 #endif
251  } /* End of while */
252 
253  /*
254  ** Check for "fatal" process error
255  */
256  if ((Status != OS_SUCCESS) || (Status != SCH_SUCCESS))
257  {
258  /*
259  ** Send an event describing the reason for the termination
260  */
261  CFE_EVS_SendEvent(SCH_APP_EXIT_EID, CFE_EVS_CRITICAL,
262  "SCH App: terminating, err = 0x%08X", (unsigned int)Status);
263 
264  /*
265  ** In case cFE Event Services is not working
266  */
267  CFE_ES_WriteToSysLog("SCH App terminating, err = 0x%08X\n", (unsigned int)Status);
268  }
269 
270  /*
271  ** Perform any custom cleanup
272  */
273  SCH_CustomCleanup();
274 
275  /*
276  ** Performance Log (stop time counter)
277  */
278  CFE_ES_PerfLogExit(SCH_APPMAIN_PERF_ID);
279 
280  /*
281  ** Let cFE kill the task (and any child tasks)
282  */
283  CFE_ES_ExitApp(RunStatus);
284 
285 } /* End of SH_AppMain() */

289 uint32 SCH_GetOutstandingActivityCount(uint32 Slot)
290 {
291  uint32 count = 0;
292  uint32 i = 0;
293 
294 
295  OS_MutSemTake(SCH_AppData.ADChildTaskMutex);
296  for(i = 0; i < SCH_DEADLINES_PER_SLOT; ++i)
297  {
298  SCH_ActivityDeadlineStatus_t *status = &SCH_AppData.DeadlineTable->Slot[Slot].Status[i];
299  if(status->State == SCH_DEADLINE_STATE_PENDED)
300  {
301  count++;
302  }
303  }
304  OS_MutSemGive(SCH_AppData.ADChildTaskMutex);
305 
306  return count;
307 }

784 /*******************************************************************
785 **
786 ** SCH_ProcessScheduleTable
787 **
788 ** NOTE: For complete prolog information, see above
789 ********************************************************************/
790 
791 int32 SCH_ProcessScheduleTable(void)
792 {
793  uint32 CurrentSlot;
794  uint32 ProcessCount;
795  int32 Result = CFE_SUCCESS;
796 
797  /*
798  ** Get the slot we should be at
799  */
800  CurrentSlot = SCH_CustomGetCurrentSlotNumber();
801 
802  /*
803  ** Compute the number of slots we need to process (watch for rollover)
804  */
805  if (CurrentSlot < SCH_AppData.NextSlotNumber)
806  {
807  ProcessCount = SCH_TOTAL_SLOTS - SCH_AppData.NextSlotNumber;
808  ProcessCount += (CurrentSlot + 1);
809  }
810  else
811  {
812  ProcessCount = (CurrentSlot - SCH_AppData.NextSlotNumber) + 1;
813  }
814 
815  /*
816  ** Correct for the following conditions observed when minor frame driven
817  ** by a clock with poor accuracy
818  **
819  ** 1) Wake up a little too late for just 1 slot
820  ** symptom = multi slots event followed by same slot event
821  **
822  ** 2) Wake up a little too early for just 1 slot
823  ** symptom = same slot event followed by multi slots event
824  */
825  if (ProcessCount == 2)
826  {
827  /*
828  ** If we want to do 2 slots but last time was OK then assume we
829  ** are seeing condition #1 above. By doing just 1 slot now,
830  ** there will still be 1 to do when the next wakeup occurs
831  ** and we will avoid both events. But, if we really are in
832  ** a delayed state, we will process both slots when we wake
833  ** up next time because then the last time will NOT be OK.
834  */
835  if (SCH_AppData.LastProcessCount == 1)
836  {
837  ProcessCount = 1;
838  }
839  SCH_AppData.LastProcessCount = 2;
840  }
841  else if (ProcessCount == SCH_TOTAL_SLOTS)
842  {
843  /*
844  ** Same as previous comment except in reverse order.
845  */
846  if (SCH_AppData.LastProcessCount != SCH_TOTAL_SLOTS)
847  {
848  ProcessCount = 1;
849  }
850  SCH_AppData.LastProcessCount = SCH_TOTAL_SLOTS;
851  }
852  else
853  {
854  SCH_AppData.LastProcessCount = ProcessCount;
855  }
856 
857  /*
858  ** If current slot = next slot - 1, assume current slot did not increment
859  */
860  if (ProcessCount == SCH_TOTAL_SLOTS)
861  {
862  SCH_AppData.SameSlotCount++;
863 
864  CFE_EVS_SendEvent(SCH_SAME_SLOT_EID, CFE_EVS_DEBUG,
865  "Slot did not increment: slot = %d",
866  (int)CurrentSlot);
867  ProcessCount = 0;
868  }
869 
870  /*
871  ** If we are too far behind, jump forward and do just the current slot
872  */
873  if (ProcessCount > SCH_MAX_LAG_COUNT)
874  {
875  SCH_AppData.SkippedSlotsCount++;
876 
877  CFE_EVS_SendEvent(SCH_SKIPPED_SLOTS_EID, CFE_EVS_ERROR,
878  "Slots skipped: slot = %d, count = %d",
879  SCH_AppData.NextSlotNumber, (int)(ProcessCount - 1));
880 
881  /*
882  ** Update the pass counter if we are skipping the rollover slot
883  */
884  if (CurrentSlot < SCH_AppData.NextSlotNumber)
885  {
886  SCH_AppData.TablePassCount++;
887  }
888 
889  /*
890  ** Process ground commands if we are skipping the time synch slot
891  ** NOTE: This assumes the Time Synch Slot is the LAST Schedule slot
892  ** (see definition of SCH_TIME_SYNC_SLOT in sch_app.h)
893  ** Ground commands should only be processed at the end of the schedule table
894  ** so that Group Enable/Disable commands do not change the state of entries
895  ** in the middle of a schedule.
896  */
897  if ((SCH_AppData.NextSlotNumber + ProcessCount) > SCH_TIME_SYNC_SLOT)
898  {
899  Result = SCH_ProcessCommands();
900  }
901 
902  SCH_AppData.NextSlotNumber = CurrentSlot;
903  ProcessCount = 1;
904  }
905 
906  /*
907  ** Don't try to catch up all at once, just do a couple
908  */
909  if (ProcessCount > SCH_MAX_SLOTS_PER_WAKEUP)
910  {
911  ProcessCount = SCH_MAX_SLOTS_PER_WAKEUP;
912  }
913 
914  /*
915  ** Keep track of multi-slot processing
916  */
917  if (ProcessCount > 1)
918  {
919  SCH_AppData.MultipleSlotsCount++;
920 
921  /* Generate an event message if not syncing to MET or when there is more than two being processed */
922  if ((ProcessCount > SCH_AppData.WorstCaseSlotsPerMinorFrame) || (SCH_AppData.SyncToMET == SCH_NOT_SYNCHRONIZED))
923  {
924  CFE_EVS_SendEvent(SCH_MULTI_SLOTS_EID, CFE_EVS_INFORMATION,
925  "Multiple slots processed: slot = %d, count = %d",
926  SCH_AppData.NextSlotNumber, (int)ProcessCount);
927  }
928  }
929 
930  /*
931  ** Process the slots (most often this will be just one)
932  */
933  while ((ProcessCount != 0) && (Result == CFE_SUCCESS))
934  {
935  Result = SCH_ProcessNextSlot();
936 
937  ProcessCount--;
938  }
939 
940  return(Result);
941 
942 } /* End of SCH_ProcessScheduleTable() */
