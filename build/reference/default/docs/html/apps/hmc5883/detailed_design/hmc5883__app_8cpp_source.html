85 /* Initialize event tables. */
86 /* */
87 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
88 int32 HMC5883::InitEvent()
89 {
90  int32 iStatus = CFE_SUCCESS;
91  int32 ind = 0;
92  int32 customEventCount = 0;
93 
94  CFE_EVS_BinFilter_t EventTbl[CFE_EVS_MAX_EVENT_FILTERS];
95 
96  /* Initialize the event filter table.
97  * Note: 0 is the CFE_EVS_NO_FILTER mask and event 0 is reserved (not used) */
98  memset(EventTbl, 0x00, sizeof(EventTbl));
99 
100  /* CFE_EVS_MAX_EVENT_FILTERS limits the number of filters per app.
101  * An explicit CFE_EVS_NO_FILTER (the default) has been provided as
102  * an example. */
103  EventTbl[ ind].EventID = HMC5883_RESERVED_EID;
104  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
105  EventTbl[ ind].EventID = HMC5883_READ_ERR_EID;
106  EventTbl[ind++].Mask = CFE_EVS_FIRST_16_STOP;
107 
108  /* Add custom events to the filter table */
109  customEventCount = HMC5883_Custom_Init_EventFilters(ind, EventTbl);
110 
111  if(-1 == customEventCount)
112  {
113  iStatus = CFE_EVS_APP_FILTER_OVERLOAD;
114  (void) CFE_ES_WriteToSysLog("Failed to init custom event filters (0x%08X)\n", (unsigned int)iStatus);
115  goto end_of_function;
116  }
117 
118  /* Register the table with CFE */
119  iStatus = CFE_EVS_Register(EventTbl, (ind + customEventCount), CFE_EVS_BINARY_FILTER);
120  if (iStatus != CFE_SUCCESS)
121  {
122  (void) CFE_ES_WriteToSysLog("HMC5883 - Failed to register with EVS (0x%08lX)\n", iStatus);
123  }
124 
125 end_of_function:
126 
127  return (iStatus);
128 }

136 int32 HMC5883::InitPipe()
137 {
138  int32 iStatus = CFE_SUCCESS;
139 
140  /* Init schedule pipe and subscribe to wakeup messages */

144  if (iStatus == CFE_SUCCESS)
145  {
146  iStatus = CFE_SB_SubscribeEx(HMC5883_WAKEUP_MID, SchPipeId, CFE_SB_Default_Qos, HMC5883_WAKEUP_MID_MAX_MSG_COUNT);

153  }
154 
155  iStatus = CFE_SB_SubscribeEx(HMC5883_SEND_HK_MID, SchPipeId, CFE_SB_Default_Qos, HMC5883_SEND_HK_MID_MAX_MSG_COUNT);
156  if (iStatus != CFE_SUCCESS)
157  {
158  (void) CFE_EVS_SendEvent(HMC5883_SUBSCRIBE_ERR_EID, CFE_EVS_ERROR,

162  }
163  }
164  else

167  "Failed to create SCH pipe (0x%08lX)",

169  goto HMC5883_InitPipe_Exit_Tag;
170  }
171 
172  /* Init command pipe and subscribe to command messages */

184  "CMD Pipe failed to subscribe to HMC5883_CMD_MID. (0x%08lX)",

243 int32 HMC5883::InitApp()
244 {
245  int32 iStatus = CFE_SUCCESS;
246  int8 hasEvents = 0;
247  boolean returnBool = TRUE;
248 
249  iStatus = InitEvent();
250  if (iStatus != CFE_SUCCESS)
251  {
252  (void) CFE_ES_WriteToSysLog("HMC5883 - Failed to init events (0x%08lX)\n", iStatus);
253  goto HMC5883_InitApp_Exit_Tag;
254  }
255  else
256  {
257  hasEvents = 1;
258  }
259 
260  iStatus = InitPipe();
261  if (iStatus != CFE_SUCCESS)
262  {
263  goto HMC5883_InitApp_Exit_Tag;
264  }
265 
266  iStatus = InitConfigTbl();
267  if (iStatus != CFE_SUCCESS)
268  {
269  goto HMC5883_InitApp_Exit_Tag;
270  }
271 
272  InitData();
273 
274  returnBool = HMC5883_Custom_Init();
275  if (FALSE == returnBool)
276  {
277  iStatus = -1;
278  (void) CFE_EVS_SendEvent(HMC5883_INIT_ERR_EID, CFE_EVS_ERROR,
279  "Custom init failed");
280  goto HMC5883_InitApp_Exit_Tag;
281  }
282  returnBool = HMC5883_Custom_ValidateID();
283  if (FALSE == returnBool)
284  {
285  iStatus = -1;
286  (void) CFE_EVS_SendEvent(HMC5883_INIT_ERR_EID, CFE_EVS_ERROR,
287  "HMC5883 Device failed validate ID");
288  goto HMC5883_InitApp_Exit_Tag;
289  }
290  returnBool = SelfCalibrate(&Diag.Calibration);
291  if (FALSE == returnBool)
292  {
293  iStatus = -1;
294  (void) CFE_EVS_SendEvent(HMC5883_INIT_ERR_EID, CFE_EVS_ERROR,
295  "HMC5883 Device failed calibration");
296  goto HMC5883_InitApp_Exit_Tag;
297  }
298  returnBool = HMC5883_Custom_Set_Range(Diag.Conversion.ConfigB);
299  if (FALSE == returnBool)
300  {
301  iStatus = -1;
302  (void) CFE_EVS_SendEvent(HMC5883_INIT_ERR_EID, CFE_EVS_ERROR,
303  "HMC5883 Device failed set range");
304  goto HMC5883_InitApp_Exit_Tag;
305  }
306  returnBool = HMC5883_Custom_Check_Range(Diag.Conversion.ConfigB);
307  if (FALSE == returnBool)
308  {
309  iStatus = -1;
310  (void) CFE_EVS_SendEvent(HMC5883_INIT_ERR_EID, CFE_EVS_ERROR,
311  "HMC5883 Device failed check range");
312  goto HMC5883_InitApp_Exit_Tag;
313  }
314  returnBool = HMC5883_Custom_Set_Config(Diag.Conversion.ConfigA);
315  if (FALSE == returnBool)
316  {
317  iStatus = -1;
318  (void) CFE_EVS_SendEvent(HMC5883_INIT_ERR_EID, CFE_EVS_ERROR,
319  "HMC5883 Device failed set config");
320  goto HMC5883_InitApp_Exit_Tag;
321  }
322  returnBool = HMC5883_Custom_Check_Config(Diag.Conversion.ConfigA);
323  if (FALSE == returnBool)
324  {
325  iStatus = -1;
326  (void) CFE_EVS_SendEvent(HMC5883_INIT_ERR_EID, CFE_EVS_ERROR,
327  "HMC5883 Device failed check config");
328  goto HMC5883_InitApp_Exit_Tag;
329  }
330 
331  /* Get the rotation from custom for diag */
332  HMC5883_Get_Rotation(&Diag.Calibration.Rotation);
333 
334  /* Register the cleanup callback */
335  iStatus = OS_TaskInstallDeleteHandler(&HMC5883_CleanupCallback);
336  if (iStatus != CFE_SUCCESS)
337  {
338  (void) CFE_EVS_SendEvent(HMC5883_INIT_ERR_EID, CFE_EVS_ERROR,
339  "Failed to init register cleanup callback (0x%08X)",
340  (unsigned int)iStatus);
341  goto HMC5883_InitApp_Exit_Tag;
342  }
343 
344 HMC5883_InitApp_Exit_Tag:
345  if (iStatus == CFE_SUCCESS)
346  {
347  (void) CFE_EVS_SendEvent(HMC5883_INIT_INF_EID, CFE_EVS_INFORMATION,
348  "Initialized. Version %d.%d.%d.%d",
349  HMC5883_MAJOR_VERSION,
350  HMC5883_MINOR_VERSION,
351  HMC5883_REVISION,
352  HMC5883_MISSION_REV);
353  /* Set the app state to initialized */
354  oHMC5883.HkTlm.State = HMC5883_INITIALIZED;
355  }
356  else
357  {
358  if (hasEvents == 1)
359  {
360  (void) CFE_ES_WriteToSysLog("HMC5883 - Application failed to initialize\n");
361  }
362  }
363 
364  return (iStatus);
365 }

370 /* Receive and Process Messages */

372 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
373 int32 HMC5883::RcvSchPipeMsg(int32 iBlocking)
374 {
375  int32 iStatus = CFE_SUCCESS;
376  CFE_SB_Msg_t* MsgPtr = NULL;
377  CFE_SB_MsgId_t MsgId;
378 
379  /* Stop Performance Log entry */
380  CFE_ES_PerfLogExit(HMC5883_MAIN_TASK_PERF_ID);
381 
382  /* Wait for WakeUp messages from scheduler */
383  iStatus = CFE_SB_RcvMsg(&MsgPtr, SchPipeId, iBlocking);
384 
385  /* Start Performance Log entry */
386  CFE_ES_PerfLogEntry(HMC5883_MAIN_TASK_PERF_ID);
387 
388  if (iStatus == CFE_SUCCESS)
389  {
390  MsgId = CFE_SB_GetMsgId(MsgPtr);
391  switch (MsgId)
392  {
393  case HMC5883_WAKEUP_MID:
394  {
395  ReadDevice();
396  SendSensorMagMsg();
397  break;
398  }
399  case HMC5883_SEND_HK_MID:
400  {
401  ProcessCmdPipe();
402  ReportHousekeeping();
403  break;
404  }
405  default:
406  {
407  (void) CFE_EVS_SendEvent(HMC5883_MSGID_ERR_EID, CFE_EVS_ERROR,
408  "Recvd invalid SCH msgId (0x%04X)", MsgId);
409  }
410  }
411  }
412  else if (iStatus == CFE_SB_NO_MESSAGE)
413  {
414  /* TODO: If there's no incoming message, you can do something here, or
415  * nothing. Note, this section is dead code only if the iBlocking arg
416  * is CFE_SB_PEND_FOREVER. */
417  iStatus = CFE_SUCCESS;
418  }
419  else if (iStatus == CFE_SB_TIME_OUT)
420  {
421  /* TODO: If there's no incoming message within a specified time (via the
422  * iBlocking arg, you can do something here, or nothing.
423  * Note, this section is dead code only if the iBlocking arg
424  * is CFE_SB_PEND_FOREVER. */
425  iStatus = CFE_SUCCESS;
426  }
427  else
428  {
429  (void) CFE_EVS_SendEvent(HMC5883_RCVMSG_ERR_EID, CFE_EVS_ERROR,
430  "SCH pipe read error (0x%08lX).", iStatus);
431  }
432 
433  return (iStatus);
434 }

439 /* Process Incoming Commands */
440 /* */
441 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
442 void HMC5883::ProcessCmdPipe()
443 {
444  int32 iStatus = CFE_SUCCESS;
445  CFE_SB_Msg_t* CmdMsgPtr = NULL;
446  CFE_SB_MsgId_t CmdMsgId;
447 
448  /* Process command messages until the pipe is empty */
449  while (1)
450  {
451  iStatus = CFE_SB_RcvMsg(&CmdMsgPtr, CmdPipeId, CFE_SB_POLL);
452  if(iStatus == CFE_SUCCESS)
453  {
454  CmdMsgId = CFE_SB_GetMsgId(CmdMsgPtr);
455  switch (CmdMsgId)
456  {
457  case HMC5883_CMD_MID:
458  {
459  ProcessAppCmds(CmdMsgPtr);
460  break;
461  }
462  default:
463  {
464  /* Bump the command error counter for an unknown command.
465  * (This should only occur if it was subscribed to with this
466  * pipe, but not handled in this switch-case.) */
467  HkTlm.usCmdErrCnt++;
468  (void) CFE_EVS_SendEvent(HMC5883_MSGID_ERR_EID, CFE_EVS_ERROR,
469  "Recvd invalid CMD msgId (0x%04X)", (unsigned short)CmdMsgId);
470  break;
471  }
472  }
473  }
474  else if (iStatus == CFE_SB_NO_MESSAGE)
475  {
476  break;
477  }
478  else
479  {
480  (void) CFE_EVS_SendEvent(HMC5883_RCVMSG_ERR_EID, CFE_EVS_ERROR,
481  "CMD pipe read error (0x%08lX)", iStatus);
482  break;
483  }
484  }
485 
486  return;
487 }

490 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
491 /* */
492 /* Process HMC5883 Commands */
493 /* */
494 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
495 void HMC5883::ProcessAppCmds(CFE_SB_Msg_t* MsgPtr)
496 {
497  uint32 uiCmdCode=0;
498 
499  if (MsgPtr != NULL)
500  {
501  uiCmdCode = CFE_SB_GetCmdCode(MsgPtr);
502  switch (uiCmdCode)
503  {
504  case HMC5883_NOOP_CC:
505  {
506  HkTlm.usCmdCnt++;
507  (void) CFE_EVS_SendEvent(HMC5883_CMD_NOOP_EID, CFE_EVS_INFORMATION,
508  "Recvd NOOP. Version %d.%d.%d.%d",
509  HMC5883_MAJOR_VERSION,
510  HMC5883_MINOR_VERSION,
511  HMC5883_REVISION,
512  HMC5883_MISSION_REV);
513  break;
514  }
515  case HMC5883_RESET_CC:
516  {
517  HkTlm.usCmdCnt = 0;
518  HkTlm.usCmdErrCnt = 0;
519  break;
520  }
521  case HMC5883_SEND_DIAG_CC:
522  {
523  SendDiag();
524  break;
525  }
526  case HMC5883_SET_CALIBRATION_CC:
527  {
528  if(CFE_SUCCESS == UpdateCalibrationValues((HMC5883_SetCalibrationCmd_t *) MsgPtr))
529  {
530  UpdateParamsFromTable();
531  HkTlm.usCmdCnt++;
532  (void) CFE_EVS_SendEvent(HMC5883_CALIBRATE_INF_EID, CFE_EVS_INFORMATION,
533  "Calibration values updated");
534  }
535  else
536  {
537  HkTlm.usCmdErrCnt++;
538  (void) CFE_EVS_SendEvent(HMC5883_CALIBRATE_ERR_EID, CFE_EVS_ERROR,
539  "Calibration values failed to update");
540  }
541  break;
542  }
543  default:
544  {
545  HkTlm.usCmdErrCnt++;
546  (void) CFE_EVS_SendEvent(HMC5883_CC_ERR_EID, CFE_EVS_ERROR,
547  "Recvd invalid command code (%u)", (unsigned int)uiCmdCode);
548  break;
549  }
550  }
551  }
552  return;
553 }

729  if(FALSE == returnBool)

737  SensorMagMsg.ZRaw = zraw_f;

741  SensorMagMsg.Y = yraw_f * (Diag.Conversion.Unit / Diag.Conversion.Divider);
742  SensorMagMsg.Z = zraw_f * (Diag.Conversion.Unit / Diag.Conversion.Divider);

746  SensorMagMsg.Y = (SensorMagMsg.Y - m_Params.y_offset) * m_Params.y_scale;

750  SensorMagMsg.Range = Diag.Conversion.Range;
751  /* Set scale */
752  SensorMagMsg.Scaling = (Diag.Conversion.Unit / Diag.Conversion.Divider);
753 
754  /* Measure temperature every 10 mag samples */
755  if(temp_count == 10)
756  {
757  returnBool = HMC5883_Custom_Measure_Temp(&temp);
758  if(FALSE == returnBool)
759  {
760  goto end_of_function;
761  }
762 
763  SensorMagMsg.Temperature = 25 + (temp / (16 * 8.0f));
764  temp_count = 0;
765  }
766  else
767  {
768  temp_count++;
769  }
770 
771 end_of_function:
772  return;
773 }

776 /* */
777 /* Run Self-Calibration Routine */
778 /* */
779 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
780 boolean HMC5883::SelfCalibrate(HMC5883_CalibrationMsg_t *Calibration)
781 {
782  uint8 range = 0;
783  uint8 config = 0;
784  uint8 i = 0;
785  uint8 good_count = 0;
786  int16 rawX = 0;
787  int16 rawY = 0;
788  int16 rawZ = 0;
789  float rangeScale = 0.0f;
790  boolean returnBool = FALSE;
791  boolean rangeSet = FALSE;
792  boolean configSet = FALSE;
793  boolean xNeg = FALSE;
794  boolean yNeg = FALSE;
795  boolean zNeg = FALSE;
796 
797  /* expected axis scaling. The datasheet says that 766 will
798  * be places in the X and Y axes and 713 in the Z
799  * axis. Experiments show that in fact 766 is placed in X,
800  * and 713 in Y and Z. This is relative to a base of 660
801  * LSM/Ga, giving 1.16 and 1.08 */
802  float expected_cal[3] = { 1.16f, 1.08f, 1.08f };
803  float sum_excited[3] = { 0.0f, 0.0f, 0.0f };
804  float cal[3] = { 0.0f, 0.0f, 0.0f };
805  float scaling[3] = { 0.0f, 0.0f, 0.0f };
806 
807  /* Save the current range setting */
808  returnBool = HMC5883_Custom_Get_Range(&range);
809  if (FALSE == returnBool)
810  {
811  goto end_of_function;
812  }
813  /* Save the current configuration */
814  returnBool = HMC5883_Custom_Get_Config(&config);
815  if (FALSE == returnBool)
816  {
817  goto end_of_function;
818  }
819  /* Set to 2.5 Gauss.*/
820  returnBool = HMC5883_Custom_Set_Range(HMC5883_BITS_CONFIG_B_RANGE_2GA5);
821  if (FALSE == returnBool)
822  {
823  rangeSet = TRUE;
824  goto end_of_function;
825  }
826  /* Set the range scale to the range setting above */
827  rangeScale = 1.0f / 660.0f;
828  /* Set negative bias enable */
829  config |= HMC5883_NEG_BIAS_ENABLE;
830  returnBool = HMC5883_Custom_Set_Config(config);
831  if (FALSE == returnBool)
832  {
833  configSet = TRUE;
834  goto end_of_function;
835  }
836  /* Set the saved config back to normal */
837  config &= ~HMC5883_NEG_BIAS_ENABLE;
838 
839  /* Discard 10 samples to let the sensor settle */
840  for (i = 0; i < 10; i++)
841  {
842  returnBool = HMC5883_Custom_Measure(&rawX, &rawY, &rawZ);
843  if(FALSE == returnBool)
844  {
845  goto end_of_function;
846  }
847  }
848 
849  /* read the sensor up to 150x, stopping when we have 50 good values */
850  for (i = 0; i < 150 && good_count < 50; i++)
851  {
852  returnBool = HMC5883_Custom_Measure(&rawX, &rawY, &rawZ);
853  if(FALSE == returnBool)
854  {
855  goto end_of_function;
856  }
857 
858  cal[0] = fabsf(expected_cal[0] / (rawX * rangeScale));
859  cal[1] = fabsf(expected_cal[1] / (rawY * rangeScale));
860  cal[2] = fabsf(expected_cal[2] / (rawZ * rangeScale));
861 
862  if (cal[0] > 0.3f && cal[0] < 1.7f &&
863  cal[1] > 0.3f && cal[1] < 1.7f &&
864  cal[2] > 0.3f && cal[2] < 1.7f)
865  {
866  good_count++;
867  sum_excited[0] += cal[0];
868  sum_excited[1] += cal[1];
869  sum_excited[2] += cal[2];
870  }
871  }
872 
873  /* Check that we have at least 5 good values for an average */
874  if (good_count < 5)
875  {
876  returnBool = FALSE;
877  goto end_of_function;
878  }
879 
880  /* Get the average */
881  scaling[0] = sum_excited[0] / good_count;
882  scaling[1] = sum_excited[1] / good_count;
883  scaling[2] = sum_excited[2] / good_count;
884 
885  /* Sanity check scale values */
886  returnBool = CheckScale(scaling[0], scaling[1], scaling[2]);
887  if (TRUE == returnBool)
888  {
889  /* Set scaling */
890  Calibration->x_scale_internal = 1.0f / scaling[0];
891  Calibration->y_scale_internal = 1.0f / scaling[1];
892  Calibration->z_scale_internal = 1.0f / scaling[2];
893  }
894 
895 end_of_function:
896 
897  if (TRUE == rangeSet)
898  {
899  /* return the range setting back to normal */
900  HMC5883_Custom_Set_Range(range);
901  }
902  if (TRUE == configSet)
903  {
904  /* return the configuration setting back to normal */
905  HMC5883_Custom_Set_Config(config);
906  }
907  return (returnBool);
908 }
