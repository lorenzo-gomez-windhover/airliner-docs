112 /* Processes a command from the ATS */
113 /* */
114 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
115 void SC_ProcessAtpCmd (void)
116 {
117  CFE_SB_MsgPtr_t CmdPtr; /* ATS command pointer */
118  SC_AtsEntryHeader_t* Entry; /* ATS entry pointer */
119  int32 EntryIndex; /* ATS entry location in table */
120  uint8 AtsIndex; /* ATS selection index */
121  uint32 CmdIndex; /* ATS command index */
122  uint16 TempAtsChar = ' ';
123  int32 Result;
124  boolean AbortATS = FALSE;
125 
126  /*
127  ** The following conditions must be met before the ATS command will be
128  ** executed:
129  ** 1.) The next time is <= the current time
130  ** 2.) The next processor number = ATP
131  ** 3.) The atp is currently EXECUTING
132  */
133 
134 
135  if ((!SC_CompareAbsTime (SC_AppData.NextCmdTime[SC_ATP], SC_AppData.CurrentTime)) &&
136  (SC_AppData.NextProcNumber == SC_ATP) &&
137  (SC_OperData.AtsCtrlBlckAddr -> AtpState == SC_EXECUTING))
138  {
139  /*
140  ** Get a pointer to the next ats command
141  */
142  AtsIndex = SC_OperData.AtsCtrlBlckAddr->AtsNumber - 1; /* remember 0..1 */
143  CmdIndex = SC_OperData.AtsCtrlBlckAddr->CmdNumber;
144  EntryIndex = SC_AppData.AtsCmdIndexBuffer[AtsIndex][CmdIndex];
145  Entry = (SC_AtsEntryHeader_t *) &SC_OperData.AtsTblAddr[AtsIndex][EntryIndex];
146  CmdPtr = (CFE_SB_MsgPtr_t) Entry->CmdHeader;
147 
148  /*
149  ** Make sure the command has not been executed, skipped or has any other bad status
150  */
151  if (SC_OperData.AtsCmdStatusTblAddr[AtsIndex][CmdIndex] == SC_LOADED)
152  {
153  /*
154  ** Make sure the command number matches what the command
155  ** number is supposed to be
156  */
157  if (Entry->CmdNumber == (CmdIndex + 1))
158  {
159  /*
160  ** Check the checksum on the command
161  **
162  */
163  if (CFE_SB_ValidateChecksum(CmdPtr) == TRUE)
164  {
165  /*
166  ** Count the command for the rate limiter
167  */
168  SC_OperData.NumCmdsSec++;
169 
170  /*
171  ** First check to see if the command is a switch command,
172  ** if it is, then execute the command now instead of sending
173  ** it out on the Software Bus (this is the only exception to
174  ** way stored commands are sent out).
175  */
176 
177  if (CFE_SB_GetMsgId(CmdPtr) == SC_CMD_MID &&
178  CFE_SB_GetCmdCode(CmdPtr) == SC_SWITCH_ATS_CC)
179  {
180  /*
181  ** call the ground switch module
182  */
183  if (SC_InlineSwitch())
184  {
185  /*
186  ** Increment the counter and update the status for
187  ** this command
188  */
189  SC_OperData.AtsCmdStatusTblAddr[AtsIndex][CmdIndex] = SC_EXECUTED;
190  SC_AppData.AtsCmdCtr++;
191  }
192  else
193  { /* the switch failed for some reason */
194 
195  SC_OperData.AtsCmdStatusTblAddr[AtsIndex][CmdIndex] = SC_FAILED_DISTRIB;
196  SC_AppData.AtsCmdErrCtr++;
197  SC_AppData.LastAtsErrSeq = SC_OperData.AtsCtrlBlckAddr -> AtsNumber;
198  SC_AppData.LastAtsErrCmd = SC_OperData.AtsCtrlBlckAddr -> CmdNumber + 1;
199 
200  } /* end if */
201  }
202  else
203  {
204  Result = CFE_SB_SendMsg(CmdPtr);
205 
206  if ( Result == CFE_SUCCESS)
207  {
208  /* The command sent OK */
209  SC_OperData.AtsCmdStatusTblAddr[AtsIndex][CmdIndex] = SC_EXECUTED;
210  SC_AppData.AtsCmdCtr++;
211 
212  }
213  else
214  { /* the command had Software Bus problems */
215  SC_OperData.AtsCmdStatusTblAddr[AtsIndex][CmdIndex] = SC_FAILED_DISTRIB;
216  SC_AppData.AtsCmdErrCtr++;
217  SC_AppData.LastAtsErrSeq = SC_OperData.AtsCtrlBlckAddr -> AtsNumber;
218  SC_AppData.LastAtsErrCmd = SC_OperData.AtsCtrlBlckAddr -> CmdNumber + 1;
219 
220  CFE_EVS_SendEvent(SC_ATS_DIST_ERR_EID, CFE_EVS_ERROR,
221  "ATS Command Distribution Failed, Cmd Number: %d, SB returned: 0x%08X",
222  Entry->CmdNumber, (unsigned int)Result);
223 
224  if (SC_OperData.AtsCtrlBlckAddr -> AtsNumber == SC_ATSA)
225  TempAtsChar = 'A';
226  else if (SC_OperData.AtsCtrlBlckAddr -> AtsNumber == SC_ATSB)
227  TempAtsChar = 'B';
228 
229  /* Mark this ATS for abortion */
230  AbortATS = TRUE;
231  }
232  }
233  }
234  else
235  { /* the checksum failed */
236  /*
237  ** Send an event message to report the invalid command status
238  */
239  CFE_EVS_SendEvent(SC_ATS_CHKSUM_ERR_EID, CFE_EVS_ERROR,
240  "ATS Command Failed Checksum: Command #%d Skipped",
241  Entry->CmdNumber);
242  /*
243  ** Increment the ATS error counter
244  */
245  SC_AppData.AtsCmdErrCtr++;
246 
247  /*
248  ** Update the last ATS error information structure
249  */
250  SC_AppData.LastAtsErrSeq = SC_OperData.AtsCtrlBlckAddr -> AtsNumber;
251  SC_AppData.LastAtsErrCmd = SC_OperData.AtsCtrlBlckAddr -> CmdNumber + 1;
252 
253  /* update the command status index table */
254  SC_OperData.AtsCmdStatusTblAddr[AtsIndex][CmdIndex] = SC_FAILED_CHECKSUM;
255 
256 
257  if (SC_AppData.ContinueAtsOnFailureFlag == FALSE)
258  { /* Stop ATS execution */
259  /*
260  ** Set the temp ATS ID if it is valid
261  */
262  if (SC_OperData.AtsCtrlBlckAddr -> AtsNumber == SC_ATSA)
263  TempAtsChar = 'A';
264  else if (SC_OperData.AtsCtrlBlckAddr -> AtsNumber == SC_ATSB)
265  TempAtsChar = 'B';
266 
267  /* Mark this ATS for abortion */
268  AbortATS = TRUE;
269  }
270  } /* end checksum test */
271  }
272  else
273  { /* the command number does not match */
274  /*
275  ** Send an event message to report the invalid command status
276  */
277 
278  CFE_EVS_SendEvent(SC_ATS_MSMTCH_ERR_EID, CFE_EVS_ERROR,
279  "ATS Command Number Mismatch: Command Skipped, expected: %d received: %d",
280  (int)CmdIndex + 1, Entry->CmdNumber);
281  /*
282  ** Increment the ATS error counter
283  */
284  SC_AppData.AtsCmdErrCtr++;
285 
286  /*
287  ** Update the last ATS error information structure
288  */
289  SC_AppData.LastAtsErrSeq = SC_OperData.AtsCtrlBlckAddr -> AtsNumber;
290  SC_AppData.LastAtsErrCmd = SC_OperData.AtsCtrlBlckAddr -> CmdNumber + 1;
291 
292  /* update the command status index table */
293  SC_OperData.AtsCmdStatusTblAddr[AtsIndex][CmdIndex] = SC_SKIPPED;
294 
295  /*
296  ** Set the temp ATS ID if it is valid
297  */
298  if (SC_OperData.AtsCtrlBlckAddr -> AtsNumber == SC_ATSA)
299  TempAtsChar = 'A';
300  else if (SC_OperData.AtsCtrlBlckAddr -> AtsNumber == SC_ATSB)
301  TempAtsChar = 'B';
302 
303  /* Mark this ATS for abortion */
304  AbortATS = TRUE;
305  } /* end if the command number does not match */
306  }
307  else /* command isn't marked as loaded */
308  {
309  /*
310  ** Send an event message to report the invalid command status
311  */
312  CFE_EVS_SendEvent(SC_ATS_SKP_ERR_EID, CFE_EVS_ERROR,
313  "Invalid ATS Command Status: Command Skipped, Status: %d",
314  SC_OperData.AtsCmdStatusTblAddr[AtsIndex][CmdIndex]);
315  /*
316  ** Increment the ATS error counter
317  */
318  SC_AppData.AtsCmdErrCtr++;
319 
320  /*
321  ** Update the last ATS error information structure
322  */
323  SC_AppData.LastAtsErrSeq = SC_OperData.AtsCtrlBlckAddr -> AtsNumber;
324  SC_AppData.LastAtsErrCmd = SC_OperData.AtsCtrlBlckAddr -> CmdNumber + 1;
325 
326  /* Do Not Mark this ATS for abortion. The command could be marked as EXECUTED
327  if we alerady jumped back in time */
328 
329  } /* end if */
330 
331  if (AbortATS == TRUE)
332  {
333  CFE_EVS_SendEvent(SC_ATS_ABT_ERR_EID, CFE_EVS_ERROR,
334  "ATS %c Aborted",
335  TempAtsChar);
336 
337  /* Stop the ATS from executing */
338  SC_KillAts();
339  SC_OperData.AtsCtrlBlckAddr -> SwitchPendFlag = FALSE;
340  }
341 
342 
343  /*
344  ** Get the next ATS command set up to execute
345  */
346  SC_GetNextAtsCommand();
347 
348 
349  } /* end if next ATS command time */
350 } /* end SC_ProcessAtpCmd */

358 void SC_ProcessRtpCommand (void)
359 {
360 
361 
362  CFE_SB_MsgPtr_t CmdPtr; /* a pointer to an RTS entry command */
363  SC_RtsEntryHeader_t* RtsEntryPtr; /* a pointer to an RTS entry header */
364  uint16 RtsNum; /* the RTS number for the cmd */
365  uint16 CmdOffset; /* the location of the cmd */
366  uint32 Result;
367 
368  /*
369  ** The following conditions must be met before a RTS command is executed:
370  ** 1.) The next command time must be <= the current time
371  ** 2.) The next processor number must be SC_RTP
372  ** 3.) The RTS number in the RTP control block must be valid and
373  ** 4.) the RTS must be EXECUTING
374  */
375 
376  if ((SC_AppData.NextCmdTime[SC_AppData.NextProcNumber] <= SC_AppData.CurrentTime) &&
377  (SC_AppData.NextProcNumber == SC_RTP) &&
378  (SC_OperData.RtsCtrlBlckAddr -> RtsNumber > 0) &&
379  (SC_OperData.RtsCtrlBlckAddr -> RtsNumber <= SC_NUMBER_OF_RTS) &&
380  (SC_OperData.RtsInfoTblAddr[SC_OperData.RtsCtrlBlckAddr -> RtsNumber - 1].RtsStatus == SC_EXECUTING))
381  {
382  /*
383  ** Count the command for the rate limiter
384  ** even if the command fails
385  */
386  SC_OperData.NumCmdsSec++;
387 
388  /* get the RTS number that can be directly indexed into the table*/
389  RtsNum = SC_OperData.RtsCtrlBlckAddr -> RtsNumber - 1;
390 
391  /*
392  ** Get the Command offset within the RTS
393  */
394  CmdOffset = SC_OperData.RtsInfoTblAddr[RtsNum].NextCommandPtr;
395 
396  /*
397  ** Get a pointer to the RTS entry using the RTS number and the offset
398  */
399  RtsEntryPtr = (SC_RtsEntryHeader_t *) &SC_OperData.RtsTblAddr[RtsNum][CmdOffset];
400  CmdPtr = (CFE_SB_MsgPtr_t) RtsEntryPtr->CmdHeader;
401 
402  if (CFE_SB_ValidateChecksum(CmdPtr) == TRUE)
403  {
404  /*
405  ** Try Sending the command on the Software Bus
406  */
407 
408  Result = CFE_SB_SendMsg(CmdPtr);
409 
410  if (Result == CFE_SUCCESS)
411  {
412  /* the command was sent OK */
413  SC_AppData.RtsCmdCtr++;
414  SC_OperData.RtsInfoTblAddr[RtsNum].CmdCtr++;
415 
416  /*
417  ** Get the next command.
418  */
419  SC_GetNextRtsCommand();
420  }
421  else
422  { /* the software bus return code was bad */
423 
424  /*
425  ** Send an event message to report the invalid command status
426  */
427  CFE_EVS_SendEvent (SC_RTS_DIST_ERR_EID,
428  CFE_EVS_ERROR,
429  "RTS %03d Command Distribution Failed: RTS Stopped. SB returned 0x%08X",
430  (int)(RtsNum + 1), (unsigned int)Result);
431 
432  SC_AppData.RtsCmdErrCtr++;
433  SC_OperData.RtsInfoTblAddr[RtsNum].CmdErrCtr++;
434  SC_AppData.LastRtsErrSeq = SC_OperData.RtsCtrlBlckAddr -> RtsNumber;
435  SC_AppData.LastRtsErrCmd = CmdOffset;
436 
437  /*
438  ** Stop the RTS from executing
439  */
440  SC_KillRts (RtsNum);
441 
442  } /* end if */
443 
444  }
445  else
446  { /* the checksum failed */
447 
448  /*
449  ** Send an event message to report the invalid command status
450  */
451  CFE_EVS_SendEvent (SC_RTS_CHKSUM_ERR_EID,
452  CFE_EVS_ERROR,
453  "RTS %03d Command Failed Checksum: RTS Stopped",
454  RtsNum + 1);
455  /*
456  ** Update the RTS command error counter and last RTS error info
457  */
458  SC_AppData.RtsCmdErrCtr++;
459  SC_OperData.RtsInfoTblAddr[RtsNum].CmdErrCtr++;
460  SC_AppData.LastRtsErrSeq = SC_OperData.RtsCtrlBlckAddr -> RtsNumber;
461  SC_AppData.LastRtsErrCmd = CmdOffset;
462 
463  /*
464  ** Stop the RTS from executing
465  */
466  SC_KillRts (RtsNum);
467  } /* end if */
468  } /* end if */
469 } /* end SC_ProcessRtpCommand */

613 void SC_ProcessRequest (CFE_SB_MsgPtr_t CmdPacket)
614 {
615  CFE_SB_MsgId_t MessageID;
616  int8 IsThereAnotherCommandToExecute = FALSE;
617 
618  /* cast the packet header pointer on the packet buffer */
619 
620  MessageID = CFE_SB_GetMsgId (CmdPacket);
621 
622  /*
623  ** Get the current system time in the global SC_AppData.CurrentTime
624  */
625  SC_GetCurrentTime();
626 
627  switch (MessageID)
628  {
629  case SC_CMD_MID:
630  /* request from the ground */
631  SC_ProcessCommand (CmdPacket);
632  break;
633 
634  case SC_SEND_HK_MID:
635  if (SC_VerifyCmdLength(CmdPacket, sizeof(SC_NoArgsCmd_t)))
636  {
637 
638  /* set during init to power on or processor reset auto-exec RTS */
639  if (SC_AppData.AutoStartRTS != 0)
640  {
641  /* make sure the selected auto-exec RTS is enabled */
642  if (SC_OperData.RtsInfoTblAddr[SC_AppData.AutoStartRTS - 1].RtsStatus == SC_LOADED)
643  {
644  SC_OperData.RtsInfoTblAddr[SC_AppData.AutoStartRTS - 1].DisabledFlag = FALSE;
645  }
646 
647  /* send ground cmd to have SC start the RTS */
648  SC_AutoStartRts(SC_AppData.AutoStartRTS);
649 
650  /* only start it once */
651  SC_AppData.AutoStartRTS = 0;
652  }
653 
654  /* request from health and safety for housekeeping status */
655  SC_SendHkPacket();
656  }
657  break;
658 
659  case SC_1HZ_WAKEUP_MID:
660  /*
661  ** Time to execute a command in the SC memory
662  */
663 
664  do
665  {
666  /*
667  ** Check to see if there is an ATS switch Pending, if so service it.
668  */
669  if (SC_OperData.AtsCtrlBlckAddr -> SwitchPendFlag == TRUE)
670  {
671  SC_ServiceSwitchPend();
672  }
673 
674  if (SC_AppData.NextProcNumber == SC_ATP)
675  {
676  SC_ProcessAtpCmd();
677  }
678  else
679  {
680  if (SC_AppData.NextProcNumber == SC_RTP)
681  {
682  SC_ProcessRtpCommand();
683  }
684  }
685 
686  SC_UpdateNextTime();
687 
688  if ((SC_AppData.NextProcNumber == SC_NONE) ||
689  (SC_AppData.NextCmdTime[SC_AppData.NextProcNumber] > SC_AppData.CurrentTime))
690  {
691  SC_OperData.NumCmdsSec = 0;
692  IsThereAnotherCommandToExecute = FALSE;
693  }
694  else /* Command needs to run immediately */
695  {
696  if (SC_OperData.NumCmdsSec >= SC_MAX_CMDS_PER_SEC)
697  {
698  SC_OperData.NumCmdsSec = 0;
699  IsThereAnotherCommandToExecute = FALSE;
700  }
701  else
702  {
703  IsThereAnotherCommandToExecute = TRUE;
704  }
705 
706  }
707  } while (IsThereAnotherCommandToExecute);
708 
709  break;
710 
711  default:
712  CFE_EVS_SendEvent (SC_MID_ERR_EID,
713  CFE_EVS_ERROR,
714  "Invalid command pipe message ID: 0x%08X",
715  MessageID);
716 
717  SC_AppData.CmdErrCtr++;
718  break;
719  } /* end switch */
