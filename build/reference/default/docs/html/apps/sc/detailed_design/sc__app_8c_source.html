100 /* SC main process loop (task entry point) */
101 /* */
102 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
103 
104 void SC_AppMain(void)
105 {
106  uint32 RunStatus = CFE_ES_APP_RUN;
107  int32 Result;
108 
109  /* Register application with cFE */
110  Result = CFE_ES_RegisterApp();
111 
112  /* Performance Log (start time counter) */
113  CFE_ES_PerfLogEntry(SC_APPMAIN_PERF_ID);
114 
115  /* Startup initialization */
116  if (Result == CFE_SUCCESS)
117  {
118  Result = SC_AppInit();
119  }
120 
121  /* Check for start-up error */
122  if (Result != CFE_SUCCESS)
123  {
124  /* Set request to terminate main loop */
125  RunStatus = CFE_ES_APP_ERROR;
126  }
127 
128  /* Main process loop */
129  while (CFE_ES_RunLoop(&RunStatus))
130  {
131  /* Performance Log (stop time counter) */
132  CFE_ES_PerfLogExit(SC_APPMAIN_PERF_ID);
133 
134  /* Pend on Software Bus for message */
135  Result = CFE_SB_RcvMsg(&SC_OperData.MsgPtr, SC_OperData.CmdPipe, CFE_SB_PEND_FOREVER);
136 
137  /* Performance Log (start time counter) */
138  CFE_ES_PerfLogEntry(SC_APPMAIN_PERF_ID);
139 
140  /* Check for Software Bus error */
141  if (Result == CFE_SUCCESS)
142  {
143  /* Invoke command handlers */
144  SC_ProcessRequest(SC_OperData.MsgPtr);
145  }
146  else
147  {
148  /* Exit main process loop */
149  RunStatus = CFE_ES_APP_ERROR;
150  }
151  }
152 
153  /* Check for "fatal" process error */
154  if (Result != CFE_SUCCESS)
155  {
156  /* Send event describing reason for termination */
157  CFE_EVS_SendEvent(SC_APP_EXIT_ERR_EID, CFE_EVS_ERROR,
158  "App terminating, Result = 0x%08X", (unsigned int)Result);
159 
160  /* In case cFE Event Services is not working */
161  CFE_ES_WriteToSysLog("SC App terminating, Result = 0x%08X\n", (unsigned int)Result);
162  }
163 
164  /* Performance Log (stop time counter) */
165  CFE_ES_PerfLogExit(SC_APPMAIN_PERF_ID);
166 
167  /* Let cFE kill the app */
168  CFE_ES_ExitApp(RunStatus);
169 
170 } /* end SC_AppMain() */

179 int32 SC_AppInit(void)
180 {
181  int32 Result;
182 
183  /* Clear global data structures */
184  CFE_PSP_MemSet(&SC_OperData, 0, sizeof(SC_OperData_t));
185  CFE_PSP_MemSet(&SC_AppData, 0, sizeof(SC_AppData_t));
186 
187  /* Number of ATS and RTS commands already executed this second */
188  SC_OperData.NumCmdsSec = 0;
189 
190  /* Continue ATS execution if ATS command checksum fails */
191  SC_AppData.ContinueAtsOnFailureFlag = SC_CONT_ON_FAILURE_START;
192 
193  /* Make sure nothing is running */
194  SC_AppData.NextProcNumber = SC_NONE;
195  SC_AppData.NextCmdTime[SC_ATP] = SC_MAX_TIME;
196  SC_AppData.NextCmdTime[SC_RTP] = SC_MAX_TIME;
197 
198  /* Initialize the SC housekeeping packet */
199  CFE_SB_InitMsg(&SC_OperData.HkPacket, SC_HK_TLM_MID, sizeof(SC_HkTlm_t), FALSE);
200 
201  /* Select auto-exec RTS to start during first HK request */
202  if (CFE_ES_GetResetType(NULL) == CFE_ES_POWERON_RESET)
203  {
204  SC_AppData.AutoStartRTS = RTS_ID_AUTO_POWER_ON;
205  }
206  else
207  {
208  SC_AppData.AutoStartRTS = RTS_ID_AUTO_PROCESSOR;
209  }
210 
211  /* Must be able to register for events */
212  Result = CFE_EVS_Register(NULL,0,CFE_EVS_NO_FILTER);
213  if (Result != CFE_SUCCESS)
214  {
215  CFE_ES_WriteToSysLog("Event Services Register returned: 0x%08X\n", (unsigned int)Result);
216  return(Result);
217  }
218 
219  /* Must be able to create Software Bus message pipe */
220  Result = CFE_SB_CreatePipe(&SC_OperData.CmdPipe, SC_PIPE_DEPTH, SC_CMD_PIPE_NAME);
221  if (Result != CFE_SUCCESS)
222  {
223  CFE_EVS_SendEvent(SC_INIT_SB_CREATE_ERR_EID, CFE_EVS_ERROR,
224  "Software Bus Create Pipe returned: 0x%08X", (unsigned int)Result);
225  return(Result);
226  }
227 
228  /* Must be able to subscribe to HK request command */
229  Result = CFE_SB_Subscribe(SC_SEND_HK_MID, SC_OperData.CmdPipe);
230  if (Result != CFE_SUCCESS)
231  {
232  CFE_EVS_SendEvent(SC_INIT_SB_SUBSCRIBE_HK_ERR_EID, CFE_EVS_ERROR,
233  "Software Bus subscribe to housekeeping returned: 0x%08X", (unsigned int)Result);
234  return(Result);
235  }
236 
237  /* Must be able to subscribe to 1Hz wakeup command */
238  Result = CFE_SB_Subscribe(SC_1HZ_WAKEUP_MID, SC_OperData.CmdPipe);
239  if (Result != CFE_SUCCESS)
240  {
241  CFE_EVS_SendEvent(SC_INIT_SB_SUBSCRIBE_1HZ_ERR_EID, CFE_EVS_ERROR,
242  "Software Bus subscribe to 1 Hz cycle returned: 0x%08X", (unsigned int)Result);
243  return(Result);
244  }
245 
246  /* Must be able to subscribe to SC commands */
247  Result = CFE_SB_Subscribe(SC_CMD_MID, SC_OperData.CmdPipe);
248  if (Result != CFE_SUCCESS)
249  {
250  CFE_EVS_SendEvent(SC_INIT_SB_SUBSCRIBE_CMD_ERR_EID, CFE_EVS_ERROR,
251  "Software Bus subscribe to command returned: 0x%08X", (unsigned int)Result);
252  return(Result);
253  }
254 
255  /* Must be able to create and initialize tables */
256  Result = SC_InitTables();
257  if (Result != CFE_SUCCESS)
258  {
259  return(Result);
260  }
261 
262  /* Send application startup event */
263  CFE_EVS_SendEvent(SC_INIT_INF_EID, CFE_EVS_INFORMATION,
264  "SC Initialized. Version %d.%d.%d.%d",
265  SC_MAJOR_VERSION, SC_MINOR_VERSION, SC_REVISION, SC_MISSION_REV);
266 
267  return(CFE_SUCCESS);
268 
269 } /* end SC_AppInit() */

278 int32 SC_InitTables(void)
279 {
280  int32 Result;
281  int32 i;
282  int32 j;
283 
284  /* Must be able to register all tables with cFE Table Services */
285  Result = SC_RegisterAllTables();
286  if (Result != CFE_SUCCESS)
287  {
288  return(Result);
289  }
290 
291  /* Must be able to get dump only table pointers */
292  Result = SC_GetDumpTablePointers();
293  if (Result != CFE_SUCCESS)
294  {
295  return(Result);
296  }
297 
298  /* ATP control block status table */
299  SC_OperData.AtsCtrlBlckAddr -> AtpState = SC_IDLE;
300  SC_OperData.AtsCtrlBlckAddr -> AtsNumber = SC_NO_ATS;
301  SC_OperData.AtsCtrlBlckAddr -> CmdNumber = 0xFFFF;
302 
303  /* RTP control block status table */
304  SC_OperData.RtsCtrlBlckAddr -> NumRtsActive = 0;
305  SC_OperData.RtsCtrlBlckAddr -> RtsNumber = SC_INVALID_RTS_NUMBER;
306 
307  /* ATS command status table(s) */
308  for (i = 0; i < SC_NUMBER_OF_ATS; i++)
309  {
310  for (j = 0; j < SC_MAX_ATS_CMDS; j++)
311  {
312  SC_OperData.AtsCmdStatusTblAddr[i][j] = SC_EMPTY;
313  }
314  }
315 
316  /* RTS information table */
317  for (i = 0; i < SC_NUMBER_OF_RTS; i++)
318  {
319  SC_OperData.RtsInfoTblAddr[i].NextCommandTime = SC_MAX_TIME;
320  SC_OperData.RtsInfoTblAddr[i].NextCommandPtr = 0;
321  SC_OperData.RtsInfoTblAddr[i].RtsStatus = SC_EMPTY;
322  SC_OperData.RtsInfoTblAddr[i].DisabledFlag = TRUE;
323  }
324 
325  /* Load default RTS tables */
326  SC_LoadDefaultTables();
327 
328  /* Must be able to get loadable table pointers */
329  Result = SC_GetLoadTablePointers();
330  if (Result != CFE_SUCCESS)
331  {
332  return(Result);
333  }
334 
335  /* Register for table update notification commands */
336  SC_RegisterManageCmds();
337 
338  return(CFE_SUCCESS);
339 
340 } /* end SC_InitTables() */
