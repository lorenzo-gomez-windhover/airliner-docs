96 /* Load the ATS from its table to memory */
97 /* */
98 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
99 void SC_LoadAts (uint16 AtsIndex)
100 {
101  uint16 AtsEntryWords; /* current ats entry length in words */
102  uint16 AtsCmdNum; /* current ats entry command number */
103  uint16 AtsEntryIndex; /* index into the load for current ats entry */
104  CFE_SB_MsgPtr_t AtsCmd; /* a pointer to an ats command */
105  SC_AtsEntryHeader_t *AtsEntryPtr; /* a pointer to an ats entry header */
106  uint16 *AtsTablePtr; /* pointer to the start of the Ats table */
107 
108  int32 Result = CFE_SUCCESS;
109  boolean StillProcessing = TRUE;
110 
111  /*
112  ** Initialize all structrures
113  */
114  SC_InitAtsTables (AtsIndex);
115 
116  /* initialize pointers and counters */
117  AtsTablePtr = SC_OperData.AtsTblAddr[AtsIndex];
118  AtsEntryIndex = 0;
119 
120  while (StillProcessing)
121  {
122  /*
123  ** Make sure that the pointer as well as the primary packet
124  ** header fit in the buffer, so a G.P fault is not caused.
125  */
126  if (AtsEntryIndex < SC_ATS_BUFF_SIZE)
127  {
128  /* get the next command number from the buffer */
129  AtsCmdNum = ((SC_AtsEntryHeader_t *)&AtsTablePtr[AtsEntryIndex]) ->CmdNumber;
130 
131  if (AtsCmdNum == 0)
132  {
133  /* end of the load reached */
134  Result = CFE_SUCCESS;
135  StillProcessing = FALSE;
136  }
137 
138  /* make sure the CmdPtr can fit in a whole Ats Cmd Header at the very least */
139  else if (AtsEntryIndex > (SC_ATS_BUFF_SIZE - (sizeof(SC_AtsEntryHeader_t)/SC_BYTES_IN_WORD)))
140  {
141  /* even the smallest command will not fit in the buffer */
142  Result = SC_ERROR;
143  StillProcessing = FALSE;
144  } /* else if the cmd number is valid and the command */
145  /* has not already been loaded */
146  else
147  if (AtsCmdNum <= SC_MAX_ATS_CMDS &&
148  SC_OperData.AtsCmdStatusTblAddr[AtsIndex][AtsCmdNum - 1] == SC_EMPTY)
149  {
150  /* get a pointer to the ats command in the table */
151  AtsEntryPtr = (SC_AtsEntryHeader_t *) &AtsTablePtr[AtsEntryIndex];
152  AtsCmd = (CFE_SB_MsgPtr_t) AtsEntryPtr->CmdHeader;
153 
154  /* if the length of the command is valid */
155  if (CFE_SB_GetTotalMsgLength(AtsCmd) >= SC_PACKET_MIN_SIZE &&
156  CFE_SB_GetTotalMsgLength(AtsCmd) <= SC_PACKET_MAX_SIZE)
157  {
158  /* get the length of the entry in WORDS (plus 1 to round byte len up to word len) */
159  AtsEntryWords = (CFE_SB_GetTotalMsgLength(AtsCmd) + 1 + SC_ATS_HEADER_SIZE) / SC_BYTES_IN_WORD;
160 
161  /* if the command does not run off of the end of the buffer */
162  if (AtsEntryIndex + AtsEntryWords <= SC_ATS_BUFF_SIZE)
163  {
164  /* set the command pointer in the command index table */
165  /* CmdNum starts at one.... */
166 
167  SC_AppData.AtsCmdIndexBuffer[AtsIndex][AtsCmdNum -1] = AtsEntryIndex;
168 
169  /* set the command status to loaded in the command status table */
170  SC_OperData.AtsCmdStatusTblAddr[AtsIndex][AtsCmdNum - 1] = SC_LOADED;
171 
172  /* increment the number of commands loaded */
173  SC_OperData.AtsInfoTblAddr[AtsIndex].NumberOfCommands++;
174 
175  /* increment the ats_entry index to the next ats entry */
176  AtsEntryIndex = AtsEntryIndex + AtsEntryWords;
177  }
178  else
179  { /* the command runs off the end of the buffer */
180  Result = SC_ERROR;
181  StillProcessing = FALSE;
182  } /* end if */
183  }
184  else
185  { /* the command length was invalid */
186  Result = SC_ERROR;
187  StillProcessing = FALSE;
188  } /* end if */
189  }
190  else
191  { /* the cmd number is invalid */
192  Result = SC_ERROR;
193  StillProcessing = FALSE;
194  } /* end if */
195  }
196  else
197  {
198  if (AtsEntryIndex == SC_ATS_BUFF_SIZE)
199  {
200  /* we encountered a load exactly as long as the buffer */
201  Result = CFE_SUCCESS;
202  StillProcessing = FALSE;
203 
204  }
205  else
206  { /* the pointer is over the end of the buffer */
207 
208  Result = SC_ERROR;
209  StillProcessing = FALSE;
210  } /* end if */
211  }/*end else */
212  } /* end while */
213 
214  /*
215  ** Now the commands are parsed through, need to build the tables
216  ** if the load was a sucess, need to build the tables
217  */
218 
219  /* if the load finished without errors and there was at least one command */
220  if ((Result == CFE_SUCCESS) && (SC_OperData.AtsInfoTblAddr[AtsIndex].NumberOfCommands > 0))
221  {
222  /* record the size of the load in the ATS info table */
223  SC_OperData.AtsInfoTblAddr[AtsIndex].AtsSize = AtsEntryIndex; /* size in WORDS */
224 
225  /* build the time index table */
226  SC_BuildTimeIndexTable(AtsIndex);
227  }
228  else
229  { /* there was an error */
230  SC_InitAtsTables (AtsIndex);
231  } /* end if */
232 
233 } /* end SC_LoadAts */

269 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
270 /* */
271 /* Inserts and element into a sorted list */
272 /* */
273 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
274 void SC_Insert (uint16 AtsIndex, uint32 NewCmdIndex, uint32 ListLength)
275 {
276  SC_AtsEntryHeader_t *Entry; /* ATS table entry pointer */
277  SC_AbsTimeTag_t NewCmdTime=0; /* new command execution time */
278  SC_AbsTimeTag_t ListCmdTime; /* list entry execution time */
279  uint32 CmdIndex; /* ATS command index (cmd num - 1) */
280  uint32 EntryIndex; /* ATS entry location in table */
281  int32 TimeBufIndex; /* this must be signed */
282 
283  /* get execution time for new list entry */
284  if (ListLength > 0)
285  {
286  /* first get the entry index in the selected ATS table for the new command */
287  EntryIndex = SC_AppData.AtsCmdIndexBuffer[AtsIndex][NewCmdIndex];
288  /* then get a pointer to the ATS entry */
289  Entry = (SC_AtsEntryHeader_t *) &SC_OperData.AtsTblAddr[AtsIndex][EntryIndex];
290  /* then get the execution time from the ATS entry for the new command */
291  NewCmdTime = SC_GetAtsEntryTime(Entry);
292  }
293 
294  /* start at last element in the sorted by time list */
295  TimeBufIndex = ListLength - 1;
296 
297  while (TimeBufIndex >= 0)
298  {
299  /* first get the cmd index for this list entry */
300  CmdIndex = SC_AppData.AtsTimeIndexBuffer[AtsIndex][TimeBufIndex];
301  /* then get the entry index from the ATS table */
302  EntryIndex = SC_AppData.AtsCmdIndexBuffer[AtsIndex][CmdIndex];
303  /* then get a pointer to the ATS entry data */
304  Entry = (SC_AtsEntryHeader_t *) &SC_OperData.AtsTblAddr[AtsIndex][EntryIndex];
305  /* then get cmd execution time from the ATS entry */
306  ListCmdTime = SC_GetAtsEntryTime(Entry);
307 
308  /* compare time for this list entry to time for new cmd */
309  if (SC_CompareAbsTime(ListCmdTime, NewCmdTime))
310  {
311  /* new cmd will execute before this list entry */
312 
313  /* move this list entry to make room for new cmd */
314  SC_AppData.AtsTimeIndexBuffer[AtsIndex][TimeBufIndex + 1] =
315  SC_AppData.AtsTimeIndexBuffer[AtsIndex][TimeBufIndex];
316 
317  /* back up to previous list entry (ok if -1) */
318  TimeBufIndex--;
319  }
320  else
321  {
322  /* new cmd will execute at same time or after this list entry */
323  break;
324  }
325  }
326 
327  /*
328  ** TimeBufIndex is now one slot before the target slot...
329  ** if new cmd time is earlier than all other entries
330  ** then TimeBufIndex is -1 and all others have been moved
331  ** else only entries with later times have been moved
332  ** In either case, there is an empty slot next to TimeBufIndex
333  */
334  SC_AppData.AtsTimeIndexBuffer[AtsIndex][TimeBufIndex + 1] = NewCmdIndex;
335 } /* end SC_Insert */

400 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
401 /* */
402 /* Parses the RTS to make sure it looks good */
403 /* */
404 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
405 boolean SC_ParseRts (uint16 Buffer[])
406 {
407  uint16 i;
408  boolean Done;
409  boolean Error;
410  CFE_SB_MsgPtr_t RtsCmd;
411  SC_RtsEntryHeader_t *RtsEntryPtr;
412  uint16 RtsCmdSize;
413 
414  i = 0;
415  Done = Error = FALSE;
416 
417  while (Error == FALSE && Done == FALSE)
418  {
419 
420  /*
421  ** Check to see if a minimum command fits within an RTS
422  */
423  if (i <= (SC_RTS_BUFF_SIZE - (sizeof(SC_RtsEntryHeader_t) / SC_BYTES_IN_WORD)))
424  {
425 
426  /*
427  ** Cast a header to the RTS buffer current location
428  ** and get the size of the packet
429  */
430 
431  RtsEntryPtr = (SC_RtsEntryHeader_t *) &Buffer[i];
432  RtsCmd = (CFE_SB_MsgPtr_t) RtsEntryPtr->CmdHeader;
433 
434  RtsCmdSize = CFE_SB_GetTotalMsgLength(RtsCmd) + SC_RTS_HEADER_SIZE;
435 
436  if ( (RtsEntryPtr->TimeTag == 0) && (CFE_SB_GetMsgId(RtsCmd) == 0))
437  {
438  Done = TRUE; /* assumed end of file */
439  }
440  else if (CFE_SB_GetMsgId(RtsCmd) == 0)
441  {
442  CFE_EVS_SendEvent (SC_RTS_INVLD_MID_ERR_EID,
443  CFE_EVS_ERROR,
444  "RTS cmd loaded with invalid MID at %d",
445  i);
446 
447  Error = TRUE; /* invalid message id */
448  }
449  else
450  {
451  /* check to see if the length field in the RTS is valid */
452  if (CFE_SB_GetTotalMsgLength(RtsCmd) < SC_PACKET_MIN_SIZE ||
453  CFE_SB_GetTotalMsgLength(RtsCmd) > SC_PACKET_MAX_SIZE)
454  {
455  CFE_EVS_SendEvent (SC_RTS_LEN_ERR_EID,
456  CFE_EVS_ERROR,
457  "RTS cmd loaded with invalid length at %d, len: %d",
458  i,
459  CFE_SB_GetTotalMsgLength(RtsCmd));
460 
461  Error = TRUE; /* Length error */
462 
463  }
464  else if ((i + ((RtsCmdSize + 1) / SC_BYTES_IN_WORD)) > SC_RTS_BUFF_SIZE)
465  {
466  CFE_EVS_SendEvent (SC_RTS_LEN_BUFFER_ERR_EID,
467  CFE_EVS_ERROR,
468  "RTS cmd at %d runs off end of buffer",
469  i);
470  Error = TRUE; /* command runs off of the end of the buffer */
471  }
472  else if ((i + ((RtsCmdSize + 1) / SC_BYTES_IN_WORD)) == SC_RTS_BUFF_SIZE)
473  {
474  Done = TRUE;
475  }
476  else
477  { /* command fits in buffer */
478 
479  i += ((RtsCmdSize + 1) / SC_BYTES_IN_WORD); /* remember 'i' is expressed in words */
480 
481  } /* end if */
482 
483  } /* endif */
484  }
485  else
486  { /* command does not fit in the buffer */
487 
488  /*
489  ** If it looks like there is data, reject the load,
490  ** if it looks empty then we are done
491  */
492  if (Buffer[i] == 0)
493  {
494  Done = TRUE;
495  }
496  else
497  {
498  CFE_EVS_SendEvent (SC_RTS_LEN_TOO_LONG_ERR_EID,
499  CFE_EVS_ERROR,
500  "RTS cmd loaded won't fit in buffer at %d",
501  i);
502  Error = TRUE;
503  }
504  } /* endif */
505 
506  } /* endwhile */
507 
508  /*
509  ** finished, report results
510  */
511 
512  /* If Error was TRUE, then SC_ParseRts must return FALSE */
513  return (!Error);
514 
515 } /* end SC_ParseRts */

564 /* */
565 /* Have new Append ATS table data, update Append ATS Info table */
566 /* */
567 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
568 void SC_UpdateAppend (void)
569 {
570  SC_AtsEntryHeader_t *Entry = {0};
571  CFE_SB_MsgPtr_t CmdPacket;
572  int32 CommandBytes;
573  int32 CommandWords;
574  int32 EntryIndex = 0;
575  int32 EntryCount = 0;
576  boolean StillProcessing = TRUE;
577 
578  /* Count Append ATS table entries and get total size */
579  while (StillProcessing)
580  {
581  if (EntryIndex >= SC_APPEND_BUFF_SIZE)
582  {
583  /* End of Append ATS table buffer */
584  StillProcessing = FALSE;
585  }
586  else
587  {
588  Entry = (SC_AtsEntryHeader_t *) &SC_OperData.AppendTblAddr[EntryIndex];
589 
590  if ((Entry->CmdNumber == 0) || (Entry->CmdNumber > SC_MAX_ATS_CMDS))
591  {
592  /* End of valid command numbers */
593  StillProcessing = FALSE;
594  }
595  else
596  {
597  /* Compute entry command packet length */
598  CmdPacket = (CFE_SB_MsgPtr_t) Entry->CmdHeader;
599  CommandBytes = CFE_SB_GetTotalMsgLength(CmdPacket);
600  CommandWords = (CommandBytes + 1) / 2;
601 
602  if ((CommandBytes < SC_PACKET_MIN_SIZE) || (CommandBytes > SC_PACKET_MAX_SIZE))
603  {
604  /* Entry command packet must have a valid length */
605  StillProcessing = FALSE;
606  }
607  else if ((EntryIndex + SC_ATS_HDR_NOPKT_WORDS + CommandWords) > SC_APPEND_BUFF_SIZE)
608  {
609  /* Entry command packet must fit within ATS append table buffer */
610  StillProcessing = FALSE;
611  }
612  else
613  {
614  /* Compute buffer index for next Append ATS table entry */
615  EntryIndex += (SC_ATS_HDR_NOPKT_WORDS + CommandWords);
616  EntryCount++;
617  }
618  }
619  }
620  }
621 
622  /* Results will also be reported in HK */
623  SC_AppData.AppendLoadCount++;
624  SC_AppData.AppendEntryCount = EntryCount;
625  SC_AppData.AppendWordCount = EntryIndex;
626 
627  CFE_EVS_SendEvent(SC_UPDATE_APPEND_EID, CFE_EVS_INFORMATION,
628  "Update Append ATS Table: load count = %d, command count = %d, byte count = %d",
629  SC_AppData.AppendLoadCount, (int)EntryCount, (int)EntryIndex * 2);
630  return;
631 
632 } /* end SC_UpdateAppend */

635 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
636 /* */
637 /* Append contents of Append ATS table to indicated ATS table */
638 /* */
639 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
640 void SC_ProcessAppend (uint16 AtsIndex)
641 {
642  SC_AtsEntryHeader_t *Entry;
643  CFE_SB_MsgPtr_t CmdPacket;
644 
645  int32 CommandBytes;
646  int32 CommandWords;
647  int32 EntryIndex;
648  int32 i;
649  uint16 CmdIndex;
650 
651  /* save index of free area at end of ATS table data */
652  EntryIndex = SC_OperData.AtsInfoTblAddr[AtsIndex].AtsSize;
653 
654  /* copy Append table data to end of ATS table data */
655  CFE_PSP_MemCpy(&SC_OperData.AtsTblAddr[AtsIndex][EntryIndex],
656  SC_OperData.AppendTblAddr, SC_AppData.AppendWordCount * 2);
657 
658  /* update size of ATS table data */
659  SC_OperData.AtsInfoTblAddr[AtsIndex].AtsSize += SC_AppData.AppendWordCount;
660 
661  /* add appended entries to ats process tables */
662  for (i = 0; i < SC_AppData.AppendEntryCount; i++)
663  {
664  /* get pointer to next appended entry */
665  Entry = (SC_AtsEntryHeader_t *) &SC_OperData.AtsTblAddr[AtsIndex][EntryIndex];
666 
667  /* convert base one cmd number to base zero index */
668  CmdIndex = Entry->CmdNumber - 1;
669 
670  /* count only new commands, not replaced commands */
671  if (SC_OperData.AtsCmdStatusTblAddr[AtsIndex][CmdIndex] == SC_EMPTY)
672  {
673  SC_OperData.AtsInfoTblAddr[AtsIndex].NumberOfCommands++;
674  }
675 
676  /* update array of pointers to ats entries */
677  SC_AppData.AtsCmdIndexBuffer[AtsIndex][CmdIndex] = EntryIndex;
678  SC_OperData.AtsCmdStatusTblAddr[AtsIndex][CmdIndex] = SC_LOADED;
679 
680  /* update entry index to point to the next entry */
681  CmdPacket = (CFE_SB_MsgPtr_t) Entry->CmdHeader;
682  CommandBytes = CFE_SB_GetTotalMsgLength(CmdPacket);
683  CommandWords = (CommandBytes + 1) / 2;
684  EntryIndex += (SC_ATS_HDR_NOPKT_WORDS + CommandWords);
685  }
686 
687  /* rebuild time sorted list of commands */
688  SC_BuildTimeIndexTable(AtsIndex);
689 
690  /* did we just append to an ats that was executing? */
691  if ((SC_OperData.AtsCtrlBlckAddr->AtpState == SC_EXECUTING) &&
692  (SC_OperData.AtsCtrlBlckAddr->AtsNumber == (AtsIndex + 1)))
693  {
694  /*
695  ** re-start the ats -- this will go thru the process of skipping
696  ** past due entries (all of the old entries that had already
697  ** been executed and all of the new entries with an old time)
698  */
699  if (SC_BeginAts(AtsIndex, 0))
700  {
701  SC_OperData.AtsCtrlBlckAddr->AtpState = SC_EXECUTING;
702  }
703  }
704 
705  /* notify cFE that we have modified the ats table */
706  CFE_TBL_Modified(SC_OperData.AtsTblHandle[AtsIndex]);
707 
708  return;
709 
710 } /* end SC_ProcessAppend */

714 /* */
715 /* Verify contents of ATS table data */
716 /* */
717 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
718 int32 SC_VerifyAtsTable (uint16 *Buffer, int32 BufferWords)
719 {
720  int32 Result = CFE_SUCCESS;
721  int32 BufferIndex = 0;
722  int32 CommandCount = 0;
723  int32 i;
724 
725  boolean StillProcessing = TRUE;
726 
727 
728  /* Initialize all command numbers as unused */
729  for (i = 0; i < SC_MAX_ATS_CMDS; i++)
730  {
731  SC_OperData.AtsDupTestArray[i] = SC_DUP_TEST_UNUSED;
732  }
733 
734  while (StillProcessing)
735  {
736  /* Verify the ATS table entry at the current buffer index */
737  Result = SC_VerifyAtsEntry(Buffer, BufferIndex, BufferWords);
738 
739  if (Result == SC_ERROR)
740  {
741  /* Entry at current buffer index is invalid */
742  StillProcessing = FALSE;
743  }
744  else if (Result == CFE_SUCCESS)
745  {
746  /* No more entries -- end of buffer or cmd num = 0 */
747  StillProcessing = FALSE;
748  }
749  else
750  {
751  /* Result is size (in words) of this entry */
752  BufferIndex += Result;
753  CommandCount++;
754  }
755  }
756 
757  if (Result == CFE_SUCCESS)
758  {
759  if (CommandCount == 0)
760  {
761  /* Table must contain at least one valid entry */
762  Result = SC_ERROR;
763 
764  CFE_EVS_SendEvent(SC_VERIFY_ATS_MPT_ERR_EID, CFE_EVS_ERROR,
765  "Verify ATS Table error: table is empty");
766  }
767  else
768  {
769  CFE_EVS_SendEvent(SC_VERIFY_ATS_EID, CFE_EVS_INFORMATION,
770  "Verify ATS Table: command count = %d, byte count = %d",
771  (int)CommandCount, (int)BufferIndex * 2);
772  }
773  }
774 
775  return(Result);
776 
777 } /* end SC_VerifyAtsTable */

780 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
781 /* */
782 /* Verify contents of one ATS table entry */
783 /* */
784 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
785 int32 SC_VerifyAtsEntry(uint16 *Buffer, int32 EntryIndex, int32 BufferWords)
786 {
787  SC_AtsEntryHeader_t *Entry = (SC_AtsEntryHeader_t *) &Buffer[EntryIndex];
788  CFE_SB_MsgPtr_t CmdPacket;
789 
790  int32 CommandBytes;
791  int32 CommandWords;
792  int32 Result = CFE_SUCCESS;
793 
794  /*
795  ** Verify the ATS table entry located at the indicated buffer offset
796  */
797  if (EntryIndex >= BufferWords)
798  {
799  /*
800  ** The process logic will prevent the index from ever exceeding
801  ** the size of the buffer due to bad table data content. Still,
802  ** we must include the "greater than" in the test above to
803  ** protect against our own potential coding errors.
804  */
805 
806  /* All done -- end of ATS buffer */
807  Result = CFE_SUCCESS;
808  }
809  else if (Entry->CmdNumber == 0)
810  {
811  /*
812  ** If there is at least one word remaining in the buffer then it
813  ** is OK to test the command number without fear of accessing
814  ** past the end of valid data because the command number is the
815  ** first element in an ATS entry structure.
816  */
817 
818  /* All done -- end of in-use portion of buffer */
819  Result = CFE_SUCCESS;
820  }
821  else if (Entry->CmdNumber > SC_MAX_ATS_CMDS)
822  {
823  /* Error -- invalid command number */
824  Result = SC_ERROR;
825 
826  CFE_EVS_SendEvent(SC_VERIFY_ATS_NUM_ERR_EID, CFE_EVS_ERROR,
827  "Verify ATS Table error: invalid command number: buf index = %d, cmd num = %d",
828  (int)EntryIndex, Entry->CmdNumber);
829  }
830  else if ((EntryIndex + SC_ATS_HDR_WORDS) > BufferWords)
831  {
832  /* Error -- not enough room for smallest possible ATS entry */
833  Result = SC_ERROR;
834 
835  CFE_EVS_SendEvent(SC_VERIFY_ATS_END_ERR_EID, CFE_EVS_ERROR,
836  "Verify ATS Table error: buffer full: buf index = %d, cmd num = %d, buf words = %d",
837  (int)EntryIndex, Entry->CmdNumber, (int)BufferWords);
838  }
839  else
840  {
841  /* Now it is OK to de-reference ATS cmd packet */
842  CmdPacket = (CFE_SB_MsgPtr_t) Entry->CmdHeader;
843 
844  /* Start with the byte length of the command packet */
845  CommandBytes = CFE_SB_GetTotalMsgLength(CmdPacket);
846 
847  /* Convert packet byte length to word length (round up odd bytes) */
848  CommandWords = (CommandBytes + 1) / 2;
849 
850  if ((CommandBytes < SC_PACKET_MIN_SIZE) || (CommandBytes > SC_PACKET_MAX_SIZE))
851  {
852  /* Error -- invalid command packet byte length */
853  Result = SC_ERROR;
854 
855  CFE_EVS_SendEvent(SC_VERIFY_ATS_PKT_ERR_EID, CFE_EVS_ERROR,
856  "Verify ATS Table error: invalid length: buf index = %d, cmd num = %d, pkt len = %d",
857  (int)EntryIndex, Entry->CmdNumber, (int)CommandBytes);
858  }
859  else if ((EntryIndex + SC_ATS_HDR_NOPKT_WORDS + CommandWords) > BufferWords)
860  {
861  /* Error -- packet must fit within buffer */
862  Result = SC_ERROR;
863 
864  CFE_EVS_SendEvent(SC_VERIFY_ATS_BUF_ERR_EID, CFE_EVS_ERROR,
865  "Verify ATS Table error: buffer overflow: buf index = %d, cmd num = %d, pkt len = %d",
866  (int)EntryIndex, Entry->CmdNumber, (int)CommandBytes);
867  }
868  else if (SC_OperData.AtsDupTestArray[Entry->CmdNumber -1] != SC_DUP_TEST_UNUSED)
869  {
870  /* Entry with duplicate command number is invalid */
871  Result = SC_ERROR;
872 
873  CFE_EVS_SendEvent(SC_VERIFY_ATS_DUP_ERR_EID, CFE_EVS_ERROR,
874  "Verify ATS Table error: dup cmd number: buf index = %d, cmd num = %d, dup index = %d",
875  (int)EntryIndex, Entry->CmdNumber,
876  (int)SC_OperData.AtsDupTestArray[Entry->CmdNumber -1]);
877  }
878  else
879  {
880  /* Compute length (in words) for this ATS table entry */
881  Result = SC_ATS_HDR_NOPKT_WORDS + CommandWords;
882 
883  /* Mark this ATS command ID as in use at this table index */
884  SC_OperData.AtsDupTestArray[Entry->CmdNumber -1] = EntryIndex;
885  }
886  }
887 
888  return(Result);
889 
890 } /* End of SC_VerifyAtsEntry */
