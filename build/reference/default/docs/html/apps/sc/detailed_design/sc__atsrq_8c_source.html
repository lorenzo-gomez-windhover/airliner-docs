88 /* Starts an ATS */
89 /* */
90 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
91 void SC_StartAtsCmd (CFE_SB_MsgPtr_t CmdPacket)
92 {
93  uint16 AtsIndex; /* ATS array index */
94 
95  if (SC_VerifyCmdLength(CmdPacket, sizeof(SC_StartAtsCmd_t)))
96  {
97 
98  /* convert ATS ID to array index */
99  AtsIndex = (((SC_StartAtsCmd_t*)CmdPacket) -> AtsId) - 1;
100 
101  /* validate ATS array index */
102  if (AtsIndex < SC_NUMBER_OF_ATS)
103  {
104  /* make sure that there is no ATS running on the ATP */
105  if (SC_OperData.AtsCtrlBlckAddr -> AtpState == SC_IDLE)
106  {
107  /* make sure the specified ATS is ready */
108  if (SC_OperData.AtsInfoTblAddr[AtsIndex].NumberOfCommands > 0)
109  {
110  /* start the ats */
111  if (SC_BeginAts (AtsIndex, 0))
112  {
113  /* finish the ATP control block .. */
114  SC_OperData.AtsCtrlBlckAddr -> AtpState = SC_EXECUTING;
115 
116  /* increment the command request counter */
117  SC_AppData.CmdCtr++;
118 
119  CFE_EVS_SendEvent(SC_STARTATS_CMD_INF_EID,
120  CFE_EVS_INFORMATION,
121  "ATS %c Execution Started",
122  (AtsIndex ? 'B' : 'A'));
123  }
124  else
125  { /* could not start the ats, all commands were skipped */
126  /* event message was sent from SC_BeginAts */
127  /* increment the command request error counter */
128  SC_AppData.CmdErrCtr++;
129 
130  } /* end if */
131  }
132  else
133  { /* the ats didn't have any commands in it */
134 
135  CFE_EVS_SendEvent(SC_STARTATS_CMD_NOT_LDED_ERR_EID,
136  CFE_EVS_ERROR,
137  "Start ATS Rejected: ATS %c Not Loaded",
138  (AtsIndex ? 'B' : 'A'));
139 
140  /* increment the command request error counter */
141  SC_AppData.CmdErrCtr++;
142 
143  } /* end if */
144 
145  }
146  else
147  { /* the ATS is being used */
148 
149  CFE_EVS_SendEvent(SC_STARTATS_CMD_NOT_IDLE_ERR_EID,
150  CFE_EVS_ERROR,
151  "Start ATS Rejected: ATP is not Idle");
152  /* increment the command request error counter */
153  SC_AppData.CmdErrCtr++;
154 
155  } /* end if */
156  }
157  else
158  { /* the specified ATS id is not valid */
159 
160  CFE_EVS_SendEvent(SC_STARTATS_CMD_INVLD_ID_ERR_EID,
161  CFE_EVS_ERROR,
162  "Start ATS %d Rejected: Invalid ATS ID",
163  ((SC_StartAtsCmd_t*)CmdPacket) -> AtsId);
164 
165  /* increment the command request error counter */
166  SC_AppData.CmdErrCtr++;
167 
168  } /* end if */
169  }
170 } /* end SC_StartAtsCmd */

229 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
230 /* */
231 /* Function for stating an ATS */
232 /* */
233 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
234 boolean SC_BeginAts (uint16 AtsIndex, uint16 TimeOffset)
235 {
236  SC_AtsEntryHeader_t *Entry; /* ATS table entry pointer */
237  int32 EntryIndex; /* ATS entry location in table */
238  SC_AbsTimeTag_t ListCmdTime; /* list entry execution time */
239  int32 TimeIndex; /* the current time buffer index */
240  int32 CmdIndex; /* ATS command index (cmd num - 1) */
241  boolean ReturnCode;
242  SC_AbsTimeTag_t TimeToStartAts; /* the REAL time to start the ATS */
243  uint16 CmdsSkipped = 0;
244 
245  TimeToStartAts = SC_ComputeAbsTime (TimeOffset);
246 
247  /*
248  ** Loop through the commands until a time tag is found that
249  ** has a time greater than or equal to the current time OR
250  ** all of the commands have been skipped
251  */
252  TimeIndex = 0; /* pointer into the time index table */
253 
254  while (TimeIndex < SC_OperData.AtsInfoTblAddr[AtsIndex].NumberOfCommands)
255  {
256  /* first get the cmd index at this list entry */
257  CmdIndex = SC_AppData.AtsTimeIndexBuffer[AtsIndex][TimeIndex];
258  /* then get the entry index from the cmd index table */
259  EntryIndex = SC_AppData.AtsCmdIndexBuffer[AtsIndex][CmdIndex];
260  /* then get a pointer to the ATS entry data */
261  Entry = (SC_AtsEntryHeader_t *) &SC_OperData.AtsTblAddr[AtsIndex][EntryIndex];
262  /* then get cmd execution time from the ATS entry */
263  ListCmdTime = SC_GetAtsEntryTime(Entry);
264 
265  /* compare ATS start time to this list entry time */
266  if (SC_CompareAbsTime(TimeToStartAts, ListCmdTime))
267  {
268  /* start time is greater than this list entry time */
269 
270  SC_OperData.AtsCmdStatusTblAddr[AtsIndex][CmdIndex] = SC_SKIPPED;
271  CmdsSkipped++;
272  TimeIndex++;
273  }
274  else
275  {
276  /* start time is less than or equal to this list entry */
277  break;
278  }
279  }
280 
281  /*
282  ** Check to see if the whole ATS was skipped
283  */
284  if (TimeIndex == SC_OperData.AtsInfoTblAddr[AtsIndex].NumberOfCommands)
285  {
286 
287  CFE_EVS_SendEvent(SC_ATS_SKP_ALL_ERR_EID,
288  CFE_EVS_ERROR,
289  "All ATS commands were skipped, ATS stopped");
290 
291  /* stop the ats */
292  SC_KillAts();
293 
294  ReturnCode = FALSE;
295 
296  }
297  else
298  { /* there is at least one command to execute */
299 
300  /*
301  ** Initialize the ATP Control Block.
302  */
303  /* leave the atp state alone, it will be updated by the caller */
304  SC_OperData.AtsCtrlBlckAddr -> AtsNumber = AtsIndex + 1;
305  SC_OperData.AtsCtrlBlckAddr -> CmdNumber = CmdIndex;
306  SC_OperData.AtsCtrlBlckAddr -> TimeIndexPtr = TimeIndex;
307 
308  /* send an event for number of commands skipped */
309  CFE_EVS_SendEvent(SC_ATS_ERR_SKP_DBG_EID,
310  CFE_EVS_DEBUG,
311  "ATS started, skipped %d commands",
312  CmdsSkipped);
313  /*
314  ** Set the next command time for the ATP
315  */
316  SC_AppData.NextCmdTime[SC_ATP] = ListCmdTime;
317 
318  ReturnCode = TRUE;
319 
320  } /* end if */
321 
322  return (ReturnCode);
323 
324 } /* end SC_BeginAts */

673  NewTime.Subseconds = 0;

686  "Jump Cmd: Skipped %d ATS commands",

688 
689  }

691  } /* end if */

697  CFE_EVS_ERROR,

701  } /* end if */

749 void SC_AppendAtsCmd (CFE_SB_MsgPtr_t CmdPacket)

752  uint16 AtsIndex; /* index (not ID) of target ATS */
753 
754  if (SC_VerifyCmdLength(CmdPacket, sizeof(SC_AppendAtsCmd_t)))
755  {

757  AtsIndex = AppendCmd->AtsId - 1;
758

760  {
761  /* invalid target ATS selection */
762  SC_AppData.CmdErrCtr++;
763 
764  CFE_EVS_SendEvent(SC_APPEND_CMD_ARG_ERR_EID, CFE_EVS_ERROR,
765  "Append ATS error: invalid ATS ID = %d", AppendCmd->AtsId);
766  }
767  else if (SC_OperData.AtsInfoTblAddr[AtsIndex].NumberOfCommands == 0)
768  {
769  /* target ATS table is empty */
770  SC_AppData.CmdErrCtr++;
771 
772  CFE_EVS_SendEvent(SC_APPEND_CMD_TGT_ERR_EID, CFE_EVS_ERROR,
773  "Append ATS %c error: ATS table is empty", 'A' + AtsIndex);
774  }
775  else if (SC_AppData.AppendEntryCount == 0)
776  {
777  /* append table is empty */
778  SC_AppData.CmdErrCtr++;
779 
780  CFE_EVS_SendEvent(SC_APPEND_CMD_SRC_ERR_EID, CFE_EVS_ERROR,
781  "Append ATS %c error: Append table is empty", 'A' + AtsIndex);
782  }
783  else if ((SC_OperData.AtsInfoTblAddr[AtsIndex].AtsSize + SC_AppData.AppendWordCount) > SC_ATS_BUFF_SIZE)

786  SC_AppData.CmdErrCtr++;

789  "Append ATS %c error: ATS size = %d, Append size = %d, ATS buffer = %d",
790  'A' + AtsIndex, (int)SC_OperData.AtsInfoTblAddr[AtsIndex].AtsSize,
791  SC_AppData.AppendWordCount, SC_ATS_BUFF_SIZE);
792  }
793  else

796  SC_AppData.AppendCmdArg = AppendCmd->AtsId;

800

802  SC_AppData.CmdCtr++;
803

806  'A' + AtsIndex, SC_AppData.AppendEntryCount);
807  }
808  }
809 } /* end SC_AppendAtsCmd */
