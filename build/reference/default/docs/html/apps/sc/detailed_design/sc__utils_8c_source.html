91 SC_AbsTimeTag_t SC_GetAtsEntryTime(SC_AtsEntryHeader_t *Entry)
92 {
93  /*
94  ** ATS Entry Header looks like this...
95  **
96  ** uint16 CmdNumber;
97  **
98  ** uint16 TimeTag1;
99  ** uint16 TimeTag2;
100  **
101  ** uint8 CmdHeader[CFE_SB_CMD_HDR_SIZE];
102  **
103  ** The command packet data is variable length,
104  ** only the command packet header is shown here.
105  */
106  SC_AbsTimeTag_t AbsTimeTag = 0;
107 
108  /* Store as bytes to avoid boundary, endian and strict-aliasing issues */
109  CFE_PSP_MemCpy(&AbsTimeTag, &Entry->TimeTag1, sizeof(SC_AbsTimeTag_t));
110 
111  return(AbsTimeTag);
112 
113 } /* End of SC_GetAtsEntryTime() */

121 SC_AbsTimeTag_t SC_ComputeAbsTime(uint16 RelTime)
122 {
123  CFE_TIME_SysTime_t AbsoluteTimeWSubs;

125  CFE_TIME_SysTime_t ResultTimeWSubs;

130  AbsoluteTimeWSubs.Subseconds = 0;

134  /*

136  */
137  ResultTimeWSubs = CFE_TIME_Add ( AbsoluteTimeWSubs, RelTimeWSubs);

139  /* We don't need subseconds */

142 } /* end of SC_ComputeAbsTime */
