482  if (CI_VerifyCmdLength (MsgPtr, ExpectedLength))
483  {
484  /* Check if command is not registered */
485  authDataPtr = ((CI_CmdAuthData_t *) MsgPtr);
486  CmdData = CI_GetRegisterdCmd(authDataPtr->msgID, authDataPtr->cmdCode, &i);
487  if (CmdData == NULL || i == CI_INVALID_VALUE)
488  {
489  (void) CFE_EVS_SendEvent(CI_CMD_AUTH_NOT_REG_EID, CFE_EVS_ERROR, "Cmd (%u) not registered for msgId (0x%04X)",
490  authDataPtr->cmdCode, (unsigned short)authDataPtr->msgID);
491  CI_AppData.HkTlm.usCmdErrCnt++;
492  goto CI_CmdAuthorize_Exit_Tag;
493  }
494 
495  /* Check if command is not 2-step */
496  if (CmdData->step != STEP_2)
497  {
498  (void) CFE_EVS_SendEvent(CI_CMD_AUTH_INV_MODE_EID, CFE_EVS_ERROR, "Cmd (%u) for msgId (0x%04X) not 2-step",
499  authDataPtr->cmdCode, (unsigned short)authDataPtr->msgID);
500  CI_AppData.HkTlm.usCmdErrCnt++;
501  goto CI_CmdAuthorize_Exit_Tag;
502  }
503 
504  /* Check if command is already authorized */
505  if (CmdData->state == AUTHORIZED)
506  {
507  (void) CFE_EVS_SendEvent(CI_CMD_AUTH_INV_STATE_EID, CFE_EVS_ERROR, "Cmd (%u) already authorized for msgId (0x%04X)",
508  authDataPtr->cmdCode, (unsigned short)authDataPtr->msgID);
509  CI_AppData.HkTlm.usCmdErrCnt++;
510  goto CI_CmdAuthorize_Exit_Tag;
511  }
512 
513  /* Lock the mutexes */
514  OS_MutSemTake(CI_AppData.ConfigTblMutex);
515  OS_MutSemTake(CI_AppData.TimeoutTblMutex);
516 
517  /* Update values */
518  CmdData->state = AUTHORIZED;
519  CI_AppData.TimeoutTbl.time[i] = CI_CMD_MAX_TIMEOUT;
520 
521  /* Unlock the mutexes */
522  OS_MutSemGive(CI_AppData.ConfigTblMutex);
523  OS_MutSemGive(CI_AppData.TimeoutTblMutex);
524 
525  (void) CFE_EVS_SendEvent(CI_CMD_AUTHORIZED_EID, CFE_EVS_INFORMATION, "Cmd (%u) authorized for msgId (0x%04X)",
526  authDataPtr->cmdCode, (unsigned short)authDataPtr->msgID);
527  CI_AppData.HkTlm.usCmdCnt++;
528  }
529 
530 CI_CmdAuthorize_Exit_Tag:
531  return;
532 }

880 /* Determine Validity of Command */
881 /* */
882 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
883 boolean CI_ValidateCmd(const CFE_SB_Msg_t* MsgPtr, uint32 MsgSize)
884 {
885  boolean bResult = FALSE;
886  uint32 usMsgLen = 0;
887  CCSDS_CmdPkt_t *cmdPkt = 0;
888 
889  /* Verify CCSDS version */
890  if (CCSDS_RD_VERS(MsgPtr->Hdr) != 0)
891  {
892  goto CI_ValidateCmd_Exit_Tag;
893  }
894 
895  /* Verify secondary header present */
896  if (CCSDS_RD_SHDR(MsgPtr->Hdr) == 0)
897  {
898  goto CI_ValidateCmd_Exit_Tag;
899  }
900 
901  /* Verify length */
902  usMsgLen = CFE_SB_GetTotalMsgLength((CFE_SB_Msg_t *) MsgPtr);
903  if (usMsgLen != MsgSize)
904  {
905  goto CI_ValidateCmd_Exit_Tag;
906  }
907 
908  /* Verify valid checksum if command packet */
909  if (CCSDS_RD_TYPE(MsgPtr->Hdr) != CCSDS_TLM)
910  {
911  cmdPkt = (CCSDS_CmdPkt_t *)MsgPtr;
912  if (CCSDS_RD_CHECKSUM(cmdPkt->SecHdr) != 0)
913  {
914  if (CFE_SB_ValidateChecksum((CFE_SB_MsgPtr_t)MsgPtr) != TRUE)
915  {
916  goto CI_ValidateCmd_Exit_Tag;
917  }
918  }
919  else
920  {
921  /* If no checksum present check the CI_CHECKSUM_REQUIRED
922  * macro defined in the mission config */
923  if (CI_CHECKSUM_REQUIRED == 1)
924  {
925  goto CI_ValidateCmd_Exit_Tag;
926  }
927  }
928  }
929 
930  bResult = TRUE;
931 
932 CI_ValidateCmd_Exit_Tag:
933  return (bResult);
934 }
