78 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
79 int32 CI_InitTbls(void)
80 {
81  int32 iStatus=0;
82 
83  /* Register Config table */
84  iStatus = CFE_TBL_Register(&CI_AppData.ConfigTblHdl,
85  CI_CONFIG_TABLENAME,
86  sizeof(CI_ConfigTblEntry_t),
87  CFE_TBL_OPT_DEFAULT,
88  CI_ValidateConfigTbl);
89  if (iStatus != CFE_SUCCESS)
90  {
91  (void) CFE_EVS_SendEvent(CI_INIT_ERR_EID, CFE_EVS_ERROR,
92  "Failed to register Config table (0x%08X)",
93  (unsigned int)iStatus);
94  goto CI_InitTbls_Exit_Tag;
95  }
96 
97  /* Register Timeout dump table */
98  iStatus = CFE_TBL_Register(&CI_AppData.TimeoutTblHdl,
99  CI_TIMEOUT_TABLENAME,
100  sizeof(CI_TimeoutTblEntry_t),
101  CFE_TBL_OPT_USR_DEF_ADDR,
102  0);
103  if (iStatus != CFE_SUCCESS)
104  {
105  (void) CFE_EVS_SendEvent(CI_INIT_ERR_EID, CFE_EVS_ERROR,
106  "Failed to register Timeout table (0x%08X)",
107  (unsigned int)iStatus);
108  goto CI_InitTbls_Exit_Tag;
109  }
110 
111  /* Load Config table file */
112  iStatus = CFE_TBL_Load(CI_AppData.ConfigTblHdl,
113  CFE_TBL_SRC_FILE,
114  CI_CONFIG_TABLE_FILENAME);
115  if (iStatus != CFE_SUCCESS)
116  {
117  (void) CFE_EVS_SendEvent(CI_INIT_ERR_EID, CFE_EVS_ERROR,
118  "Failed to load Config Table (0x%08X)",
119  (unsigned int)iStatus);
120  goto CI_InitTbls_Exit_Tag;
121  }
122 
123  /* Load Timeout table file */
124  iStatus = CFE_TBL_Load(CI_AppData.TimeoutTblHdl,
125  CFE_TBL_SRC_ADDRESS,
126  &CI_AppData.TimeoutTbl);
127  if (iStatus != CFE_SUCCESS)
128  {
129  (void) CFE_EVS_SendEvent(CI_INIT_ERR_EID, CFE_EVS_ERROR,
130  "Failed to load Timeout Table (0x%08X)",
131  (unsigned int)iStatus);
132  goto CI_InitTbls_Exit_Tag;
133  }
134 
135  iStatus = CI_AcquireConfigPointers();
136 
137 CI_InitTbls_Exit_Tag:

144 /* Validate Config Table */
145 /* */
146 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
147 int32 CI_ValidateConfigTbl(void* ConfigTblPtr)
148 {
149  int32 iStatus = 0;
150  uint32 i = 0;
151  CI_ConfigTblEntry_t* CI_ConfigTblPtr = (CI_ConfigTblEntry_t*)(ConfigTblPtr);
152 
153  if (ConfigTblPtr == NULL)
154  {
155  iStatus = -1;
156  goto CI_ValidateConfigTbl_Exit_Tag;
157  }
158 
159  /* Perform simple validation on registered command states */
160  for(i = 0; i < CI_MAX_RGST_CMDS; ++i)
161  {
162  /* If message is non-zero perform validation. Skip otherwise. */
163  if(CI_ConfigTblPtr->cmds[i].mid != 0)
164  {
165  /* Check if stepping is valid */
166  if(CI_ConfigTblPtr->cmds[i].step != STEP_1 &&
167  CI_ConfigTblPtr->cmds[i].step != STEP_2)
168  {
169  iStatus = -1;
170  break;
171  }
172 
173  /* Check if state is valid */
174  if(CI_ConfigTblPtr->cmds[i].state != UNAUTHORIZED &&
175  CI_ConfigTblPtr->cmds[i].state != AUTHORIZED)
176  {
177  iStatus = -1;
178  break;
179  }
180  }
181  }
182 
183 CI_ValidateConfigTbl_Exit_Tag:
184  return (iStatus);
185 }

190 /* */
191 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
192 int32 CI_AcquireConfigPointers(void)
193 {
194  int32 iStatus = CFE_SUCCESS;
195 
196  /*
197  ** Release the table
198  */
199  (void) CFE_TBL_ReleaseAddress(CI_AppData.ConfigTblHdl);
200  (void) CFE_TBL_ReleaseAddress(CI_AppData.TimeoutTblHdl);
201 
202  /*
203  ** Manage the tables
204  */
205  iStatus = CFE_TBL_Manage(CI_AppData.ConfigTblHdl);
206  if ((iStatus != CFE_SUCCESS) && (iStatus != CFE_TBL_INFO_UPDATED))
207  {
208  (void) CFE_EVS_SendEvent(CI_CONFIG_TABLE_ERR_EID, CFE_EVS_ERROR,
209  "Failed to manage Config table (0x%08X)",
210  (unsigned int)iStatus);
211  goto CI_AcquireConfigPointers_Exit_Tag;
212  }
213 
214  iStatus = CFE_TBL_Manage(CI_AppData.TimeoutTblHdl);
215  if ((iStatus != CFE_SUCCESS) && (iStatus != CFE_TBL_INFO_UPDATED))
216  {
217  (void) CFE_EVS_SendEvent(CI_TIMEOUT_TABLE_ERR_EID, CFE_EVS_ERROR,
218  "Failed to manage Timeout table (0x%08X)",
219  (unsigned int)iStatus);
220  goto CI_AcquireConfigPointers_Exit_Tag;
221  }
222 
223  /*
224  ** Get a pointer to the tables
225  */
226  iStatus = CFE_TBL_GetAddress((void*)&CI_AppData.ConfigTblPtr,
227  CI_AppData.ConfigTblHdl);
228  if (iStatus == CFE_TBL_INFO_UPDATED)
229  {
230  iStatus = CFE_SUCCESS;
231  }
232  else if(iStatus != CFE_SUCCESS)
233  {
234  CI_AppData.ConfigTblPtr = 0;
235  (void) CFE_EVS_SendEvent(CI_CONFIG_TABLE_ERR_EID, CFE_EVS_ERROR,
236  "Failed to get Config table's address (0x%08X)",
237  (unsigned int)iStatus);
238  }
239 
240  iStatus = CFE_TBL_GetAddress((void*)&CI_AppData.TimeoutTbl,
241  CI_AppData.TimeoutTblHdl);
242  if (iStatus == CFE_TBL_INFO_UPDATED)
243  {
244  iStatus = CFE_SUCCESS;
245  }
246  else if(iStatus != CFE_SUCCESS)
247  {
248  (void) CFE_EVS_SendEvent(CI_TIMEOUT_TABLE_ERR_EID, CFE_EVS_ERROR,
249  "Failed to get Timeout table's address (0x%08X)",
250  (unsigned int)iStatus);
251  }
252 
253 CI_AcquireConfigPointers_Exit_Tag:
254  return (iStatus);
255 
256 } /* End of CI_AcquirePointers */
