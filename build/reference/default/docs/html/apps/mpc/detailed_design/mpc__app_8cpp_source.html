435

438

442  switch (MsgId)

445  {
446  /* If vehicle local position has been received begin

448  if(ProcessDataPipe() == true) {
449  Execute();

455  {
456  ProcessCmdPipe();
457  ReportHousekeeping();
458  break;
459  }
460

465  }
466  }
467  }
468  else if (iStatus == CFE_SB_NO_MESSAGE)
469  {
470  /* If there's no incoming message, you can do something here, or
471  * nothing. Note, this section is dead code only if the iBlocking arg

473  iStatus = CFE_SUCCESS;
474  }
475  else if (iStatus == CFE_SB_TIME_OUT)
476  {
477  /* If there's no incoming message within a specified time (via the
478  * iBlocking arg, you can do something here, or nothing.
479  * Note, this section is dead code only if the iBlocking arg
480  * is CFE_SB_PEND_FOREVER. */
481  iStatus = CFE_SUCCESS;
482  }
483  else
484  {
485  (void) CFE_EVS_SendEvent(MPC_RCVMSG_ERR_EID, CFE_EVS_ERROR,
486  "SCH pipe read error (0x%08lX).", iStatus);

630  }

634  break;
635  }

638  (void) CFE_EVS_SendEvent(MPC_RCVMSG_ERR_EID, CFE_EVS_ERROR,

641  }

1116  uint64 t = PX4LIB_GetPX4TimeUs();
1117  float dt = t_prev != 0 ? (t - t_prev) / 1e6f : 0.004f;
1118  t_prev = t;
1119 
1120  /* Set default max velocity in xy to vel_max */
1121  m_VelMaxXy = ConfigTblPtr->XY_VEL_MAX;
1122 
1123  /* Reset flags when landed */
1124  if (m_VehicleLandDetectedMsg.Landed)
1125  {
1126  m_ResetPositionSetpoint = TRUE;
1127  m_ResetAltitudeSetpoint = TRUE;
1128  m_DoResetAltPos = TRUE;
1129  m_ModeAuto = FALSE;
1130  m_PositionHoldEngaged = FALSE;

1136  m_ResetYawSetpoint = TRUE;
1137  m_HoldOffboardXY = FALSE;
1138  m_HoldOffboardZ = FALSE;
1139 
1140  /* Also reset previous setpoints */
1141  m_YawTakeoff = m_Yaw;
1142  m_VelocitySetpointPrevious.Zero();
1143  m_VelocityPrevious.Zero();
1144 
1145  /* Make sure attitude setpoint output "disables" attitude control */
1146  m_VehicleAttitudeSetpointMsg.Thrust = 0.0f;
1147  m_VehicleAttitudeSetpointMsg.Timestamp = PX4LIB_GetPX4TimeUs();
1148  }
1149 
1150  if (!m_InTakeoff && (!m_VehicleLandDetectedMsg.Landed && m_WasLanded) && m_VehicleControlModeMsg.Armed == PX4_ARMING_STATE_ARMED)
1151  {
1152  m_InTakeoff = TRUE;
1153  /* This ramp starts negative and goes to positive later because we want to
1154  * be as smooth as possible. If we start at 0, we alrady jump to hover throttle. */
1155  m_TakeoffVelLimit = -.5f;
1156  }
1157 
1158  else if (!m_VehicleControlModeMsg.Armed == PX4_ARMING_STATE_ARMED) {
1159  /* If we're disarmed and for some reason were in a smooth takeoff, we reset that. */
1160  m_InTakeoff = FALSE;
1161  }
1162

1421 void MPC::GenerateAttitudeSetpoint(float dt)
1422 {
1423  /* Yaw setpoint is integrated over time, but we don't want to integrate the offset's */
1424  m_VehicleAttitudeSetpointMsg.YawBody -= m_ManYawOffset;
1425  m_ManYawOffset = 0.0f;
1426 
1427  /* Reset yaw setpoint to current position if needed. */
1428  if (m_ResetYawSetpoint)
1429  {
1430  m_ResetYawSetpoint = FALSE;
1431  m_VehicleAttitudeSetpointMsg.YawBody = m_Yaw;
1432  }
1433  else if (!m_VehicleLandDetectedMsg.Landed &&
1434  !(!m_VehicleControlModeMsg.ControlAltitudeEnabled && m_ManualControlSetpointMsg.Z < 0.1f))
1435  {
1436  float YawTarget;
1437  float YawOffs;
1438 
1439  /* Do not move yaw while sitting on the ground. */
1440 
1441  /* We want to know the real constraint, and global overrides manual. */
1442  const float YawRateMax = (math::radians(ConfigTblPtr->MAN_Y_MAX) < math::radians(ConfigTblPtr->MC_YAWRATE_MAX)) ? math::radians(ConfigTblPtr->MAN_Y_MAX) :
1443  math::radians(ConfigTblPtr->MC_YAWRATE_MAX);
1444  const float YawOffsetMax = YawRateMax / math::radians(ConfigTblPtr->MC_YAW_P);
1445 
1446  m_VehicleAttitudeSetpointMsg.YawSpMoveRate = m_ManualControlSetpointMsg.R * YawRateMax;
1447  YawTarget = _wrap_pi(m_VehicleAttitudeSetpointMsg.YawBody + m_VehicleAttitudeSetpointMsg.YawSpMoveRate * dt);
1448  YawOffs = _wrap_pi(YawTarget - m_Yaw);
1449 
1450  /* If the yaw offset became too big for the system to track stop
1451  * shifting it, only allow if it would make the offset smaller again. */
1452  if (fabsf(YawOffs) < YawOffsetMax ||
1453  (m_VehicleAttitudeSetpointMsg.YawSpMoveRate > 0 && YawOffs < 0) ||
1454  (m_VehicleAttitudeSetpointMsg.YawSpMoveRate < 0 && YawOffs > 0))
1455  {
1456  m_VehicleAttitudeSetpointMsg.YawBody = YawTarget;
1457  }
1458  }
1459 
1460  /* Control throttle directly if no climb rate controller is active */
1461  if (!m_VehicleControlModeMsg.ControlClimbRateEnabled)
1462  {
1463  float ThrVal = ThrottleCurve(m_ManualControlSetpointMsg.Z, ConfigTblPtr->THR_HOVER);
1464  m_VehicleAttitudeSetpointMsg.Thrust = fmin(ThrVal, ConfigTblPtr->MANTHR_MAX);
1465 
1466  /* Enforce minimum throttle if not landed */
1467  if (!m_VehicleLandDetectedMsg.Landed)
1468  {
1469  m_VehicleAttitudeSetpointMsg.Thrust = fmax(m_VehicleAttitudeSetpointMsg.Thrust, ConfigTblPtr->MANTHR_MIN);
1470  }
1471  }
1472 
1473  /* Control roll and pitch directly if no aiding velocity controller is active. */
1474  if (!m_VehicleControlModeMsg.ControlVelocityEnabled)
1475  {
1476  m_VehicleAttitudeSetpointMsg.RollBody = m_ManualControlSetpointMsg.Y * math::radians(ConfigTblPtr->MAN_TILT_MAX);
1477  m_VehicleAttitudeSetpointMsg.PitchBody = -m_ManualControlSetpointMsg.X * math::radians(ConfigTblPtr->MAN_TILT_MAX);
1478 
1479  /* Only if optimal recovery is not used, modify roll/pitch. */
1480  if (ConfigTblPtr->VT_OPT_RECOV_EN <= 0)
1481  {
1482  /* Construct attitude setpoint rotation matrix. modify the setpoints for roll
1483  * and pitch such that they reflect the user's intention even if a yaw error
1484  * (yaw_sp - yaw) is present. In the presence of a yaw error constructing a rotation matrix
1485  * from the pure euler angle setpoints will lead to unexpected attitude behaviour from
1486  * the user's view as the euler angle sequence uses the yaw setpoint and not the current
1487  * heading of the vehicle.
1488  */
1489 
1490  /* Calculate our current yaw error. */
1491  float yaw_error = _wrap_pi(m_VehicleAttitudeSetpointMsg.YawBody - m_Yaw);
1492 
1493  // Compute the vector obtained by rotating a z unit vector by the rotation
1494  // given by the roll and pitch commands of the user
1495  math::Vector3F zB = {0, 0, 1};
1496  math::Matrix3F3 R_sp_roll_pitch = math::Matrix3F3::FromEuler(m_VehicleAttitudeSetpointMsg.RollBody, m_VehicleAttitudeSetpointMsg.PitchBody, 0);
1497  math::Vector3F z_roll_pitch_sp = R_sp_roll_pitch * zB;
1498 
1499  /* Transform the vector into a new frame which is rotated around the z axis
1500  * by the current yaw error. this vector defines the desired tilt when we look
1501  * into the direction of the desired heading.
1502  */
1503  math::Matrix3F3 R_yaw_correction = math::Matrix3F3::FromEuler(0.0f, 0.0f, -yaw_error);
1504  z_roll_pitch_sp = R_yaw_correction * z_roll_pitch_sp;
1505 
1506  /* Use the formula z_roll_pitch_sp = R_tilt * [0;0;1]
1507  * R_tilt is computed from_euler; only TRUE if cos(roll) not equal zero
1508  * -> valid if roll is not +-pi/2;
1509  */
1510  m_VehicleAttitudeSetpointMsg.RollBody = -asinf(z_roll_pitch_sp[1]);
1511  m_VehicleAttitudeSetpointMsg.PitchBody = atan2f(z_roll_pitch_sp[0], z_roll_pitch_sp[2]);
1512  }
1513 
1514  /* Copy quaternion setpoint to attitude setpoint topic. */
1515  math::Quaternion q_sp(m_VehicleAttitudeSetpointMsg.RollBody, m_VehicleAttitudeSetpointMsg.PitchBody, m_VehicleAttitudeSetpointMsg.YawBody);
1516  q_sp.copyTo(m_VehicleAttitudeSetpointMsg.Q_D);
1517  m_VehicleAttitudeSetpointMsg.Q_D_Valid = TRUE;
1518  }
1519 
1520  m_VehicleAttitudeSetpointMsg.Timestamp = PX4LIB_GetPX4TimeUs();
1521 }

1588  ManVelSp[1] /= ManVelHorLength;
1589  }

1592  ResetPosSetpoint();

1600

1665  osalbool SmoothPosTransition = PosHoldDesired
1666  && (fabsf(ConfigTblPtr->ACC_HOR_MAX - m_AccelerationStateLimitXY) < FLT_EPSILON) &&
1667  (ConfigTblPtr->HOLD_MAX_XY < FLT_EPSILON || VelXyMag < ConfigTblPtr->HOLD_MAX_XY);
1668

1670  if (SmoothPosTransition)
1671  {
