48 /* Initialize Config Table */
49 /* */
50 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
51 int32 MPC::InitConfigTbl()
52 {
53  int32 iStatus=0;
54 
55  /* Register Config table */
56  iStatus = CFE_TBL_Register(&ConfigTblHdl,
57  MPC_CONFIG_TABLENAME,
58  (sizeof(MPC_ConfigTbl_t)),
59  CFE_TBL_OPT_DEFAULT,
60  MPC::ValidateConfigTbl);
61  if (iStatus != CFE_SUCCESS)
62  {
63  /* Note, a critical table could return another nominal code. If this table is
64  * made critical this logic would have to change. */
65  (void) CFE_EVS_SendEvent(MPC_CFGTBL_REG_ERR_EID, CFE_EVS_ERROR,
66  "Failed to register config table (0x%08lX)",
67  iStatus);
68  goto MPC_InitConfigTbl_Exit_Tag;
69  }
70 
71  /* Load Config table file */
72  iStatus = CFE_TBL_Load(ConfigTblHdl,
73  CFE_TBL_SRC_FILE,
74  MPC_CONFIG_TABLE_FILENAME);
75  if (iStatus != CFE_SUCCESS)
76  {
77  /* Note, CFE_SUCCESS is for a successful full table load. If a partial table
78  load is desired then this logic would have to change. */
79  (void) CFE_EVS_SendEvent(MPC_CFGTBL_LOAD_ERR_EID, CFE_EVS_ERROR,
80  "Failed to load Config Table (0x%08lX)",
81  iStatus);
82  goto MPC_InitConfigTbl_Exit_Tag;
83  }
84 
85  iStatus = AcquireConfigPointers();
86 
87 MPC_InitConfigTbl_Exit_Tag:
88  return iStatus;
89 }

95 /* Acquire Config Pointers */
96 /* */
97 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
98 int32 MPC::AcquireConfigPointers(void)
99 {
100  int32 iStatus = CFE_SUCCESS;
101 
102  /*
103  ** Release the table
104  */
105  /* TODO: This return value can indicate success, error, or that the info has been
106  * updated. We ignore this return value in favor of checking CFE_TBL_Manage(), but
107  * be sure this is the behavior you want. */
108  (void) CFE_TBL_ReleaseAddress(ConfigTblHdl);
109 
110  /*
111  ** Manage the table
112  */
113  iStatus = CFE_TBL_Manage(ConfigTblHdl);
114  if ((iStatus != CFE_SUCCESS) && (iStatus != CFE_TBL_INFO_UPDATED))
115  {
116  (void) CFE_EVS_SendEvent(MPC_CFGTBL_MANAGE_ERR_EID, CFE_EVS_ERROR,
117  "Failed to manage MPC Config table (0x%08lX)",
118  iStatus);
119  goto MPC_AcquireConfigPointers_Exit_Tag;
120  }
121 
122  /*
123  ** Get a pointer to the table
124  */
125  iStatus = CFE_TBL_GetAddress((void**)&ConfigTblPtr, ConfigTblHdl);
126  if (iStatus == CFE_TBL_INFO_UPDATED)
127  {
128  iStatus = CFE_SUCCESS;
129  UpdateParamsFromTable();
130  }
131  else if(iStatus != CFE_SUCCESS)
132  {
133  ConfigTblPtr = 0;
134  (void) CFE_EVS_SendEvent(MPC_CFGTBL_GETADDR_ERR_EID, CFE_EVS_ERROR,
135  "Failed to get Config table's address (0x%08lX)",
136  iStatus);
137  }
138 
139 MPC_AcquireConfigPointers_Exit_Tag:
140  return iStatus;
141 }
