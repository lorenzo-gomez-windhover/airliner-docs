266 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
267 void VM_Navigation::DoAction()
268 {
269  /* Set common fields in vehicle control mode message */
270  App.VehicleControlModeMsg.Armed = App.VehicleStatusMsg.ArmingState;
271  App.VehicleControlModeMsg.SystemHilEnabled = false;
272  App.VehicleControlModeMsg.ExternalManualOverrideOk = false;
273  App.VehicleControlModeMsg.ControlOffboardEnabled = false;
274 
275  switch (FSM.getState().getId())
276  {
277  case VM_NavSM_StateType::VM_NAVSM_MANUAL_STATE:
278  {
279  App.VehicleControlModeMsg.ControlManualEnabled = true;
280  App.VehicleControlModeMsg.ControlAutoEnabled = false;
281  App.VehicleControlModeMsg.ControlRatesEnabled = IsStabilizationRequired();
282  App.VehicleControlModeMsg.ControlAttitudeEnabled = IsStabilizationRequired();
283  App.VehicleControlModeMsg.ControlRattitudeEnabled = false;
284  App.VehicleControlModeMsg.ControlAltitudeEnabled = false;
285  App.VehicleControlModeMsg.ControlClimbRateEnabled = false;
286  App.VehicleControlModeMsg.ControlPositionEnabled = false;
287  App.VehicleControlModeMsg.ControlVelocityEnabled = false;
288  App.VehicleControlModeMsg.ControlAccelerationEnabled = false;
289  App.VehicleControlModeMsg.ControlTerminationEnabled = false;
290  break;
291  }
292 
293  case VM_NavSM_StateType::VM_NAVSM_ALTCTL:
294  {
295  App.VehicleControlModeMsg.ControlManualEnabled = true;
296  App.VehicleControlModeMsg.ControlAutoEnabled = false;
297  App.VehicleControlModeMsg.ControlRatesEnabled = true;
298  App.VehicleControlModeMsg.ControlAttitudeEnabled = true;
299  App.VehicleControlModeMsg.ControlRattitudeEnabled = false;
300  App.VehicleControlModeMsg.ControlAltitudeEnabled = true;
301  App.VehicleControlModeMsg.ControlClimbRateEnabled = true;
302  App.VehicleControlModeMsg.ControlPositionEnabled = false;
303  App.VehicleControlModeMsg.ControlVelocityEnabled = false;
304  App.VehicleControlModeMsg.ControlAccelerationEnabled = false;
305  App.VehicleControlModeMsg.ControlTerminationEnabled = false;
306  break;
307  }
308 
309  case VM_NavSM_StateType::VM_NAVSM_POSCTL:
310  {
311  App.VehicleControlModeMsg.ControlManualEnabled = true;
312  App.VehicleControlModeMsg.ControlAutoEnabled = false;
313  App.VehicleControlModeMsg.ControlRatesEnabled = true;
314  App.VehicleControlModeMsg.ControlAttitudeEnabled = true;
315  App.VehicleControlModeMsg.ControlRattitudeEnabled = false;
316  App.VehicleControlModeMsg.ControlAltitudeEnabled = true;
317  App.VehicleControlModeMsg.ControlClimbRateEnabled = true;
318  App.VehicleControlModeMsg.ControlPositionEnabled = !App.VehicleStatusMsg.InTransitionMode;
319  App.VehicleControlModeMsg.ControlVelocityEnabled = !App.VehicleStatusMsg.InTransitionMode;
320  App.VehicleControlModeMsg.ControlAccelerationEnabled = false;
321  App.VehicleControlModeMsg.ControlTerminationEnabled = false;
322  break;
323  }
324 
325  /* Fall through */
326  case VM_NavSM_StateType::VM_NAVSM_AUTO_RTL:
327  case VM_NavSM_StateType::VM_NAVSM_AUTO_LAND:
328  case VM_NavSM_StateType::VM_NAVSM_AUTO_LOITER:
329  case VM_NavSM_StateType::VM_NAVSM_AUTO_TAKEOFF:
330  {
331  App.VehicleControlModeMsg.ControlManualEnabled = false;
332  App.VehicleControlModeMsg.ControlAutoEnabled = true;
333  App.VehicleControlModeMsg.ControlRatesEnabled = true;
334  App.VehicleControlModeMsg.ControlAttitudeEnabled = true;
335  App.VehicleControlModeMsg.ControlRattitudeEnabled = false;
336  App.VehicleControlModeMsg.ControlAltitudeEnabled = true;
337  App.VehicleControlModeMsg.ControlClimbRateEnabled = true;
338  App.VehicleControlModeMsg.ControlPositionEnabled = !App.VehicleStatusMsg.InTransitionMode;
339  App.VehicleControlModeMsg.ControlVelocityEnabled = !App.VehicleStatusMsg.InTransitionMode;
340  App.VehicleControlModeMsg.ControlAccelerationEnabled = false;
341  App.VehicleControlModeMsg.ControlTerminationEnabled = false;
342  break;
343  }
344 
345  case VM_NavSM_StateType::VM_NAVSM_ACRO:
346  {
347  App.VehicleControlModeMsg.ControlManualEnabled = true;
348  App.VehicleControlModeMsg.ControlAutoEnabled = false;
349  App.VehicleControlModeMsg.ControlRatesEnabled = true;
350  App.VehicleControlModeMsg.ControlAttitudeEnabled = false;
351  App.VehicleControlModeMsg.ControlRattitudeEnabled = false;
352  App.VehicleControlModeMsg.ControlAltitudeEnabled = false;
353  App.VehicleControlModeMsg.ControlClimbRateEnabled = false;
354  App.VehicleControlModeMsg.ControlPositionEnabled = false;
355  App.VehicleControlModeMsg.ControlVelocityEnabled = false;
356  App.VehicleControlModeMsg.ControlAccelerationEnabled = false;
357  App.VehicleControlModeMsg.ControlTerminationEnabled = false;
358  break;
359  }
360 
361  case VM_NavSM_StateType::VM_NAVSM_STAB:
362  {
363  App.VehicleControlModeMsg.ControlManualEnabled = true;
364  App.VehicleControlModeMsg.ControlAutoEnabled = false;
365  App.VehicleControlModeMsg.ControlRatesEnabled = true;
366  App.VehicleControlModeMsg.ControlAttitudeEnabled = true;
367  App.VehicleControlModeMsg.ControlRattitudeEnabled = false;
368  App.VehicleControlModeMsg.ControlAltitudeEnabled = false;
369  App.VehicleControlModeMsg.ControlClimbRateEnabled = false;
370  App.VehicleControlModeMsg.ControlPositionEnabled = false;
371  App.VehicleControlModeMsg.ControlVelocityEnabled = false;
372  App.VehicleControlModeMsg.ControlAccelerationEnabled = false;
373  App.VehicleControlModeMsg.ControlTerminationEnabled = false;
374  App.VehicleControlModeMsg.ExternalManualOverrideOk = false;
375  break;
376  }
377 
378  case VM_NavSM_StateType::VM_NAVSM_RATTITUDE:
379  {
380  App.VehicleControlModeMsg.ControlManualEnabled = true;
381  App.VehicleControlModeMsg.ControlAutoEnabled = false;
382  App.VehicleControlModeMsg.ControlRatesEnabled = true;
383  App.VehicleControlModeMsg.ControlAttitudeEnabled = true;
384  App.VehicleControlModeMsg.ControlRattitudeEnabled = true;
385  App.VehicleControlModeMsg.ControlAltitudeEnabled = false;
386  App.VehicleControlModeMsg.ControlClimbRateEnabled = false;
387  App.VehicleControlModeMsg.ControlPositionEnabled = false;
388  App.VehicleControlModeMsg.ControlVelocityEnabled = false;
389  App.VehicleControlModeMsg.ControlAccelerationEnabled = false;
390  App.VehicleControlModeMsg.ControlTerminationEnabled = false;

418  osalbool SensorAccelMsgReady = (App.SensorAccelMsg.Timestamp > 0);
419  osalbool SensorCombinedMsgReady = (App.SensorCombinedMsg.Timestamp > 0);
420  osalbool VehicleAttitudeMsgReady = (App.VehicleAttitudeMsg.Timestamp > 0);
421  osalbool VehicleLocalPositionMsg = (App.VehicleLocalPositionMsg.Timestamp > 0);
422  osalbool VehicleLandDetectedMsgReady = (App.VehicleLandDetectedMsg.Timestamp > 0);
423  osalbool VehicleGlobalPositionMsgReady = (App.VehicleGlobalPositionMsg.Timestamp > 0);
424  osalbool VehicleGpsPositionMsgReady = (App.VehicleGpsPositionMsg.Timestamp > 0);
425 
426  if (SensorMagMsgReady && SensorGyroMsgReady && SensorAccelMsgReady
427  && SensorCombinedMsgReady && VehicleAttitudeMsgReady
428  && VehicleLocalPositionMsg && VehicleLandDetectedMsgReady
429  && VehicleGlobalPositionMsgReady && VehicleGpsPositionMsgReady)
430  {
431  validity = true;
432  }
433 
434  if (!validity)
435  {
436  /* Send event */
437  CFE_EVS_SendEvent(VM_SEN_NOT_READY_INFO_EID, CFE_EVS_INFORMATION,
438  "Sensors not ready");
439  }
440 
441  return validity;
442 }
