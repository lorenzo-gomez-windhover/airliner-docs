482 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
483 int32 VM::RcvSchPipeMsg(int32 iBlocking)
484 {
485  int32 iStatus = CFE_SUCCESS;
486  CFE_SB_Msg_t* MsgPtr = 0;
487  CFE_SB_MsgId_t MsgId;
488 
489  /* Stop Performance Log entry */
490  CFE_ES_PerfLogExit(VM_MAIN_TASK_PERF_ID);
491 
492  /* Wait for WakeUp messages from scheduler */
493  iStatus = CFE_SB_RcvMsg(&MsgPtr, SchPipeId, iBlocking);
494 
495  /* Start Performance Log entry */
496  CFE_ES_PerfLogEntry(VM_MAIN_TASK_PERF_ID);
497 
498  if (iStatus == CFE_SUCCESS)
499  {
500  MsgId = CFE_SB_GetMsgId(MsgPtr);
501  switch (MsgId)
502  {
503  case VM_WAKEUP_MID:
504  {
505  uint64 timestamp;
506 
507  CheckParams();
508  ProcessDataPipe();
509 
510  /* Update status in caution and warning */
511  m_caws.SetStatus(&VehicleStatusMsg);
512 
513  /* Cyclic maintainance loop */
514  Execute();
515 
516  /* Get a common timestamp. */
517  timestamp = PX4LIB_GetPX4TimeUs();
518  ActuatorArmedMsg.Timestamp = timestamp;
519  VehicleStatusMsg.Timestamp = timestamp;
520  VehicleManagerStateMsg.Timestamp = timestamp;
521  VehicleControlModeMsg.Timestamp = timestamp;
522 
523  /* Execute all stateful behavior. */
524  ArmingSM.DoAction();
525  NavigationSM.DoAction();
526 
527  /* Publish the messages. */
528  SendVehicleManagerStateMsg();
529  SendVehicleControlModeMsg();
530  break;
531  }
532 
533  case VM_SEND_HK_MID:
534  {
535  ProcessCmdPipe();
536  ReportHousekeeping();
537  break;
