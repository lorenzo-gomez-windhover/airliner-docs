45 /* Initialize Config Table */
46 /* */
47 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
48 int32 VM::InitConfigTbl()
49 {
50  int32 iStatus = 0;
51 
52  /* Register Config table */
53  iStatus = CFE_TBL_Register(&ConfigTblHdl,
54  VM_CONFIG_TABLENAME,
55  (sizeof(VM_ConfigTbl_t)),
56  CFE_TBL_OPT_DEFAULT,
57  VM::ValidateConfigTbl);
58  if (iStatus != CFE_SUCCESS)
59  {
60  /* Note, a critical table could return another nominal code. If this table is
61  * made critical this logic would have to change. */
62  (void) CFE_EVS_SendEvent(VM_CFGTBL_REG_ERR_EID, CFE_EVS_ERROR,
63  "Failed to register config table (0x%08lX)",
64  iStatus);
65  goto VM_InitConfigTbl_Exit_Tag;
66  }
67 
68  /* Load Config table file */
69  iStatus = CFE_TBL_Load(ConfigTblHdl,
70  CFE_TBL_SRC_FILE,
71  VM_CONFIG_TABLE_FILENAME);
72  if (iStatus != CFE_SUCCESS)
73  {
74  /* Note, CFE_SUCCESS is for a successful full table load. If a partial table
75  load is desired then this logic would have to change. */
76  (void) CFE_EVS_SendEvent(VM_CFGTBL_LOAD_ERR_EID, CFE_EVS_ERROR,
77  "Failed to load Config Table (0x%08lX)",
78  iStatus);
79  goto VM_InitConfigTbl_Exit_Tag;
80  }
81 
82  iStatus = AcquireConfigPointers();
83 
84 VM_InitConfigTbl_Exit_Tag:
85  return iStatus;
86 }

160 /* Acquire Config Pointers */
161 /* */
162 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
163 int32 VM::AcquireConfigPointers(void)
164 {
165  int32 iStatus = CFE_SUCCESS;
166 
167  /*
168  ** Release the table
169  */
170  /* TODO: This return value can indicate success, error, or that the info has been
171  * updated. We ignore this return value in favor of checking CFE_TBL_Manage(), but
172  * be sure this is the behavior you want. */
173  (void) CFE_TBL_ReleaseAddress(ConfigTblHdl);
174 
175  /*
176  ** Manage the table
177  */
178  iStatus = CFE_TBL_Manage(ConfigTblHdl);
179  if ((iStatus != CFE_SUCCESS) && (iStatus != CFE_TBL_INFO_UPDATED))
180  {
181  (void) CFE_EVS_SendEvent(VM_CFGTBL_MANAGE_ERR_EID, CFE_EVS_ERROR,
182  "Failed to manage VM Config table (0x%08lX)",
183  iStatus);
184  goto VM_AcquireConfigPointers_Exit_Tag;
185  }
186 
187  /*
188  ** Get a pointer to the table
189  */
190  iStatus = CFE_TBL_GetAddress((void**)&ConfigTblPtr, ConfigTblHdl);
191  if (iStatus == CFE_TBL_INFO_UPDATED)
192  {
193  iStatus = CFE_SUCCESS;
194  }
195  else if(iStatus != CFE_SUCCESS)
196  {
197  ConfigTblPtr = 0;
198  (void) CFE_EVS_SendEvent(VM_CFGTBL_GETADDR_ERR_EID, CFE_EVS_ERROR,
199  "Failed to get Config table's address (0x%08lX)",
200  iStatus);
201  }
202 
203 VM_AcquireConfigPointers_Exit_Tag:
204  return iStatus;
205 }
