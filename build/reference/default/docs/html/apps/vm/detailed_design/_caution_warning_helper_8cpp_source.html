235 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
236 /* */
237 /* Set status of the Caution and Warning System from the */
238 /* PX4_VehicleStatusMsg. */
239 /* */
240 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
241 osalbool CautionWarningHelper::SetStatus(const PX4_VehicleStatusMsg_t *status)
242 {
243  osalbool returnBool = TRUE;
244  osalbool decodeNavState = TRUE;
245  /* Initialize final state to the current state */
246  CAWS_States_t finalState = m_State;
247  static uint8 callCount = 0;
248 
249  /* Currently this function is called from VM wakeup so throttle the rate */
250  if (callCount >= 250)
251  {
252  callCount = 0;
253  goto end_of_function;
254  }
255  callCount++;
256 
257  if (0 == status)
258  {
259  returnBool = FALSE;
260  goto end_of_function;
261  }
262 
263  if(TRUE == status->RcSignalLost)
264  {
265  finalState = CAWS_FAILSAFE;
266  goto end_of_function;
267  }
268 
269  /* Decode armed state */
270  switch(status->ArmingState)
271  {
272  case PX4_ARMING_STATE_INIT:
273  {
274  finalState = CAWS_NEUTRAL;
275  decodeNavState = FALSE;
276  break;
277  }
278 
279  case PX4_ARMING_STATE_STANDBY:
280  {
281  finalState = CAWS_NEUTRAL;
282  decodeNavState = FALSE;
283  break;
284  }
285 
286  case PX4_ARMING_STATE_ARMED:
287  {
288  finalState = CAWS_POSITIVE;
289  decodeNavState = TRUE;
290  break;
291  }
292 
293  case PX4_ARMING_STATE_ARMED_ERROR:
294  {
295  finalState = CAWS_NEGATIVE;
296  decodeNavState = FALSE;
297  break;
298  }
299 
300  case PX4_ARMING_STATE_STANDBY_ERROR:
301  {
302  finalState = CAWS_NEGATIVE;
303  decodeNavState = FALSE;
304  break;
305  }
306 
307  case PX4_ARMING_STATE_REBOOT:
308  {
309  finalState = CAWS_NEGATIVE;
310  decodeNavState = FALSE;
311  break;
312 
313  }
314 
315  case PX4_ARMING_STATE_IN_AIR_RESTORE:
316  {
317  finalState = CAWS_NEGATIVE;
318  decodeNavState = FALSE;
319  break;
320  }
321 
322  default:
323  {
324  returnBool = FALSE;
325  decodeNavState = FALSE;
326  break;
327  }
328  }
329 
330  if(TRUE == decodeNavState)
331  {
332  /* Next decode navigation state */
333  switch(status->NavState)
334  {
335  case PX4_NAVIGATION_STATE_MANUAL:
336  {
337  /* Do nothing for state manual */
338  break;
339  }
340 
341  case PX4_NAVIGATION_STATE_ALTCTL:
342  {
343  finalState = CAWS_ALTCTL;
344  break;
345  }
346 
347  case PX4_NAVIGATION_STATE_POSCTL:
348  {
349  finalState = CAWS_POSCTL;
350  break;
351  }
352 
353  case PX4_NAVIGATION_STATE_AUTO_MISSION:
354  {
355  finalState = CAWS_AUTO;
356  break;
357  }
358 
359  case PX4_NAVIGATION_STATE_AUTO_LOITER:
360  {
361  finalState = CAWS_AUTO;
362  break;
363  }
364 
365  case PX4_NAVIGATION_STATE_AUTO_RTL:
366  {
367  finalState = CAWS_FAILSAFE;
368  break;
369  }
370 
371  case PX4_NAVIGATION_STATE_AUTO_RCRECOVER:
372  {
373  finalState = CAWS_AUTO;
374  break;
375  }
376 
377  case PX4_NAVIGATION_STATE_AUTO_RTGS:
378  {
379  finalState = CAWS_AUTO;
380  break;
381  }
382 
383  case PX4_NAVIGATION_STATE_AUTO_LANDENGFAIL:
384  {
385  finalState = CAWS_AUTO;
386  break;
387  }
388 
389  case PX4_NAVIGATION_STATE_AUTO_LANDGPSFAIL:
390  {
391  finalState = CAWS_AUTO;
392  break;
393  }
394 
395  case PX4_NAVIGATION_STATE_ACRO:
396  case PX4_NAVIGATION_STATE_UNUSED:
397  case PX4_NAVIGATION_STATE_DESCEND:
398  case PX4_NAVIGATION_STATE_TERMINATION:
399  case PX4_NAVIGATION_STATE_OFFBOARD:
400  case PX4_NAVIGATION_STATE_STAB:
401  case PX4_NAVIGATION_STATE_RATTITUDE:
402  /* Fallthru */
403  {
404  break;
405  }
406 
407  case PX4_NAVIGATION_STATE_AUTO_TAKEOFF:
408  {
409  finalState = CAWS_AUTO;
410  break;
411  }
412 
413  case PX4_NAVIGATION_STATE_AUTO_LAND:
414  {
415  finalState = CAWS_AUTO;
416  break;
417  }
418 
419  case PX4_NAVIGATION_STATE_AUTO_FOLLOW_TARGET:
420  {
421  finalState = CAWS_AUTO;
422  break;
423  }
424 
425  case PX4_NAVIGATION_STATE_AUTO_PRECLAND:
426  {
427  finalState = CAWS_AUTO;
428  break;
429  }
430 
431  default:
432  {
433  returnBool = FALSE;
434  break;
435  }
436  }
437  }
438 
439 end_of_function:
440  /* If the current state is not the decoded state... */
441  if(finalState != m_State)
442  {
443  /* Set the new state */
444  SetState(finalState);
445  }
446 
447  return returnBool;
448 } /* End CautionWarningHelper::SetState() */
