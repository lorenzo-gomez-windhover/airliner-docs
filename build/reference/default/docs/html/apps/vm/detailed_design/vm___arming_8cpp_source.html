217 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
218 osalbool VM_Arming::PreFlightCheckCleared()
219 {
220  osalbool cleared = true;
221 
222  /* Warn about global position */
223  if (App.VehicleGlobalPositionMsg.Timestamp == 0)
224  {
225  (void) CFE_EVS_SendEvent(VM_NOPE_ERR_EID, CFE_EVS_INFORMATION, "Warning: Position estimator not initialized");
226  }
227 
228  /* Battery warning check */
229  if (App.BatteryStatusMsg.Warning >= PX4_BatteryWarningSeverity_t::PX4_BATTERY_WARNING_LOW)
230  {
231  CFE_EVS_SendEvent(VM_PRE_ARM_BAT_CHECK_INFO_EID, CFE_EVS_ERROR, "Low battery, cannot arm.");
232  cleared = false;
233  goto end_of_function;
234  }
235 
236  /* Safety check */
237  if (App.SafetyMsg.SafetySwitchAvailable && !App.SafetyMsg.SafetyOff)
238  {
239  CFE_EVS_SendEvent(VM_PRE_ARM_SAFETY_CHECK_INFO_EID, CFE_EVS_ERROR, "Safety is ON, cannot arm");
240  cleared = false;
241  goto end_of_function;
242  }
243 
244  /* Sensors check */
245  if (!App.HkTlm.StatusFlags.SensorsInitialized)
246  {
247  CFE_EVS_SendEvent(VM_PRE_ARM_SENSORS_CHECK_INFO_EID, CFE_EVS_ERROR, "Sensors not setup correctly");
248  cleared = false;
249  }
250 
251 end_of_function:
252  return cleared;
253 }
