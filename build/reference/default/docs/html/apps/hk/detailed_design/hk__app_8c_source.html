86 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
87 /* */
88 /* HK application entry point and main process loop */
89 /* */
90 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
91 void HK_AppMain(void)
92 {
93  int32 Status;
94 
95  /*
96  ** Register the Application with Executive Services
97  */
98  CFE_ES_RegisterApp();
99 
100  /*
101  ** Create the first Performance Log entry
102  */
103  CFE_ES_PerfLogEntry(HK_APPMAIN_PERF_ID);
104 
105 
106  /* Perform Application Initialization */
107  Status = HK_AppInit();
108  if (Status != CFE_SUCCESS)
109  {
110  HK_AppData.RunStatus = CFE_ES_APP_ERROR;
111  }
112 
113  /*
114  ** Application Main Loop.
115  */
116  while(CFE_ES_RunLoop(&HK_AppData.RunStatus) == TRUE)
117  {
118  /*
119  ** Performance Log Exit Stamp.
120  */
121  CFE_ES_PerfLogExit(HK_APPMAIN_PERF_ID);
122 
123  /*
124  ** Pend on the arrival of the next Software Bus message.
125  */
126  Status = CFE_SB_RcvMsg(&HK_AppData.MsgPtr,HK_AppData.CmdPipe,CFE_SB_PEND_FOREVER);
127 
128  if(Status != CFE_SUCCESS)
129  {
130  CFE_EVS_SendEvent(HK_RCV_MSG_ERR_EID, CFE_EVS_ERROR,
131  "HK_APP Exiting due to CFE_SB_RcvMsg error 0x%08X", (unsigned int)Status);
132 
133  /* Write to syslog in case there is a problem with event services */
134  CFE_ES_WriteToSysLog("HK_APP Exiting due to CFE_SB_RcvMsg error 0x%08X\n", (unsigned int)Status);
135 
136  HK_AppData.RunStatus = CFE_ES_APP_ERROR;
137  }
138  else /* Success */
139  {
140  /*
141  ** Performance Log Entry Stamp.
142  */
143  CFE_ES_PerfLogEntry(HK_APPMAIN_PERF_ID);
144 
145 
146  /* Perform Message Processing */
147  HK_AppPipe(HK_AppData.MsgPtr);
148  }
149  } /* end while */
150 
151  /*
152  ** Performance Log Exit Stamp.
153  */
154  CFE_ES_PerfLogExit(HK_APPMAIN_PERF_ID);
155 
156  /*
157  ** Exit the Application.
158  */
159  CFE_ES_ExitApp(HK_AppData.RunStatus);
160 
161 } /* End of HK_AppMain() */

367 /* Process a command pipe message */
368 /* */
369 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
370 void HK_AppPipe (CFE_SB_MsgPtr_t MessagePtr)
371 {
372  CFE_SB_MsgId_t MessageID;
373  uint16 CommandCode;
374 
375  MessageID = CFE_SB_GetMsgId (MessagePtr);
376  switch (MessageID)
377  {
378 
379  case HK_SEND_COMBINED_PKT_MID:
380  if (CFE_SB_GetTotalMsgLength(MessagePtr) != sizeof(HK_Send_Out_Msg_t))
381  {
382  CFE_EVS_SendEvent( HK_MSG_LEN_ERR_EID, CFE_EVS_ERROR,
383  "Msg with Bad length Rcvd: ID = 0x%04X, Exp Len = %d, Len = %d",
384  MessageID,
385  sizeof(HK_Send_Out_Msg_t),
386  CFE_SB_GetTotalMsgLength(MessagePtr));
387  }
388  else
389  {
390  HK_SendCombinedHKCmd(MessagePtr);
391  }
392  break;
393 
394  /* Request for HK's Housekeeping data... */
395  case HK_SEND_HK_MID:
396  if (CFE_SB_GetTotalMsgLength(MessagePtr) != CFE_SB_CMD_HDR_SIZE)
397  {
398  CFE_EVS_SendEvent( HK_MSG_LEN_ERR_EID, CFE_EVS_ERROR,
399  "Msg with Bad length Rcvd: ID = 0x%04X, Exp Len = %d, Len = %d",
400  MessageID,
401  CFE_SB_CMD_HDR_SIZE,
402  CFE_SB_GetTotalMsgLength(MessagePtr));
403  }
404  else
405  {
406  /* Send out HK's housekeeping data */
407  HK_HousekeepingCmd(MessagePtr);
408  }
409  /* Check for copy table load and runtime dump request */
410  HK_CheckStatusOfTables();
411  break;
412 
413  /* HK ground commands */
414  case HK_CMD_MID:
415 
416  CommandCode = CFE_SB_GetCmdCode(MessagePtr);
417  switch (CommandCode)
418  {
419  case HK_NOOP_CC:
420  HK_NoopCmd (MessagePtr);
421  break;
422 
423  case HK_RESET_CC:
424  HK_ResetCtrsCmd (MessagePtr);
425  break;
426 
427  default:
428  CFE_EVS_SendEvent(HK_CC_ERR_EID, CFE_EVS_ERROR,
429  "Cmd Msg with Invalid command code Rcvd -- ID = 0x%04X, CC = %d",
430  MessageID, CommandCode);
431 
432  HK_AppData.ErrCounter++;
433  break;
434  }
435  break;
436 
437  /* Incoming housekeeping data from other Subsystems... */
438  default:
439 
440  HK_ProcessIncomingHkData (MessagePtr);
441  break;
442  }
443 
444  return;
445 
446 } /* End of HK_AppPipe() */
