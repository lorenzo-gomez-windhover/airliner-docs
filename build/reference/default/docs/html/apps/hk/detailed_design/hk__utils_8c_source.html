69 /* HK process incoming housekeeping data */
70 /* */
71 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
72 void HK_ProcessIncomingHkData (CFE_SB_MsgPtr_t MessagePtr)
73 {
74  hk_copy_table_entry_t * StartOfCopyTable;
75  hk_copy_table_entry_t * CpyTblEntry;
76  hk_runtime_tbl_entry_t * StartOfRtTable;
77  hk_runtime_tbl_entry_t * RtTblEntry;
78  uint16 Loop;
79  CFE_SB_MsgId_t MessageID;
80  uint8 * DestPtr;
81  uint8 * SrcPtr;
82  int32 MessageLength=0;
83  int32 MessageErrors;
84  int32 LastByteAccessed;
85 
86 
87  StartOfCopyTable = (hk_copy_table_entry_t *) HK_AppData.CopyTablePtr;
88  StartOfRtTable = (hk_runtime_tbl_entry_t *) HK_AppData.RuntimeTablePtr;
89  MessageID = CFE_SB_GetMsgId (MessagePtr);
90  MessageErrors = 0;
91 
92  /* Spin thru the entire table looking for matches */
93  for (Loop=0; Loop < HK_COPY_TABLE_ENTRIES; Loop++)
94  {
95  CpyTblEntry = & StartOfCopyTable [Loop];
96  RtTblEntry = & StartOfRtTable [Loop];
97 
98  /* Does the inputMID for this table entry match what we're looking for */
99  if (MessageID == CpyTblEntry->InputMid)
100  {
101  /* Ensure that we don't reference past the end of the input packet */
102  MessageLength = CFE_SB_GetTotalMsgLength(MessagePtr);
103  LastByteAccessed = CpyTblEntry->InputOffset + CpyTblEntry->NumBytes;
104  if (MessageLength >= LastByteAccessed)
105  {
106  /* We have a match. Build the Source and Destination addresses
107  and move the data */
108  DestPtr = ( (uint8 *) RtTblEntry->OutputPktAddr) + CpyTblEntry->OutputOffset;
109  SrcPtr = ( (uint8 *) MessagePtr) + CpyTblEntry->InputOffset;
110 
111  CFE_PSP_MemCpy (DestPtr, SrcPtr, CpyTblEntry->NumBytes);
112 
113  /* Set the data present field to indicate the data is there */
114  RtTblEntry->DataPresent = HK_DATA_PRESENT;
115 
116  }
117  else
118  {
119  /* Error: copy data is past the end of the input packet */
120  MessageErrors++;
121  }
122  }
123  }
124 
125  /* Send, at most, one error event per input packet */
126  if (MessageErrors != 0)
127  {
128  CFE_EVS_SendEvent (HK_ACCESSING_PAST_PACKET_END_EID, CFE_EVS_ERROR,
129  "HK table definition exceeds packet length. MID:0x%04X, Length:%d, Count:%d",
130  MessageID, (int)MessageLength, (int)MessageErrors);
131  }
132 
133  return;
134 
135 } /* end HK_ProcessIncomingHkData */

153 /* HK process new copy table */
154 /* */
155 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
156 void HK_ProcessNewCopyTable (hk_copy_table_entry_t * CpyTblPtr,
157  hk_runtime_tbl_entry_t * RtTblPtr)
158 {
159  hk_copy_table_entry_t * StartOfCopyTable = &(hk_copy_table_entry_t){0};
160  hk_copy_table_entry_t * OuterCpyEntry = &(hk_copy_table_entry_t){0};
161  hk_copy_table_entry_t * InnerDefEntry = &(hk_copy_table_entry_t){0};
162  hk_runtime_tbl_entry_t * StartOfRtTable = &(hk_runtime_tbl_entry_t){0};
163  hk_runtime_tbl_entry_t * OuterRtEntry = &(hk_runtime_tbl_entry_t){0};
164  hk_runtime_tbl_entry_t * InnerRtEntry = &(hk_runtime_tbl_entry_t){0};
165  int32 Loop1 = 0;
166  int32 Loop2 = 0;
167  CFE_SB_MsgId_t MidOfThisPacket = 0;
168  int32 SizeOfThisPacket = 0;
169  int32 FurthestByteFromThisEntry = 0;
170  CFE_SB_MsgPtr_t NewPacketAddr = 0;
171  int32 Result = CFE_SUCCESS;
172 
173  StartOfCopyTable = CpyTblPtr;
174  StartOfRtTable = RtTblPtr;
175 
176  /* Loop thru the RunTime table initializing the fields */
177  for (Loop1 = 0; Loop1 < HK_COPY_TABLE_ENTRIES; Loop1++)
178  {
179  OuterRtEntry = & StartOfRtTable [Loop1];
180 
181  OuterRtEntry->OutputPktAddr = NULL;
182  OuterRtEntry->InputMidSubscribed = HK_INPUTMID_NOT_SUBSCRIBED;
183  OuterRtEntry->DataPresent = HK_DATA_NOT_PRESENT;
184 
185  }
186 
187  /* Loop thru the table looking for all of the SB packets that need to be built */
188  for (Loop1 = 0; Loop1 < HK_COPY_TABLE_ENTRIES; Loop1++)
189  {
190  OuterCpyEntry = & StartOfCopyTable [Loop1];
191  OuterRtEntry = & StartOfRtTable [Loop1];
192 
193  /* If the both MIDs exists but the Packet Address has yet to be assigned,
194  we need to build an SB packet, so compute the size */
195  if ( (OuterCpyEntry->OutputMid != HK_UNDEFINED_ENTRY) &&
196  (OuterCpyEntry->InputMid != HK_UNDEFINED_ENTRY) &&
197  (OuterRtEntry->OutputPktAddr == NULL) )
198  {
199  /* We have a table entry that needs a SB message to be built */
200  MidOfThisPacket = OuterCpyEntry->OutputMid;
201  SizeOfThisPacket = 0;
202 
203  /* Spin thru entire table looking for duplicate OutputMid's. This will let
204  us find the byte offset furthest from the beginning of the packet */
205  for (Loop2=0; Loop2 < HK_COPY_TABLE_ENTRIES; Loop2++)
206  {
207  InnerDefEntry = & StartOfCopyTable [Loop2];
208 
209  /* If this entry's MID matches the one we're looking for */
210  if (InnerDefEntry->OutputMid == MidOfThisPacket)
211  {
212  /* The byte furthest away from the section described by this entry */
213  FurthestByteFromThisEntry = InnerDefEntry->OutputOffset +
214  InnerDefEntry->NumBytes;
215 
216  /* Save the byte offset of the byte furthest from the packet start */
217  if (FurthestByteFromThisEntry > SizeOfThisPacket)
218  {
219  SizeOfThisPacket = FurthestByteFromThisEntry;
220 
221  }
222  }
223  }
224 
225  /* Build the packet with the size computed above */
226  NewPacketAddr = NULL;
227  if (SizeOfThisPacket > 0)
228  {
229  Result = CFE_ES_GetPoolBuf ((uint32 **) & NewPacketAddr,
230  HK_AppData.MemPoolHandle,
231  SizeOfThisPacket);
232 
233  if (Result >= CFE_SUCCESS)
234  {
235  /* Spin thru entire table (again) looking for duplicate OutputMid's.
236  This will let us assign the packet created above to all
237  of the table entries that need to use it */
238  for (Loop2=0; Loop2 < HK_COPY_TABLE_ENTRIES; Loop2++)
239  {
240  InnerDefEntry = & StartOfCopyTable [Loop2];
241  InnerRtEntry = & StartOfRtTable [Loop2];
242 
243  /* If this entry's MID matches the one we're looking for */
244  if (InnerDefEntry->OutputMid == MidOfThisPacket)
245  {
246  InnerRtEntry->OutputPktAddr = NewPacketAddr;
247  }
248  }
249 
250  /* Init the SB Packet only once regardless of how many times its in the table */
251  CFE_SB_InitMsg (NewPacketAddr, MidOfThisPacket, SizeOfThisPacket, TRUE);
252 
253  }
254  else
255  {
256  CFE_EVS_SendEvent (HK_MEM_POOL_MALLOC_FAILED_EID, CFE_EVS_ERROR,
257  "HK Processing New Table: ES_GetPoolBuf for size %d returned 0x%04X",
258  (int)SizeOfThisPacket, (unsigned int)Result);
259  }
260  }
261  }
262 
263  /* If HK needs to subscribe to this Input packet... */
264  if ( (OuterRtEntry->InputMidSubscribed == HK_INPUTMID_NOT_SUBSCRIBED) &&
265  (OuterCpyEntry->InputMid != HK_UNDEFINED_ENTRY) )
266  {
267  Result = CFE_SB_Subscribe (OuterCpyEntry->InputMid, HK_AppData.CmdPipe);
268 
269  if (Result == CFE_SUCCESS)
270  {
271  /* Spin thru entire table (again) looking for duplicate InputMid's.
272  This will let us mark each duplicate as already having been subscribed */
273  for (Loop2=0; Loop2 < HK_COPY_TABLE_ENTRIES; Loop2++)
274  {
275  InnerDefEntry = & StartOfCopyTable [Loop2];
276  InnerRtEntry = & StartOfRtTable [Loop2];
277 
278  /* If this entry's MID matches the one we're looking for */
279  if (OuterCpyEntry->InputMid == InnerDefEntry->InputMid)
280  {
281  InnerRtEntry->InputMidSubscribed = HK_INPUTMID_SUBSCRIBED;
282  }
283  }
284  }
285  else
286  {
287  CFE_EVS_SendEvent (HK_CANT_SUBSCRIBE_TO_SB_PKT_EID, CFE_EVS_ERROR,
288  "HK Processing New Table:SB_Subscribe for Mid 0x%04X returned 0x%04X",
289  OuterCpyEntry->InputMid, (unsigned int)Result);
290  }
291  }
292  }
293 
294 } /* end HK_ProcessNewCopyTable */

299 /* HK Tear down old copy table */
300 /* */
301 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
302 void HK_TearDownOldCopyTable (hk_copy_table_entry_t * CpyTblPtr,
303  hk_runtime_tbl_entry_t * RtTblPtr)
304 {
305  hk_copy_table_entry_t * StartOfCopyTable;
306  hk_copy_table_entry_t * OuterCpyEntry;
307  hk_copy_table_entry_t * InnerDefEntry;
308  hk_runtime_tbl_entry_t * StartOfRtTable;
309  hk_runtime_tbl_entry_t * OuterRtEntry;
310  hk_runtime_tbl_entry_t * InnerRtEntry;
311  int32 Loop1;
312  int32 Loop2;
313  CFE_SB_MsgId_t MidOfThisPacket;
314  void * OutputPktAddr;
315  void * SavedPktAddr;
316  int32 Result = CFE_SUCCESS;
317 
318  StartOfCopyTable = CpyTblPtr;
319  StartOfRtTable = RtTblPtr;
320 
321  /* Loop thru the table looking for all of the SB packets that need to be freed */
322  for (Loop1 = 0; Loop1 < HK_COPY_TABLE_ENTRIES; Loop1++)
323  {
324  OuterCpyEntry = & StartOfCopyTable [Loop1];
325  OuterRtEntry = & StartOfRtTable [Loop1];
326 
327  /* If a Packet Address has been assigned, it needs to get deleted */
328  if (OuterRtEntry->OutputPktAddr != NULL)
329  {
330  OutputPktAddr = OuterRtEntry->OutputPktAddr;
331  MidOfThisPacket = OuterCpyEntry->OutputMid;
332 
333  SavedPktAddr = OutputPktAddr;
334  Result = CFE_ES_PutPoolBuf (HK_AppData.MemPoolHandle, (uint32 *) OutputPktAddr);
335  if (Result >= CFE_SUCCESS)
336  {
337  /* Spin thru the entire table looking for entries that used the same SB packets */
338  for (Loop2=0; Loop2 < HK_COPY_TABLE_ENTRIES; Loop2++)
339  {
340  InnerDefEntry = & StartOfCopyTable [Loop2];
341  InnerRtEntry = & StartOfRtTable [Loop2];
342 
343  if ( (InnerDefEntry->OutputMid == MidOfThisPacket) &&
344  (InnerRtEntry->OutputPktAddr == SavedPktAddr) )
345  {
346  /* NULL out the table entry whose packet was freed above */
347  InnerRtEntry->OutputPktAddr = (CFE_SB_MsgPtr_t) NULL;
348  }
349  }
350  }
351  else
352  {
353  CFE_EVS_SendEvent (HK_MEM_POOL_FREE_FAILED_EID, CFE_EVS_ERROR,
354  "HK TearDown: ES_putPoolBuf Err pkt:0x%08lX ret 0x%04X, hdl 0x%08lx",
355  (unsigned long)SavedPktAddr, (unsigned int)Result, (unsigned long)HK_AppData.MemPoolHandle);
356 
357  }
358  }
359 
360  /* If the InputMid for this Table Entry has been subscribed, it needs to
361  get Unsubscribed as do any other identical InputMids throughout the table.
362  We don't have to worry about leaving any Mid's subscribed since the entire table
363  is getting clobbered. */
364  if (OuterRtEntry->InputMidSubscribed == HK_INPUTMID_SUBSCRIBED)
365  {
366  CFE_SB_Unsubscribe (OuterCpyEntry->InputMid, HK_AppData.CmdPipe);
367 
368  /* Spin thru the entire table looking for entries that used the same SB packets */
369  for (Loop2=0; Loop2 < HK_COPY_TABLE_ENTRIES; Loop2++)
370  {
371  InnerDefEntry = & StartOfCopyTable [Loop2];
372  InnerRtEntry = & StartOfRtTable [Loop2];
373 
374  if (InnerDefEntry->InputMid == OuterCpyEntry->InputMid)
375  {
376  InnerRtEntry->InputMidSubscribed = HK_INPUTMID_NOT_SUBSCRIBED;
377  }
378  }
379  }
380 
381  }
382 
383 } /* end HK_TearDownOldCopyTable */

394 {
395  boolean PacketFound = FALSE;
396  hk_runtime_tbl_entry_t * StartOfRtTable;
397  hk_runtime_tbl_entry_t * RtTblEntry;

400  CFE_SB_MsgId_t InputMidMissing;
401 
402  StartOfRtTable = (hk_runtime_tbl_entry_t *) HK_AppData.RuntimeTablePtr;
403 
404  /* Look thru each item in this Table, but only send this packet once, at most */
405  for (Loop = 0; ( (Loop < HK_COPY_TABLE_ENTRIES) && (PacketFound == FALSE) ); Loop++)
406  {
407  RtTblEntry = & StartOfRtTable [Loop];
408

410  if (RtTblEntry->OutputPktAddr != NULL)

422 
423  }

425  else /* This clause is only exclusive if discarding incomplete packets */

431

433  }
434 
435  HK_SetFlagsToNotPresent(ThisEntrysOutMid);
436 
437  PacketFound = TRUE;
438 
439  }
440  }
441  }
442 
443  if (PacketFound == FALSE)
444  {
445  CFE_EVS_SendEvent (HK_UNKNOWN_COMBINED_PACKET_EID, CFE_EVS_INFORMATION,

448  }
449 
450 } /* end HK_SendCombinedHkPacket */

468  {
469  /* Validate the specified Table */
470  CFE_TBL_Validate(HK_AppData.CopyTableHandle);
471 
472  }
473  else if (Status == CFE_TBL_INFO_UPDATE_PENDING)

479  CFE_TBL_ReleaseAddress (HK_AppData.CopyTableHandle);

490  }

493 
494  CFE_EVS_SendEvent (HK_UNEXPECTED_GETSTAT_RET_EID, CFE_EVS_ERROR,

530 
531  CFE_EVS_SendEvent (HK_UNEXPECTED_GETSTAT2_RET_EID, CFE_EVS_ERROR,

542 /* HK Check for missing combined output message data */
543 /* */
544 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
545 int32 HK_CheckForMissingData(CFE_SB_MsgId_t OutPktToCheck, CFE_SB_MsgId_t *MissingInputMid)
546 {
547  int32 Loop = 0;
548  int32 Status = HK_NO_MISSING_DATA;
549  hk_copy_table_entry_t * StartOfCopyTable;
550  hk_copy_table_entry_t * CpyTblEntry;
551  hk_runtime_tbl_entry_t * StartOfRtTable;
552  hk_runtime_tbl_entry_t * RtTblEntry;
553 
554  StartOfCopyTable = (hk_copy_table_entry_t *) HK_AppData.CopyTablePtr;
555  StartOfRtTable = (hk_runtime_tbl_entry_t *) HK_AppData.RuntimeTablePtr;
556 
557  /* Loop thru each item in the runtime table until end is reached or
558  * data-not-present detected */
559  do
560  {
561  CpyTblEntry = &StartOfCopyTable[Loop];
562  RtTblEntry = &StartOfRtTable[Loop];
563
