39 void MD_ProcessStartCmd(CFE_SB_MsgPtr_t MessagePtr)
40 {
41  uint16 TableId;
42  uint16 TableIndex;
43  MD_CmdStartStop_t *Start;
44  boolean AnyTablesInMask;
45 
46  AnyTablesInMask = FALSE;
47 
48  Start = (MD_CmdStartStop_t * ) MessagePtr;
49 
50  /* Preview tables specified by command: */
51  /* Check that there's at least one valid table specified. */
52  for (TableId=1; TableId <= MD_NUM_DWELL_TABLES; TableId++)
53  {
54  if (MD_TableIsInMask(TableId, Start->TableMask))
55  {
56  /* At least one valid Table Id is in Mask */
57  AnyTablesInMask=TRUE;
58  }
59  }
60 
61  /*
62  ** Handle specified operation.
63  ** If nominal, start each of the specified tables.
64  ** If error case was encountered, issue error message.
65  */
66  if (AnyTablesInMask)
67  { /* Handle Nominal Case */
68 
69  for (TableId=1; TableId <= MD_NUM_DWELL_TABLES; TableId++)
70  {
71  if (MD_TableIsInMask(TableId, Start->TableMask))
72  {
73  /* Setting Countdown to 1 causes a dwell packet to be issued */
74  /* on first wakeup call received. */
75  TableIndex = TableId-1;
76  MD_AppData.MD_DwellTables[ TableIndex ].Enabled = MD_DWELL_STREAM_ENABLED;
77  MD_AppData.MD_DwellTables[ TableIndex ].Countdown = 1;
78  MD_AppData.MD_DwellTables[ TableIndex ].CurrentEntry = 0;
79  MD_AppData.MD_DwellTables[ TableIndex ].PktOffset = 0;
80 
81  /* Change value in Table Services managed buffer */
82  MD_UpdateTableEnabledField (TableIndex, MD_DWELL_STREAM_ENABLED);
83 
84  /* If table contains a rate of zero, report that no processing will occur */
85  if (MD_AppData.MD_DwellTables[ TableIndex ].Rate == 0)
86  {
87  CFE_EVS_SendEvent(MD_ZERO_RATE_CMD_INF_EID, CFE_EVS_INFORMATION,
88  "Dwell Table %d is enabled with a delay of zero so no processing will occur", TableId);
89  }
90  }
91  }
92 
93  MD_AppData.CmdCounter++;
94 
95  CFE_EVS_SendEvent(MD_START_DWELL_INF_EID, CFE_EVS_INFORMATION,
96  "Start Dwell Table command processed successfully for table mask 0x%04X",
97  Start->TableMask);
98  }
99  else /* No valid table id's specified in mask */
100  {
101  MD_AppData.ErrCounter++;
102  CFE_EVS_SendEvent(MD_EMPTY_TBLMASK_ERR_EID, CFE_EVS_ERROR,
103  "%s command rejected because no tables were specified in table mask (0x%04X)",
104  "Start Dwell", Start->TableMask );
105  }
106  return;
107 
108 } /* End of MD_ProcessStartCmd */

112 void MD_ProcessStopCmd(CFE_SB_MsgPtr_t MessagePtr)
113 {
114  MD_CmdStartStop_t *Stop;
115  uint16 TableId;
116  uint16 TableIndex;
117  boolean AnyTablesInMask;
118 
119  AnyTablesInMask = FALSE;
120  Stop = (MD_CmdStartStop_t * ) MessagePtr;
121 
122 
123  for (TableId=1; TableId <= MD_NUM_DWELL_TABLES; TableId++)
124  {
125  if (MD_TableIsInMask(TableId, Stop->TableMask))
126  {
127  TableIndex = TableId-1;
128  MD_AppData.MD_DwellTables[ TableIndex ].Enabled = MD_DWELL_STREAM_DISABLED;
129  MD_AppData.MD_DwellTables[ TableIndex ].Countdown = 0;
130  MD_AppData.MD_DwellTables[ TableIndex ].CurrentEntry = 0;
131  MD_AppData.MD_DwellTables[ TableIndex ].PktOffset = 0;
132 
133  AnyTablesInMask=TRUE;
134 
135  /* Change value in Table Services managed buffer */
136  MD_UpdateTableEnabledField (TableIndex, MD_DWELL_STREAM_DISABLED);
137 
138  }
139  }
140 
141  if (AnyTablesInMask)
142  {
143  CFE_EVS_SendEvent(MD_STOP_DWELL_INF_EID, CFE_EVS_INFORMATION,
144  "Stop Dwell Table command processed successfully for table mask 0x%04X",
145  Stop->TableMask );
146 
147  MD_AppData.CmdCounter++;
148  }
149  else
150  {
151  CFE_EVS_SendEvent(MD_EMPTY_TBLMASK_ERR_EID, CFE_EVS_ERROR,
152  "%s command rejected because no tables were specified in table mask (0x%04X)",
153  "Stop Dwell", Stop->TableMask );
154  MD_AppData.ErrCounter++;
155  }
156  return;
157 } /* End of MD_ProcessStopCmd */

161 void MD_ProcessJamCmd(CFE_SB_MsgPtr_t MessagePtr)
162 {
163  /* Local variables */
164  MD_CmdJam_t *Jam = 0;
165  boolean AllInputsValid = TRUE;
166  cpuaddr ResolvedAddr=0;
167  MD_DwellControlEntry_t *DwellEntryPtr = 0; /* points to local task data */
168  uint16 EntryIndex = 0;
169  uint8 TableIndex = 0;
170  CFS_SymAddr_t NewDwellAddress;
171 
172  /*
173  ** Cast message to Jam Command.
174  */
175  Jam = (MD_CmdJam_t * ) MessagePtr;
176 
177  /* In case Dwell Address sym name isn't null terminated, do it now. */
178  Jam->DwellAddress.SymName[OS_MAX_SYM_LEN - 1] = '\0';
179 
180  /*
181  ** Check that TableId and EntryId command arguments,
182  ** which are used as array indexes, are valid.
183  */
184  if ( !MD_ValidTableId ( Jam->TableId) )
185  {
186  CFE_EVS_SendEvent(MD_INVALID_JAM_TABLE_ERR_EID, CFE_EVS_ERROR,
187  "Jam Cmd rejected due to invalid Tbl Id arg = %d (Expect 1.. %d)",
188  Jam->TableId, MD_NUM_DWELL_TABLES);
189 
190  AllInputsValid = FALSE;
191  }
192  else if ( !MD_ValidEntryId ( Jam->EntryId))
193  {
194  CFE_EVS_SendEvent(MD_INVALID_ENTRY_ARG_ERR_EID, CFE_EVS_ERROR,
195  "Jam Cmd rejected due to invalid Entry Id arg = %d (Expect 1.. %d)",
196  Jam->EntryId, MD_DWELL_TABLE_SIZE);
197 
198  AllInputsValid = FALSE;
199  }
200 
201  /*
202  ** If all inputs checked so far are valid, continue.
203  */
204  if (AllInputsValid == TRUE)
205  {
206  TableIndex = Jam->TableId-1;
207  EntryIndex = Jam->EntryId-1;
208 
209  DwellEntryPtr = (MD_DwellControlEntry_t *) &MD_AppData.MD_DwellTables[TableIndex].Entry[EntryIndex];
210 
211  if (Jam->FieldLength == 0)
212  /*
213  ** Jam a null entry. Set all entry fields to zero.
214  */
215  {
216  /* Assign local values */
217  DwellEntryPtr->ResolvedAddress = 0;
218  DwellEntryPtr->Length = 0;
219  DwellEntryPtr->Delay = 0;
220 
221  /* Update Table Services buffer */
222  NewDwellAddress.Offset = 0;
223  NewDwellAddress.SymName[0] = '\0';
224  MD_UpdateTableDwellEntry (TableIndex, EntryIndex, 0, 0, NewDwellAddress);
225 
226  /* Issue event */
227  CFE_EVS_SendEvent(MD_JAM_NULL_DWELL_INF_EID, CFE_EVS_INFORMATION,
228  "Successful Jam of a Null Dwell Entry to Dwell Tbl#%d Entry #%d",
229  Jam->TableId, Jam->EntryId );
230  }
231  else
232  /*
233  ** Process non-null entry.
234  */
235  {
236  /*
237  ** Check that address and field length arguments pass all validity checks.
238  */
239 
240  /* Resolve and Validate Dwell Address */
241  if (CFS_ResolveSymAddr(&Jam->DwellAddress,&ResolvedAddr) == FALSE)
242  {
243  /* If DwellAddress argument couldn't be resolved, issue error event */
244  CFE_EVS_SendEvent(MD_CANT_RESOLVE_JAM_ADDR_ERR_EID,
245  CFE_EVS_ERROR,
246  "Jam Cmd rejected because symbolic address '%s' couldn't be resolved",
247  Jam->DwellAddress.SymName);
248  AllInputsValid = FALSE;
249  }
250  else if (!MD_ValidFieldLength(Jam->FieldLength))
251  {
252  CFE_EVS_SendEvent(MD_INVALID_LEN_ARG_ERR_EID, CFE_EVS_ERROR,
253  "Jam Cmd rejected due to invalid Field Length arg = %d (Expect 0,1,2,or 4)",
254  Jam->FieldLength);
255  AllInputsValid = FALSE;
256  }
257  else if (!MD_ValidAddrRange(ResolvedAddr, Jam->FieldLength))
258  {
259  /* Issue event message that ResolvedAddr is invalid */
260  CFE_EVS_SendEvent(MD_INVALID_JAM_ADDR_ERR_EID, CFE_EVS_ERROR,
261  "Jam Cmd rejected because address 0x%08X is not in a valid range",
262  (unsigned int)ResolvedAddr);
263  AllInputsValid = FALSE;
264  }
265 #if MD_ENFORCE_DWORD_ALIGN == 0
266  else if ((Jam->FieldLength == 4) &&
267  CFS_Verify16Aligned(ResolvedAddr, (uint32)Jam->FieldLength) != TRUE)
268  {
269  CFE_EVS_SendEvent(MD_JAM_ADDR_NOT_16BIT_ERR_EID, CFE_EVS_ERROR,
270  "Jam Cmd rejected because address 0x%08X is not 16-bit aligned",
271  ResolvedAddr);
272  AllInputsValid = FALSE;
273  }
274 #else
275  else if ((Jam->FieldLength == 4) &&
276  CFS_Verify32Aligned(ResolvedAddr, (uint32)Jam->FieldLength) != TRUE)
277  {
278  CFE_EVS_SendEvent(MD_JAM_ADDR_NOT_32BIT_ERR_EID, CFE_EVS_ERROR,
279  "Jam Cmd rejected because address 0x%08X is not 32-bit aligned",
280  (unsigned int)ResolvedAddr);
281  AllInputsValid = FALSE;
282  }
283 #endif
284  else if ((Jam->FieldLength == 2) && CFS_Verify16Aligned(ResolvedAddr, (uint32)Jam->FieldLength) != TRUE)
285  {
286  CFE_EVS_SendEvent(MD_JAM_ADDR_NOT_16BIT_ERR_EID, CFE_EVS_ERROR,
287  "Jam Cmd rejected because address 0x%08X is not 16-bit aligned",
288  (unsigned int)ResolvedAddr);
289  AllInputsValid = FALSE;
290  }
291 
292 
293  if (AllInputsValid == TRUE)
294  /*
295  ** Perform Jam Operation : Copy Resolved DwellAddress, Length, and Delay to
296  ** local control structure.
297  */
298  {
299  /* Jam the new values into Application control structure */
300  DwellEntryPtr->ResolvedAddress = ResolvedAddr;
301  DwellEntryPtr->Length = Jam->FieldLength;
302  DwellEntryPtr->Delay = Jam->DwellDelay;
303 
304  /* Update values in Table Services buffer */
305  NewDwellAddress.Offset = Jam->DwellAddress.Offset;
306 
307  strncpy (NewDwellAddress.SymName, Jam->DwellAddress.SymName, OS_MAX_SYM_LEN);
308 
309  MD_UpdateTableDwellEntry (TableIndex, EntryIndex, Jam->FieldLength, Jam->DwellDelay, NewDwellAddress);
310 
311  /* Issue event */
312  CFE_EVS_SendEvent(MD_JAM_DWELL_INF_EID, CFE_EVS_INFORMATION,
313  "Successful Jam to Dwell Tbl#%d Entry #%d",
314  Jam->TableId, Jam->EntryId );
315  }
316 
317  } /* end else Process non-null entry */
318 
319  } /* end if AllInputsValid */
320 
321  /*
322  ** Handle bookkeeping.
323  */
324  if (AllInputsValid == TRUE)
325  {
326  MD_AppData.CmdCounter++;
327 
328  /* Update Dwell Table Control Info, including rate */
329  MD_UpdateDwellControlInfo(TableIndex);
330 
331  /* If table contains a rate of zero, and it enabled report that no processing will occur */
332  if ((MD_AppData.MD_DwellTables[ TableIndex ].Rate == 0) &&
333  (MD_AppData.MD_DwellTables[ TableIndex ].Enabled == MD_DWELL_STREAM_ENABLED))
334  {
335  CFE_EVS_SendEvent(MD_ZERO_RATE_CMD_INF_EID, CFE_EVS_INFORMATION,
336  "Dwell Table %d is enabled with a delay of zero so no processing will occur", Jam->TableId);
337  }
338  }
339  else
340  {
341  MD_AppData.ErrCounter++;
342  }
343 
344  return;
345 
346 } /* End of MD_ProcessJamCmd */

353 void MD_ProcessSignatureCmd(CFE_SB_MsgPtr_t MessagePtr)
354 {
355  MD_CmdSetSignature_t *SignatureCmd;
356  uint16 TblId;
357  uint16 StringLength;
358 
359  /*
360  ** Cast message to Set Signature Command.
361  */
362  SignatureCmd = (MD_CmdSetSignature_t * ) MessagePtr;
363 
364  TblId = SignatureCmd->TableId;
365 
366  /*
367  ** Check for Null Termination of string
368  */
369  for(StringLength = 0; StringLength < MD_SIGNATURE_FIELD_LENGTH; StringLength++)
370  {
371  if(SignatureCmd->Signature[StringLength] == '\0')
372  break;
373  }
374 
375  if (StringLength >= MD_SIGNATURE_FIELD_LENGTH)
376  {
377  CFE_EVS_SendEvent(MD_INVALID_SIGNATURE_LENGTH_ERR_EID, CFE_EVS_ERROR,
378  "Set Signature cmd rejected due to invalid Signature length");
379 
380  MD_AppData.ErrCounter++;
381  }
382 
383  /*
384  ** Check for valid TableId argument
385  */
386  else if ( !MD_ValidTableId ( TblId) )
387  {
388  CFE_EVS_SendEvent(MD_INVALID_SIGNATURE_TABLE_ERR_EID, CFE_EVS_ERROR,
389  "Set Signature cmd rejected due to invalid Tbl Id arg = %d (Expect 1.. %d)",
390  TblId, MD_NUM_DWELL_TABLES);
391 
392  MD_AppData.ErrCounter++;
393  }
394 
395  else
396 
397  /*
398  ** Handle nominal case.
399  */
400  {
401  /* Copy signature field to local dwell control structure */
402  strncpy(MD_AppData.MD_DwellTables[TblId-1].Signature,
403  SignatureCmd->Signature, MD_SIGNATURE_FIELD_LENGTH);
404 
405  /* Update signature in Table Services buffer */
406  MD_UpdateTableSignature(TblId-1,SignatureCmd->Signature);
407 
408  CFE_EVS_SendEvent(MD_SET_SIGNATURE_INF_EID, CFE_EVS_INFORMATION,
409  "Successfully set signature for Dwell Tbl#%d to '%s'",
410  TblId, SignatureCmd->Signature );
411 
412  MD_AppData.CmdCounter++;
413  }
414  return;
415 
416 }
