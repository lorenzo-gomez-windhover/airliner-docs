34 void MD_DwellLoop( void )
35 {
36  uint16 TblIndex;
37  uint16 EntryIndex;
38  uint16 NumDwellAddresses;
39  MD_DwellPacketControl_t *TblPtr;
40 
41  /* Check each dwell table */
42  for (TblIndex = 0; TblIndex < MD_NUM_DWELL_TABLES ; TblIndex++)
43  {
44 
45  TblPtr = &MD_AppData.MD_DwellTables[TblIndex];
46  NumDwellAddresses = TblPtr->AddrCount;
47 
48  /* Process enabled dwell tables */
49  if ((TblPtr->Enabled == MD_DWELL_STREAM_ENABLED) && (TblPtr->Rate > 0))
50  {
51 
52  /*
53  ** Handle special case that dwell pkt is already full because
54  ** pkt size was shortened after data had been written to the pkt.
55  */
56 
57  if ( TblPtr->CurrentEntry >= NumDwellAddresses)
58  {
59 
60  MD_SendDwellPkt( TblIndex );
61 
62  /* Initialize CurrentEntry index */
63  TblPtr->CurrentEntry = 0;
64  TblPtr->PktOffset = 0;
65  TblPtr->Countdown = TblPtr->Entry[NumDwellAddresses - 1 ].Delay;
66 
67  }
68 
69  else
70  /*
71  ** Handle nominal processing
72  */
73  {
74  /* Decrement counter */
75  TblPtr->Countdown--;
76 
77 
78  /* Check if it's time to collect data */
79  while (TblPtr->Countdown == 0)
80  {
81  EntryIndex = TblPtr->CurrentEntry;
82 
83  /* Read data for next address and write it to dwell pkt */
84  MD_GetDwellData(TblIndex, EntryIndex);
85 
86  /* Check if the dwell pkt is now full */
87  if (EntryIndex == NumDwellAddresses - 1)
88 
89  /* Case: Just filled last active entry of dwell table */
90  {
91 
92  /*
93  ** Send dwell packet
94  */
95 
96  MD_SendDwellPkt( TblIndex );
97 
98  /*
99  ** Assign control values to cause dwell processing to
100  ** continue at beginning of dwell control structure.
101  */
102 
103  /* Reset countdown timer based on current Delay field */
104  TblPtr->Countdown = TblPtr->Entry[EntryIndex ].Delay;
105 
106  /* Initialize CurrentEntry index */
107  TblPtr->CurrentEntry = 0;
108  TblPtr->PktOffset = 0;
109  }
110 
111  else
112  /* Case: There are more addresses to read for current pkt.*/
113  {
114  /*
115  ** Assign control values to cause dwell processing to
116  ** continue at next entry in dwell control structure.
117  */
118 
119  /* Reset countdown timer based on current Delay field */
120  TblPtr->Countdown = TblPtr->Entry[EntryIndex ].Delay;
121 
122  /* Increment CurrentEntry index */
123  TblPtr->CurrentEntry++;
124  }
125 
126  } /* end while Countdown == 0 */
127 
128  } /* end else handle nominal processing */
129 
130  } /* end if current dwell stream enabled */
131 
132  } /* end for each dwell table */
133 
134 } /* End of MD_DwellLoop */

138 int32 MD_GetDwellData( uint16 TblIndex, uint16 EntryIndex )
139 {
140  uint8 NumBytes; /* Num of bytes to read */
141  uint32 MemReadVal; /* 1-, 2-, or 4-byte value */
142  MD_DwellPacketControl_t *TblPtr; /* Points to table struct */
143  cpuaddr DwellAddress; /* dwell address */
144  int32 Status;
145 
146  Status = CFE_SUCCESS;
147 
148  /* Initialize pointer to current table */
149  TblPtr = (MD_DwellPacketControl_t *)&MD_AppData.MD_DwellTables[TblIndex];
150 
151  /* How many bytes to read?*/
152  NumBytes = TblPtr->Entry[EntryIndex].Length;
153 
154  /* fetch data pointed to by this address */
155  DwellAddress = TblPtr->Entry[EntryIndex].ResolvedAddress;
156 
157  if (NumBytes == 1)
158  {
159  if (CFE_PSP_MemRead8( DwellAddress, (uint8 *) &MemReadVal ) != CFE_SUCCESS)
160  {
161  Status = -1;
162  }
163  }
164 
165  else if (NumBytes == 2)
166  {
167  if (CFE_PSP_MemRead16( DwellAddress, (uint16 *) &MemReadVal ) != CFE_SUCCESS)
168  {
169  Status = -1;
170  }
171  }
172 
173  else if (NumBytes == 4)
174  {
175  if (CFE_PSP_MemRead32( DwellAddress, &MemReadVal ) != CFE_SUCCESS)
176  {
177  Status = -1;
178  }
179  }
180 
181  else /* Invalid dwell length */
182  /* Shouldn't ever get here unless length value was corrupted. */
183  {
184  Status = -1;
185  }
186 
187 
188  /* If value was read successfully, copy value to dwell packet. */
189  /* Wouldn't want to copy, if say, there was an invalid length & we */
190  /* didn't read. */
191  if (Status == CFE_SUCCESS)
192  {
193  CFE_PSP_MemCpy( (void*) &MD_AppData.MD_DwellPkt[TblIndex].Data[TblPtr->PktOffset],
194  (void*) &MemReadVal,
195  NumBytes);
196  }
197 
198  /* Update write location in dwell packet */
199  TblPtr->PktOffset += NumBytes;
200 
201  return Status;
202 
203 } /* End of MD_GetDwellData */

208 void MD_SendDwellPkt( uint16 TableIndex )
209 {
210  uint16 DwellPktSize; /* Dwell Packet Size, in bytes */
211 
212  /* Assign pointers to structures */
213  MD_DwellPacketControl_t *TblPtr = &MD_AppData.MD_DwellTables[TableIndex];
214  MD_DwellPkt_t *PktPtr = &MD_AppData.MD_DwellPkt[TableIndex];
215 
216  /*
217  ** Assign packet fields.
218  */
219  PktPtr->TableId = TableIndex + 1;
220  PktPtr->AddrCount = TblPtr->AddrCount;
221  PktPtr->Rate = TblPtr->Rate;
222 #if MD_SIGNATURE_OPTION == 1
223  strncpy(PktPtr->Signature, TblPtr->Signature, MD_SIGNATURE_FIELD_LENGTH - 1);
224  /* Make sure string is null-terminated. */
225  PktPtr->Signature[MD_SIGNATURE_FIELD_LENGTH - 1] = '\0';
226 #endif
227  PktPtr->ByteCount = TblPtr->DataSize;
228 
229  /*
230  ** Set packet length in header.
231  */
232 
233  DwellPktSize = MD_DWELL_PKT_LNGTH - MD_DWELL_TABLE_SIZE * 4 + TblPtr->DataSize;
234 
235 
236  CFE_SB_SetTotalMsgLength((CFE_SB_Msg_t *)PktPtr, DwellPktSize);
237 
238  /*
239  ** Send housekeeping telemetry packet.
240  */
241  CFE_SB_TimeStampMsg((CFE_SB_Msg_t *) PktPtr);
242  CFE_SB_SendMsg((CFE_SB_Msg_t *) PktPtr);
243 
244 } /* End of MD_SendDwellPkt */
