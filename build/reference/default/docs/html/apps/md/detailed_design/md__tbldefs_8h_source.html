40 /*
41 ** \brief Memory Dwell structure for individual memory dwell specifications
42 **
43 ** \par Description
44 ** To be valid, entry must either be a null entry
45 ** (specified by a zero field length) or all of its
46 ** address and length fields must pass various checks.
47 **
48 ** The following checks must be met for a valid non-null entry:
49 **
50 ** * Length fields must be 1, 2, or 4.
51 ** * Delay can be any value in a uint16 field.
52 ** * DwellAddress.SymName must be a null string or must be found in the Symbol Table.
53 ** * The sum of DwellAddress.SymName plus DwellAddress.Offset must be in a valid range,
54 ** as determined by OSAL routines.
55 ** * If the Length field is 2, the sum of DwellAddress.SymName plus DwellAddress.Offset
56 ** must be even.
57 ** * If the Length field is 4, the sum of DwellAddress.SymName plus DwellAddress.Offset
58 ** must be evenly divisable by 4.
59 */
60 typedef struct
61 {
62  uint16 Length;
63  uint16 Delay;
64  CFS_SymAddr_t DwellAddress;
65 } MD_TableLoadEntry_t;
