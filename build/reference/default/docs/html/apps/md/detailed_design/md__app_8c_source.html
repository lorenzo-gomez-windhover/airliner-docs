48 void MD_AppMain ( void )
49 {
50  CFE_SB_MsgId_t MessageID;
51  int32 Status;
52  uint8 TblIndex;
53  boolean IsRegistered;
54 
55  /*
56  ** Register the Application with Executive Services
57  */
58  Status = CFE_ES_RegisterApp();
59  if(Status != CFE_SUCCESS)
60  {
61  IsRegistered = FALSE;
62  CFE_ES_WriteToSysLog
63  ("MD_APP: Call to CFE_ES_RegisterApp Failed:RC=0x%08X\n",(unsigned int)Status);
64  MD_AppData.RunStatus = CFE_ES_APP_ERROR;
65  }/* end if */
66  else
67  { /* Successfully registered */
68  IsRegistered = TRUE;
69  MD_AppData.RunStatus = CFE_ES_APP_RUN;
70 
71  /* Create the first Performance Log entry */
72  CFE_ES_PerfLogEntry(MD_APPMAIN_PERF_ID);
73 
74  /* Initialize the task */
75  Status = MD_AppInit();
76 
77  if (Status != CFE_SUCCESS)
78  {
79  CFE_ES_WriteToSysLog("MD:Application Init Failed,RC=0x%08X\n", (unsigned int)Status);
80  MD_AppData.RunStatus = CFE_ES_APP_ERROR;
81  }
82  }
83 
84 
85  /* Loop Forever, or until ES exit command, or SB error */
86  while ( CFE_ES_RunLoop(&MD_AppData.RunStatus) == TRUE)
87  {
88  /* Copy any newly loaded tables */
89  for (TblIndex=0; TblIndex < MD_NUM_DWELL_TABLES; TblIndex++)
90  {
91 
92  MD_ManageDwellTable(TblIndex);
93 
94  } /* end for each table loop */
95 
96  CFE_ES_PerfLogExit(MD_APPMAIN_PERF_ID);
97 
98  /* Wait for next message */
99  Status = CFE_SB_RcvMsg (&MD_AppData.MsgPtr, MD_AppData.CmdPipe, CFE_SB_PEND_FOREVER );
100 
101  if (Status != CFE_SUCCESS)
102  {
103  /*
104  ** Exit on pipe read error
105  */
106  CFE_EVS_SendEvent(MD_PIPE_ERR_EID, CFE_EVS_ERROR,
107  "SB Pipe Read Error, App will exit. Pipe Return Status = 0x%08X", (unsigned int)Status);
108 
109  MD_AppData.RunStatus = CFE_ES_APP_ERROR;
110 
111  }
112 
113  CFE_ES_PerfLogEntry(MD_APPMAIN_PERF_ID);
114 
115  /* Process Executive Request */
116  if(Status == CFE_SUCCESS)
117  {
118  MessageID = CFE_SB_GetMsgId(MD_AppData.MsgPtr);
119 
120  switch(MessageID)
121  {
122  case MD_WAKEUP_MID:
123 
124  if (CFE_SB_GetTotalMsgLength(MD_AppData.MsgPtr) != sizeof(MD_NoArgsCmd_t))
125  {
126  CFE_EVS_SendEvent( MD_MSG_LEN_ERR_EID, CFE_EVS_ERROR,
127  "Msg with Bad length Rcvd: ID = 0x%04X, Exp Len = %d, Len = %d",
128  MessageID,
129  sizeof(MD_NoArgsCmd_t),
130  CFE_SB_GetTotalMsgLength(MD_AppData.MsgPtr));
131  }
132  else
133  {
134  MD_DwellLoop();
135  }
136  break ;
137 
138  case MD_CMD_MID:
139 
140  MD_ExecRequest( MD_AppData.MsgPtr );
141  break ;
142 
143  case MD_SEND_HK_MID:
144 
145  if (CFE_SB_GetTotalMsgLength(MD_AppData.MsgPtr) != sizeof(MD_NoArgsCmd_t))
146  {
147  CFE_EVS_SendEvent( MD_MSG_LEN_ERR_EID, CFE_EVS_ERROR,
148  "Msg with Bad length Rcvd: ID = 0x%04X, Exp Len = %d, Len = %d",
149  MessageID,
150  sizeof(MD_NoArgsCmd_t),
151  CFE_SB_GetTotalMsgLength(MD_AppData.MsgPtr));
152  }
153  else
154  {
155  MD_HkStatus();
156  }
157  break;
158 
159  default:
160  CFE_EVS_SendEvent(MD_MID_ERR_EID, CFE_EVS_ERROR,
161  "Msg with Invalid message ID Rcvd -- ID = 0x%04X",
162  MessageID );
163  break;
164  }
165  }
166 
167 
168  } /* end while forever loop */
169 
170 
171  /*
172  ** Performance Log Exit Stamp.
173  */
174  if (IsRegistered == TRUE)
175  {
176  CFE_ES_PerfLogExit(MD_APPMAIN_PERF_ID);
177  }
178 
179  /*
180  ** Exit the Application
181  */
182  CFE_ES_ExitApp(MD_AppData.RunStatus);
183 
184 
185 } /* End of MD_AppMain */

382 int32 MD_InitTableServices( void )
383 {
384  int32 Status;
385  int32 GetAddressResult;
386  uint8 TblIndex;
387  boolean RecoveredValidTable; /* for current table */
388  boolean TableInitValidFlag = TRUE; /* for all tables so far*/
389  MD_DwellTableLoad_t InitMemDwellTable;
390  MD_DwellTableLoad_t* MD_LoadTablePtr = 0;
391  uint16 TblRecos = 0; /* Number of Tables Recovered */
392  uint16 TblInits = 0; /* Number of Tables Initialized */
393  char TblFileName[OS_MAX_PATH_LEN];
394 
395  /* Prepare Data Structure used for loading Initial Table Data */
396 
397  CFE_PSP_MemSet(&InitMemDwellTable,0,sizeof(MD_DwellTableLoad_t));
398 
399  /*
400  ** For each table, load recovered data if available and valid.
401  ** Otherwise, load initial data.
402  */
403 
404  for ( TblIndex = 0;
405  (TblIndex < MD_NUM_DWELL_TABLES) && (TableInitValidFlag == TRUE);
406  TblIndex++)
407  {
408  RecoveredValidTable = FALSE;
409 
410  /* Prepare Table Name */
411  snprintf(MD_AppData.MD_TableName[TblIndex],
412  CFE_TBL_MAX_NAME_LENGTH + 1, /* allows total of CFE_TBL_MAX_NAME_LENGTH characters to be copied */
413  "%s%d", MD_DWELL_TABLE_BASENAME, TblIndex + 1);
414 
415  snprintf(TblFileName,
416  OS_MAX_PATH_LEN, /* allows total of CFE_TBL_MAX_NAME_LENGTH characters to be copied */
417  MD_TBL_FILENAME_FORMAT, TblIndex + 1);
418 
419  /* Register Dwell Table #tblnum */
420  Status = CFE_TBL_Register(
421  &MD_AppData.MD_TableHandle[TblIndex], /* Table Handle (to be returned) */
422  MD_AppData.MD_TableName[TblIndex], /* Table Name */
423  sizeof(MD_DwellTableLoad_t), /* Size of Table being Registered */
424  CFE_TBL_OPT_DEFAULT | CFE_TBL_OPT_CRITICAL,
425  &MD_TableValidationFunc); /* Ptr to table validation function */
426 
427  /* Check for recovered table data */
428  if (Status == CFE_TBL_INFO_RECOVERED_TBL)
429  {
430  /* Restore dwell table contents from CDS */
431 
432  GetAddressResult = CFE_TBL_GetAddress ( (void *) &MD_LoadTablePtr,
433  MD_AppData.MD_TableHandle[TblIndex]);
434 
435  if(GetAddressResult != CFE_TBL_INFO_UPDATED)
436  {
437  CFE_EVS_SendEvent(MD_NO_TBL_COPY_ERR_EID, CFE_EVS_ERROR,
438 "Didn't update MD tbl #%d due to unexpected CFE_TBL_GetAddress return: 0x%08X",
439  TblIndex+1, (unsigned int)GetAddressResult);
440  }
441 
442  else /* GetAddressResult == CFE_TBL_INFO_UPDATED*/
443  {
444  if(MD_TableValidationFunc((void*)MD_LoadTablePtr) ==
445  CFE_SUCCESS)
446  {
447  MD_CopyUpdatedTbl( MD_LoadTablePtr, TblIndex);
448 
449  if (MD_LoadTablePtr->Enabled == MD_DWELL_STREAM_ENABLED)
450  {
451  MD_StartDwellStream ( (uint16) TblIndex);
452  }
453 
454  RecoveredValidTable=TRUE;
455  TblRecos++;
456  CFE_EVS_SendEvent(MD_RECOVERED_TBL_VALID_INF_EID, CFE_EVS_INFORMATION,
457  "Recovered Dwell Table #%d is valid and has been copied to the MD App",
458  TblIndex+1);
459  }
460  else
461  {
462  CFE_EVS_SendEvent(MD_RECOVERED_TBL_NOT_VALID_ERR_EID, CFE_EVS_ERROR,
463  "MD App will reinitialize Dwell Table #%d because recovered table is not valid",
464  TblIndex+1);
465  }
466 
467  } /* end if(GetAddressResult != CFE_TBL_INFO_UPDATED) */
468 
469  /* Unlock Table */
470  CFE_TBL_ReleaseAddress(MD_AppData.MD_TableHandle[TblIndex] );
471 
472  } /* end if Register indicates recovered data */
473 
474  else if (Status == CFE_TBL_ERR_INVALID_SIZE)
475  {
476  CFE_EVS_SendEvent (MD_DWELL_TBL_TOO_LARGE_CRIT_EID, CFE_EVS_CRITICAL,
477  "Dwell Table(s) are too large to register: %d > %d bytes, %d > %d entries",
478  MD_TBL_LOAD_LNGTH, CFE_TBL_MAX_SNGL_TABLE_SIZE,
479  MD_DWELL_TABLE_SIZE,
480  (uint16) ((CFE_TBL_MAX_SNGL_TABLE_SIZE - sizeof(uint32) ) / sizeof (MD_TableLoadEntry_t)));
481  TableInitValidFlag = FALSE;
482  } /* end if */
483 
484  else if (Status != CFE_SUCCESS)
485  {
486  CFE_EVS_SendEvent (MD_TBL_REGISTER_CRIT_EID, CFE_EVS_CRITICAL,
487  "CFE_TBL_Register error 0x%08X received for tbl#%d",
488  (unsigned int)Status,TblIndex+1);
489  TableInitValidFlag = FALSE;
490  } /* end if */
491 
492  /*
493  ** Load initial values if needed
494  */
495  if((RecoveredValidTable == FALSE) && (TableInitValidFlag == TRUE))
496  {
497  Status = CFE_TBL_Load(MD_AppData.MD_TableHandle[TblIndex],
498  CFE_TBL_SRC_FILE, /* following ptr is memory ptr */
499  (const void *) TblFileName); /* Pointer to data to be loaded */
500 
501  MD_AppData.MD_DwellTables[TblIndex].Enabled = MD_DWELL_STREAM_DISABLED;
502 
503  if(Status != CFE_SUCCESS)
504  {
505  CFE_ES_WriteToSysLog
506  ("MD_APP: Error 0x%08X received loading tbl#%d\n",
507  (unsigned int)Status,TblIndex+1);
508  TableInitValidFlag = FALSE;
509  }/* end if */
510  else
511  {
512  TblInits++;
513  }
514 
515  }
516 
517  if( TableInitValidFlag == TRUE)
518  {
519  /* Update Dwell Table Control Info, including rate */
520  MD_UpdateDwellControlInfo(TblIndex);
521  }
522 
523  } /* end for loop */
524 
525  /* Output init and recovery event message */
526  CFE_EVS_SendEvent(MD_TBL_INIT_INF_EID, CFE_EVS_INFORMATION,
527  "Dwell Tables Recovered: %d, Dwell Tables Initialized: %d",
528  TblRecos, TblInits);
529 
530 
531  if (TableInitValidFlag == TRUE)
532  {
533  return CFE_SUCCESS;
534  }
535  else
536  {
537  return Status;
538  }
539 
540 } /* End of MD_InitTableServices */

544 /******************************************************************************/
545 int32 MD_ManageDwellTable (uint8 TblIndex)
546 {
547  int32 Status = CFE_SUCCESS;
548  int32 GetAddressResult;
549  boolean FinishedManaging = FALSE;
550  MD_DwellTableLoad_t *MD_LoadTablePtr = 0;
551 
552  while (!FinishedManaging)
553  {
554  /* Determine if the table has a pending validation or update */
555  Status = CFE_TBL_GetStatus( MD_AppData.MD_TableHandle[TblIndex]);
556 
557 
558  if (Status == CFE_TBL_INFO_VALIDATION_PENDING)
559  {
560  /* Validate the specified Table */
561  Status = CFE_TBL_Validate(MD_AppData.MD_TableHandle[TblIndex]);
562 
563  if (Status != CFE_SUCCESS)
564  {
565  /* If an error occurred during Validate, */
566  /* then do not perform any more managing */
567  FinishedManaging = TRUE;
568  }
569  }
570  else if (Status == CFE_TBL_INFO_UPDATE_PENDING)
571  {
572  Status = CFE_TBL_Update ( MD_AppData.MD_TableHandle[TblIndex]);
573 
574  /* If table was updated, copy contents to local structure */
575  if (Status == CFE_SUCCESS)
576  {
577  GetAddressResult = CFE_TBL_GetAddress ( (void *) &MD_LoadTablePtr,
578  MD_AppData.MD_TableHandle[TblIndex]);
579 
580  if(GetAddressResult == CFE_TBL_INFO_UPDATED)
581  {
582  MD_CopyUpdatedTbl(MD_LoadTablePtr, TblIndex);
583 
584  if (MD_LoadTablePtr->Enabled == MD_DWELL_STREAM_ENABLED)
585  {
586  MD_StartDwellStream ( (uint16) TblIndex);
587  }
588  }
589  else
590  {
591  CFE_EVS_SendEvent(MD_NO_TBL_COPY_ERR_EID, CFE_EVS_ERROR,
592 "Didn't update MD tbl #%d due to unexpected CFE_TBL_GetAddress return: 0x%08X",
593  TblIndex+1, (unsigned int)GetAddressResult);
594  }
595 
596  /* Unlock Table */
597  CFE_TBL_ReleaseAddress(MD_AppData.MD_TableHandle[TblIndex] );
598  }
599 
600  /* After an Update, always assume we are done */
601  /* and return Update Status */
602  FinishedManaging = TRUE;
603  }
604  else if((Status & CFE_SEVERITY_BITMASK) == CFE_SEVERITY_ERROR)
605  {
606  CFE_EVS_SendEvent(MD_TBL_STATUS_ERR_EID, CFE_EVS_ERROR,
607  "Received unexpected error 0x%08X from CFE_TBL_GetStatus for tbl #%d",
608  (unsigned int)Status, TblIndex+1);
609  FinishedManaging = TRUE;
610  }
611 
612  else
613  {
614  FinishedManaging = TRUE;
615  }
616  }
617 
618  return Status;
619 
620 } /* End of MD_ManageDwellTable */

625 void MD_ExecRequest(CFE_SB_MsgPtr_t MessagePtr )
626 {
627  uint16 CommandCode;
628  CFE_SB_MsgId_t MessageID;
629  int16 CmdIndx;
630  uint16 ActualLength;
631 
632  /* Extract Command Code and Message Id */
633  CommandCode = CFE_SB_GetCmdCode(MessagePtr);
634  MessageID = CFE_SB_GetMsgId(MessagePtr);
635 
636  /* Find index which will be used to access ExpectedLength data */
637  CmdIndx = MD_SearchCmdHndlrTbl(MessageID, CommandCode);
638 
639  if (CmdIndx < 0)
640  {
641 
642  /* If match wasn't found in command info structure, */
643  /* issue an error event, increment error counter, and return. */
644  CFE_EVS_SendEvent (MD_CC_NOT_IN_TBL_ERR_EID, CFE_EVS_ERROR,
645  "Command Code %d not found in MD_CmdHandlerTbl structure",
646  CommandCode);
647  MD_AppData.ErrCounter++;
648  return;
649  }
650 
651  /* Get Command Length */
652  ActualLength = CFE_SB_GetTotalMsgLength(MessagePtr);
653 
654  /* If Command Length is inconsistent with command type, */
655  /* send error event, increment error count, and return. */
656  if (ActualLength != MD_CmdHandlerTbl[CmdIndx].ExpectedLength)
657  {
658  CFE_EVS_SendEvent( MD_CMD_LEN_ERR_EID, CFE_EVS_ERROR,
659  "Cmd Msg with Bad length Rcvd: ID = 0x%04X, CC = %d, Exp Len = %d, Len = %d",
660  MessageID, CommandCode,
661  (int)MD_CmdHandlerTbl[CmdIndx].ExpectedLength, ActualLength);
662 
663  MD_AppData.ErrCounter++;
664  return;
665  }
666 
667 
668  /* Process command */
669  switch(CommandCode)
670  {
671  case MD_NOOP_CC:
672 
673  CFE_EVS_SendEvent (MD_NOOP_INF_EID, CFE_EVS_INFORMATION,
674  "No-op command, Version %d.%d.%d.%d",
675  MD_MAJOR_VERSION,
676  MD_MINOR_VERSION,
677  MD_REVISION,
678  MD_MISSION_REV);
679 
680  MD_AppData.CmdCounter++;

787 int16 MD_SearchCmdHndlrTbl( CFE_SB_MsgId_t MessageID, uint16 CommandCode )
788 {
789  int16 TblIndx = -1; /* need index to be 0 after it is incremented for 1st time */
790  boolean MatchedMsgId = FALSE;
791  boolean FoundMatch = FALSE;
792 
793  do
794  {
795  /* Point to next entry in Command Handler Table */
796  TblIndx++;
797 
798  /* Check to see if we found a matching Message ID */
799  if ((MD_CmdHandlerTbl[TblIndx].MsgId == MessageID) &&
800  (MD_CmdHandlerTbl[TblIndx].MsgTypes != MD_TERM_MSGTYPE))
801  { /* MessageID matches and this isn't last Table entry */
802 
803  /* Flag any found message IDs so that if there's an error, we can */
804  /* determine if it was a bad message ID or bad command code */
805  MatchedMsgId = TRUE;
806 
807  /* If entry in the Command Handler Table is a command entry, */
808  /* then check for a matching command code */
809  if (MD_CmdHandlerTbl[TblIndx].MsgTypes == MD_CMD_MSGTYPE)
810  {
811  if (MD_CmdHandlerTbl[TblIndx].CmdCode == CommandCode)
812  {
813  /* Found matching message ID and Command Code */
814  FoundMatch = TRUE;
815  }
816  }
817  else
818  /* Message is not a command message with specific command code */
819  {
820  /* Matching Message ID is all that is required */
821  FoundMatch = TRUE;
822  }
823  }
824 
825 
826  } while ((!FoundMatch) &&
827  (MD_CmdHandlerTbl[TblIndx].MsgTypes != MD_TERM_MSGTYPE));
828 
829  /* If we failed to find a match, return a negative index */
830  if (!FoundMatch)
831  {
832  /* Determine if the message ID was bad or the command code */
833  if (MatchedMsgId)
834  {
835  /* A matching message ID was found, so command code must be bad */
836  TblIndx = MD_BAD_CMD_CODE;
837  }
838  else /* No matching message ID was found */
839  {
840  TblIndx = MD_BAD_MSG_ID;
841  }
842  }
843 
844  return TblIndx;
845 } /* End of MD_SearchCmdHndlrTbl() */
