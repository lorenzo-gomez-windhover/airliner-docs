36 int32 MD_TableValidationFunc (void *TblPtr)
37 {
38  uint16 ActiveEntryCount;
39  uint16 Size;
40  uint32 Rate;
41  int32 Status = CFE_SUCCESS; /* Initialize to valid table */
42  MD_DwellTableLoad_t* LocalTblPtr = (MD_DwellTableLoad_t*) TblPtr;
43  uint16 TblErrorEntryIndex; /* Zero-based entry number for error; */
44  /* valid if there is an error. */
45 
46 #if MD_SIGNATURE_OPTION == 1
47 
48  uint16 StringLength;
49 
50  /*
51  ** Check for Null Termination of string
52  */
53  for(StringLength = 0; StringLength < MD_SIGNATURE_FIELD_LENGTH; StringLength++)
54  {
55  if(LocalTblPtr->Signature[StringLength] == '\0')
56  break;
57  }
58 
59 #endif
60 
61 
62  /* Check that Enabled flag has valid value. */
63  if ((LocalTblPtr->Enabled != MD_DWELL_STREAM_DISABLED) &&
64  (LocalTblPtr->Enabled != MD_DWELL_STREAM_ENABLED))
65  {
66  CFE_EVS_SendEvent(MD_TBL_ENA_FLAG_EID, CFE_EVS_ERROR,
67  "Dwell Table rejected because value of enable flag (%d) is invalid",
68  (int)LocalTblPtr->Enabled);
69  Status = MD_TBL_ENA_FLAG_ERROR;
70  }
71 
72 #if MD_SIGNATURE_OPTION == 1
73 
74  else if(StringLength >= MD_SIGNATURE_FIELD_LENGTH)
75  {
76  CFE_EVS_SendEvent(MD_TBL_SIG_LEN_ERR_EID, CFE_EVS_ERROR,
77  "Dwell Table rejected because Signature length was invalid");
78 
79  Status = MD_SIG_LEN_TBL_ERROR;
80  }
81 
82 #endif
83 
84  else
85  {
86  /* Read Dwell Table to get entry count, byte size, and dwell rate */
87  MD_ReadDwellTable(LocalTblPtr, &ActiveEntryCount, &Size, &Rate);
88 
89 
90  /* Validate entry contents */
91  if ((Status = MD_CheckTableEntries(LocalTblPtr,
92  &TblErrorEntryIndex))
93  != CFE_SUCCESS )
94  {
95  if (Status == MD_RESOLVE_ERROR)
96  {
97  CFE_EVS_SendEvent(MD_RESOLVE_ERR_EID, CFE_EVS_ERROR,
98  "Dwell Table rejected because address (sym='%s'/offset=0x%08X) in entry #%d couldn't be resolved",

104  {

106  "Dwell Table rejected because address (sym='%s'/offset=0x%08X) in entry #%d was out of range",
107  LocalTblPtr->Entry[TblErrorEntryIndex].DwellAddress.SymName,
108  (unsigned int)LocalTblPtr->Entry[TblErrorEntryIndex].DwellAddress.Offset,
109  TblErrorEntryIndex + 1);
110 
111  }
112  else if (Status == MD_INVALID_LEN_ERROR)
113  {
114  CFE_EVS_SendEvent(MD_TBL_HAS_LEN_ERR_EID, CFE_EVS_ERROR,
115  "Dwell Table rejected because length (%d) in entry #%d was invalid",
116  LocalTblPtr->Entry[TblErrorEntryIndex].Length,
117  TblErrorEntryIndex + 1);
118  }
119  else if(Status == MD_NOT_ALIGNED_ERROR)
120  {
121  CFE_EVS_SendEvent(MD_TBL_ALIGN_ERR_EID, CFE_EVS_ERROR,
122  "Dwell Table rejected because address (sym='%s'/offset=0x%08X) in entry #%d not properly aligned for %d-byte dwell",
123  LocalTblPtr->Entry[TblErrorEntryIndex].DwellAddress.SymName,
124  (unsigned int)LocalTblPtr->Entry[TblErrorEntryIndex].DwellAddress.Offset,

138 
139

141 
142

148 int32 MD_ReadDwellTable(MD_DwellTableLoad_t *TblPtr,
149  uint16 *ActiveAddrCountPtr,
150  uint16 *SizePtr,
151  uint32 *RatePtr)
152 {
153  uint16 EntryId;
154  *ActiveAddrCountPtr = 0;
155  *SizePtr = 0;
156  *RatePtr = 0;
157 
158  for ( EntryId = 0;
159  (EntryId < MD_DWELL_TABLE_SIZE) &&
160  ( TblPtr->Entry[EntryId].Length != 0);
161  EntryId++
162  )
163  {
164  /* *ActiveAddrCountPtr++; */
165  (*ActiveAddrCountPtr)++;
166  *SizePtr+= TblPtr->Entry[EntryId].Length;
167  *RatePtr+= TblPtr->Entry[EntryId].Delay;
168  } /* end while */
169 
170  return CFE_SUCCESS;
171 } /* End of MD_ReadDwellTable */
172 /******************************************************************************/
173 int32 MD_CheckTableEntries(MD_DwellTableLoad_t *TblPtr,
174  uint16 *ErrorEntryArg)
175 {
176  int32 Status = CFE_SUCCESS;
177  uint16 EntryIndex;
178  int32 FirstBadIndex = -1;
179 
180  int32 GoodCount = 0;
181  int32 BadCount = 0;
182  int32 UnusedCount = 0;
183 
184  *ErrorEntryArg = 0;
185 
186  /*
187  ** Check each Dwell Table entry for valid address range
188  */
189  for ( EntryIndex=0 ;
190  (EntryIndex < MD_DWELL_TABLE_SIZE) && (Status == CFE_SUCCESS) ;
191  EntryIndex++)
192  {
193 
194  Status = MD_ValidTableEntry(&TblPtr->Entry[EntryIndex]);
195 
196  if (Status == CFE_SUCCESS)
197  {
198  if (TblPtr->Entry[EntryIndex].Length == 0)
199  {
200  /*
201  ** Stop reading through the table when the entry length is
202  ** zero, remaining entries count as unused
203  */
204  break;
205  }
206  else
207  {
208  GoodCount++;
209  }
210  }
211  else
212  {
213  /* Status != CFE_SUCCESS */
214  BadCount++;
215  if (FirstBadIndex == -1)
216  {
217  FirstBadIndex = EntryIndex;
218  *ErrorEntryArg = FirstBadIndex;
219  /* Keep counting good,bad,unused, don't exit immediately */
220  }
221  }
222  }
223 
224  UnusedCount = MD_DWELL_TABLE_SIZE - EntryIndex;
225 
226  /*
227  ** Generate informational event with error totals
228  */
229  CFE_EVS_SendEvent(MD_DWELL_TBL_INF_EID, CFE_EVS_INFORMATION,
230  "MD Dwell Tbl verify results: good = %d, bad = %d, unused = %d",
231  (int)GoodCount, (int)BadCount, (int)UnusedCount);
232  return Status;
233 
234 } /* End of MD_CheckTableEntries */

236 /******************************************************************************/
237 int32 MD_ValidTableEntry (MD_TableLoadEntry_t *TblEntryPtr)
238 {
239  int32 Status;
240  uint16 DwellLength;
241  cpuaddr ResolvedAddr = 0;
242 
243  DwellLength= TblEntryPtr->Length ;
244 
245  if ( DwellLength == 0)
246  {
247  Status = CFE_SUCCESS;
248  }
249  else
250  {
251  if (CFS_ResolveSymAddr(&TblEntryPtr->DwellAddress,
252  &ResolvedAddr) != TRUE)
253  { /* Symbol was non-null AND was not in Symbol Table */
254  Status = MD_RESOLVE_ERROR;
255  } /* end CFS_ResolveSymAddr */
256 
257  else if (MD_ValidAddrRange( ResolvedAddr, (uint32)DwellLength) != TRUE)
258  { /* Address is in invalid range */
259  Status = MD_INVALID_ADDR_ERROR;
260  }
261  else if (MD_ValidFieldLength(DwellLength) != TRUE)
262  {
263  Status = MD_INVALID_LEN_ERROR;
264  }
265 #if MD_ENFORCE_DWORD_ALIGN == 0
266  else if ((DwellLength == 4)
267  && CFS_Verify16Aligned(ResolvedAddr, (uint32)DwellLength) != TRUE)
268  {
269  Status = MD_NOT_ALIGNED_ERROR;
270  }
271 #else
272  else if ((DwellLength == 4)
273  && CFS_Verify32Aligned(ResolvedAddr, (uint32)DwellLength) != TRUE)
274  {
275  Status = MD_NOT_ALIGNED_ERROR;
276  }
277 #endif
278  else if ((DwellLength == 2)
279  && CFS_Verify16Aligned(ResolvedAddr, (uint32)DwellLength) != TRUE)
280  {
281  Status = MD_NOT_ALIGNED_ERROR;
282  }
283  else
284  {
285  Status = CFE_SUCCESS;
286  }
287 
288  } /* end else */
289 
290  return Status;
291 
292 } /* End of MD_ValidTableEntry */

295 void MD_CopyUpdatedTbl(MD_DwellTableLoad_t *MD_LoadTablePtr, uint8 TblIndex)
296 {
297  uint8 EntryIndex;
298  cpuaddr ResolvedAddr;
299  MD_TableLoadEntry_t *ThisLoadEntry;
300  MD_DwellPacketControl_t *LocalControlStruct;
301 
302  /* Assign pointer to internal control structure. */
303  LocalControlStruct = &MD_AppData.MD_DwellTables[TblIndex];
304 
305  /* Copy 'Enabled' field from load structure to internal control structure. */
306  LocalControlStruct->Enabled = MD_LoadTablePtr->Enabled;
307 
308 #if MD_SIGNATURE_OPTION == 1
309  /* Copy 'Signature' field from load structure to internal control structure. */
310  strncpy (LocalControlStruct->Signature, MD_LoadTablePtr->Signature, MD_SIGNATURE_FIELD_LENGTH);
311 
312  /* Ensure that resulting string is null-terminated */
313  LocalControlStruct->Signature[MD_SIGNATURE_FIELD_LENGTH - 1] = '\0';
314 #endif
315 
316  /* For each row in the table load, */
317  /* copy length, delay, and address fields from load structure to */
318  /* internal control structure. */
319  for(EntryIndex=0; EntryIndex < MD_DWELL_TABLE_SIZE; EntryIndex++)
320  {
321  /* Get ResolvedAddr & insert in local control structure */
322 
323  ThisLoadEntry = &MD_LoadTablePtr->Entry[EntryIndex];
324 
325  CFS_ResolveSymAddr(&ThisLoadEntry->DwellAddress, &ResolvedAddr);
326  CFE_PSP_MemCpy( &MD_AppData.MD_DwellTables[TblIndex].Entry[EntryIndex].ResolvedAddress,
327  (void *) &ResolvedAddr, sizeof(uint32) );
328 
329  /* Copy length */
330  CFE_PSP_MemCpy( &MD_AppData.MD_DwellTables[TblIndex].Entry[EntryIndex].Length,
331  (void *) &ThisLoadEntry->Length, sizeof(uint16) );
332 
333  /* Copy delay */
334  CFE_PSP_MemCpy( &MD_AppData.MD_DwellTables[TblIndex].Entry[EntryIndex].Delay,
335  (void *) &ThisLoadEntry->Delay, sizeof(uint16) );
336 
337  } /* end for loop */
338 
339  /* Update Dwell Table Control Info, used to process dwell packets */
340  MD_UpdateDwellControlInfo((uint16)TblIndex);
341 
342  return;
343 
344 } /* End of MD_CopyUpdatedTbl */

374  uint16 EntryIndex,
375  uint16 NewLength,
376  uint16 NewDelay,
377  CFS_SymAddr_t NewDwellAddress)
378 {
379  int32 GetAddressResult;

382 
383  /* Get pointer to Table */
384  GetAddressResult = CFE_TBL_GetAddress((void*)&MD_LoadTablePtr,
385  MD_AppData.MD_TableHandle[TableIndex]);
386  /* Get pointer to specific entry */
387  EntryPtr = &MD_LoadTablePtr->Entry[EntryIndex];
388 
389  /* Copy new numerical values to Table Services buffer */
390  EntryPtr->Length = NewLength;
391  EntryPtr->Delay = NewDelay;
392  EntryPtr->DwellAddress.Offset = NewDwellAddress.Offset;
393 
394  /* Copy symbol name to Table Services buffer */
395  strncpy(EntryPtr->DwellAddress.SymName,
396  NewDwellAddress.SymName,
397  OS_MAX_SYM_LEN);
398  /* Ensure string is null terminated. */
399  EntryPtr->DwellAddress.SymName[OS_MAX_SYM_LEN - 1] = '\0';
400 
401  /* Notify Table Services that buffer was modified */
402  CFE_TBL_Modified (MD_AppData.MD_TableHandle[TableIndex]);
403 
404  /* Release access to Table Services buffer */
405  CFE_TBL_ReleaseAddress (MD_AppData.MD_TableHandle[TableIndex]);
406 
407  return;
