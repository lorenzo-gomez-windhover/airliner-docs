32 boolean MD_TableIsInMask(int16 TableId, uint16 TableMask)
33 {
34  uint16 LocalMask = TableMask;
35  boolean Status;
36 
37  /* Shift TableId - 1 times */
38  if (TableId - 1)
39  {
40  LocalMask = LocalMask >> (TableId -1);
41  }
42 
43  /* If result is odd, */
44  /* then table is in mask. */
45  if ( (LocalMask & (uint16) 1) == (uint16) 1)
46  Status = TRUE;
47  else
48  Status = FALSE;
49 
50  return Status;
51 
52 } /* End MD_TableIsInMask */

56 void MD_UpdateDwellControlInfo (uint16 TableIndex)
57 {
58  /* Initialize local data */
59  uint16 EntryIndex = 0;
60  uint16 NumDwellAddresses = 0;
61  uint16 NumDwellDataBytes = 0;
62  uint32 NumDwellDelayCounts = 0;
63  MD_DwellPacketControl_t *TblPtr = &MD_AppData.MD_DwellTables[TableIndex];
64 
65  /*
66  ** Sum Address Count, Data Size, and Rate from Dwell Entries
67  */
68  while ((EntryIndex < MD_DWELL_TABLE_SIZE) &&
69  (TblPtr->Entry[EntryIndex].Length != 0))
70  {
71  NumDwellAddresses++;
72  NumDwellDataBytes += TblPtr->Entry[EntryIndex].Length;
73  NumDwellDelayCounts += TblPtr->Entry[EntryIndex].Delay;
74  EntryIndex++;
75  }
76 
77  /* Copy totals to dwell control structure. */
78  TblPtr->AddrCount = NumDwellAddresses;
79  TblPtr->DataSize = NumDwellDataBytes;
80  TblPtr->Rate = NumDwellDelayCounts;
81 
82  return;
83 } /* MD_UpdateDwellControlInfo */
