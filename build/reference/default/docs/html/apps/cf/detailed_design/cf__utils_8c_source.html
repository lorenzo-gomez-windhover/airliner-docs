321 CF_QueueEntry_t *CF_FindNodeByTransId(char *TransIdStr)
322 {
323  char *TmpPtr;
324  uint8 DownlinkTrans = 0;
325  uint8 UplinkTrans = 0;
326  char GivenEntityId[CF_MAX_CFG_VALUE_CHARS];
327  int32 GivenTransNum = 0;
328  uint32 i,Chan,Q;
329  CF_QueueEntry_t *NodePtr;
330 
331 
332  /* Verify the given EntityId is valid, this will also tell the code what */
333  /* queues to search (uplink or downlink) */
334  /* To verify the given EntityId, we need to..*/
335  /* separate the SrcEntityId (0.24) from the transaction id (0.24_3) so we */
336  /* can compare the given SrcEntityId string with the FlightEntityId string*/
337  /* in the table. If strings are equal, SrcEntityId is valid and downlink, */
338  /* otherwise compare to PeerEntityId for each channel that is In-use */
339 
340  /* For uplink trans, peer entity id is the source */
341  /* For downlink trans, flight entity id is the source */
342 
343  /* copy the string so that this function can manipulate the string as needed */
344  strncpy(&GivenEntityId[0],TransIdStr,CF_MAX_CFG_VALUE_CHARS);
345 
346 
347  /* Extract source EntityId (ex. 0.24) from TransId (ex. 0.24_13) by */
348  /* replacing the underscore with a null terminator */
349  TmpPtr = &GivenEntityId[0];
350 
351  for(i=0;i<CF_MAX_CFG_VALUE_CHARS;i++)
352  {
353  if(*TmpPtr == '_')
354  {
355  *TmpPtr = '\0';
356 
357  /* Extract the TransNum (ex.13) from the TransId (ex. 0.24_13) */
358  TmpPtr++;
359  GivenTransNum = atoi(TmpPtr);
360 
361  break;
362  }
363 
364  TmpPtr++;
365  }
366 
367 
368  /* Validate the Entity Id */
369  if(CF_ValidateEntityId(&GivenEntityId[0]) != CF_SUCCESS)
370  {
371  return NULL;
372  }
373 
374  /* determine if transaction is uplink or downlink */
375  if(strncmp(&GivenEntityId[0], &CF_AppData.Tbl->FlightEntityId[0],
376  CF_MAX_CFG_VALUE_CHARS) == 0)
377  {
378  /* match found, flight is source, must be downlink trans */
379  DownlinkTrans = 1;
380 
381  }else{
382  /* if not downlink, must be uplink */
383  UplinkTrans = 1;
384 
385  }
386 
387  if(UplinkTrans)
388  {
389  /* check uplink, active queue */
390  NodePtr = CF_FindUpNodeByTransID(CF_UP_ACTIVEQ,GivenEntityId,GivenTransNum);
391  if(NodePtr != NULL) return NodePtr;
392 
393  /* check uplink, history queue */
394  NodePtr = CF_FindUpNodeByTransID(CF_UP_HISTORYQ,GivenEntityId,GivenTransNum);
395  if(NodePtr != NULL) return NodePtr;
396 
397  }else if(DownlinkTrans){
398 
399  for(Chan=0;Chan<CF_MAX_PLAYBACK_CHANNELS;Chan++)
400  {
401  if(CF_AppData.Tbl->OuCh[Chan].EntryInUse == CF_ENTRY_IN_USE)
402  {
403  for(Q=0;Q<CF_QUEUES_PER_CHAN;Q++)
404  {
405  NodePtr = CF_FindPbNodeByTransNum(Chan, Q, GivenTransNum);
406  if(NodePtr != NULL) return NodePtr;
407  }
408  }
409  }
410  }
411 
412  return NULL;
413 
414 }/* CF_FindNodeByTransId */

630 /* first value (before the dot) can be 1,2 or 3 digits. */
631 /* second value (after the dot) can be 1,2 or 3 digits. */
632 /* each value must be zero to 255 */
633 int32 CF_ValidateEntityId(char *EntityIdStrPtr)
634 {
635  char TmpBuf[CF_MAX_TRANSID_CHARS];
636  char *TmpPtr = &TmpBuf[0];
637  uint32 DotLocation;/* zero based */
638  char *PtrToDot;
639  uint32 i;
640  uint32 FirstValue;
641  uint32 SecondValue;
642  uint32 LengthIsGood;
643 
644  /* copy input string into temporary buffer */
645  strncpy(TmpBuf,EntityIdStrPtr,CF_MAX_TRANSID_CHARS);
646 
647  /* string length must be between 3 bytes (ex. 2.5) and 7 bytes (ex. 255.255) */
648 
649  /* check that string is not too short */
650  for(i=0;i<3;i++)
651  {
652  if(*TmpPtr == '\0')
653  return CF_ERROR;
654  TmpPtr++;
655  }/* end for */
656  /* TmpPtr at 4th char */
657 
658  /* Check that the null terminator is in the 4th - 8th position */
659  LengthIsGood = CF_FALSE;
660  for(i=0;i<5;i++)
661  {
662  if(*TmpPtr == '\0')LengthIsGood = CF_TRUE;
663  TmpPtr++;
664 
665  }/* end for */
666 
667  if(LengthIsGood == CF_FALSE)
668  return CF_ERROR;
669 
670  /* length is good, reset pointer */
671  TmpPtr = &TmpBuf[0];
672 
673  /* check for a dot in the string */
674  PtrToDot = strstr(TmpPtr,".");
675  if(PtrToDot==NULL)
676  return CF_ERROR;
677 
678  /* DotLocation is zero based, zero is first character */
679  DotLocation = (uint32)(PtrToDot - TmpPtr);
680 
681  if((DotLocation < 1) || (DotLocation > 3))
682  return CF_ERROR;
683 
684  /* temporarily replace the dot with a null terminator */
685  *PtrToDot = '\0';
686 
687  /* get first value */
688  FirstValue = atoi(TmpPtr);
689 
690  /* put back to original */
691  *PtrToDot = '.';
692 
693  /* Set TmpPtr to first char of second value */
694  TmpPtr = PtrToDot;
695  TmpPtr++;
696 
697  /* get second value */
698  SecondValue = atoi(TmpPtr);
699 
700  /* Values must fit in a byte */
701  if((FirstValue > 255) || (SecondValue > 255))
702  return CF_ERROR;
703 
704  return CF_SUCCESS;
705
