83  /* WHAT IT DOES: Determines (and returns) the layout of the given
84  * raw PDU.

237 static void m__gen_pdu_header (HDR hdr, u_int_2 data_field_length,
238  DATA *raw_header)
239  /* WHAT IT DOES: Given the PDU-header expressed as a C structure,
240  * and given the length of the data-field, this routine returns
241  * a raw PDU-header.
242  */
243  {
244  u_int_1 byte;
245  u_int_1 entity_id_length;
246  int i;
247  u_int_1 lower_byte;
248  u_int_1 upper_byte;
249 
250  /*------------------------------------------------------------*/
251  /* Initialize the raw Header */
252  raw_header->length = 0;
253 
254  /*----------------------*/
255  /* Byte 1 of the header */
256  /*----------------------*/
257  byte = 0;
258  if (hdr.pdu_type == FILE_DATA_PDU)
259  byte = byte | 0x10;
260  if (hdr.direction == TOWARD_SENDER)
261  byte = byte | 0x08;
262  if (hdr.mode == UNACK_MODE)
263  byte = byte | 0x04;
264  m__append_n_bytes (raw_header, &byte, 1);
265 
266  /*----------------------------------------------------*/
267  /* Generate and append the 2-byte "Data field length" */
268  /*----------------------------------------------------*/
269  upper_byte = data_field_length / 256;
270  m__append_n_bytes (raw_header, &upper_byte, 1);
271  lower_byte = data_field_length % 256;
272  m__append_n_bytes (raw_header, &lower_byte, 1);
273 
274  /* <NOT_SUPPORTED> CRC in PDUs */
275 
276  /*-------------------------------------------------------------------*/
277  /* Generate and append the 1-byte Entity-id length & trans-id length */
278  /*-------------------------------------------------------------------*/
279 
280  /* First, determine the appropriate length for entity-IDs */
281  if (hdr.dest_id.length >= hdr.trans.source_id.length)
282  entity_id_length = hdr.dest_id.length;
283  else
284  entity_id_length = hdr.trans.source_id.length;
285 
286  byte = 0;
287  byte = byte | ((entity_id_length - 1) << 4);
288  byte = byte | (HARD_CODED_TRANS_SEQ_NUM_LENGTH - 1);
289  m__append_n_bytes (raw_header, &byte, 1);
290 
291  /*------------------------------------------------*/
292  /* Append the var-length "source entity-id" field */
293  /*------------------------------------------------*/
294 
295  /* Because both ID lengths (source & destination) have to be the same,
296  * it may be necessary to add zeros (they are added on the left)
297  */
298  byte = 0;
299  for (i=hdr.trans.source_id.length; i<hdr.dest_id.length; i++)
300  m__append_n_bytes (raw_header, &byte, 1);
301  m__append_n_bytes (raw_header, &hdr.trans.source_id.value,
302  hdr.trans.source_id.length);
303 
304  /*-----------------------------------------------------*/
305  /* Append the var-length "trans sequence number" field */
306  /*-----------------------------------------------------*/
307 
308  /* The trans-sequence-number field length (for outgoing PDUs) is
309  * hard-coded within this implementation of CFDP.
310  * (See HARD_CODED_TRANS_SEQ_NUM_LENGTH).
311  */
312  m__append_4_byte_integer (raw_header, hdr.trans.number);
313 
314  /*-----------------------------------------------------*/
315  /* Append the var-length "destination entity-id" field */
316  /*-----------------------------------------------------*/
317 
318  /* Because both ID lengths (source & destination) have to be the same,
319  * it may be necessary to add zeros (they are added on the left)
320  */
321  byte = 0;
322  for (i=hdr.dest_id.length; i<hdr.trans.source_id.length; i++)
323  m__append_n_bytes (raw_header, &byte, 1);
324  m__append_n_bytes (raw_header, &hdr.dest_id.value, hdr.dest_id.length);
325  }

343 /*=r=************************************************************************/
344 static TRANSACTION m__transaction (const DATA *pdu)
345  /* WHAT IT DOES: Returns the transaction info from the given raw PDU. */
346  {
347  int i;
348  PDU_LAYOUT layout;
349  TRANSACTION trans;
350  int trans_seq_num_length;
351  /*------------------------------------------------------------*/
352  /* Determine where the Source-ID field is located in this pdu */
353  layout = m__pdu_layout (pdu);
354 
355  /* Copy its contents */
356  trans.source_id.length = layout.length_of_source_id;
357  for (i=0; i<trans.source_id.length; i++)
358  trans.source_id.value[i] = pdu->content[i+layout.start_of_source_id];
359 
360  trans_seq_num_length = layout.length_of_trans_seq_num;
361  if (trans_seq_num_length > HARD_CODED_TRANS_SEQ_NUM_LENGTH)
362  /* Uh-oh, our assumption has proven wrong. We assumed that no
363  * CFDP partner would use a transaction-sequence-number field
364  * longer than 4 bytes.
365  */
366  {
367  e_msg__ ("cfdp_engine: Truncating trans-seq-num "
368  "(%u bytes -> %u bytes).\n",
369  trans_seq_num_length, HARD_CODED_TRANS_SEQ_NUM_LENGTH);
370  /* Do the best we can (use least significant 4 bytes) */
371  while (trans_seq_num_length > HARD_CODED_TRANS_SEQ_NUM_LENGTH)
372  {
373  layout.start_of_trans_seq_num ++;
374  trans_seq_num_length --;
375  }
376  }
377  trans.number = 0;
378  for (i=0; i<trans_seq_num_length; i++)
379  trans.number = (trans.number * 256) +
380  pdu->content[i+layout.start_of_trans_seq_num];
381 
382  return (trans);
383  }

388 /*=r=*************************************************************************/
389 boolean pdu__is_this_pdu_acceptable (PDU *pdu)
390 {
391  int data_field_length;
392  int dir_code;
393  int entity_id_length;
394  int i;
395  int length_of_pdu_header;
396  ID my_id;
397  ID pdu_id;
398  int pdu_type;
399  int start_of_dest_id;
400  int trans_id_length;
401  /*------------------------------------------------------------*/
402 
403  /* NOTE: Since the point of this routine is to screen out unacceptable
404  * pdus before they are passed to the "real" pdu conversion routines,
405  * this routine is self-contained (ie. it doesn't use any of the
406  * utility routines). Also, efficiency is given priority over
407  * "easy to understand".
408  */
409 
410  /*A valid pdu must be at least 'MINIMUM_VALID_PDU_LENGTH' bytes long */
411  if (pdu->length < MINIMUM_VALID_PDU_LENGTH)
412  {
413  e_msg__ ("cfdp_engine: PDU rejected due to length "
414  "(%lu bytes); must be >= %lu.\n",
415  pdu->length, MINIMUM_VALID_PDU_LENGTH);
416  return (0);
417  }
418 
419  /* (Pull out some information from the pdu-header) */
420  data_field_length = (pdu->content[1] * 256) + pdu->content[2];
421  entity_id_length = ((pdu->content[3] & 0x70) >> 4) + 1;
422  trans_id_length = (pdu->content[3] & 0x07) + 1;
423  length_of_pdu_header = 4 + (2 * entity_id_length) + trans_id_length;
424 
425  /* A valid pdu must be as long as the "header + data_field" */
426  if (pdu->length < (length_of_pdu_header + data_field_length))
427  {
428  e_msg__ ("cfdp_engine: PDU rejected due to length "
429  "(%lu bytes); must be >= %lu.\n",
430  pdu->length, length_of_pdu_header + data_field_length);
431  return (0);
432  }
433 
434  /* A valid pdu must have some data following the header */
435  if (data_field_length == 0)
436  {
437  e_msg__ ("cfdp_engine: PDU rejected due to data-field-length (0); "
438  "must be > 0.\n");
439  return (0);
440  }
441 
442  /* A valid File Directive PDU must specify a valid File Directive Code */
443  pdu_type = pdu->content[0] && 0x10;
444  if (pdu_type == 0x00)
445  /* This is a File Directive PDU */
446  {
447  /* First byte after pdu-header is the File Directive Code */
448  dir_code = pdu->content[length_of_pdu_header];
449  /* See Table 5-4 in the CFDP Blue Book for dir-code values */
450  if ((dir_code < 4) || (dir_code > 0x0C))
451  {
452  e_msg__ ("cfdp_engine: PDU rejected due to unrecognized "
453  "Directive Code (%2.2x).\n",
454  dir_code);
455  return (NO);
456  }
457  }
458 
459  /* Our 'PDU_AS_STRUCT' structure can't hold an entity-id whose length is
460  * greater than 'MAX_ID_LENGTH'
461  */
462  if (entity_id_length > MAX_ID_LENGTH)
463  {
464  e_msg__ ("cfdp_engine: PDU rejected due to entity-id length "
465  "(%u); MAX_ID_LENGTH=%u.\n",
466  entity_id_length, MAX_ID_LENGTH);
467  return (NO);
468  }
469 
470  /* Our 'PDU_AS_STRUCT' structure can't hold a trans-seq-num whose length
471  * is greater than 4.
472  */
473  if (trans_id_length > 4)
474  {
475  e_msg__ ("cfdp_engine: PDU rejected due to trans length "
476  "(%u); max supported is 4.\n",
477  trans_id_length);
478  return (NO);
479  }
480 
481  /*--------------------------------------------*/
482  /* The PDU must be addressed to our entity-id */
483  /*--------------------------------------------*/
484 
485  /* Get our entity-id */
486  my_id = mib__get_my_id ();
487 
488  /* Extract the (appropriate) entity-id from the pdu-header */
489  if (pdu->content[0] & 0x08)
490  /* The direction of this pdu is "towards the file sender" --
491  * i.e. we are the source of the transaction.
492  * So extract the source-id from the pdu.
493  */
494  {
495  pdu_id.length = entity_id_length;
496  /* (The source-id always starts at byte 4 of the header) */
497  for (i=0; i<entity_id_length; i++)
498  pdu_id.value[i] = pdu->content[i+4];
499  }
500  else
501  /* The direction of this pdu is "towards the file receiver" --
502  * i.e. we are the destination of the transaction.
503  * So extract the dest-id from the pdu.
504  */
505  {
506  pdu_id.length = entity_id_length;
507  /* (The dest-id is after the source-id and trans-seq-number) */
508  start_of_dest_id = 4 + entity_id_length + trans_id_length;
509  for (i=0; i<entity_id_length; i++)
510  pdu_id.value[i] = pdu->content[i+start_of_dest_id];
511  }
512 
513  /* The 2 entity-ids should match */
514  if (!cfdp_are_these_ids_equal (my_id, pdu_id))
515  {
516  e_msg__ ("cfdp_engine: PDU rejected due to invalid dest id (%s).\n",
517  cfdp_id_as_string (pdu_id));
518  return (NO);
519  }
520 
521  return (YES);
522 }
