79 static char *m__convert_string_to_uppercase (const char *string)
80  /* WHAT IT DOES: Converts the given char string to uppercase letters,
81  * and returns the result via the return status.
82  */
83 #define MAX_LENGTH 256
84  {
85  int i;
86  int length;
87  static char uppercase [MAX_LENGTH+1];
88  /*------------------------------------------------------------*/
89 
90  /* Avoid writing past the allocated memory */
91  length = strlen(string);
92  if (length > MAX_LENGTH)
93  {
94  w_msg__ ("cfdp_engine: MIB string truncated (%u->%u)\n",
95  length, MAX_LENGTH);
96  }
97 
98  /* Make the conversion */
99  memset (uppercase, STRING_TERMINATOR, sizeof(uppercase));
100  for (i=0; i<length; i++)
101  uppercase[i] = (char) toupper ((int) string[i]);
102  return (uppercase);
103  }
104 #undef MAX_LENGTH

142 static RESPONSE m__string_as_response (char *string)
143  /* WHAT IT DOES: Given a char string, it returns a Response.
144  * EXAMPLE: If given "CANCEL" it returns RESPONSE_CANCEL.
145  */
146  {
147  static RESPONSE response;
148  /*------------------------------------------------------------*/
149  if (!strcmp (string, "ABANDON"))
150  response = RESPONSE_ABANDON;
151  else if (!strcmp (string, "CANCEL"))
152  response = RESPONSE_CANCEL;
153  else if (!strcmp (string, "IGNORE"))
154  response = RESPONSE_IGNORE;
155  else if (!strcmp (string, "SUSPEND"))
156  response = RESPONSE_SUSPEND;
157  else
158  {
159  w_msg__ ("cfdp_engine: MIB: unrecognized fault response (%s) "
160  "converted to 'Cancel'\n",
161  string);
162  response = RESPONSE_CANCEL;
163  }
164 
165  return (response);
166  }

381  {
382  int i;
383  ID id;
384  unsigned long int long_int;
385  char param [MAX_MIB_PARAMETER_LENGTH+1];
386  u_int_4 requested_size;
387  char value [MAX_MIB_VALUE_LENGTH+1];
388  /*------------------------------------------------------------*/
389  INITIALIZE_ONCE;
390 
391  /*--------------------------*/
392  /* Validate input arguments */
393  /*--------------------------*/
394 
395  if (strlen(param_in) == 0)
396  {
397  e_msg__ ("cfdp_engine: can't set MIB parameter; "
398  "no parameter given.\n");
399  return (0);
400  }
401  else if (strlen(param_in) > MAX_MIB_PARAMETER_LENGTH)
402  {
403  e_msg__ ("cfdp_engine: can't set MIB parameter; "
404  "parameter-string (%s) is too long.\n",
405  param_in);
406  return (0);
407  }
408  else if (strlen(value_in) == 0)
409  {
410  e_msg__ ("cfdp_engine: can't set MIB parameter; no value given.\n");
411  return (0);
412  }
413  else if (strlen(value_in) > MAX_MIB_PARAMETER_LENGTH)
414  {
415  e_msg__ ("cfdp_engine: can't set MIB parameter; "
416  "value-string (%s) is too long.\n",
417  value_in);
418  return (0);
419  }
420 
421  COPY (param, m__convert_string_to_uppercase (param_in));
422  COPY (value, m__convert_string_to_uppercase (value_in));
423 
424  /*-----------------------------------------*/
425  /* Compare input with each Local parameter */
426  /*-----------------------------------------*/
427 
428  if (!strcmp (param, MIB_MY_ID))
429  {
430  if (cfdp_id_from_string (value, &id))
431  {
432  m_mib_local.my_id = id;
433  i_msg__ ("cfdp_engine: entity-id set to '%s'.\n",
434  cfdp_id_as_string (m_mib_local.my_id));
435  }
436  else
437  {
438  e_msg__ ("cfdp_engine: can't set entity-id to illegal value "
439  "(%s).\n",
440  value);
441  return (0);
442  }
443  }
444 
445  else if (!strcmp (param, MIB_ISSUE_EOF_RECV))
446  {
447  if (!strcmp (value, "YES"))
448  {
449  m_mib_local.issue_eof_recv = YES;
450  i_msg__ ("cfdp_engine: 'issue_eof_recv' set to 'yes'.\n");
451  }
452  else if (!strcmp (value, "NO"))
453  {
454  m_mib_local.issue_eof_recv = NO;
455  i_msg__ ("cfdp_engine: 'issue_eof_recv' set to 'no'.\n");
456  }
457  else
458  {
459  e_msg__ ("cfdp_engine: can't set 'issue_eof_recv'; "
460  "value (%s) must be 'yes' or 'no'.\n",
461  value_in);
462  return (0);
463  }
464  }
465
