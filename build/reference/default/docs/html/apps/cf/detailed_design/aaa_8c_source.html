54 /* ***** FILE CHECKSUM *****
55  * Background: The file checksum is not your typical checksum. The
56  * typical checksum is one byte long and is an exclusive-or of all the
57  * data bytes. The file checksum is 4 bytes long. It is calculated by
58  * adding together all the bytes in a file using a 4-byte-wide addition.
59  * Strategy used here: For outgoing files, the file checksum calculation
60  * is performed on one chunk of file-data at a time AS PART OF THE
61  * INITIAL TRANSMISSION OF THE FILE. This strategy uses much less
62  * system resources than performing the calculation in a SEPARATE READ
63  * of the whole file, but the algorithm is more complicated...
64  * Complication: The code here gets complicated because there is no
65  * guarantee that each chunk of file-data will be a multiple of 4 bytes
66  * in length. So, here are 2 examples. The first illustrates the
67  * conceptual calculation. The second shows how it is actually
68  * performed in this module. Both examples use a file that is 12 bytes
69  * long containing a "ramp" pattern (byte 1 is "01", byte 2 is "02", etc).
70  * Example 1 -- Every chunk of filedata is a multiple of 4 bytes (this allows
71  * the checksum calculation to be simple):
72  * 00 00 00 00 // Initial checksum value
73  * 01 02 03 04 // First 4 bytes of the file
74  * + -----------
75  * 01 02 03 04 // Interim checksum result
76  * 05 06 07 08 // Second 4 bytes of the file
77  * + ----------- //
78  * 06 08 0A 0C // Interim checksum result
79  * 09 0A 0B 0C // Final 4 bytes of the file
80  * + -----------
81  * 0F 12 15 18 // Final checksum result
82  * Example 2 -- Chunks of filedata are not a multiple of 4 bytes (in this
83  * example, every chunk is 6 bytes long):
84  * 00 00 00 00 // Initial checksum value
85  * 01 02 03 04 // first 4 bytes of file-data
86  * 05 06 00 00 // 5th and 6th byte of file plus zero-fill to *right*
87  * + -----------
88  * 06 08 03 04 // interim checksum result
89  * 00 00 07 08 // 7th and 8th byte of file plus zero-fill to *left*
90  * 09 0A 0B 0C // Final 4 bytes of the file
91  * + -----------
92  * 0F 12 15 18 // Final checksum result.
93  * Summary: This module does as much of the checksum calculation as it
94  * can in 4-byte operations. It may handle a few bytes at the beginning
95  * and/or end of a chunk of file-data as a special case as shown in the
96  * above examples.
97  */

248 /*=r=************************************************************************/
249 u_int_4 aaa__calculate_file_checksum (char *file_name)
250  {
251 #define BUFFER_SIZE 1024
252  unsigned char buffer [BUFFER_SIZE];
253  u_int_4 checksum = 0;
254  FILE *fp;
255  int i;
256  u_int_4 *iptr;
257  u_int_4 length;
258  /*------------------------------------------------------------*/
259 
260  /*---------------------------------------*/
261  /* Open the specified 'file' for reading */
262  /*---------------------------------------*/
263 
264  fp = fopen_callback (file_name, "rb");
265  if (fp == NULL)
266  {
267  e_msg__ ("cfdp_engine: Unable to open file (%s) to calculate "
268  "checksum.\n",
269  file_name);
270  ASSERT__ (fp != NULL);
271  /* Return a bogus answer */
272  return (0xdeadbeef);
273  }
274 
275  /*---------------------------------------------------------------------*/
276  /* Perform the checksum calculation on one chunk of the file at a time */
277  /*---------------------------------------------------------------------*/
278 
279  for (;;)
280  {
281  /* This 'memset' is critical. It allows us to use the simple
282  * "4 bytes at a time" method even if the length of the file
283  * is not a multiple of 4 bytes.
284  * (If the very last "4 byte" piece of the file is actually
285  * less than 4 bytes, the zeros will ensure that the checksum
286  * result is still correct).
287  */
288  memset (buffer, 0, sizeof(buffer));
289 
290  /* Read the next chunk of the file */
291  length = fread_callback (buffer, 1, BUFFER_SIZE, fp);
292  if (length == 0)
293  /* We've read the whole file, so we're done */
294  break;
295 
296  /* We've read a chunk of the file -- all chunks (except the last)
297  * will be 'BUFFER_SIZE' bytes in length; the last chunk will
298  * probably be less than 'BUFFER_SIZE' bytes.
299  */
300  /* Set up a pointer to walk through the buffer 4 bytes at a time */
301  iptr = (u_int_4 *) buffer;
302  for (i=0; i<length; i+=4)
303  /* Add each 4 byte piece to the checksum. */
304  {
305  checksum += ntohl (*iptr);
306  iptr ++;
307  }
308  }
309 
310  /* Close the file and return the checksum */
311  fclose_callback (fp);
312  return (checksum);
313  }

800 /*=r=************************************************************************/
801 void aaa__send_one_file_data_pdu (MACHINE *m, FD *fd)
802  {
803  u_int_4 buffer_length;
804  register u_int_4 checksum_in_register;
805  u_int_1 four_byte_buffer [4];
806  u_int_4 *four_byte_pointer;
807  int how_many_bytes_till_boundary;
808  int how_many_bytes_of_zero_fill;
809  int i;
810  u_int_4 *i4ptr;
811  TRANS_STATUS *mp = &(m->publik); /* useful shorthand */
812  int offset_modulo_four;
813  static PDU pdu;
814  NODE *ptr;
815  size_t result;
816  /*------------------------------------------------------------*/
817 
818  /* Determine how much file-data to send: send as much of the first
819  * data gap as will fit in one PDU (the whole data gap, if possible).
820  */
821  ptr = m->nak.head; /* Point to the first data gap on the list */
822  if (ptr == NULL)
823  /* Avoid crashing if nak-list is empty (shouldn't ever happen) */
824  {
825  e_msg__ ("cfdp_engine: bug. See 'aaa__send_one_file_data_pdu'.\n");
826  /* Avoid crashing... */
827  return;
828  }
829  fd->offset = ptr->begin;
830  if (ptr->end - ptr->begin <= mib__outgoing_file_chunk_size (NULL))
831  /* It all fits; take it all */
832  fd->buffer_length = ptr->end - ptr->begin;
833  else
834  /* Won't all fit; use as much as allowed */
835  fd->buffer_length = mib__outgoing_file_chunk_size (NULL);
836 
837  /* If this is a retransmission, seek to the required file offset
838  * (during the first transmission, the end of the previous file data
839  * is always the start of the current file data).
840  */
841  if (m->has_eof_been_sent)
842  {
843  if (cfdp_is_message_class_enabled (CFDP_MSG_PDU_RETRANSMITTED_FD))
844  d_msg__ ("<aaa> Resending FD: %u bytes at offset %lu.\n",
845  fd->buffer_length, fd->offset);
846  if (fseek_callback (m->fp, fd->offset, SEEK_SET))
847  e_msg__ ("cfdp_engine: 'fseek' error (%s/%u).\n",
848  __func__, __LINE__);
849  }
850 
851  /* ***OPTIMIZATION***
852  * This optimization is not yet implemented (it would break the
853  * current checksum-calculation algorithm). Consider implementing
854  * it in the future...
855  * Instead of reading the file-data into a buffer and then
856  * building a PDU from that buffer, build the skeleton of the
857  * PDU and then read the file-data directly into the PDU.
858  * The straightforward approach uses 2 memcpy, the optimization 1.
859  * Careful - the checksum-calculation algorithm requires up to 3
860  * bytes that follow the file-data to be zeroed out.
861  */
862 
863  /* Read the data from the file and store it in the filedata-buffer.
864  * Note: It's important to zero out the filedata buffer before reading
865  * (so that the algorithm used for checksum calculation can
866  * remain simple -- it assumes that zeros follow the real file-data).
867  */
868 
869  memset (fd->buffer, 0, sizeof(fd->buffer));
870  result = fread_callback (fd->buffer, (size_t) fd->buffer_length,
871  1, m->fp);
872  if (result != 1)
873  e_msg__ ("cfdp_engine: 'fread' error (%s/%u).\n", __func__, __LINE__);
874 
875  /* Generate a FD Pdu */
876  pdu__make_fd_pdu (m->hdr, fd, &pdu);
877 
878  /*-----------------------------------------------------------------*/
879  /* Please refer to the detailed note near the top of this file */
880  /* for an explanation of the file checksum and the algorithm used. */
881  /*-----------------------------------------------------------------*/
882 
883  if (!m->has_eof_been_sent)
884  /* We are still blasting the file (initial transmission) */
885  {
886 
887  /*----------------------------------------------------------------*/
888  /* Before entering the normal "4 bytes at a time" checksum */
889  /* calculation loop, get back onto a 4-byte boundary. */
890  /* (see example 2 in the detailed note near the top of this file) */
891  /*----------------------------------------------------------------*/
892 
893  /* Determine how many bytes of file-data need to be processed to
894  * reach a 4-byte boundary.
895  * Example: If our current file-data starts at offset 5, then the
896  * offset_modulo_four is 1, and 3 more bytes are needed to reach
897  * the next 4-byte boundary (at offset = 8).
898  */
899  offset_modulo_four = fd->offset & 0x03;
900  if (offset_modulo_four == 0)
901  how_many_bytes_till_boundary = 0;
902  else
903  how_many_bytes_till_boundary = 4 - offset_modulo_four;
904 
905  /* Perform the (custom) processing required (if any) to get
906  * back to a 4-byte boundary.
907  */
908  if (how_many_bytes_till_boundary > 0)
909  {
910  /* Set up a 4 byte buffer with zero-fill on the left... */
911  four_byte_buffer[0] = 0;
912  four_byte_buffer[1] = 0;
913  four_byte_buffer[2] = 0;
914  four_byte_buffer[3] = 0;
915  /* ... and the appropriate number of file-data bytes copied in a
916  * "right-justified" position.
917  */
918  how_many_bytes_of_zero_fill = 4 - how_many_bytes_till_boundary;
919  memcpy (&four_byte_buffer[how_many_bytes_of_zero_fill],
920  fd->buffer,
921  how_many_bytes_till_boundary);
922  /* Add the 4-byte buffer to the current checksum */
923  four_byte_pointer = (u_int_4 *) four_byte_buffer;
924  mp->file_checksum_as_calculated += ntohl (*four_byte_pointer);
925  }
926 
927  /* Adjust the file-data buffer length and data-pointer */
928  buffer_length = fd->buffer_length - how_many_bytes_till_boundary;
929  i4ptr = (u_int_4 *) &fd->buffer[how_many_bytes_till_boundary];
930 
931  /*------------------------------------------------------------*/
932  /* This is the normal "4 bytes at a time" checksum algorithm. */
933  /*------------------------------------------------------------*/
934 
935  /* Add one 4-byte wide piece at a time to the running checksum.
936  * Note: This algorithm assumes that there are at least 3 bytes
937  * of zeros immediately beyond the file-data. That's why the
938  * file-data buffer is 3 bytes longer than "necessary", and why
939  * the file-data buffer is zeroed out before each read.
940  * ***OPTIMIZATION***
941  * use a temporary register variable inside the 'for' loop.
942  */
943  checksum_in_register = mp->file_checksum_as_calculated;
944  for (i=0; i<buffer_length; i+=4)
945  {
946  /* Use 'ntohl' to avoid big-endian versus little-endian issues */
947  checksum_in_register += ntohl (*i4ptr);
948  /* Advance 4 more bytes into the buffer of file data */
949  i4ptr ++;
950  }
951  mp->file_checksum_as_calculated = checksum_in_register;
952  }
953 
954  /* If configured to do so, output a detailed message */
955  if (cfdp_is_message_class_enabled (CFDP_MSG_DEBUG_PDU))
956  pdu__display_disassembled_pdu (pdu);
957 
958  /* Send the Pdu */
959  pdu_output__send (m->hdr.trans, m->hdr.dest_id, &pdu);
960  mp->fd_length = fd->buffer_length;
961 
962  /* If configured to do so, issue an Indication */
963  if (mib__issue_file_segment_sent ())
964  indication__ (IND_FILE_SEGMENT_SENT, &(m->publik));
965 
966  /* During the initial send of the file, the next chunk of file data
967  * to be sent will always immediately follow the current chunk
968  */
969  if (!m->has_eof_been_sent)
970  mp->fd_offset = fd->offset + fd->buffer_length;
971  }

975 /*=r=************************************************************************/
976 void aaa__shutdown (MACHINE *m)
977  {
978  GAP *current;
979  GAP *next;
980  TRANS_STATUS *mp = &(m->publik); /* useful shorthand */
981  /*------------------------------------------------------------*/

984  if (m->is_there_an_open_file)
985  {
986  fclose_callback (m->fp);
987  m->is_there_an_open_file = NO;
988  }
989 
990  /* If a temp file exists at this point, then we are a Receiver in a
991  * transaction that wasn't entirely successful. Delete the temp-file unless
992  * the engine user has told us not to (i.e. they may want to look at what
993  * they got even if it is is not complete).
994  */
995  if (m->is_there_a_temp_file)
996  {
997  if (mib__save_incomplete_files (NULL))
998  /* Engine user wants us to save incomplete files, so don't delete it */
999  d_msg__ ("cfdp_engine: saving incomplete temp-file (%s).\n",
1000  m->publik.temp_file_name);
1001  else
1002  /* Ok to delete the temp-file */
1003  remove_callback (mp->temp_file_name);
1004  }
1005 
1006  mp->state = FINISHED;
1007  m->is_outgoing_md_buffered = NO;
1008  m->is_outgoing_eof_buffered = NO;
1009  m->is_outgoing_ack_buffered = NO;
1010  m->is_outgoing_nak_buffered = NO;
1011  m->is_outgoing_fin_buffered = NO;
1012 
1013  /* Free all memory in the Nak-list */
1014  for (current=m->nak.head; current!=NULL; current=next)
1015  {
1016  next = current->next;
1017  nak_mem__free (&(m->nak), current);
1018  }
1019 
1020  /* Stop all timers */
1021  timer__cancel (&m->ack_timer);
1022  timer__cancel (&m->nak_timer);
1023  timer__cancel (&m->inactivity_timer);
1024 
1025  /* Finally, set a flag indicating that this machine has finished. */
1026  m->has_this_state_machine_finished = YES;
1027  }
