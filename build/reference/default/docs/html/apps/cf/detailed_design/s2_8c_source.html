83 static boolean m__fault_handler (MACHINE *m, CONDITION_CODE fault)
84 /* WHAT IT DOES: Responds to the given fault in accordance with the
85  * engine user's setting of "response to fault" (in the MIB).
86  * RETURN STATUS: If '1', the response to the given fault was 'ignore',
87  * and the calling state table routine may continue with whatever it
88  * was doing. If '0', the response was suspend, cancel, or abandon
89  * (and the calling state table routine should return immediately).
90  */
91 {
92  boolean ok_for_caller_to_continue;
93  /*------------------------------------------------------------*/
94 
95  /* Save the fault-type */
96  m->publik.condition_code = fault;
97  /* 'Indication' callback allows engine user to respond to events */
98  indication__ (IND_FAULT, &(m->publik));
99 
100  /* Unless the user has configured to 'ignore' this fault, the calling
101  * routine should not continue with what it was doing.
102  */
103  ok_for_caller_to_continue = NO;
104 
105  if (mib__response (fault) == RESPONSE_CANCEL)
106  {
107  m->have_we_initiated_a_cancel = YES;
108  aaa__cancel_locally (m);
109  m__ask_partner_to_cancel (m);
110  m__move_to_state_s2 (m);
111  }
112 
113  else if (mib__response (fault) == RESPONSE_SUSPEND)
114  aaa__notice_of_suspension (m);
115 
116  else if (mib__response (fault) == RESPONSE_IGNORE)
117  ok_for_caller_to_continue = YES;
118 
119  else if (mib__response (fault) == RESPONSE_ABANDON)
120  aaa__abandon_this_transaction (m);
121 
122  else
123  e_msg__ ("cfdp_engine: problem with the Fault mechanism in <S2>.\n");
124 
125  return (ok_for_caller_to_continue);
126 }

131 static void m__release_an_ack_pdu (MACHINE *m)
132  /* WHAT IT DOES: Outputs an Ack PDU,
133  * and performs the actions required after the PDU is released.
134  */
135 {
136  /*------------------------------------------------------------*/
137 
138  /*----------------*/
139  /* Output the PDU */
140  /*----------------*/
141 
142  aaa__send_ack (m->hdr, m->ack);
143  m->is_outgoing_ack_buffered = NO;
144 
145  /*----------------------------------------------------*/
146  /* Perform actions required after the PDU is released */
147  /*----------------------------------------------------*/
148 
149  /* We have just sent an Ack-Finished PDU to our partner. Normally, this
150  * is the last PDU of the transaction (and we can shut down).
151  * However, if we have initiated a Cancel, we can't shut down until our
152  * partner Acks our cancel.
153  */
154 
155  if (m->have_we_initiated_a_cancel)
156  /* Keep running until our Cancel is ack'd. */
157  ;
158  else
159  /* We're done, wrap up and shut down. */
160  aaa__transaction_has_finished (m);
161 }

166 static void m__release_a_metadata_pdu (MACHINE *m)
167  /* WHAT IT DOES: Outputs a Metadata PDU,
168  * and performs the actions required after the PDU is released.
169  */
170 {
171  TRANS_STATUS *mp = &(m->publik); /* useful shorthand */
172  /*------------------------------------------------------------*/
173 
174  /*----------------*/
175  /* Output the PDU */
176  /*----------------*/
177 
178  aaa__send_metadata (m->hdr, m->publik.md);
179  m->is_outgoing_md_buffered = NO;
180 
181  /*----------------------------------------------------*/
182  /* Perform actions required after the PDU is released */
183  /*----------------------------------------------------*/
184 
185  /* Update our bookkeeping (of what data still needs to be sent) */
186  nak__metadata_sent_or_received (&(m->nak));
187 
188  /* 'Indication' callback allows engine user to respond to events */
189  indication__ (IND_METADATA_SENT, &(m->publik));
190 
191  if (mp->attempts == 0)
192  /* First transmission of Metadata */
193  {
194 
195  if (mp->md.file_transfer && !m->is_external_xfer_open)
196  /* This is the typical case -- we have been asked to send a file,
197  * and the engine user is letting us build all the PDUs.
198  */
199  {
200  if (!aaa__open_source_file (m))
201  {
202  e_msg__ ("cfdp_engine: unable to open source file '%s'.\n",
203  m->publik.md.source_file_name);
204  if (!m__fault_handler (m, FILESTORE_REJECTION))
205  return;
206  }
207  if (!aaa__is_file_structure_valid (m))
208  {
209  if (!m__fault_handler (m, INVALID_FILE_STRUCTURE))
210  return;
211  }
212 
213  /* Queue the entire file for metered released */
214  nak__set_end_of_scope (&(m->nak), mp->md.file_size);
215  }
216 
217  else if (mp->md.file_transfer && m->is_external_xfer_open)
218  /* Special case: Although a file is to be transferred, we are
219  * not supposed to send Filedata PDUs (unless retransmissions
220  * are necessary). The engine user will send the initial set
221  * of Filedata PDUs.
222  * So, just chill until the engine user finishes sending the file.
223  */
224  ;
225 
226  else
227  /* No file transfer, so initiate transfer of EOF */
228  {
229  m->eof = aaa__build_eof (mp->condition_code, 0, 0);
230  m->is_outgoing_eof_buffered = YES;
231  }
232 
233  }
234 }

239 static void m__release_an_eof_pdu (MACHINE *m)
240  /* WHAT IT DOES: Outputs an EOF PDU,
241  * and performs the actions required after the PDU is released.
242  */
243 {
244  TRANS_STATUS *mp = &(m->publik); /* useful shorthand */
245  /*------------------------------------------------------------*/
246 
247  /*----------------*/
248  /* Output the PDU */
249  /*----------------*/
250 
251  aaa__send_eof (m->hdr, m->eof);
252  m->is_outgoing_eof_buffered = NO;
253 
254  /*----------------------------------------------------*/
255  /* Perform actions required after the PDU is released */
256  /*----------------------------------------------------*/
257 
258  /* Bookkeeping */
259  mp->attempts ++;
260  if (m->eof.condition_code == NO_ERROR)
261  mp->phase = 2;
262  else
263  mp->phase = 4;
264 
265  /* Give our partner ack-timeout seconds to ack this pdu */
266  m->ack_timer = timer__start (mib__ack_timeout(NULL));
267 
268  /* Issue the EOF-Sent Indication once */
269  if (!m->has_eof_been_sent)
270  {
271  m->has_eof_been_sent = YES;
272  /* If configured to do so, let the engine user know EOF was sent */
273  if (mib__issue_eof_sent())
274  indication__ (IND_EOF_SENT, &(m->publik));
275  /* At this point, our partner is supposed to begin sending PDUs back,
276  * so start the Inactivity-timer.
277  */
278  m->inactivity_timer = timer__start (mib__inactivity_timeout(NULL));
279  }

285 static void m__release_a_file_data_pdu (MACHINE *m, FD *fd)
286  /* WHAT IT DOES: Outputs a Filedata PDU */
287 {
288  TRANS_STATUS *mp = &(m->publik); /* useful shorthand */
289  /*------------------------------------------------------------*/
290 
291  /*----------------*/
292  /* Output the PDU */
293  /*----------------*/
294  aaa__send_one_file_data_pdu (m, fd);
295 
296  /*----------------------------------------------------*/
297  /* Perform actions required after the PDU is released */
298  /*----------------------------------------------------*/
299 
300  /* Remove this data range from our list of data gaps */
301  nak__data_sent (&(m->nak), fd->offset, fd->offset+fd->buffer_length);
302 
303  if (nak__how_many_filedata_gaps (&(m->nak)) == 0)
304  /* All data has been (re)transmitted */
305  {
306  /* Note: the file is closed each time all data-output
307  * completes, and re-opened each time a Nak is received.
308  */
309  fclose_callback (m->fp);
310  m->is_there_an_open_file = NO;
311 
312  /* If we haven't already sent an EOF PDU, queue one up now */
313  if (!m->has_eof_been_sent)
314  {
315  m->eof = aaa__build_eof (mp->condition_code,
316  mp->file_checksum_as_calculated,
317  mp->md.file_size);
318  m->is_outgoing_eof_buffered = YES;
319  }
320  }

326 static void m__ack_fin_from_previous_machine (MACHINE *m,
327  PDU_AS_STRUCT *pdu_ptr)
328  /* Special case:
329  * It's possible that a Class 2 Sender's final outgoing PDU
330  * (an Ack-Finished PDU) is not delivered to the Receiver. In that
331  * case, the Sender will shut down, but the Receiver will not. The
332  * Receiver will resend the Finished PDU. That will cause
333  * another Class 2 Sender state machine to be created.
334  * This routine (re)sends the Ack-Finished PDU so the Receiver
335  * can complete.
336  */
337 {
338  TRANS_STATUS *mp = &(m->publik); /* useful shorthand */
339  /*------------------------------------------------------------*/
340 
341  /* Bookkeeping */
342  mp->is_this_trans_solely_for_ack_fin = YES;
343  mp->phase = 4;
344 
345  /* Save this Finished PDU */
346  m->fin = pdu_ptr->data_field.fin;
347 
348  mp->condition_code = m->fin.condition_code;
349  if (mp->condition_code != NO_ERROR)
350  /* Our partner is cancelling the transaction */
351  aaa__cancel_locally (m);
352 
353  /* Reuse the incoming PDU-header (with 'direction' field reversed )*/
354  m->hdr = pdu_ptr->hdr;
355  m->hdr.direction = TOWARD_RECEIVER;
356  mp->trans = m->hdr.trans;
357  mp->partner_id = m->hdr.dest_id;
358 
359  /* 'Indication' callback allows engine user to respond to events */
360  indication__ (IND_TRANSACTION, &(m->publik));
361  indication__ (IND_MACHINE_ALLOCATED, &(m->publik));
362 
363  /* Build and queue an Ack of the Finished PDU */
364  m->ack = aaa__build_ack_fin (m);
365  m->is_outgoing_ack_buffered = YES;
366 
367  /* (After the Ack-Finished PDU is released, we will shut down) */
368 }

377 /*=r=************************************************************************/
378 void s2__state_table (MACHINE *m, int event,
379  PDU_AS_STRUCT *pdu_ptr, REQUEST *req_ptr)
380  /* NOTE: Conceptually, this state table has 2 states:
381  * S1 = Nominal
382  * S2 = Transaction Cancelled
383  * So, the state is always S1 unless the transaction is cancelled.
384  * Transactions can be cancelled in 3 ways:
385  * 1) the local User can issue a Cancel Request
386  * 2) our CFDP partner can send a Finished-Cancel PDU.
387  * 3) A fault may occur, and the response to that fault may be
388  * to cancel.
389  */
390 {
391  static FD fd;
392  TRANS_STATUS *mp = &(m->publik); /* useful shorthand */
393  /*------------------------------------------------------------*/
394 
395  /* Possibly display some debug info */
396  if (cfdp_is_message_class_enabled (CFDP_MSG_STATE_ALL))
397  aaa__display_state_and_event ("S2", m->publik.state, event);
398 
399 
400  /*---------------------------------------------------------------------*/
401  /* Here is the response to each event. Generally, the response is the */
402  /* same for both states. If not, the logic within a single event will */
403  /* be broken into two parts. */
404  /*---------------------------------------------------------------------*/
405 
406 
407  if (event == THROTTLE_OUTGOING_FILE_DIR_PDU)
408  /* Release, at most, one outgoing File Directive PDU */
409  {
410  if (mp->frozen)
411  /* We can't release any PDUs while frozen */
412  ;
413 
414  else if (mp->suspended)
415  /* The only PDU that can released while suspended is an Ack */
416  {
417  if (m->is_outgoing_ack_buffered)
418  /* An Ack PDU is ready. Do we have the "green light"? */
419  if (pdu_output__ready (FILE_DIR_PDU, mp->trans, m->hdr.dest_id))
420  /* Light is green. Send it */
421  m__release_an_ack_pdu (m);
422  }
423 
424  else if (m->is_outgoing_ack_buffered)
425  /* We're not frozen or suspended, so we can release any PDU.
426  * An Ack is ready. Do we have the "green light"?
427  */
428  {
429  if (pdu_output__ready (FILE_DIR_PDU, mp->trans, m->hdr.dest_id))
430  /* Light is green. Send it */
431  m__release_an_ack_pdu (m);
432  }
433 
434  else if (m->is_outgoing_md_buffered)
435  /* We're not frozen or suspended, so we can release any PDU.
436  * A Metadata is ready. Do we have the "green light"?
437  */
438  {
439  if (pdu_output__ready (FILE_DIR_PDU, mp->trans, m->hdr.dest_id))
440  /* Light is green. Send it. */
441  m__release_a_metadata_pdu (m);
442  }
443 
444  else if (m->is_outgoing_eof_buffered)
445  /* We're not frozen or suspended, so we can release any PDU.
446  * An EOF is ready. Do we have the "green light"?
447  */
448  {
449  if (pdu_output__ready (FILE_DIR_PDU, mp->trans, m->hdr.dest_id))
450  /* Light is green. Send it. */
451  m__release_an_eof_pdu (m);
452  }
453 
454  }
455 
456 
457 
458  else if (event == THROTTLE_OUTGOING_FILE_DATA_PDU)
459  /* Release, at most, one outgoing Filedata PDU. */
460  {
461  if (mp->frozen || mp->suspended)
462  /* We can't release Filedata PDUs while frozen or suspended */
463  ;
464 
465  else if (nak__how_many_filedata_gaps (&(m->nak)) > 0)
466  /* There is File Data ready. Do we have the "green light?" */
467  {
468  /* Careful: If we have sent an EOF-Cancel and are awaiting
469  * an Ack-EOF-Cancel, don't send any more Filedata!
470  */
471  if (!mp->cancelled)
472  {
473  if (pdu_output__ready (FILE_DATA_PDU, mp->trans, m->hdr.dest_id))
474  /* Light is green. Send it. */
475  m__release_a_file_data_pdu (m, &fd);
476  }
477  }
478  }
479 
480 
481 
482  else if (event == RECEIVED_PUT_REQUEST)
483  /* A Put Request was received NOMINAL PATH STARTS HERE... */
484  {
485  m->has_a_put_request_been_received = YES;
486 
487  /* Generate a Metadata pdu */
488  aaa__build_metadata_from_put_req (*req_ptr, &m->hdr, &mp->md);
489 
490  /* Make some info available to the engine user */
491  mp->trans = m->hdr.trans;
492  mp->partner_id = req_ptr->info.put.dest_id;
493 
494  /* 'Indication' callback allows engine user to respond to events */
495  indication__ (IND_TRANSACTION, &(m->publik));
496  indication__ (IND_MACHINE_ALLOCATED, &(m->publik));
497 
498  /* Queue the Metadata PDU for metered release */
499  m->is_outgoing_md_buffered = YES;
500  }
501 
502 
503 
504  else if (event == RECEIVED_SUSPEND_REQUEST)
505  /* A Suspend Request was received; suspend timers */
506  aaa__notice_of_suspension (m);
507 
508 
509 
510  else if (event == RECEIVED_RESUME_REQUEST)
511  /* A Resume Request was received; possibly resume timers */
512  {
513  mp->suspended = NO;
514  /* 'Indication' callback allows engine user to respond to events */
515  indication__ (IND_RESUMED, &(m->publik));
516  /* Timers remain paused if either 'suspended' or 'frozen' */
517  if (!mp->frozen)
518  aaa__resume_timers (m);
519  }
520 
521 
522 
523  else if (event == RECEIVED_CANCEL_REQUEST)
524  /* Received a Cancel Request (from our User, not our CFDP partner) */
525  {
526  /* If the transaction isn't already being cancelled, begin doing so */
527  if (mp->state == S1)
528  {
529  mp->condition_code = CANCEL_REQUEST_RECEIVED;
530  m->have_we_initiated_a_cancel = YES;
531  aaa__cancel_locally (m);
532  m__ask_partner_to_cancel (m);
533  m__move_to_state_s2 (m);
534  }
535  }
536 
537 
538 
539  else if (event == RECEIVED_ABANDON_REQUEST)
540  /* Abandon this transaction */
541  aaa__abandon_this_transaction (m);
542 
543 
544 
545  else if (event == RECEIVED_REPORT_REQUEST)
546  /* A Report Request was received; issue a Report */
547  indication__ (IND_REPORT, &(m->publik));
548 
549 
550 
551  else if (event == RECEIVED_FREEZE_REQUEST)
552  /* A "Freeze" was received, so suspend all timers */
553  {
554  mp->frozen = YES;
555  aaa__suspend_timers (m);
556  }
557 
558 
559 
560  else if (event == RECEIVED_THAW_REQUEST)
561  /* A "Thaw" was received; possibly resume all timers */
562  {
563  mp->frozen = NO;
564  /* Timers remain paused if either 'suspended' or 'frozen' */
565  if (!mp->suspended)
566  aaa__resume_timers (m);
567  }
568 
569 
570 
571  else if (event == RECEIVED_ACK_EOF_NO_ERROR_PDU)
572  /* An Ack-EOF-No-error PDU was received. Nominally, this lets us know
573  * that the Receiver got our EOF, and they will Nak any missing data.
574  */
575  {
576  if (mp->state == S1)
577  {
578  if (mp->attempts == 0)
579  /* Protocol violation */
580  w_msg__ ("cfdp_engine: protocol violation: "
581  "ack of non-existent EOF-no-error.\n");
582  else
583  {
584  /* We are now in Phase 3 (the handoff is complete) */
585  mp->phase = 3;
586  timer__cancel (&m->ack_timer);
587  /* If an EOF is queued, don't send it */
588  m->is_outgoing_eof_buffered = NO;
589  }
590  }
591 
592  else if (mp->state == S2)
593  /* The transaction is being cancelled, so ignore this input */
594  ;
595  }
596 
597 
598 
599  else if (event == RECEIVED_ACK_EOF_CANCEL_PDU)
600  /* An Ack-EOF-Cancel PDU was received. When we initiate a Cancel,
601  * this is the expected response from our partner, and we can shut down.
602  */
603  {
604  if (mp->state == S1)
605  /* Protocol violation */
606  w_msg__ ("cfdp_engine: protocol violation: "
607  "ack of non-existent EOF-cancel.\n");
608 
609  else if (mp->state == S2)
610  /* This Ack is in response to our EOF-Cancel; so shut down */
611  {
612  m->ack = pdu_ptr->data_field.ack;
613  aaa__transaction_has_finished (m);
614  }
615  }
616 
617 
618 
619  else if (event == RECEIVED_NAK_PDU)
620  /* A Nak PDU was received. Our partner has told us what data is missing.
621  * Nominally, we queue up the missing data for retransmission.
622  * If the transaction has been cancelled, we don't bother.
623  */
624  {
625  mp->how_many_naks ++;
626 
627  if (mp->state == S1)
628  {
629  /* Note: The file is closed each time all data-output completes,
630  * and re-opened each time a Nak is received.
631  */
632  if (mp->md.file_transfer && !m->is_there_an_open_file)
633  {
634  if (!aaa__open_source_file (m))
635  {
636  e_msg__ ("cfdp_engine: unable to open source file '%s'.\n",
637  m->publik.md.source_file_name);
638  if (!m__fault_handler (m, FILESTORE_REJECTION))
639  return;
640  }
641  }
642 
643  /* Initiate the requested Filedata retransmission by giving a
644  * copy of this Nak to our bookkeeper (the 'nak' module).
645  * (the requested retransmissions will be metered out one PDU at
646  * a time by the 'throttle_outgoing_pdus' event)
647  */
648  nak__copy_nak (&m->nak, &pdu_ptr->data_field.nak);
649 
650  /* Queue Metadata retransmission (if requested in the Nak) */
651  if (m->nak.is_metadata_missing)
652  m->is_outgoing_md_buffered = YES;
653  }
654  }
655 
656 
657 
658  else if (event == RECEIVED_FINISHED_NO_ERROR_PDU)
659  /* Received a Finished (no error) PDU. Nominally, this tells us that
660  * our partner has received all data; we Ack their Finished and shut
661  * down.
662  */
663  {
664 
665  /* If the final Ack-Finished that we sent from a previous (now dead)
666  * state machine was dropped, our partner would re-send the
667  * Finished PDU, and we would have the special case that is
668  * handled here.
669  */
670  if (!m->has_a_put_request_been_received)
671  {
672  m__ack_fin_from_previous_machine (m, pdu_ptr);
673  return;
674  }
675 
676  /* Check for protocol violation; if so, don't respond */
677  if (!m->has_eof_been_sent)

679  "received Finished-no-error before sending EOF.\n");
680 
681  /* If no protocol violation, then respond */
682  else
683  {
684  /* Save this Finished PDU */
685  m->fin = pdu_ptr->data_field.fin;
686  if (m->fin.delivery_code != 0)
687  w_msg__ ("cfdp_engine: Finished-No-error from partner "
688  "shouldn't say delivery was incomplete.\n");
689 
690  /* Acknowledge the Finished PDU with an Ack-Finished PDU */
691  m->ack = aaa__build_ack_fin (m);
692  m->is_outgoing_ack_buffered = YES;
693 
694  if (mp->state == S1)
695  {
696  /* Move to Phase 4 (Shut Down) */
697  mp->phase = 4;
698 
699  /* If Metadata and/or EOF are buffered, don't send them */
700  m->is_outgoing_md_buffered = NO;
701  m->is_outgoing_eof_buffered = NO;
702  }
703 
704  else
705  /* Ignore this pdu in state S2 (we are already cancelling) */
706  ;
707  }
708  }
709 
710 
711 
712  else if (event == RECEIVED_FINISHED_CANCEL_PDU)
713  /* Received a Finished-Cancel Pdu. This is our partner's way of
714  * initiating a cancellation of the transaction. We always Ack
715  * this pdu. If we haven't initiated a Cancel from our side, then
716  * we shut down after sending the Ack. If we have, we keep running
717  * until our own EOF-Cancel is acked.
718  */
719  {
720  /* If the final Ack-Finished that we sent from a previous (now dead)
721  * state machine was dropped, our partner would re-send the
722  * Finished PDU, and we would have the special case that is
723  * handled here.
724  */
725  if (!m->has_a_put_request_been_received)
726  {
727  m__ack_fin_from_previous_machine (m, pdu_ptr);
728  return;
729  }
730 
731  /* Note: a Finished-Cancel PDU can be received at any time.
732  * Therefore, there is no check for a protocol violation here.
733  */
734 
735  /* Save a copy of the Finished PDU */
736  m->fin = pdu_ptr->data_field.fin;
737 
738  /* If we haven't already cancelled locally, accept partner's request */
739  if (mp->state == S1)
740  {
741  aaa__cancel_locally (m);
742  mp->condition_code = m->fin.condition_code;
743  mp->phase = 4;
744  m__move_to_state_s2 (m);
745  }
746 
747  /* Regardless, we must acknowledge receipt of the Finished PDU */
748  m->ack = aaa__build_ack_fin (m);
749  m->is_outgoing_ack_buffered = YES;
750  }
751 
752 
753 
754  else if (event == ACK_TIMER_EXPIRED)
755  /* Ack-timer expired. Resend the EOF Pdu (if permitted). */
756  {
757  /* 'Indication' callback allows engine user to respond to events */
758  indication__ (IND_ACK_TIMER_EXPIRED, &(m->publik));
759

761  /* This is the nominal situation */
762  {
763  /* If we've already sent the EOF-no-error PDU as many times as
764  * allowed, then declare a Fault.
765  */
766  if (mp->attempts >= mib__ack_limit (NULL))
767  /* Let the Fault mechanism decide what to do */
768  if (!m__fault_handler (m, POSITIVE_ACK_LIMIT_REACHED))
769  return;
770 
771  /* Try again -- Re-queue the EOF for metered release */
772  m->is_outgoing_eof_buffered = YES;
773  }
774 
775  else
776  /* Transaction is being cancelled */
777  {
778  /* If we've already sent the EOF-Cancel PDU as many times as
779  * allowed, the only choice is to give up (abandon the transaction).
780  */
781  if (mp->attempts >= mib__ack_limit (NULL))
782  /* "Positive Ack Limit" has been reached; give up */
783  aaa__abandon_this_transaction (m);
784  else
785  /* Try again -- Re-queue the EOF for metered release */
786  m->is_outgoing_eof_buffered = YES;
787  }
788  }
789 
790 
791 
792  else if (event == INACTIVITY_TIMER_EXPIRED)
793  /* We are waiting for a response from our partner, and there is none. */
794  {
795  /* 'Indication' callback allows engine user to respond to events */
796  indication__ (IND_INACTIVITY_TIMER_EXPIRED, &(m->publik));
797 
798  if (mp->state == S1)
799  {
800  /* Restart the Inactivity-timer */
801  m->inactivity_timer = timer__start (mib__inactivity_timeout(NULL));
802  /* Let the Fault mechanism decide what else to do */
803  if (!m__fault_handler (m, INACTIVITY_DETECTED))
804  return;
805  }
806  else
807  /* We are already cancelling; just give up completely */
808  aaa__abandon_this_transaction (m);
809  }
810 
811 
812 
813  else if (event == EXTERNAL_FILE_TRANSFER_COMPLETED)
814  /* This event was added in May 2007 to support high-speed (external)
815  * file transfer by the engine user. When this event fires, we know
816  * that the engine user has completed sending the first round of
817  * Filedata PDUs. Therefore, we can send the EOF PDU and take over
818  * responsibility for handling acks and naks.
819  */
820  {
821  /* Queue an outgoing EOF-No-error PDU */
822  m->eof = aaa__build_eof (mp->condition_code,
823  mp->file_checksum_as_calculated,
824  mp->md.file_size);
825  m->is_outgoing_eof_buffered = YES;
826 
827  /* Clear the flags related to external xfer */
828  m->is_external_xfer_open = NO;
829  m->should_external_xfer_be_closed = NO;
830  }
831 
832 
833 
834  else
835  /* The above events are the only ones that should occur. If some
836  * other event occurs, ignore it (but warn the user).
837  */
838  w_msg__ ("cfdp_engine: ignored event '%s' in state %s "
839  "for trans '%s' <S1>.\n",
840  event__event_as_string(event), mp->state,
841  cfdp_trans_as_string (mp->trans));
842 }
