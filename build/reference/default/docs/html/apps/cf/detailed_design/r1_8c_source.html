49 static boolean m__fault_handler (MACHINE *m, CONDITION_CODE fault)
50 /* WHAT IT DOES: Responds to the given fault in accordance with the
51  * engine user's setting of "response to fault" (in the MIB).
52  * RETURN STATUS: If '1', the response to the given fault was 'ignore',
53  * and the calling state table routine may continue with whatever it
54  * was doing. If '0', the response was suspend, cancel, or abandon
55  * (and the calling state table routine should return immediately).
56  */
57 {
58  boolean ok_for_caller_to_continue;
59  /*------------------------------------------------------------*/
60 
61  /* Save the fault-type */
62  m->publik.condition_code = fault;
63  /* 'Indication' callback allows engine user to respond to events */
64  indication__ (IND_FAULT, &(m->publik));
65 
66  /* Unless the user has configured to 'ignore' this fault, the calling
67  * routine should not continue with what it was doing.
68  */
69  ok_for_caller_to_continue = NO;
70 
71  if (mib__response (fault) == RESPONSE_CANCEL)
72  {
73  m->have_we_initiated_a_cancel = YES;
74  aaa__cancel_locally (m);
75  /* We never give feedback to our partner, so just wrap up. */
76  aaa__transaction_has_finished (m);
77  }
78 
79  else if (mib__response (fault) == RESPONSE_SUSPEND)
80  aaa__notice_of_suspension (m);
81 
82  else if (mib__response (fault) == RESPONSE_IGNORE)
83  ok_for_caller_to_continue = YES;
84 
85  else if (mib__response (fault) == RESPONSE_ABANDON)
86  aaa__abandon_this_transaction (m);
87 
88  else
89  e_msg__ ("cfdp_engine: problem with the Fault mechanism in <R1>.\n");
90 
91  return (ok_for_caller_to_continue);
92 }

99 void r1__state_table (MACHINE *m, int event,
100  PDU_AS_STRUCT *pdu_ptr, REQUEST *req_ptr)
101  /* NOTE: Conceptually, this state table has 2 states:
102  * S1 = Waiting for a Metadata PDU to arrive
103  * S2 = Waiting for an EOF PDU to arrive
104  */
105 {
106  TRANS_STATUS *mp = &(m->publik); /* useful shorthand */
107  /*------------------------------------------------------------*/
108 
109  /* Possibly display some debug info */
110  if (cfdp_is_message_class_enabled (CFDP_MSG_STATE_ALL))
111  aaa__display_state_and_event ("R1", m->publik.state, event);
112 
113 
114  /*------------------------------------------------------------*/
115  /* Here is the logic for responding to each Event in State S1 */
116  /*------------------------------------------------------------*/
117 
118 
119  if (mp->state == S1)
120  /* In state S1, we are waiting for a Metadata PDU. As soon as a
121  * Metadata PDU is received, we move to state S2. Normally, the
122  * transition to state S2 happens quickly because the first PDU
123  * received is typically Metadata. However, there are many other
124  * possibilities, and each of these is captured as an event below.
125  */
126  {
127 
128 
129 
130  if (event == RECEIVED_CANCEL_REQUEST)
131  /* Received a Cancel Request, meaning that someone/something
132  * on *our* end (as opposed to our partner) decided to cancel
133  * this transaction.
134  */
135  {
136  mp->condition_code = CANCEL_REQUEST_RECEIVED;
137  aaa__cancel_locally (m);
138  /* We never give feedback to our partner, so just wrap up. */
139  aaa__transaction_has_finished (m);
140  }
141 
142 
143 
144  else if (event == RECEIVED_ABANDON_REQUEST)
145  /* Abandon this transaction */
146  aaa__abandon_this_transaction (m);
147 
148 
149 
150  else if (event == RECEIVED_REPORT_REQUEST)
151  /* Received a Report Request */
152  indication__ (IND_REPORT, &(m->publik));
153 
154 
155 
156  else if (event == RECEIVED_METADATA_PDU)
157  /* Received Metadata Pdu. NOMINAL PATH STARTS HERE... */
158  {
159  /* Store both the pdu-header and the MD data-field */
160  m->hdr = pdu_ptr->hdr;
161  mp->trans = m->hdr.trans;
162  mp->partner_id = mp->trans.source_id;
163  ASSERT__ (pdu_ptr->dir_code == MD_PDU);
164  mp->md = pdu_ptr->data_field.md;
165  mp->has_md_been_received = YES;
166 
167  /* Update the Nak-list to indicate that Metadata has been received */
168  nak__metadata_sent_or_received (&(m->nak));
169 
170  /* Issue the Machine-Allocated Indication once per transaction */
171  if (!m->has_a_pdu_been_received)
172  {
173  indication__ (IND_MACHINE_ALLOCATED, &(m->publik));
174  m->has_a_pdu_been_received = YES;
175  }
176 
177  /* 'Indication' callback allows engine user to respond to events */
178  indication__ (IND_METADATA_RECV, &(m->publik));
179 
180  /* If a file is to be transferred, open a temp file to hold the
181  * incoming file-data.
182  */
183  if (mp->md.file_transfer)
184  {
185  if (aaa__open_temp_file (m))
186  m->is_there_an_open_file = YES;
187  else
188  {
189  e_msg__ ("cfdp_engine: unable to open a temp file (MD).\n");
190  if (!m__fault_handler (m, FILESTORE_REJECTION))
191  return;
192  }
193  }
194 
195  /* <NOT_SUPPORTED> Metadata TLVs */
196 
197  /* Once we receive a Metadata PDU, we move to state S2 */
198  mp->state = S2;
199  if (cfdp_is_message_class_enabled (CFDP_MSG_STATE_CHANGE))
200  d_msg__ ("<R1> (S2) 'Wait for an EOF Pdu'\n");
201  }
202 
203 
204 
205  else if (event == RECEIVED_FILEDATA_PDU)
206  /* Received File-data Pdu (before Metadata) */
207  {
208  /* Save the header info */
209  m->hdr = pdu_ptr->hdr;
210  mp->trans = m->hdr.trans;
211  mp->partner_id = mp->trans.source_id;
212 
213  /* Issue the Machine-Allocated Indication once per transaction */
214  if (!m->has_a_pdu_been_received)
215  {
216  indication__ (IND_MACHINE_ALLOCATED, &(m->publik));
217  m->has_a_pdu_been_received = YES;
218  }
219 
220  /* "Accept" (copy) the received Pdu */
221  ASSERT__ (pdu_ptr->is_this_a_file_data_pdu);
222  mp->fd_offset = pdu_ptr->data_field.fd.offset;
223  mp->fd_length = pdu_ptr->data_field.fd.buffer_length;
224 
225  if (mib__issue_file_segment_recv())
226  indication__ (IND_FILE_SEGMENT_RECV, &(m->publik));
227 
228  /* Open a temp file (once) to hold the incoming file-data. */
229  if (!m->is_there_an_open_file)
230  {
231  if (aaa__open_temp_file (m))
232  m->is_there_an_open_file = YES;
233  else
234  {
235  e_msg__ ("cfdp_engine: unable to open a temp file (FD).\n");
236  if (!m__fault_handler (m, FILESTORE_REJECTION))
237  return;
238  }
239  }
240 
241  /* Store the incoming file-data */
242  aaa__store_file_data (m->fp, &pdu_ptr->data_field.fd);
243 
244  /* Update the received file size */
245  if (mp->fd_offset + mp->fd_length > mp->received_file_size)
246  /* This FD has made the file bigger than before */
247  mp->received_file_size = mp->fd_offset + mp->fd_length;
248 
249  /* If this Filedata segment exposes a gap, add gap to nak-list.
250  * (the 'nak' module takes care of all this bookkeeping stuff)
251  */
252  nak__data_received (&m->nak, mp->fd_offset,
253  mp->fd_offset + mp->fd_length);
254  }
255 
256 
257 
258  else if (event == RECEIVED_EOF_NO_ERROR_PDU)
259  /* Received EOF Pdu (before Metadata); shut down */
260  {
261  m->hdr = pdu_ptr->hdr;
262  mp->trans = m->hdr.trans;
263  mp->partner_id = mp->trans.source_id;
264 
265  /* Issue the Machine-Allocated Indication once per transaction */
266  if (!m->has_a_pdu_been_received)
267  {
268  indication__ (IND_MACHINE_ALLOCATED, &(m->publik));
269  m->has_a_pdu_been_received = YES;
270  }
271 
272  i_msg__ ("cfdp_engine: Class 1 Receiver finished, "
273  "but never received Metadata.\n");
274  aaa__transaction_has_finished (m);
275  }
276 
277 
278 
279  else if (event == RECEIVED_EOF_CANCEL_PDU)
280  /* Received EOF-Cancel Pdu, meaning our partner has cancelled
281  * this transaction.
282  */
283  {
284  mp->cancelled = YES;
285 
286  /* Copy the Condition Code, which tells us *why* they cancelled */
287  mp->condition_code =
288  pdu_ptr->data_field.eof.condition_code;
289 
290  /* Issue the Machine-Allocated Indication once per transaction */
291  if (!m->has_a_pdu_been_received)
292  {
293  indication__ (IND_MACHINE_ALLOCATED, &(m->publik));
294  m->has_a_pdu_been_received = YES;
295  }
296 
297  /* Wrap up and shut down */
298  aaa__transaction_has_finished (m);
299  }
300 
301 
302 
303  else if (event == INACTIVITY_TIMER_EXPIRED)
304  /* Inactivity-timer expired */
305  {
306  m->inactivity_timer = timer__start
307  (mib__inactivity_timeout(NULL));
308  if (!m__fault_handler (m, INACTIVITY_DETECTED))
309  return;
310  }
311 
312 
313 
314  else
315  w_msg__ ("cfdp_engine: ignored event '%s' in state S1 "
316  "for trans '%s' <R1>.\n",
317  event__event_as_string(event),
318  cfdp_trans_as_string (mp->trans));
319  }
320 
321 
322 
323 
324  /*------------------------------------------------------------*/
325  /* Here is the logic for responding to each Event in State S2 */
326  /*------------------------------------------------------------*/
327 
328  else if (mp->state == S2)
329  /* Wait for an EOF Pdu */
330  {
331 
332 
333

335  /* Received a Cancel Request */
336  {
337  mp->condition_code = CANCEL_REQUEST_RECEIVED;

339  /* We never give feedback to our partner, so just wrap up. */
340  aaa__transaction_has_finished (m);
341  }
342

346  /* Abandon this transaction */
347  aaa__abandon_this_transaction (m);
348 
349 
350 
351  else if (event == RECEIVED_REPORT_REQUEST)
352  /* Received a Report Request */
353  indication__ (IND_REPORT, &(m->publik));
354 
355 
356 
357  else if (event == RECEIVED_FILEDATA_PDU)
358  /* Received a File-data Pdu; store the file data. ...NOMINAL PATH... */
359  {
360  /* "Accept" (copy) the received Pdu */
361  ASSERT__ (pdu_ptr->is_this_a_file_data_pdu);
362  mp->fd_offset = pdu_ptr->data_field.fd.offset;
363  mp->fd_length = pdu_ptr->data_field.fd.buffer_length;
364 
365  /* The rest matches the published state table */
366  if (mp->md.file_transfer)
367  {
368  if (mib__issue_file_segment_recv())
369  indication__ (IND_FILE_SEGMENT_RECV, &(m->publik));
370  aaa__store_file_data (m->fp, &pdu_ptr->data_field.fd);
371

374  /* This FD has made the file bigger than before */

385  * PDU told us that there would be no file transfer.

400

402  {
403  /* Close the temporary file */

406  /* This should never happen, but if it does, avoid crashing */
407  ;
408  else

410  m->is_there_an_open_file = NO;
411 
412  if (!aaa__is_file_size_valid (m))

414  if (!m__fault_handler (m, FILE_SIZE_ERROR))
415  return;
416  }
417  if (!aaa__is_file_checksum_valid (m))
418  {

420  return;
421  }

436

438  }

442  }
