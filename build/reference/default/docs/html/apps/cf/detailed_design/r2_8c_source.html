88 static boolean m__fault_handler (MACHINE *m, CONDITION_CODE fault)
89 /* WHAT IT DOES: Responds to the given fault in accordance with the
90  * engine user's setting of "response to fault" (in the MIB).
91  * RETURN STATUS: If '1', the response to the given fault was 'ignore',
92  * and the calling state table routine may continue with whatever it
93  * was doing. If '0', the response was suspend, cancel, or abandon
94  * (and the calling state table routine should return immediately).
95  */
96 {
97  boolean ok_for_caller_to_continue;
98  /*------------------------------------------------------------*/
99 
100  /* Save the fault-type */
101  m->publik.condition_code = fault;
102  /* 'Indication' callback allows engine user to respond to events */
103  indication__ (IND_FAULT, &(m->publik));
104 
105  /* Unless the user has configured to 'ignore' this fault, the calling
106  * routine should not continue with what it was doing.
107  */
108  ok_for_caller_to_continue = NO;
109 
110  if (mib__response (fault) == RESPONSE_CANCEL)
111  {
112  m->have_we_initiated_a_cancel = YES;
113  aaa__cancel_locally (m);
114  m__ask_partner_to_cancel (m);
115  m__move_to_state_s2 (m);
116  }
117 
118  else if (mib__response (fault) == RESPONSE_SUSPEND)
119  aaa__notice_of_suspension (m);
120 
121  else if (mib__response (fault) == RESPONSE_IGNORE)
122  ok_for_caller_to_continue = YES;
123 
124  else if (mib__response (fault) == RESPONSE_ABANDON)
125  aaa__abandon_this_transaction (m);
126 
127  else
128  e_msg__ ("cfdp_engine: problem with the Fault mechanism in <R2>.\n");
129 
130  return (ok_for_caller_to_continue);
131 }

136 static void m__all_data_has_been_received (MACHINE *m)
137  /* WHAT IT DOES: Performs the actions required after all data has
138  * been delivered. Typically, validates the delivered file, moves
139  * it to the destination specified in the Metadata PDU, and
140  * builds/queues an outgoing Finished-No-error PDU.
141  */
142 {
143  TRANS_STATUS *mp = &(m->publik); /* useful shorthand */
144  /*------------------------------------------------------------*/
145 
146  /* IMPORTANT: The program will crash if the actions in this routine are
147  * carried out more than once, so prevent that from happening.
148  */
149  if (mp->delivery_code == DATA_COMPLETE)
150  /* Return immediately! */
151  {
152  w_msg__ ("cfdp_engine: 'all_data_has_been_received' called 2+ times.\n");
153  return;
154  }
155 
156  /* We've received everything, so delivery is complete... */
157  mp->delivery_code = DATA_COMPLETE;
158 
159  /* ... and there is no need to send a Nak back to the sender. */
160  m->is_outgoing_nak_buffered = NO;
161  timer__cancel (&m->nak_timer);
162 
163  /* If a file has been transferred, do something with it */
164  if (mp->md.file_transfer)
165  {
166  /* If the temporary file is still open, then close it */
167  if (m->is_there_an_open_file)
168  {
169  ASSERT__ (m->fp != NULL);
170  if (m->fp == NULL)
171  /* This should never happen, but if it does, avoid crashing */
172  ;
173  else
174  fclose_callback (m->fp);
175  m->is_there_an_open_file = NO;
176  }
177 
178  /* Validate the received file (via the checksum) */
179  if (!aaa__is_file_checksum_valid (m))
180  {
181  if (!m__fault_handler (m, FILE_CHECKSUM_FAILURE))
182  return;
183  }
184 
185  /* Move the temporary file to the real destination */
186  if (!rename_callback (mp->temp_file_name, mp->md.dest_file_name))
187  /* Success */
188  m->is_there_a_temp_file = NO;
189  else
190  /* Oops. Our attempt to move the file failed. */
191  {
192  e_msg__ ("cfdp_engine: unable to rename '%s' to '%s'.\n",
193  mp->temp_file_name, mp->md.dest_file_name);
194  e_msg__ ("cfdp_engine: are they on separate filesystems?\n");
195  if (!m__fault_handler (m, FILESTORE_REJECTION))
196  return;
197  }
198  }
199 
200  /* <NOT_SUPPORTED> Filestore Requests (executed here) */
201 
202  /* Build and queue a Finished Pdu */
203  m->fin = aaa__build_fin (NO_ERROR, mp->delivery_code);
204  m->is_outgoing_fin_buffered = YES;
205  mp->attempts = 0;
206 }

211 static void m__release_an_ack_pdu (MACHINE *m)
212  /* WHAT IT DOES: Outputs an Ack PDU,
213  * and performs the actions required after the PDU is released.
214  */
215 {
216  TRANS_STATUS *mp = &(m->publik); /* useful shorthand */
217  /*------------------------------------------------------------*/
218 
219  /*----------------*/
220  /* Output the PDU */
221  /*----------------*/
222 
223  aaa__send_ack (m->hdr, m->ack);
224  m->is_outgoing_ack_buffered = NO;
225 
226  /*----------------------------------------------------*/
227  /* Perform actions required after the PDU is released */
228  /*----------------------------------------------------*/
229 
230  if (m->ack.condition_code == NO_ERROR)
231  /* An Ack-EOF-no-error PDU was just sent to our CFDP partner.
232  * The transaction remains in-progress, and enters Phase 3
233  * (Fill Gaps) unless we have initiated a Cancel.
234  */
235  {
236  if (!m->have_we_initiated_a_cancel)
237  mp->phase = 3;
238  }
239 
240  else
241  /* An Ack-EOF-Cancel PDU was just sent to our CFDP partner.
242  * The transaction was cancelled by our partner and we acknowledged
243  * their cancellation. Typically, we can shut down now.
244  * However, if we have also initiated a Cancel, we can't shut down until
245  * our partner Acks our cancel.
246  */
247  {
248  if (m->have_we_initiated_a_cancel)
249  /* Keep running until our own Cancel is ack'd. */
250  ;
251  else
252  /* We're done, wrap up and shut down. */
253  aaa__transaction_has_finished (m);
254  }
255 }

290 static void m__release_a_nak_pdu (MACHINE *m)
291  /* WHAT IT DOES: Outputs a Nak PDU,
292  * and performs the actions required after the PDU is released.
293  */
294 {
295  TRANS_STATUS *mp = &(m->publik); /* useful shorthand */
296  /*------------------------------------------------------------*/
297 
298  /*----------------*/
299  /* Output the PDU */
300  /*----------------*/
301 
302  aaa__send_nak (m->hdr, m->nak);
303  m->is_outgoing_nak_buffered = NO;
304 
305  /* If the gap-list will not fit in one Nak PDU, we will need to
306  * spread the gaps across multiple Nak PDUs...
307  * <NOT_SUPPORTED> A gap-list that spans multiple Nak PDUs
308  */
309 
310  /*----------------------------------------------------*/
311  /* Perform actions required after the PDU is released */
312  /*----------------------------------------------------*/
313 
314  /* Keep track of how many Naks we send */
315  mp->how_many_naks ++;
316 
317  /* If we are nominal, give our partner nak-timeout seconds to respond */
318  if (mp->state == S1)
319  {
320  mp->attempts ++;
321  m->nak_timer = timer__start (mib__nak_timeout(NULL));
322  }
323 
324  /* If the transaction is being cancelled, don't bother */
325  else
326  ;
327 }

336 /*=r=************************************************************************/
337 void r2__state_table (MACHINE *m, int event,
338  PDU_AS_STRUCT *pdu_ptr, REQUEST *req_ptr)
339  /* NOTE: Conceptually, this state table has 2 states:
340  * S1 = Nominal
341  * S2 = Transaction Cancelled
342  * So, the state is always S1 unless the transaction is cancelled.
343  * Transactions can be cancelled in 3 ways:
344  * 1) the local User can issue a Cancel Request
345  * 2) our CFDP partner can send an EOF-Cancel PDU.
346  * 3) A fault may occur, and the response to that fault may be
347  * to cancel.
348  */
349 {
350  TRANS_STATUS *mp = &(m->publik); /* useful shorthand */
351  /*------------------------------------------------------------*/
352 
353  /* Possibly display some debug info */
354  if (cfdp_is_message_class_enabled (CFDP_MSG_STATE_ALL))
355  aaa__display_state_and_event ("R2", m->publik.state, event);
356 
357 
358  /*-----------------------*/
359  /* Respond to each Event */
360  /*-----------------------*/
361 
362 
363 
364  if (event == THROTTLE_OUTGOING_FILE_DIR_PDU)
365  /* Release, at most, one outgoing PDU. */
366  {
367  if (mp->frozen)
368  /* We can't release any PDUs while frozen */
369  ;
370 
371  else if (mp->suspended)
372  /* The only PDU that can released while suspended is an Ack */
373  {
374  if (m->is_outgoing_ack_buffered)
375  /* An Ack PDU is ready. Do we have the "green light"? */
376  if (pdu_output__ready (FILE_DIR_PDU, mp->trans, m->hdr.dest_id))
377  /* Light is green. Send it */
378  m__release_an_ack_pdu (m);
379  }
380 
381  else if (m->is_outgoing_ack_buffered)
382  /* We're not frozen or suspended, so we can release any PDU.
383  * An Ack is ready. Do we have the "green light"?
384  */
385  {
386  if (pdu_output__ready (FILE_DIR_PDU, mp->trans, m->hdr.dest_id))
387  /* Light is green. Send it */
388  m__release_an_ack_pdu (m);
389  }
390 
391  else if (m->is_outgoing_nak_buffered)
392  /* We're not frozen or suspended, so we can release any PDU.
393  * A Nak is ready. Do we have the "green light"?
394  */
395  {
396  if (pdu_output__ready (FILE_DIR_PDU, mp->trans, m->hdr.dest_id))
397  /* Light is green. Send it */
398  m__release_a_nak_pdu (m);
399  }
400 
401  else if (m->is_outgoing_fin_buffered)
402  /* We're not frozen or suspended, so we can release any PDU.
403  * A Finished is ready. Do we have the "green light"?
404  */
405  {
406  if (pdu_output__ready (FILE_DIR_PDU, mp->trans, m->hdr.dest_id))
407  /* Light is green. Send it */
408  m__release_a_fin_pdu (m);
409  }
410  }
411 
412 
413 
414  else if (event == RECEIVED_SUSPEND_REQUEST)
415  /* A Suspend Request was received; suspend timers */
416  aaa__notice_of_suspension (m);
417 
418 
419 
420  else if (event == RECEIVED_RESUME_REQUEST)
421  /* A Resume Request was received; possibly resume timers */
422  {
423  /* 'Indication' callback allows engine user to respond to events */
424  indication__ (IND_RESUMED, &(m->publik));
425  mp->suspended = NO;
426  /* Timers remain paused if either 'suspended' or 'frozen' */
427  if (!mp->frozen)
428  aaa__resume_timers (m);
429  }
430 
431 
432 
433  else if (event == RECEIVED_CANCEL_REQUEST)
434  /* Received a Cancel Request (from our User, not from our partner) */
435  {
436  /* If the transaction isn't already being cancelled, begin doing so */
437  if (mp->state == S1)
438  {
439  mp->condition_code = CANCEL_REQUEST_RECEIVED;
440  m->have_we_initiated_a_cancel = YES;
441  /* Cancel both locally... */
442  aaa__cancel_locally (m);
443  /* ... and remotely (i.e. have our partner cancel too) */
444  m__ask_partner_to_cancel (m);
445  m__move_to_state_s2 (m);
446  }
447  }
448 
449 
450 
451  else if (event == RECEIVED_ABANDON_REQUEST)
452  /* Abandon this transaction */
453  aaa__abandon_this_transaction (m);
454 
455 
456 
457  else if (event == RECEIVED_REPORT_REQUEST)
458  /* A Report Request was received; issue a Report */
459  indication__ (IND_REPORT, &(m->publik));
460 
461 
462 
463  else if (event == RECEIVED_FREEZE_REQUEST)
464  /* A "Freeze" was received, so suspend all timers. */
465  {
466  mp->frozen = YES;
467  aaa__suspend_timers (m);
468  }
469 
470 
471 
472  else if (event == RECEIVED_THAW_REQUEST)
473  /* A "Thaw" was received; possibly resume all timers. */
474  {
475  mp->frozen = NO;
476  /* Timers remain paused if either 'suspended' or 'frozen' */
477  if (!mp->suspended)
478  aaa__resume_timers (m);
479  }
480 
481 
482 
483  else if (event == RECEIVED_METADATA_PDU)
484  /* Received a Metadata PDU. This is normally the first PDU we receive
485  * but not always. And its possible that we might receive more than one
486  * Metadata pdu; We only respond to the first Metadata we receive.
487  */
488  {
489  /* Ignore PDUs received while cancelling */
490  if (mp->state == S2)
491  return;
492 
493  /* CFDP messages that we send back to our partner will reuse the
494  * header that we receive in the first message from them (except
495  * that the "direction" field will be reversed).
496  * Note: The first message we receive may be Metadata, Filedata, or EOF.
497  */
498 
499  /* Our outgoing PDU header will be the same as the first incoming
500  * PDU header (but with the 'direction' field reversed).
501  */
502  if (!m->has_a_pdu_been_received)
503  {
504  aaa__reuse_senders_first_hdr (pdu_ptr->hdr, m);
505  /* 'Indication' callback allows engine user to respond to events */
506  indication__ (IND_MACHINE_ALLOCATED, &(m->publik));
507  m->has_a_pdu_been_received = YES;
508  }
509 
510  /* Note: Metadata may arrive more than once (it shouldn't, but it
511  * can). Respond to the first Metadata and ignore all the others.
512  */
513  if (!mp->has_md_been_received)
514  /* This is the first Metadata PDU for this transaction */
515  {
516  mp->has_md_been_received = YES;
517  mp->md = pdu_ptr->data_field.md;
518  /* 'Indication' callback allows engine user to respond to events */
519  indication__ (IND_METADATA_RECV, &(m->publik));
520 
521  /* If this transaction will transfer a file, prepare */
522  if (mp->md.file_transfer)
523  {
524  /* Open a temporary file (unless a file was already opened
525  * in response to a Filedata or EOF PDU).
526  * Note: Normally, Metadata is the first PDU received,
527  * but that is not guaranteed.
528  */
529  if (!m->is_there_an_open_file)
530  {
531  if (aaa__open_temp_file (m))
532  m->is_there_an_open_file = YES;
533  else
534  {
535  e_msg__ ("cfdp_engine: unable to open a temp file "
536  "(MD).\n");
537  if (!m__fault_handler (m, FILESTORE_REJECTION))
538  return;
539  }
540  }
541  }
542 
543  /* Update the Nak-list to indicate that Metadata has been received */
544  nak__metadata_sent_or_received (&(m->nak));
545 
546  /* <NOT_SUPPORTED> Metadata TLVs (processed here) */
547 
548  /* Protocol Optimization: Don't wait for the Nak-timer to expire
549  * before checking to see if retransmission is complete.
550  */
551  if (m->has_eof_been_received)
552  {
553  if (nak__is_list_empty (&m->nak))
554  /* All data has been received; begin nominal Finish */
555  m__all_data_has_been_received (m);
556  }
557  }
558  }
559 
560 
561 
562  else if (event == RECEIVED_FILEDATA_PDU)
563  /* Received a Filedata Pdu. Nominally, we store it in a file, and
564  * update our bookkeeping. But if the transaction is being cancelled,
565  * don't bother.
566  */
567  {
568  /* Ignore PDUs received while cancelling */
569  if (mp->state == S2)
570  return;
571 
572  /* Our outgoing PDU header will be the same as the first incoming
573  * PDU header (but with the 'direction' field reversed).
574  */
575  if (!m->has_a_pdu_been_received)
576  {
577  aaa__reuse_senders_first_hdr (pdu_ptr->hdr, m);
578  /* 'Indication' callback allows engine user to respond to events */
579  indication__ (IND_MACHINE_ALLOCATED, &(m->publik));
580  m->has_a_pdu_been_received = YES;
581  }
582 
583  /* If we haven't already opened a file to receive the data, do so now */
584  if (!m->is_there_an_open_file)
585  {
586  if (aaa__open_temp_file (m))
587  m->is_there_an_open_file = YES;
588  else
589  {
590  e_msg__ ("cfdp_engine: unable to open a temp file (FD).\n");
591  if (!m__fault_handler (m, FILESTORE_REJECTION))
592  return;
593  }
594  }
595 
596  /* Save the File-data in the temporary file */
597  mp->fd_offset = pdu_ptr->data_field.fd.offset;
598  mp->fd_length = pdu_ptr->data_field.fd.buffer_length;
599  if (mib__issue_file_segment_recv())
600  /* 'Indication' callback allows engine user to respond to events */
601  indication__ (IND_FILE_SEGMENT_RECV, &(m->publik));
602  aaa__store_file_data (m->fp, &pdu_ptr->data_field.fd);
603 
604  /* Update the received file size */
605  if (mp->fd_offset + mp->fd_length > mp->received_file_size)
606  /* This FD has made the file bigger than before */
607  mp->received_file_size = mp->fd_offset + mp->fd_length;
608 
609  /* If this Filedata segment exposes a gap, add gap to nak-list.
610  * (the 'nak' module takes care of all this bookkeeping stuff)
611  */
612  nak__data_received (&m->nak, mp->fd_offset,
613  mp->fd_offset + mp->fd_length);
614 
615  /* If this Filedata PDU arrived *after* the EOF PDU (typically, not
616  * the case), check for a "file size error".
617  */
618  if (m->has_eof_been_received)
619  /* A "file_size_error" means that the received file-data cannot be
620  * stored without extending the file's length beyond the length
621  * specified in the EOF PDU.
622  */
623  if (!aaa__is_file_size_valid (m))
624  if (!m__fault_handler (m, FILE_SIZE_ERROR))
625  return;
626 
627  /* Protocol Optimization: Don't wait for the Nak-timer to expire
628  * before checking to see if retransmission is complete.
629  */
630  if (m->has_eof_been_received)
631  {
632  if (nak__is_list_empty (&m->nak))
633  /* All data has been received; begin nominal Finish */
634  m__all_data_has_been_received (m);
635  }
636  }
637 
638 
639 
640  else if (event == RECEIVED_EOF_NO_ERROR_PDU)
641  /* Received an EOF-no-error Pdu. This tells us that the Sender has
642  * sent all data once. We always Ack this pdu. Nominally, we
643  * take stock and respond with either a Nak or Finished PDU. However,
644  * if the transaction is being cancelled, don't bother.
645  */
646  {
647 
648  /*-------------------------------*/
649  /* Always acknowledge an EOF PDU */
650  /*-------------------------------*/
651 
652  /* Save a copy of the EOF */
653  m->eof = pdu_ptr->data_field.eof;
654 
655  /* Our outgoing PDU header will be the same as the first incoming
656  * PDU header (but with the 'direction' field reversed).
657  */
658  aaa__reuse_senders_first_hdr (pdu_ptr->hdr, m);
659 
660  /* Build and queue an Ack of the EOF (every EOF must be acked) */
661  m->ack = aaa__build_ack_eof (m);
662  m->is_outgoing_ack_buffered = YES;
663 
664  /*---------------------------------------*/
665  /* Ignore PDUs received while cancelling */
666  /*---------------------------------------*/
667  if (mp->state == S2)
668  return;
