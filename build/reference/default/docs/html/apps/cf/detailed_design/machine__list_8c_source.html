60 MACHINE *machine_list__allocate_slot (void)
61 {
62  MACHINE *answer;
63  u_int_4 i;
64  /*------------------------------------------------------------*/
65 
66  INITIALIZE_ONCE;
67 
68  /* Linear search until we find an unused slot in the machine-array
69  * (or until all slots are checked).
70  */
71  i = 0;
72  while (m_is_this_slot_in_use[i] && (i < MAX_CONCURRENT_TRANSACTIONS))
73  i ++;
74 
75  if (i < MAX_CONCURRENT_TRANSACTIONS)
76  /* Success. We found a free slot. */
77  {
78  m_is_this_slot_in_use[i] = YES;
79  answer = &(m_machine[i]);
80  }
81  else
82  /* Failure. No free slots */
83  answer = NULL;
84 
85  return (answer);
86 }

91 boolean machine_list__deallocate_slot (MACHINE *machine_to_delete)
92 {
93  u_int_4 i;
94  MACHINE *machine_ptr;
95  boolean were_we_successful;
96  /*------------------------------------------------------------*/
97 
98  INITIALIZE_ONCE;
99

101  * address of the given machine (or until all slots are checked).
102  */
103  were_we_successful = NO;
104  i = 0;
105  while (!were_we_successful && (i < MAX_CONCURRENT_TRANSACTIONS))
106  {
107  machine_ptr = &(m_machine[i]);

112  i ++;
113  }
114 
115  if (were_we_successful)
116  m_is_this_slot_in_use[i] = NO;
117 
118  return (were_we_successful);

136 MACHINE *machine_list__get_next_machine (void)
137 {
138  static MACHINE *answer;
139  /*------------------------------------------------------------*/
140 
141  INITIALIZE_ONCE;
142 
143  /* Pick up where we left off, and check one slot at a time until
144  * we find one that is being used (or until all slots are checked).
145  */
146  while (!m_is_this_slot_in_use[m_current_walk_index] &&
147  (m_current_walk_index < MAX_CONCURRENT_TRANSACTIONS))
148  m_current_walk_index ++;
149 
150  if (m_current_walk_index < MAX_CONCURRENT_TRANSACTIONS)
151  /* We found the next in-use slot */
152  {
153  answer = &(m_machine[m_current_walk_index]);
154  /* Next time, pick up from the slot after this one */
155  m_current_walk_index ++;
156  }
157  else
158  /* We've reached the end of this walk */
159  answer = NULL;
160 
161  return (answer);
162 }
