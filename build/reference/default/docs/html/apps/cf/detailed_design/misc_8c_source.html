135 /*=r=*************************************************************************/
136 void misc__update_summary_statistics (MACHINE *m)
137  /* WHAT IT DOES: Given the status of a state machine that has
138  * finished, this routine updates the summary statistics.
139  */
140 {
141  /*------------------------------------------------------------*/
142 
143  if ((m->publik.role == S_1) || (m->publik.role == S_2))
144  {
145  /* If the transaction existed solely to ack a (retransmitted)
146  * Finished PDU, then ignore it.
147  */
148  if (m->publik.is_this_trans_solely_for_ack_fin)
149  ;
150  /* Otherwise, increment the appropriate Sender count */
151  else
152  {
153  /* A successful transaction is neither cancelled nor abandoned */
154  if ((m->publik.condition_code == NO_ERROR) &&
155  (!m->publik.abandoned))
156  m_statistics.successful_sender_count ++;
157  else
158  m_statistics.unsuccessful_sender_count ++;
159  }
160  }
161 
162  else
163  /* Increment the appropriate Receiver count */
164  {
165  /* A successful transaction is neither cancelled nor abandoned */
166  if ((m->publik.condition_code == NO_ERROR) &&
167  (!m->publik.abandoned))
168  m_statistics.successful_receiver_count ++;
169  else
170  m_statistics.unsuccessful_receiver_count ++;
171  }
172 }
