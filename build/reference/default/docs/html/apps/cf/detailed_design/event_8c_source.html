33 CFDP_EVENT event__determine_from_pdu_struct (const PDU_AS_STRUCT
34  *pdu_as_struct)
35  {
36  CFDP_EVENT event;
37  boolean is_msg_needed;
38  FILE_DIR_CODE which_dir_is_being_acked;
39  /*------------------------------------------------------------*/
40 
41  /*---------------------------------------------------------------*/
42  /* If configured to do so, output a message summarizing this PDU */
43  /*---------------------------------------------------------------*/
44  if (pdu_as_struct->is_this_a_file_data_pdu)
45  {
46  if (cfdp_is_message_class_enabled (CFDP_MSG_PDU_FILEDATA))
47  is_msg_needed = YES;
48  else
49  is_msg_needed = NO;
50  }
51  else if (cfdp_is_message_class_enabled (CFDP_MSG_PDU_NON_FILEDATA))
52  is_msg_needed = YES;
53  else
54  is_msg_needed = NO;
55  if (is_msg_needed)
56  d_msg__ ("----------> %s\n", pdu_struct_as_string (*pdu_as_struct));
57 
58  /*----------------------------------------------------*/
59  /* Determine which event corresponds to the given PDU */
60  /*----------------------------------------------------*/
61 
62  if (pdu_as_struct->is_this_a_file_data_pdu)
63  event = RECEIVED_FILEDATA_PDU;
64 
65  else if (pdu_as_struct->dir_code == MD_PDU)
66  event = RECEIVED_METADATA_PDU;
67 
68  else if (pdu_as_struct->dir_code == EOF_PDU)
69  {
70  /* Is the PDU an EOF (No Error) or an EOF (Cancel)? */
71  if (pdu_as_struct->data_field.eof.condition_code == NO_ERROR)
72  event = RECEIVED_EOF_NO_ERROR_PDU;
73  else
74  event = RECEIVED_EOF_CANCEL_PDU;
75  }
76 
77  else if (pdu_as_struct->dir_code == FIN_PDU)
78  {
79  /* Is the PDU is a Finished (No Error) or a Finished (Cancel)? */
80  if (pdu_as_struct->data_field.fin.condition_code == NO_ERROR)
81  event = RECEIVED_FINISHED_NO_ERROR_PDU;
82  else
83  event = RECEIVED_FINISHED_CANCEL_PDU;
84  }
85 
86  else if (pdu_as_struct->dir_code == ACK_PDU)
87  {
88  /* Is this PDU an Ack of an EOF PDU or an Ack of a Finished PDU? */
89  which_dir_is_being_acked =
90  pdu_as_struct->data_field.ack.directive_code;
91  if (which_dir_is_being_acked == EOF_PDU)
92  {
93  /* Is this Ack-EOF PDU an ack of an EOF-no-error or EOF-Cancel?*/
94  if (pdu_as_struct->data_field.ack.condition_code == NO_ERROR)
95  event = RECEIVED_ACK_EOF_NO_ERROR_PDU;
96  else
97  event = RECEIVED_ACK_EOF_CANCEL_PDU;
98  }
99  else if (which_dir_is_being_acked == FIN_PDU)
100  {
101  /* Is this Ack-Fin PDU an ack of a Fin-no-error or Fin-Cancel?*/
102  if (pdu_as_struct->data_field.ack.condition_code == NO_ERROR)
103  event = RECEIVED_ACK_FIN_NO_ERROR_PDU;
104  else
105  event = RECEIVED_ACK_FIN_CANCEL_PDU;
106  }
107  else
108  /* Invalid Ack - this "can't happen unless there is a bug" */
109  {
110  e_msg__ ("cfdp_engine: received Ack of unknown dir-type (%u).\n",
111  pdu_as_struct->dir_code);
112  ASSERT__BUG_DETECTED;
113  event = CFDP_EVENT_INVALID_ACK;
114  }
115  }
116 
117  else if (pdu_as_struct->dir_code == NAK_PDU)
118  event = RECEIVED_NAK_PDU;
119 
120  else
121  /* The Directive Code is unrecognized */
122  {
123  w_msg__ ("cfdp_engine: received PDU with unknown dir-code (%u).\n",
124  pdu_as_struct->dir_code);
125  event = CFDP_EVENT_INVALID_DIR_CODE;
126  }
127 
128  return (event);
129  }

134 CFDP_EVENT event__determine_from_request (REQUEST request)
135  {
136  int event;
137  /*------------------------------------------------------------*/
138 
139  if (request.type == REQ_PUT)

144

146  event = RECEIVED_SUSPEND_REQUEST;
147 
148  else if (request.type == REQ_RESUME)
149  event = RECEIVED_RESUME_REQUEST;
150 
151  else if (request.type == REQ_CANCEL)
152  event = RECEIVED_CANCEL_REQUEST;
153 
154  else if (request.type == REQ_REPORT)
155  event = RECEIVED_REPORT_REQUEST;
156 
157  else if (request.type == REQ_FREEZE)
158  event = RECEIVED_FREEZE_REQUEST;
159 
160  else if (request.type == REQ_THAW)
161  event = RECEIVED_THAW_REQUEST;
162 
163  else
164  {
165  w_msg__ ("cfdp_engine: User Request (%u) is unknown.\n",
166  request.type);
167  ASSERT__BUG_DETECTED;
168  event = CFDP_EVENT_UNKNOWN_REQUEST;
169  }
170 
171  return (event);
