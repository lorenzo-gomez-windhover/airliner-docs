26 char *ack_struct_as_string (ACK ack)
27  /* WHAT IT DOES: Returns a character-string that summarizes the given
28  * ACK structure.
29  */
30 {
31  static char string [1024];
32  char substring [32];
33  /*------------------------------------------------------------*/
34 
35  strcpy (string, "\0");
36 
37  /* Include whether the Ack is of an EOF-PDU or a Finished-PDU */
38  if (ack.directive_code == EOF_PDU)
39  APPEND (string, "Ack-EOF");
40  else if (ack.directive_code == FIN_PDU)
41  APPEND (string, "Ack-Fin");
42  else
43  {
44  APPEND (string, "INVALID ACK PDU - ");
45  APPEND (string, "Directive Code = ");
46  sprintf (substring, "%d", ack.directive_code);
47  APPEND (string, substring);
48  APPEND (string, ".");
49  return (string);
50  }
51 
52  /* Is the PDU being acked effectively a 'Cancel' PDU? -- i.e. an Ack of
53  * an EOF (Cancel) or a Finished (Cancel) PDU?
54  * Background: Instead of having a separate 'Cancel' PDU, CFDP Senders
55  * use an EOF PDU to cancel, and CFDP Receivers use a Finished PDU to
56  * cancel.
57  */
58  if (ack.condition_code != NO_ERROR)
59  APPEND (string, "-Cancel");
60  APPEND (string, ": ");
61 
62 
63  /* The Directive Subtype only applies to multi-hop CFDP, so ignore it */
64  ;
65 
66  /* Include the Transaction Status */
67  if (ack.transaction_status == 0)
68  APPEND (string, " (undefined transaction)");
69  else if (ack.transaction_status == 2)
70  APPEND (string, " (terminated transaction)");
71  else if (ack.transaction_status == 3)
72  APPEND (string, " (unrecognized transaction)");
73 
74  APPEND (string, ".");
75  return (string);
76 }

234  /* WHAT IT DOES: Returns a char-string that summarizes the given PDU. */
235 {

239

245  /* Include the appropriate info for each PDU-type */
246  if (pdu_as_struct.is_this_a_file_data_pdu)

249  APPEND (string, md_struct_as_string (pdu_as_struct.data_field.md));
250  else if (pdu_as_struct.dir_code == EOF_PDU)
251  APPEND (string, eof_struct_as_string (pdu_as_struct.data_field.eof));
252  else if (pdu_as_struct.dir_code == FIN_PDU)
253  APPEND (string, fin_struct_as_string (pdu_as_struct.data_field.fin));

257  APPEND (string, nak_struct_as_string (pdu_as_struct.data_field.nak));
