83 static void m__state_table (MACHINE *m, int event,
84  PDU_AS_STRUCT *pdu_ptr, REQUEST *req_ptr)
85  /* WHAT IT DOES: Calls the appropriate state table routine (based
86  * upon the role that we are to play - e.g. Class 2 Sender).
87  * The given machine, event, PDU, and Request are all passed along
88  * to the state table routine.
89  * NOTE: Also enforces the restriction that the state tables are
90  * not re-entrant for any particular state machine.
91  * NOTE: If the given state machine finishes, it is deleted.
92  */
93 {
94  /*------------------------------------------------------------*/
95 
96  /* Avoid re-entering the engine core for any particular transaction */
97  if (m->is_user_inside_the_engine_core)
98  {
99  w_msg__ ("cfdp_engine: user attempt to re-enter state tables denied.\n");
100  return;
101  }
102  m->is_user_inside_the_engine_core = YES;
103 
104  /* Run the given state machine in the appropriate state table routine */
105  if (m->publik.role == S_1)
106  s1__state_table (m, event, pdu_ptr, req_ptr);
107  else if (m->publik.role == S_2)
108  s2__state_table (m, event, pdu_ptr, req_ptr);
109  else if (m->publik.role == R_1)
110  r1__state_table (m, event, pdu_ptr, req_ptr);
111  else if (m->publik.role == R_2)
112  r2__state_table (m, event, pdu_ptr, req_ptr);
113  else
114  e_msg__ ("cfdp_engine: unable to run state table; unknown Role.\n");
115 
116  m->is_user_inside_the_engine_core = NO;
117 
118  /* If the state machine has finished, delete it */
119  if (m->has_this_state_machine_finished)
120  {
121  /* Engine-wide bookeeping */
122  misc__update_summary_statistics (m);
123  misc__set_last_condition_code (m->publik.condition_code);
124  misc__set_last_trans_abandoned (m->publik.abandoned);
125  /* Let the engine user know what has happened */
126  indication__ (IND_MACHINE_DEALLOCATED, &(m->publik));
127  /* Delete the state machine (give up the slot associated with it) */
128  if (!machine_list__deallocate_slot (m))
129  e_msg__ ("cfdp_engine: "
130  "attempt to delete completed state machine failed.\n");
131  }
132 }

137 static MACHINE *m__restart_old_sender_machine (HDR hdr)
138  /* WHAT IT DOES: Given a PDU-header (that is presumably a Finished
139  * PDU), it attempts to open a comm-link to the CFDP partner,
140  * allocate a state machine slot, and initialize the state machine.
141  * If successful, returns a pointer to the new state machine;
142  * otherwise, returns NULL.
143  */
144 {
145  MACHINE *m = NULL;
146  ID partner_id;
147  /*------------------------------------------------------------*/
148 
149  /* Attempt to open a comm-link to the CFDP partner entity. Since we were
150  * (and are) the sender, our partner is the destination of this transaction.
151  */
152  partner_id = hdr.dest_id; /* We're the sender, partner is dest-id */
153  if (!pdu_output__open (mib__get_my_id(), partner_id))
154  e_msg__ ("cfdp_engine: discarded incoming Fin PDU - "
155  "unable to communicate with node '%s'\n",
156  cfdp_id_as_string (partner_id));
157 
158  /* If successful, attempt to allocate a slot in the machine-list */
159  else if ((m = machine_list__allocate_slot ()) == NULL)
160  e_msg__ ("cfdp_engine: discarded incoming Fin PDU - "
161  "can't exceed %lu transactions\n",
162  MAX_CONCURRENT_TRANSACTIONS);
163 
164  /* If successful, initialize the state machine's MACHINE structure */

393 /*=r=************************************************************************/
394 boolean cfdp_give_request (const char *request_string)
395 {
396  /* CFDP_FILE *fp;*/ /* See NOTE below */
397  MACHINE *m;
398  REQUEST request;
399  boolean successful = YES;
400  /*------------------------------------------------------------*/
401 
402  REPORT_VERSION_ONCE;
403 
404  /*---------------------------------*/
405  /* Validate the given User Request */
406  /*---------------------------------*/
407 
408  if (request_string == NULL)
409  {
410  e_msg__ ("cfdp_engine: cfdp_give_request: given a null-pointer.\n");
411  successful = NO;
412  }
413 
414  else if (!utils__request_from_string (request_string, &request))
415  /* The string does not represent a valid request */
416  {
417  e_msg__ ("cfdp_engine: request is invalid (%s).\n", request_string);
418  successful = NO;
419  }
420 
421  else if ((request.type == REQ_PUT) &&
422  (cfdp_are_these_ids_equal (request.info.put.dest_id,
423  mib__get_my_id())))
424  /* This Put Request is invalid */
425  {
426  e_msg__ ("cfdp_engine: Won't allow you to send a file to yourself!\n");
427  successful = NO;
428  }
429 
430  /* If the User Request is an "empty" one, just ignore it */
431  else if (request.type == REQ_NONE)
432  /* No action needed */
433  d_msg__ ("cfdp_engine: ignoring an empty User Request.\n");
434 
435 
436  /*---------------------------------------------------------
437  * If the User Request is a Put Request, then create a new
438  * state machine, and pass the Put Request to that machine
439  *---------------------------------------------------------*/
440 
441  else if (request.type == REQ_PUT)
442  {
443 
444 /*
445 * NOTE: Because file opens are taking 1ms(GPM rad750) and 99% of the time,
446 * this check passes, we'll do away with this open/close check. This check
447 * detects a non-existent file before the transfer begins. Without this check,
448 * the transfer will begin with a meta data pdu sent. Before the first file-data
449 * pdu is sent, a filestore rejection will occur followed by an EOF-Cancel.
450 */
451 
452  /* Start a new state machine */
453  if ((m = m__start_new_sender_machine (request)) != NULL)
454  /* Fire the state table event associated with a Put Request */
455  m__state_table (m, RECEIVED_PUT_REQUEST, NULL, &request);
456  else
457  successful = NO;
458 
459 
460 #if 0
461  /* Optimization: If the source-file doesn't exist, catch it now */
462  fp = fopen_callback (request.info.put.source_file_name, "rb");
463  if (fp == NULL)
464  /* The source-file doesn't exist; no point in trying to transfer it */
465  {
466 
467  e_msg__ ("cfdp_engine: can't transfer non-existent file (%s).\n",
468  request.info.put.source_file_name);
469  successful = NO;
470  }
471  else
472  {
473  /* It does exist; close it and let the state machine re-open it */
474  fclose_callback (fp);
475 
476  /* Start a new state machine */
477  if ((m = m__start_new_sender_machine (request)) != NULL)
478  /* Fire the state table event associated with a Put Request */
479  m__state_table (m, RECEIVED_PUT_REQUEST, NULL, &request);
480  else
481  successful = NO;
482  }
483 #endif
484 
485  }
486 
487 
488  /*------------------------------------------------------------------
489  * If the User Request applies to *all* transactions, then pass it
490  * to all active state machines
491  *------------------------------------------------------------------*/
492 
493  else if (request.type == REQ_REPORT)
494  /* Report all active transactions */
495  m__give_request_to_all_trans (request.type);
496 
497  else if (request.type == REQ_FREEZE)
498  /* Freeze all active transactions, and any future ones too */
499  {
500  m__give_request_to_all_trans (request.type);
501  misc__freeze_all_partners ();
502  }
503 
504  else if (request.type == REQ_THAW)
505  /* Thaw all active transactions, and any future ones too */
506  {
507  m__give_request_to_all_trans (request.type);
508  misc__thaw_all_partners ();
509  }
510 
511  else if (request.type == REQ_ABANDON_ALL_TRANSACTIONS)
512  m__give_request_to_all_trans (REQ_ABANDON);
513 
514  else if (request.type == REQ_CANCEL_ALL_TRANSACTIONS)
515  m__give_request_to_all_trans (REQ_CANCEL);
516 
517  else if (request.type == REQ_SUSPEND_ALL_TRANSACTIONS)
518  m__give_request_to_all_trans (REQ_SUSPEND);
519 
520  else if (request.type == REQ_RESUME_ALL_TRANSACTIONS)
521  m__give_request_to_all_trans (REQ_RESUME);
522 
523 
524  /*-----------------------------------------------------------------
525  * If the User Request applies to one transaction, then attempt to
526  * find an active state machine assigned to that transaction, and
527  * pass the Request to that state machine.
528  *-----------------------------------------------------------------*/
529 
530  else if ((request.type == REQ_ABANDON) ||
531  (request.type == REQ_CANCEL) ||
532  (request.type == REQ_SUSPEND) ||
533  (request.type == REQ_RESUME))
534  {
535  if ((m = machine_list__get_this_trans (request.info.trans)) != NULL)
536  /* Success (a machine is assigned to this transaction) */
537  m__give_request_to_one_trans (m, request.type);
538  else
539  /* No state machine assigned */
540  {
541  e_msg__ ("cfdp_engine: ignoring User-Request that references "
542  "unknown transaction (%s).\n",
543  cfdp_trans_as_string (request.info.trans));
544  successful = NO;
545  }
546  }
547 
548 
549  else
550  {
551  w_msg__ ("cfdp_engine: ignoring unrecognized User-Request (%s).\n",
552  request_string);
553  successful = NO;
554  }
555 
556  return (successful);
557 }

561 /*=r=************************************************************************/
562 boolean cfdp_give_pdu (PDU pdu)
563  /* NOTE: The CFDP protocol requires a transaction's Inactivity-timer
564  * to be started (or restarted) each time a PDU is received.
565  * That action is taken here rather than in 25 different places in
566  * all the state table modules.
567  */
568 {
569  HDR hdr;
570  MACHINE *m;
571  ROLE my_role;
572  _PDU_TYPE_ pdu_type;
573  boolean successful;
574  /*------------------------------------------------------------*/
575 
576  REPORT_VERSION_ONCE;
577 
578  /*------------------*/
579  /* Validate the pdu */
580  /*------------------*/
581 
582  if (!pdu__is_this_pdu_acceptable (&pdu))
583  return (0);
584 
585  pdu_type = pdu__what_type (&pdu);
586  if (pdu_type == _DONT_KNOW_)
587  /* PDUs such as 'Keepalive' are simply ignored */
588  {
589  d_msg__ ("cfdp_engine: ignored incoming PDU of unknown type.\n");
590  return (1);
591  }
592  hdr = pdu__hdr_struct_from_pdu (&pdu);
593  my_role = pdu__receiver_role_from_pdu_hdr (hdr);
594  /* If we are the sender of the file, there are some pdu-types that our
595  * partner should never send (for example, EOF). If we are the receiver
596  * of the file, there are other pdu-types that our partner should never
597  * send (for example, Ack-EOF).
598  */
599  if (!m__is_pdu_sane (pdu_type, my_role))
600  return (0);
601 
602  /* Perhaps put out some debug messages */
603  if (cfdp_is_message_class_enabled (CFDP_MSG_DEBUG_PDU))
604  {
605  pdu__display_raw_pdu (pdu);
606  pdu__display_disassembled_pdu (pdu);
607  }
608 
609  /* Validation was successful; assume the response will also be successful
610  * unless/until proven otherwise.
611  */
612  successful = YES;
613 
614  /*----------------------------------------------------------------*/
615  /* If an existing state machine is assigned to this transaction, */
616  /* pass the PDU to that state machine. */
617  /*----------------------------------------------------------------*/
618 
619  if ((m = machine_list__get_this_trans (hdr.trans)) != NULL)
620  {
621  /* Conceptually, re-starting the inactivity-timer belongs in each
622  * of the state tables, but it's done here for practical purposes
623  * (it would be duplicated in 20+ places otherwise).
624  */
625  if (!m->publik.frozen && !m->publik.suspended)
626  m->inactivity_timer = timer__start (mib__inactivity_timeout(NULL));
627  m__give_pdu_to_trans (m, &pdu);
628  }
629 
630  /*----------------------------------------------------------------*/
631  /* If the incoming PDU is a MD/FD/EOF for a Receiver, then create */
632  /* a new state machine and pass the PDU to that state machine. */
633  /*----------------------------------------------------------------*/
634 
635  else if (((pdu_type == _MD_) || (pdu_type == _FD_) || (pdu_type == _EOF_)) &&
636  ((my_role == R_1) || (my_role == R_2)))
637  {
638 
639  if ((m = m__start_new_receiver_machine (hdr)) != NULL)
640  m__give_pdu_to_trans (m, &pdu);
641  else
642  successful = NO;
643  }
644 
645  /*--------------------------------------------------------------------*/
646  /* If the incoming PDU is a Finished PDU for a Class 2 Sender, then */
647  /* create a new state machine and pass the PDU to that state machine. */
648  /*--------------------------------------------------------------------*/
649 
650  else if ((pdu_type == _FIN_) && (my_role == S_2))
651  {
652  if ((m = m__restart_old_sender_machine (hdr)) != NULL)
653  m__give_pdu_to_trans (m, &pdu);
654  else
655  successful = NO;
656  }
657 
658  /*----------------------------*/
659  /* Any other PDUs are ignored */
660  /*----------------------------*/
661

673 /*=r=************************************************************************/
674 void cfdp_cycle_each_transaction (void)
675  /* NOTE: In order to ensure that outgoing File-Directives get a
676  * higher priority than outgoing File-Data, it is necessary to
677  * loop through all the state machines twice.
678  */
679 {
680  MACHINE *m;
681  /*------------------------------------------------------------*/
682 
683  REPORT_VERSION_ONCE;
684 
685  /*--------------------------------------------------------*/
686  /* Give each machine a chance to do higher-priority stuff */
687  /*--------------------------------------------------------*/
688 
689  machine_list__open_a_walk ();
690  while ((m=machine_list__get_next_machine()) != NULL)
691  {
692 
693  /*------------------------------------------------------*/
694  /* If a protocol timer has expired, initiate a response */
695  /*------------------------------------------------------*/
696 
697  if (timer__expired (&m->inactivity_timer))
698  /* The inactivity-timer expired */
699  {
700  timer__cancel (&m->inactivity_timer);
701  m__state_table (m, INACTIVITY_TIMER_EXPIRED, NULL, NULL);
702  }
703 
704  else if (timer__expired (&m->ack_timer))
705  /* The ack-timer expired */
706  {
707  timer__cancel (&m->ack_timer);
708  m__state_table (m, ACK_TIMER_EXPIRED, NULL, NULL);
709  }
710 
711  else if (timer__expired (&m->nak_timer))
712  /* The nak-timer expired */
713  {
714  timer__cancel (&m->nak_timer);
715  m__state_table (m, NAK_TIMER_EXPIRED, NULL, NULL);
716  }
717 
718  /*-------------------------------------------------------------*/
719  /* Special case -- support high-speed (external) file transfer */
720  /*-------------------------------------------------------------*/
721  else if (m->should_external_xfer_be_closed)
722  {
723  m__state_table (m, EXTERNAL_FILE_TRANSFER_COMPLETED, NULL, NULL);
724  m->should_external_xfer_be_closed = NO;
725  }
726 
727  /*------------------------------------------------------------*/
728  /* Allow up to one outgoing File-Directive PDU to be released */
729  /* (Class 1 Receiver does not output PDUs) */
730  /*------------------------------------------------------------*/
731  else if (m->publik.role != R_1)
732  m__state_table (m, THROTTLE_OUTGOING_FILE_DIR_PDU, NULL, NULL);
733  }
734 
735  /*---------------------------------------------------*/
736  /* Then allow each machine to output a File-Data PDU */
737  /*---------------------------------------------------*/
738 
739  machine_list__open_a_walk ();
740  while ((m=machine_list__get_next_machine()) != NULL)
741  {
742  /* (Only Senders output File-Data) */
743  if ((m->publik.role == S_2) || (m->publik.role == S_1))
744  m__state_table (m, THROTTLE_OUTGOING_FILE_DATA_PDU, NULL, NULL);
745  }
746 }
