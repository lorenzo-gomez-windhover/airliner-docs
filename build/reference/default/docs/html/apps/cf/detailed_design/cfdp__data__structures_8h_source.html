322 /**************************/
323 /*** Transaction status ***/
324 /**************************/
325 
326 /* This info applies to a single transaction (each transaction has its
327  * own status info). This info may be useful for reporting purposes.
328  * To get general information about the transaction:
329  * 1) Look at 'trans' to determine the transaction-id.
330  * 2) Look at 'role' to determine the role of the local entity
331  * (e.g. Class 2 Sender)
332  * 3) Look inside 'md' to determine the source and destination files.
333  * (But, if local entity is the receiver, 'md' will not contain
334  * real info until 'has_md_been_received' is set)
335  * 4) Look at 'start_time' to determine when the transaction started.
336  * To see if the transaction has finished:
337  * 1) Look at 'finished'. If it is set, the transaction is over;
338  * otherwise, the transaction is still in-progress.
339  * To monitor an in-progress transaction:
340  * 1) Look at 'frozen' to see if the transaction is frozen.
341  * 2) Look at 'suspended' to see if the transaction is suspended.
342  * 3) Look at 'cancelled' to see if the transaction is being cancelled.
343  * 3a) If the transaction is being cancelled, look at 'condition_code'
344  * to determine why (e.g. 'Nak limit reached').
345  * 4) Look at 'phase' to see what phase the transaction is in.
346  * 5) Look at 'how_many_naks' to see how many Nak PDUs have been
347  * sent (by the Receiver) or received (by the Sender).
348  * 6) Call one or more of the 'cfdp_transaction_...' routines
349  * to get transaction-progress, see what timers are running, etc.
350  * To determine the outcome of a transaction that has finished:
351  * 1) Look at 'final_status' (should be either successful, cancelled,
352  * or abandoned).
353  * 2) If the final status is not successful, look at 'condition_code'
354  * to find out why (e.g. 'Nak limit reached').
355  * Note: If the transaction was abandoned due to a local User Request
356  * to abandon, then the condition-code will be 'no-error'.
357  */
358 
359 typedef struct
360 {
361  TRANSACTION trans; /* This identifies the transaction */
362  boolean abandoned:1; /* Has this transaction been abandoned? */
363  u_int_4 attempts; /* How many attempts to send current PDU? */
364  boolean cancelled:1; /* Has this transaction been cancelled? */
365  CONDITION_CODE condition_code; /* See the 'Blue Book' */
366  DELIVERY_CODE delivery_code; /* See the 'Blue Book' */
367  boolean external_file_xfer:1; /* Is engine user sending Filedata? */
368  u_int_4 fd_offset; /* Offset of last Filedata sent/received */
369  u_int_4 fd_length; /* Length of last Filedata sent/received */
370  u_int_4 file_checksum_as_calculated;
371  FINAL_STATUS final_status;
372  boolean finished:1; /* Has this trans finished? */
373  boolean frozen:1; /* Is this transaction currently frozen? */
374  boolean has_md_been_received:1;
375  u_int_1 how_many_naks; /* How many Nak PDUs have been sent/recd? */
376  boolean is_this_trans_solely_for_ack_fin:1; /* re-start of old trans */
377  MD md; /* Metadata; includes dest file name */
378  ID partner_id; /* Who is this transaction with? */
379  u_int_1 phase; /* Either 1, 2, 3, or 4 */
380  u_int_4 received_file_size;
381  ROLE role; /* (e.g. Receiver Class 1) */
382  STATE state;
383  time_t start_time; /* When was this transaction started? */
384  boolean suspended:1; /* Is this trans currently suspended? */
385  char temp_file_name [MAX_TEMP_FILE_NAME_LENGTH+1];
386 } TRANS_STATUS;
