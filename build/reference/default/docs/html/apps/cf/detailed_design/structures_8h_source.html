14  * SUMMARY: (PDU = Protocol Data Unit; i.e. a CFDP message)
15  * This file defines a single data structure that can hold any CFDP PDU,
16  * plus data structures to represent various CFDP concepts.
17  * Each PDU contains a header plus data, where data is either
18  * metadata (MD), file-data (FD), EOF, ack, nak, etc. I have defined
19  * one structure to hold the pdu-header, structures to hold what follows
20  * the pdu-header (i.e. one structure each for metadata, file-data,
21  * EOF, etc), and one "bottom-line" structure that can contain
22  * any CFDP PDU. (Plus there are "building block" data-types that go
23  * inside PDUs; e.g. the CFDP Condition Code)
24  * PURPOSE: These definitions exist in the hope that their use allows
25  * the CFDP source code to be simpler (and easier to read).
26  * The intent is that each incoming PDU is immediately converted to a
27  * structure (which allows easier access to fields within the PDU).
28  * Similarly, each outgoing PDU is generated using the structure, and is
29  * converted to bits just prior to release.
30  * CHANGES:
31  * 2007_07_31 Tim Ray (TR)
32  * - Official release. (see pre_release_changes.log for earlier changes)
33  * 2007_11_19 TR
34  * - Bug fix. 'max_pdu_header_length' calculation was fixed.
35  * (Previously, assumption was that entity-id length = trans-num length)
36  */

130 /* In order for CFDP to reliably deliver files, the Receiver must keep
131  * track of gaps in the received file data (and report those gaps to
132  * the Sender).
133  */

136 struct gap

139  u_int_4 end;
140  struct gap *next;
141 } ;

144 /* If there are lots of gaps, they may not all fit in one NAK PDU.
145  * This next macro defines the max gaps that will fit in one Nak PDU.
146  * To understand the calculation below:
147  * 1) The part of a CFDP PDU that follows the PDU-header is called the
148  * 'PDU data field'.
149  * 2) The PDU data-field for a Nak PDU has 9 bytes, plus 8 bytes per gap.
150  * (Byte #1 specifies that this is a Nak PDU)
151  * (Bytes #2-9 specify the start-of-scope and end-of-scope;
152  * i.e. to what portion of the file does this Nak apply?)
153  * (Each additional 8 bytes specify start-offset and end-offset of a gap)
154  */
155 #define MAX_GAPS_PER_NAK_PDU ((MAX_PDU_DATA_FIELD_LENGTH - 9) / 8)

159 /* For outgoing PDUs, the transaction-sequence-number field is hard-coded.
160  * Strictly speaking, this violates the CFDP standard because this field
161  * is variable-length, and up to 8 bytes long.
162  * However, a 4-byte counter can accomodate one new transaction every second
163  * for over 100 years before rolling over (i.e. 100 years worth of
164  * transactions will each have a unique ID).
165  */
