96 /* Background: The entity-id identifies a particular CFDP entity.
97  * A typical Goddard mission will have 2 CFDP entities - one on the spacecraft
98  * and one on the ground.
99  * This parameter tells the library how many bytes to allow for storage of
100  * entity-ids. CFDP allows the entity-id to be anywhere from 1 to 8 bytes
101  * in length. The safest value for this parameter is 8, but if memory
102  * usage is an issue, it can be set to a smaller value (if there are only two
103  * entities, a one-byte entity-id is sufficient).
104  */
105 #define MAX_ID_LENGTH 2
106 
107 /* This parameter indicates the amount of storage required (in bytes) to
108  * hold the largest chunk of file-data that will ever be contained in a
109  * single incoming or outgoing Filedata PDU.
110  * (CFDP limits this to just under 64K bytes, but any particular mission
111  * may have a further limitation).
112  */
113 #define MAX_FILE_CHUNK_SIZE CF_MAX_OUTGOING_CHUNK_SIZE
114 
115 /* This parameter indicates the amount of storage required (in bytes) to hold

120  */

253 /* An assertion fires when the library encounters a situation that
254  * "can't ever happen" -- it usually indicates a bug in the program,
255  * and a program crash is likely. The user of the CFDP library is
256  * free to decide what response to take in response to assertions.
257  * The intent is to use assertions to help find bugs during testing.
258  * Once CFDP is operational, they can be 'disabled' (i.e. they still fire,
259  * but the program is not exited).
260  * Note: Other than through assertions, the CFDP library does not ever
261  * perform an exit.
262  */
263 
264 /* Set this parameter to '1' for 'yes' or '0' for 'no'. */
265 #define IS_ASSERT_ALLOWED_TO_EXIT 1
266 
267 #define ASSERT__(s) \
268 { \
269  if (!(s)) \
270  { \
271  msg__ ("Assertion failed in %s at line %u.\n", __func__, __LINE__); \
272  if (IS_ASSERT_ALLOWED_TO_EXIT) \
273  abort (); \
274  } \
275 }
276 
277 #define ASSERT__BUG_DETECTED \
278 { \
279  msg__ ("Bug detected in %s at line %u.\n", __func__, __LINE__); \
280  if (IS_ASSERT_ALLOWED_TO_EXIT) \
281  abort (); \
282 } \
