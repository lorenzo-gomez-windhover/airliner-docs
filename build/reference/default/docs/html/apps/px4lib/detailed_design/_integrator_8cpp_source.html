91 Integrator::put(uint64 timestamp, math::Vector3F &val, math::Vector3F &integral, uint64 &integral_dt)
92 {
93  if (0 == _last_integration_time)
94  {
95  /* this is the first item in the integrator */
96  _last_integration_time = timestamp;
97  _last_reset_time = timestamp;
98  _last_val = val;
99 
100  return FALSE;
101  }
102 
103  double dt = 0.0;
104 
105  // Integrate:
106  // Leave dt at 0 if the integration time does not make sense.
107  // Without this check the integral is likely to explode.
108  if (timestamp >= _last_integration_time)
109  {
110  dt = (double)(timestamp - _last_integration_time) / 1000000.0;
111  }
112 
113  // Use trapezoidal integration to calculate the delta integral
114  math::Vector3F delta_alpha = (val + _last_val) * dt * 0.5f;
115  _last_val = val;
116 
117  // Calculate coning corrections if required
118  if (TRUE == _coning_comp_on)
119  {
120  // Coning compensation derived by Paul Riseborough and Jonathan Challinger,
121  // following:
122  // Tian et al (2010) Three-loop Integration of GPS and Strapdown INS with Coning and Sculling Compensation
123  // Sourced: http://www.sage.unsw.edu.au/snap/publications/tian_etal2010b.pdf
124  // Simulated: https://github.com/priseborough/InertialNav/blob/master/models/imu_error_modelling.m
125  _beta = _beta + ((_last_alpha + _last_delta_alpha * (1.0f / 6.0f)) % delta_alpha) * 0.5f;
126  _last_delta_alpha = delta_alpha;
127  _last_alpha = _alpha;
128  }
129 
130  // accumulate delta integrals
131  _alpha = _alpha + delta_alpha;
132  _last_integration_time = timestamp;
133 
134  // Only do auto reset if auto reset interval is not 0.
135  if (_auto_reset_interval > 0 && (timestamp - _last_reset_time) > _auto_reset_interval)
136  {
137  // apply coning corrections if required
138  if (TRUE == _coning_comp_on)
139  {
140  integral = _alpha + _beta;
141  }
142  else
143  {
144  integral = _alpha;
145  }
146  // reset the integrals and coning corrections
147  _reset(integral_dt);
148 
149  return TRUE;
150  }
151  else
152  {
153  return FALSE;
154  }
155 }
