79 unsigned get_lookup_table_index(float *val, float min, float max)
80 {
81  /* for the rare case of hitting the bounds exactly
82  * the rounding logic wouldn't fit, so enforce it.
83  */
84  /* limit to table bounds - required for maxima even when table spans full globe range */
85  if (*val < min) {
86  *val = min;
87  }
88 
89  /* limit to (table bounds - 1) because bilinear interpolation requires checking (index + 1) */
90  if (*val > max) {
91  *val = max - SAMPLING_RES;
92  }
93 
94  return (-(min) + *val) / SAMPLING_RES;
95 }

97 float get_mag_declination(float lat, float lon)
98 {
99  /*
100  * If the values exceed valid ranges, return zero as default
101  * as we have no way of knowing what the closest real value
102  * would be.
103  */
104  if (lat < -90.0f || lat > 90.0f ||
105  lon < -180.0f || lon > 180.0f) {
106  return 0.0f;
107  }
108 
109  /* round down to nearest sampling resolution */
110  float min_lat = (int)(lat / SAMPLING_RES) * SAMPLING_RES;
111  float min_lon = (int)(lon / SAMPLING_RES) * SAMPLING_RES;
112 
113  /* find index of nearest low sampling point */
114  unsigned min_lat_index = get_lookup_table_index(&min_lat, SAMPLING_MIN_LAT, SAMPLING_MAX_LAT);
115  unsigned min_lon_index = get_lookup_table_index(&min_lon, SAMPLING_MIN_LON, SAMPLING_MAX_LON);
116 
117  float declination_sw = get_lookup_table_val(min_lat_index, min_lon_index);
118  float declination_se = get_lookup_table_val(min_lat_index, min_lon_index + 1);
119  float declination_ne = get_lookup_table_val(min_lat_index + 1, min_lon_index + 1);
120  float declination_nw = get_lookup_table_val(min_lat_index + 1, min_lon_index);
121 
122  /* perform bilinear interpolation on the four grid corners */
123  float lat_scale = constrain((lat - min_lat) / SAMPLING_RES, 0.0f, 1.0f);
124  float lon_scale = constrain((lon - min_lon) / SAMPLING_RES, 0.0f, 1.0f);
125 
126  float declination_min = lon_scale * (declination_se - declination_sw) + declination_sw;
127  float declination_max = lon_scale * (declination_ne - declination_nw) + declination_nw;
128 
129  return lat_scale * (declination_max - declination_min) + declination_min;
130 }
