49 void LowPassFilter2p::set_cutoff_frequency(float sample_freq, float cutoff_freq)
50 {
51  _cutoff_freq = cutoff_freq;
52 
53  if (_cutoff_freq <= 0.0f) {
54  // no filtering
55  return;
56  }
57 
58  float fr = sample_freq / _cutoff_freq;
59  float ohm = tanf(M_PI_F / fr);
60  float c = 1.0f + 2.0f * cosf(M_PI_F / 4.0f) * ohm + ohm * ohm;
61  _b0 = ohm * ohm / c;
62  _b1 = 2.0f * _b0;
63  _b2 = _b0;
64  _a1 = 2.0f * (ohm * ohm - 1.0f) / c;
65  _a2 = (1.0f - 2.0f * cosf(M_PI_F / 4.0f) * ohm + ohm * ohm) / c;
66 }

68 float LowPassFilter2p::apply(float sample)
69 {
70  if (_cutoff_freq <= 0.0f) {
71  // no filtering
72  return sample;
73  }
74 
75  // do the filtering
76  float delay_element_0 = sample - _delay_element_1 * _a1 - _delay_element_2 * _a2;
77 
78  //if (!PX4_ISFINITE(delay_element_0)) {
79  if (!std::isfinite(delay_element_0)) {
80  // don't allow bad values to propagate via the filter
81  delay_element_0 = sample;
82  }
83 
84  float output = delay_element_0 * _b0 + _delay_element_1 * _b1 + _delay_element_2 * _b2;
85 
86  _delay_element_2 = _delay_element_1;
87  _delay_element_1 = delay_element_0;
88 
89  // return the value. Should be no need to check limits
90  return output;
91 }

93 float LowPassFilter2p::reset(float sample)
94 {
95  float dval = sample / (_b0 + _b1 + _b2);
96  _delay_element_1 = dval;
97  _delay_element_2 = dval;
