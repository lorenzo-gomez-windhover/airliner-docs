412 {
413  int32 iStatus = CFE_SUCCESS;
414  CFE_SB_Msg_t* CmdMsgPtr=NULL;
415  CFE_SB_MsgId_t CmdMsgId;
416 
417  /* Process command messages until the pipe is empty */
418  while (1)
419  {
420  iStatus = CFE_SB_RcvMsg(&CmdMsgPtr, CmdPipeId, CFE_SB_POLL);
421  if(iStatus == CFE_SUCCESS)
422  {
423  CmdMsgId = CFE_SB_GetMsgId(CmdMsgPtr);
424  switch (CmdMsgId)

426  case SENS_CMD_MID:
427  ProcessAppCmds(CmdMsgPtr);
428  break;
429 
430  default:
431  /* Bump the command error counter for an unknown command.

434  HkTlm.usCmdErrCnt++;
435  (void) CFE_EVS_SendEvent(SENS_MSGID_ERR_EID, CFE_EVS_ERROR,
436  "Recvd invalid CMD msgId (0x%04X)", (unsigned short)CmdMsgId);
437  break;
438  }
439  }
440  else if (iStatus == CFE_SB_NO_MESSAGE)
441  {
442  break;
443  }
444  else
445  {
446  (void) CFE_EVS_SendEvent(SENS_RCVMSG_ERR_EID, CFE_EVS_ERROR,
447  "CMD pipe read error (0x%08lX)", iStatus);

775 void SENS::ProcessRCInput(void)
776 {
777  if(CVT.InputRcMsg.Timestamp > CVT.LastInputRcTime)
778  {
779  /* Read low-level values from FMU or IO RC inputs (PPM, Spektrum, S.Bus) */
780  //struct rc_input_values rc_input;
781 
782  /* Detect RC signal loss */
783  bool signal_lost;
784 
785  /* Check flags and require at least four channels to consider the signal valid */
786  if (CVT.InputRcMsg.RcLost || CVT.InputRcMsg.RcFailsafe || CVT.InputRcMsg.ChannelCount < 4)
787  {
788  /* Signal is lost or no enough channels */
789  signal_lost = true;
790  }
791  else
792  {
793  /* Signal looks good */
794  signal_lost = false;
795 
796  /* Check failsafe */
797  int8 fs_ch = RcChannelsMsg.Function[ConfigTblPtr->MapFailsafe];
798 
799  /* If not 0, use channel number instead of rc.function mapping */
800  if (ConfigTblPtr->MapFailsafe > 0)
801  {
802  fs_ch = ConfigTblPtr->MapFailsafe - 1;
803  }
804 
805  if (ConfigTblPtr->FailsThr > 0 && fs_ch >= 0)
806  {
807  /* Failsafe configured */
808  if ((ConfigTblPtr->FailsThr < ConfigTblPtr->Min[fs_ch] && CVT.InputRcMsg.Values[fs_ch] < ConfigTblPtr->FailsThr) ||
809  (ConfigTblPtr->FailsThr > ConfigTblPtr->Max[fs_ch] && CVT.InputRcMsg.Values[fs_ch] > ConfigTblPtr->FailsThr))
810  {
811  /* Failsafe triggered, signal is lost by receiver */
812  signal_lost = true;
813  }
814  }
815  }
816 
817  uint32 channel_limit = CVT.InputRcMsg.ChannelCount;
818 
819  if (channel_limit > PX4_RC_CHANNELS_FUNCTION_COUNT) {
820  channel_limit = PX4_RC_CHANNELS_FUNCTION_COUNT;
821  }
822 
823  /* Read out and scale values from raw message even if signal is invalid */
824  for (uint32 i = 0; i < channel_limit; i++)
825  {
826  /*
827  * 1) Constrain to min/max values, as later processing depends on bounds.
828  */
829  if (CVT.InputRcMsg.Values[i] < ConfigTblPtr->Min[i]) {
830  CVT.InputRcMsg.Values[i] = ConfigTblPtr->Min[i];
831  }
832 
833  if (CVT.InputRcMsg.Values[i] > ConfigTblPtr->Max[i]) {
834  CVT.InputRcMsg.Values[i] = ConfigTblPtr->Max[i];
835  }
836 
837  /*
838  * 2) Scale around the mid point differently for lower and upper range.
839  *
840  * This is necessary as they don't share the same endpoints and slope.
841  *
842  * First normalize to 0..1 range with correct sign (below or above center),
843  * the total range is 2 (-1..1).
844  * If center (trim) == min, scale to 0..1, if center (trim) == max,
845  * scale to -1..0.
846  *
847  * As the min and max bounds were enforced in step 1), division by zero
848  * cannot occur, as for the case of center == min or center == max the if
849  * statement is mutually exclusive with the arithmetic NaN case.
850  *
851  * DO NOT REMOVE OR ALTER STEP 1!
852  */
853  if (CVT.InputRcMsg.Values[i] > (ConfigTblPtr->Trim[i] + ConfigTblPtr->DZ[i]))
854  {
855  RcChannelsMsg.Channels[i] = (CVT.InputRcMsg.Values[i] - ConfigTblPtr->Trim[i] - ConfigTblPtr->DZ[i]) / (float)(
856  ConfigTblPtr->Max[i] - ConfigTblPtr->Trim[i] - ConfigTblPtr->DZ[i]);
857 
858  RcChannelsMsg.ChannelCount = CVT.InputRcMsg.ChannelCount;
859  RcChannelsMsg.RSSI = CVT.InputRcMsg.RSSI;
860  RcChannelsMsg.SignalLost = signal_lost;
861  RcChannelsMsg.Timestamp = CVT.InputRcMsg.Timestamp;
862  RcChannelsMsg.TimestampLastValid = CVT.InputRcMsg.LastSignal;
863  RcChannelsMsg.FrameDropCount = CVT.InputRcMsg.RcLostFrameCount;
864  }
865  else if (CVT.InputRcMsg.Values[i] < (ConfigTblPtr->Trim[i] - ConfigTblPtr->DZ[i]))
866  {
867  RcChannelsMsg.Channels[i] = (CVT.InputRcMsg.Values[i] - ConfigTblPtr->Trim[i] + ConfigTblPtr->DZ[i]) / (float)(
868  ConfigTblPtr->Trim[i] - ConfigTblPtr->Min[i] - ConfigTblPtr->DZ[i]);
869 
870  }
871  else
872  {
873  /* in the configured dead zone, output zero */
874  RcChannelsMsg.Channels[i] = 0.0f;
875  }
876 
877  RcChannelsMsg.Channels[i] *= ConfigTblPtr->Rev[i];
878 
879  /* handle any parameter-induced blowups */
880  if (!isfinite(RcChannelsMsg.Channels[i])) {
881  RcChannelsMsg.Channels[i] = 0.0f;
882  }
883  }
884 
885  RcChannelsMsg.ChannelCount = CVT.InputRcMsg.ChannelCount;
886  RcChannelsMsg.RSSI = CVT.InputRcMsg.RSSI;
887  RcChannelsMsg.SignalLost = signal_lost;
888  RcChannelsMsg.Timestamp = CVT.InputRcMsg.LastSignal;
889  RcChannelsMsg.FrameDropCount = CVT.InputRcMsg.RcLostFrameCount;

891  /* Publish rc_channels topic even if signal is invalid, for debug */
892  SendRcChannelsMsg();
893 
894  /* Only publish manual control if the signal is still present and was present once */
895  if (!signal_lost && CVT.InputRcMsg.LastSignal > 0)
896  {
897  /* Set mode slot to unassigned */
898  ManualControlSetpointMsg.ModeSlot = PX4_MODE_SLOT_NONE;
899 
900  /* Set the timestamp to the last signal time */
901  ManualControlSetpointMsg.Timestamp = CVT.InputRcMsg.LastSignal;
902  ManualControlSetpointMsg.DataSource = PX4_MANUAL_CONTROL_SOURCE_RC;
903 
904  /* Limit controls */
905  ManualControlSetpointMsg.Y = GetRcValue(PX4_RC_CHANNELS_FUNCTION_ROLL, -1.0, 1.0);
906  ManualControlSetpointMsg.X = GetRcValue(PX4_RC_CHANNELS_FUNCTION_PITCH, -1.0, 1.0);
907  ManualControlSetpointMsg.R = GetRcValue(PX4_RC_CHANNELS_FUNCTION_YAW, -1.0, 1.0);
