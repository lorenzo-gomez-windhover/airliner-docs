96 void CS_ReportBaselineAppCmd(CFE_SB_MsgPtr_t MessagePtr)
97 {
98  /* command verification variables */
99  uint16 ExpectedLength = sizeof(CS_AppNameCmd_t);
100 
101  CS_AppNameCmd_t * CmdPtr;
102  CS_Res_App_Table_Entry_t * ResultsEntry;
103  uint32 Baseline;
104 
105  /* Verify command packet length */
106  if ( CS_VerifyCmdLength (MessagePtr,ExpectedLength) )
107  {
108  CmdPtr = (CS_AppNameCmd_t *) MessagePtr;
109 
110  CmdPtr -> Name[OS_MAX_API_NAME - 1] = '\0';
111  if ( CS_GetAppResTblEntryByName(&ResultsEntry, CmdPtr -> Name))
112  {
113  if (ResultsEntry -> ComputedYet == TRUE)
114  {
115  Baseline = ResultsEntry -> ComparisonValue;
116  CFE_EVS_SendEvent (CS_BASELINE_APP_INF_EID,
117  CFE_EVS_INFORMATION,
118  "Report baseline of app %s is 0x%08X",
119  CmdPtr -> Name,
120  (unsigned int)Baseline);
121  }
122  else
123  {
124  CFE_EVS_SendEvent (CS_NO_BASELINE_APP_INF_EID,
125  CFE_EVS_INFORMATION,
126  "Report baseline of app %s has not been computed yet",
127  CmdPtr -> Name);
128  }
129  CS_AppData.CmdCounter++;
130  }
131  else
132  {
133  CFE_EVS_SendEvent (CS_BASELINE_INVALID_NAME_APP_ERR_EID,
134  CFE_EVS_ERROR,
135  "App report baseline failed, app %s not found",

137  CS_AppData.CmdErrCounter++;
138  }
139  }
140  return;
141 } /* End of CS_ReportBaselineAppCmd () */

147 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
148 void CS_RecomputeBaselineAppCmd (CFE_SB_MsgPtr_t MessagePtr)
149 {
150  /* command verification variables */
151  uint16 ExpectedLength = sizeof(CS_AppNameCmd_t);
152 
153  uint32 ChildTaskID;
154  int32 Status;
155  CS_AppNameCmd_t * CmdPtr;
156  CS_Res_App_Table_Entry_t * ResultsEntry;
157 
158  /* Verify command packet length */
159 
160  if ( CS_VerifyCmdLength (MessagePtr,ExpectedLength) )
161  {
162  CmdPtr = (CS_AppNameCmd_t *) MessagePtr;
163 
164  if (CS_AppData.RecomputeInProgress == FALSE && CS_AppData.OneShotInProgress == FALSE)
165  {
166 
167  /* make sure the entry is a valid number and is defined in the table */
168 
169  CmdPtr -> Name[OS_MAX_API_NAME - 1] = '\0';
170 
171  if (CS_GetAppResTblEntryByName(&ResultsEntry, CmdPtr -> Name))
172  {
173  /* There is no child task running right now, we can use it*/
174  CS_AppData.RecomputeInProgress = TRUE;
175 
176  /* fill in child task variables */
177  CS_AppData.ChildTaskTable = CS_APP_TABLE;
178 
179 
180  CS_AppData.RecomputeAppEntryPtr = ResultsEntry;
181 
182  Status= CFE_ES_CreateChildTask(&ChildTaskID,
183  CS_RECOMP_APP_TASK_NAME,
184  CS_RecomputeAppChildTask,
185  NULL,
186  CFE_ES_DEFAULT_STACK_SIZE,
187  CS_CHILD_TASK_PRIORITY,
188  CS_CHILD_TASK_FLAGS);
189  if (Status ==CFE_SUCCESS)
190  {
191  CFE_EVS_SendEvent (CS_RECOMPUTE_APP_STARTED_DBG_EID,
192  CFE_EVS_DEBUG,
193  "Recompute baseline of app %s started",
194  CmdPtr -> Name);
195  CS_AppData.CmdCounter++;
196  }
197  else/* child task creation failed */
198  {
199  CFE_EVS_SendEvent (CS_RECOMPUTE_APP_CREATE_CHDTASK_ERR_EID,
200  CFE_EVS_ERROR,
201  "Recompute baseline of app %s failed, CFE_ES_CreateChildTask returned: 0x%08X",
202  CmdPtr -> Name,
203  (unsigned int)Status);
204  CS_AppData.CmdErrCounter++;
205  CS_AppData.RecomputeInProgress = FALSE;
206  }
207  }
208  else
209  {
210  CFE_EVS_SendEvent (CS_RECOMPUTE_UNKNOWN_NAME_APP_ERR_EID,
211  CFE_EVS_ERROR,
212  "App recompute baseline failed, app %s not found",
213  CmdPtr -> Name);
214  CS_AppData.CmdErrCounter++;
215  }
216  }
217  else
218  {
219  /*send event that we can't start another task right now */
220  CFE_EVS_SendEvent (CS_RECOMPUTE_APP_CHDTASK_ERR_EID,
221  CFE_EVS_ERROR,
222  "App recompute baseline for app %s failed: child task in use",
223  CmdPtr -> Name);
224  CS_AppData.CmdErrCounter++;
225  }
226  }
227  return;
228 }/* end CS_RecomputeBaselineAppCmd */
