497 /* Background check cFE core */
498 /* */
499 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
500 boolean CS_BackgroundCfeCore(void)
501 {
502  CS_Res_EepromMemory_Table_Entry_t * ResultsEntry;
503  boolean DoneWithCycle = FALSE;
504  boolean DoneWithEntry = FALSE;
505  uint32 ComputedCSValue = 0;
506  int32 Status;
507 
508  if( CS_AppData.CfeCoreCSState == CS_STATE_ENABLED)
509  {
510  ResultsEntry = & CS_AppData.CfeCoreCodeSeg;
511 
512  /* Check if the code segment state is enabled (could be disabled for
513  a ground-commanded recompute) */
514  if (ResultsEntry->State == CS_STATE_ENABLED)
515  {
516 
517  /* If we complete an entry's checksum, this function will update it for us */
518  Status = CS_ComputeEepromMemory (ResultsEntry, &ComputedCSValue, &DoneWithEntry);
519 
520  /* We need to avoid the case of finishing a table, moving on to the next one
521  and computing an entry in that table, since it could put us above the
522  maximum bytes per cycle */
523  DoneWithCycle = TRUE;
524 
525  if(Status == CS_ERROR)
526  {
527  /* we had a miscompare */
528 
529  CS_AppData.CfeCoreCSErrCounter++;
530 
531  CFE_EVS_SendEvent (CS_CFECORE_MISCOMPARE_ERR_EID,
532  CFE_EVS_ERROR,
533  "Checksum Failure: cFE Core, Expected: 0x%08X, Calculated: 0x%08X",
534  (unsigned int)(ResultsEntry -> ComparisonValue),
535  (unsigned int)ComputedCSValue);
536  }
537 
538  if (DoneWithEntry == TRUE)
539  {
540  CS_AppData.CurrentEntryInTable ++;
541  }
542 
543  /* only one entry to do */
544  if ( CS_AppData.CurrentEntryInTable > 0)
545  {
546  /* We are done with this table */
547  CS_AppData.CfeCoreBaseline = ResultsEntry -> ComparisonValue;
548  CS_GoToNextTable();
549  }
550  }
551  else
552  {
553  /* Currently disabled, so skip it */
554  CS_GoToNextTable();
555  }
556  }
557  else /* Table is disabled, skip it */
558  {
559  CS_GoToNextTable();
560  }
561 
562  return DoneWithCycle;
563 }/* end CS_BackgroundCfeCore */

638 boolean CS_BackgroundEeprom(void)
639 {
640  CS_Res_EepromMemory_Table_Entry_t * StartOfResultsTable;
641  CS_Res_EepromMemory_Table_Entry_t * ResultsEntry;
642  boolean DoneWithCycle = FALSE;
643  boolean DoneWithEntry = FALSE;
644  uint32 ComputedCSValue = 0;
645  int32 Loop;
646  uint32 EntireEepromCS;
647  uint16 CurrEntry;
648  int32 Status;
649 
650  if (CS_AppData.EepromCSState == CS_STATE_ENABLED)
651  {
652  if(CS_FindEnabledEepromEntry( &CurrEntry) == TRUE)
653  {
654  /* we found an enabled entry to checksum */
655  StartOfResultsTable = CS_AppData.ResEepromTblPtr;
656  ResultsEntry = & StartOfResultsTable[ CurrEntry];
657 
658  /* If we complete an entry's checksum, this function will update it for us */
659  Status = CS_ComputeEepromMemory (ResultsEntry, &ComputedCSValue, &DoneWithEntry);
660 
661  /* We need to avoid the case of finishing a table, moving on to the next one
662  and computing an entry in that table, since it could put us above the
663  maximum bytes per cycle */
664  DoneWithCycle = TRUE;
665 
666  if(Status == CS_ERROR)
667  {
668  /* we had a miscompare */
669 
670  CS_AppData.EepromCSErrCounter++;
671 
672  CFE_EVS_SendEvent (CS_EEPROM_MISCOMPARE_ERR_EID,
673  CFE_EVS_ERROR,
674  "Checksum Failure: Entry %d in Eeprom Table, Expected: 0x%08X, Calculated: 0x%08X",
675  CurrEntry,
676  (unsigned int)(ResultsEntry -> ComparisonValue),
677  (unsigned int)ComputedCSValue);
678  }
679 
680  if (DoneWithEntry == TRUE)
681  {
682  CS_AppData.CurrentEntryInTable ++;
683  }
684 
685  if ( CS_AppData.CurrentEntryInTable >= CS_MAX_NUM_EEPROM_TABLE_ENTRIES)
686  {
687  /* Since we are done CS'ing the entire Eeprom table, update the baseline
688  number for telemetry */
689  EntireEepromCS = 0;
690  for (Loop = 0; Loop < CS_MAX_NUM_EEPROM_TABLE_ENTRIES; Loop++)
691  {
692  EntireEepromCS += CS_AppData.ResEepromTblPtr[Loop].ComparisonValue;
693  }
694 
695  CS_AppData.EepromBaseline = EntireEepromCS;
696 
697  /* We are done with this table */
698  CS_GoToNextTable();
699  }
700  }
701  else
702  {
703  /* If we don't have a full table, the above set of code won't get
704  executed, so we do it if there aren't any more full entries left */
705  if ( CS_AppData.CurrentEntryInTable >= CS_MAX_NUM_EEPROM_TABLE_ENTRIES)
706  {
707  /* Since we are done CS'ing the entire Eeprom table, update the baseline
708  number for telemetry */
709  EntireEepromCS = 0;
710  for (Loop = 0; Loop < CS_MAX_NUM_EEPROM_TABLE_ENTRIES; Loop++)
711  {
712  EntireEepromCS += CS_AppData.ResEepromTblPtr[Loop].ComparisonValue;
713  }
714  CS_AppData.EepromBaseline = EntireEepromCS;
715  }
716 
717  /* There are no enabled entries in this table */
718  CS_GoToNextTable();
719  }
720  }
721  else /* Table is disabled, skip it */
722  {
723  CS_GoToNextTable();
724  }
725 
726  return DoneWithCycle;
727 
728 }/* end CS_BackgroundEeprom */
