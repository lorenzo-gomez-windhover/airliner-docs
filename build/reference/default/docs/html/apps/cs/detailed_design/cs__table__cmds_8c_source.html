96 void CS_ReportBaselineTablesCmd(CFE_SB_MsgPtr_t MessagePtr)
97 {
98  /* command verification variables */
99  uint16 ExpectedLength = sizeof(CS_TableNameCmd_t);
100 
101  CS_TableNameCmd_t * CmdPtr;
102  CS_Res_Tables_Table_Entry_t * ResultsEntry;
103  uint32 Baseline;
104 
105  /* Verify command packet length */
106  if ( CS_VerifyCmdLength (MessagePtr,ExpectedLength) )
107  {
108  CmdPtr = (CS_TableNameCmd_t *) MessagePtr;
109 
110  CmdPtr -> Name[CFE_TBL_MAX_FULL_NAME_LEN - 1] = '\0';
111  if ( CS_GetTableResTblEntryByName(&ResultsEntry, CmdPtr -> Name))
112  {
113  if (ResultsEntry -> ComputedYet == TRUE)
114  {
115  Baseline = ResultsEntry -> ComparisonValue;
116  CFE_EVS_SendEvent (CS_BASELINE_TABLES_INF_EID,
117  CFE_EVS_INFORMATION,
118  "Report baseline of table %s is 0x%08X",
119  CmdPtr -> Name,
120  (unsigned int)Baseline);
121  }
122  else
123  {
124  CFE_EVS_SendEvent (CS_NO_BASELINE_TABLES_INF_EID,
125  CFE_EVS_INFORMATION,
126  "Report baseline of table %s has not been computed yet",
127  CmdPtr -> Name);
128  }
129  CS_AppData.CmdCounter++;
130  }
131  else
132  {
133  CFE_EVS_SendEvent (CS_BASELINE_INVALID_NAME_TABLES_ERR_EID,
134  CFE_EVS_ERROR,
135  "Tables report baseline failed, table %s not found",
136  CmdPtr -> Name);
137  CS_AppData.CmdErrCounter++;
138  }
139  }
140  return;
141 } /* End of CS_ReportBaselineTablesCmd () */

145 /* CS Recompute the baseline of an entry in the Tables table cmd */

147 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
148 void CS_RecomputeBaselineTablesCmd (CFE_SB_MsgPtr_t MessagePtr)
149 {
150  /* command verification variables */
151  uint16 ExpectedLength = sizeof(CS_TableNameCmd_t);
152 
153  uint32 ChildTaskID;
154  int32 Status;
155  CS_TableNameCmd_t * CmdPtr;
156  CS_Res_Tables_Table_Entry_t * ResultsEntry;
157 
158  /* Verify command packet length */
159 
160  if ( CS_VerifyCmdLength (MessagePtr,ExpectedLength) )
161  {
162  CmdPtr = (CS_TableNameCmd_t *) MessagePtr;
163 
164  if (CS_AppData.RecomputeInProgress == FALSE && CS_AppData.OneShotInProgress == FALSE)
165  {
166  /* make sure the entry is a valid number and is defined in the table */
167 
168  CmdPtr -> Name[CFE_TBL_MAX_FULL_NAME_LEN - 1] = '\0';
169 
170  if (CS_GetTableResTblEntryByName(&ResultsEntry, CmdPtr -> Name))
171  {
172  /* There is no child task running right now, we can use it*/
173  CS_AppData.RecomputeInProgress = TRUE;
174 
175  /* fill in child task variables */
176  CS_AppData.ChildTaskTable = CS_TABLES_TABLE;
177 
178 
179 
180  CS_AppData.RecomputeTablesEntryPtr = ResultsEntry;
181 
182  Status= CFE_ES_CreateChildTask(&ChildTaskID,
183  CS_RECOMP_TABLES_TASK_NAME,
184  CS_RecomputeTablesChildTask,
185  NULL,
186  CFE_ES_DEFAULT_STACK_SIZE,
187  CS_CHILD_TASK_PRIORITY,
188  CS_CHILD_TASK_FLAGS);
189  if (Status ==CFE_SUCCESS)
190  {
191  CFE_EVS_SendEvent (CS_RECOMPUTE_TABLES_STARTED_DBG_EID,
192  CFE_EVS_DEBUG,
193  "Recompute baseline of table %s started",
194  CmdPtr -> Name);
195  CS_AppData.CmdCounter++;
196  }
197  else/* child task creation failed */
198  {
199  CFE_EVS_SendEvent (CS_RECOMPUTE_TABLES_CREATE_CHDTASK_ERR_EID,
200  CFE_EVS_ERROR,
201  "Recompute baseline of table %s failed, CFE_ES_CreateChildTask returned: 0x%08X",
202  CmdPtr -> Name,
203  (unsigned int)Status);
204  CS_AppData.CmdErrCounter++;
205  CS_AppData.RecomputeInProgress = FALSE;
206  }
207  }
208  else
209  {
210  CFE_EVS_SendEvent (CS_RECOMPUTE_UNKNOWN_NAME_TABLES_ERR_EID,
211  CFE_EVS_ERROR,
212  "Tables recompute baseline failed, table %s not found",
213  CmdPtr -> Name);
214  CS_AppData.CmdErrCounter++;
215  }
216  }
217  else
218  {
219  /*send event that we can't start another task right now */
220  CFE_EVS_SendEvent (CS_RECOMPUTE_TABLES_CHDTASK_ERR_EID,
221  CFE_EVS_ERROR,
222  "Tables recompute baseline for table %s failed: child task in use",
223  CmdPtr -> Name);
224  CS_AppData.CmdErrCounter++;
225  }
226  }
227  return;
228 }/* end CS_RecomputeBaselineTablesCmd */
