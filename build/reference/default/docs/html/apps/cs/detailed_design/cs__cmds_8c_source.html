97 /* CS's background checksumming command */
98 /* */
99 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
100 void CS_BackgroundCheckCmd (CFE_SB_MsgPtr_t MessagePtr)
101 {
102  /* command verification variables */
103  uint16 ExpectedLength = sizeof(CS_NoArgsCmd_t);
104  boolean DoneWithCycle = FALSE;
105  boolean EndOfList = FALSE;
106  CFE_SB_MsgId_t MessageID;
107  uint16 CommandCode;
108  uint16 ActualLength = CFE_SB_GetTotalMsgLength(MessagePtr);
109 
110  /* Verify the command packet length */
111  if (ExpectedLength != ActualLength)
112  {
113  CommandCode = CFE_SB_GetCmdCode(MessagePtr);
114  MessageID= CFE_SB_GetMsgId(MessagePtr);
115 
116  CFE_EVS_SendEvent(CS_LEN_ERR_EID,
117  CFE_EVS_ERROR,
118  "Invalid msg length: ID = 0x%04X, CC = %d, Len = %d, Expected = %d",
119  MessageID,
120  CommandCode,
121  ActualLength,
122  ExpectedLength);
123  }
124  else
125  {
126  if (CS_AppData.ChecksumState == CS_STATE_ENABLED)
127  {
128  DoneWithCycle = FALSE;
129  EndOfList = FALSE;
130 
131  /* We check for end-of-list because we don't necessarily know the
132  order in which the table entries are defined, and we don't
133  want to keep looping through the list */
134 
135  while ((DoneWithCycle != TRUE) && (EndOfList != TRUE))
136  {
137  /* We need to check the current table value here because
138  it is updated (and possibly reset to zero) inside each
139  function called */
140  if (CS_AppData.CurrentCSTable >= (CS_NUM_TABLES - 1))
141  {
142  EndOfList = TRUE;
143  }
144 
145  /* Call the appropriate background function based on the current table
146  value. The value is updated inside each function */
147  switch (CS_AppData.CurrentCSTable)
148  {
149  case (CS_CFECORE):
150  DoneWithCycle = CS_BackgroundCfeCore();
151  break;
152 
153  case(CS_OSCORE):
154 
155  DoneWithCycle = CS_BackgroundOS();
156  break;
157 
158  case (CS_EEPROM_TABLE):
159  DoneWithCycle = CS_BackgroundEeprom();
160  break;
161 
162  case (CS_MEMORY_TABLE):
163  DoneWithCycle = CS_BackgroundMemory();
164  break;
165 
166  case (CS_TABLES_TABLE):
167  DoneWithCycle = CS_BackgroundTables();
168  break;
169 
170  case (CS_APP_TABLE):
171 
172  DoneWithCycle = CS_BackgroundApp();
173  break;
174 
175  /* default case in case CS_AppData.CurrentCSTable is some random bad value */
176  default:
177 
178  /* We are at the end of the line */
179  CS_AppData.CurrentCSTable = 0;
180  CS_AppData.CurrentEntryInTable = 0;
181  CS_AppData.PassCounter++;
182  DoneWithCycle = TRUE;
183  break;
184 
185 
186  }/* end switch */
187  } /* end while */
188  }
189  else
190  {
191  /* CS is disabled, Application-wide */
192  }
193  }
194  return;
195 } /* End of CS_BackgroundCheckCmd () */
