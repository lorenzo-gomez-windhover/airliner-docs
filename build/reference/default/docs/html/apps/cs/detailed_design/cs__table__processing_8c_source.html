42 int32 CS_ValidateEepromChecksumDefinitionTable (void * TblPtr)
43 {
44  int32 Result = CFE_SUCCESS;
45  int32 Status = OS_ERROR;
46  CS_Def_EepromMemory_Table_Entry_t * StartOfTable = NULL;
47  CS_Def_EepromMemory_Table_Entry_t * OuterEntry = NULL;
48  int32 OuterLoop = 0;
49  uint32 StateField = 0;
50  uint32 Address = 0;
51  uint32 Size = 0;
52  int32 GoodCount = 0;
53  int32 BadCount = 0;
54  int32 EmptyCount = 0;
55 
56  StartOfTable = (CS_Def_EepromMemory_Table_Entry_t *) TblPtr;
57 
58  for (OuterLoop = 0; OuterLoop < CS_MAX_NUM_EEPROM_TABLE_ENTRIES; OuterLoop++)
59  {
60  OuterEntry = & StartOfTable [OuterLoop];
61 
62  StateField = OuterEntry->State;
63  Address = OuterEntry->StartAddress;
64  Size = OuterEntry->NumBytesToChecksum;
65 
66  if ((StateField == CS_STATE_EMPTY) ||
67  (StateField == CS_STATE_ENABLED) ||
68  (StateField == CS_STATE_DISABLED))
69  {
70  /* If the StateField is within this range, it's check if it's not empty. */
71  if (StateField == CS_STATE_DISABLED || StateField == CS_STATE_ENABLED)
72  {
73  Status = CFE_PSP_MemValidateRange(Address,Size,CFE_PSP_MEM_EEPROM);
74  if (Status != OS_SUCCESS)
75  {
76  BadCount++;
77  if (Result != CS_TABLE_ERROR)
78  {
79  CFE_EVS_SendEvent (CS_VAL_EEPROM_RANGE_ERR_EID,
80  CFE_EVS_ERROR,
81  "Eeprom Table Validate: Illegal checksum range found in Entry ID %d, CFE_PSP_MemValidateRange returned: 0x%08X",
82  (int)OuterLoop,
83  (unsigned int)Status);
84  Result = CS_TABLE_ERROR;
85  }
86  }
87  else
88  {
89  /* Valid range for non-empty entry */
90  GoodCount++;
91  }
92  }
93  else
94  {
95  /* Entry is marked as empty */
96  EmptyCount++;
97  }
98  }
99  else
100  {
101  /* Invalid state definition */
102  BadCount++;
103  if (Result != CS_TABLE_ERROR)

105  CFE_EVS_SendEvent (CS_VAL_EEPROM_STATE_ERR_EID,
106  CFE_EVS_ERROR,
107  "Eeprom Table Validate: Illegal State Field (0x%04X) found in Entry ID %d",
108  (unsigned short)StateField,
109  (int)OuterLoop);
110  Result = CS_TABLE_ERROR;
111  }
112  }
113  } /* for (OuterLoop = 0; OuterLoop < CS_MAX_NUM_EEPROM_TABLE_ENTRIES; OuterLoop++) */
114 
115  CFE_EVS_SendEvent(CS_VAL_EEPROM_INF_EID,
116  CFE_EVS_INFORMATION,
117  "CS Eeprom Table verification results: good = %d, bad = %d, unused = %d",
118  (int)GoodCount,

120  (int)EmptyCount);
121 
122  return (Result);
123 } /* CS_ValidateEEPROMCheckSumDefinitionTable */

469 /* CS processing new definition tables for Eeprom or Memory */
470 /* */
471 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
472 void CS_ProcessNewEepromMemoryDefinitionTable (CS_Def_EepromMemory_Table_Entry_t * DefinitionTblPtr,
473  CS_Res_EepromMemory_Table_Entry_t * ResultsTblPtr,
474  uint16 NumEntries,
475  uint16 Table)
476 {
477  CS_Def_EepromMemory_Table_Entry_t * StartOfDefTable = NULL;
478  CS_Def_EepromMemory_Table_Entry_t * DefEntry = NULL;
479  CS_Res_EepromMemory_Table_Entry_t * StartOfResultsTable = NULL;
480  CS_Res_EepromMemory_Table_Entry_t * ResultsEntry = NULL;
481  uint16 Loop = 0;
482  uint16 NumRegionsInTable = 0;
483  uint16 PreviousState = CS_STATE_EMPTY;
484  char TableType[CS_TABLETYPE_NAME_SIZE];
485 
486  CFE_PSP_MemCpy(&StartOfResultsTable, ResultsTblPtr, sizeof(StartOfResultsTable));
487  CFE_PSP_MemCpy(&StartOfDefTable, DefinitionTblPtr, sizeof(StartOfDefTable));
488 
489  strncpy(&TableType[0], "Undef Tbl", CS_TABLETYPE_NAME_SIZE); /* Init the table type string */
490 
491  /* We don't want to be doing chekcksums while changing the table out */
492  if( Table == CS_EEPROM_TABLE)
493  {
494  PreviousState = CS_AppData.EepromCSState;
495  CS_AppData.EepromCSState = CS_STATE_DISABLED;
496  }
497  if( Table == CS_MEMORY_TABLE)
498  {
499  PreviousState = CS_AppData.MemoryCSState;
500  CS_AppData.MemoryCSState = CS_STATE_DISABLED;
501  }
502 
503  for (Loop = 0; Loop < NumEntries; Loop++)
504  {
505  DefEntry = & (StartOfDefTable [Loop]);
506  ResultsEntry = &(StartOfResultsTable [Loop]);
507 
508  if (DefEntry -> State != CS_STATE_EMPTY )
509  {
510  /* This is just a real simple test, because all of the exception handling
511  has already been done by the Validation routine above */
512  NumRegionsInTable++;
513 
514  ResultsEntry -> State = DefEntry -> State;
515  ResultsEntry -> ComputedYet = FALSE;
516  ResultsEntry -> NumBytesToChecksum = DefEntry -> NumBytesToChecksum;
517  ResultsEntry -> ComparisonValue = 0;
518  ResultsEntry -> ByteOffset = 0;
519  ResultsEntry -> TempChecksumValue = 0;
520  ResultsEntry -> StartAddress = DefEntry -> StartAddress;
521  }
522  else
523  {
524  ResultsEntry -> State = CS_STATE_EMPTY;
525  ResultsEntry -> ComputedYet = FALSE;
526  ResultsEntry -> NumBytesToChecksum = 0;
527  ResultsEntry -> ComparisonValue = 0;
528  ResultsEntry -> ByteOffset = 0;
529  ResultsEntry -> TempChecksumValue = 0;
530  ResultsEntry -> StartAddress = 0;
531  }
532  }
533 
534  /* Reset the table back to the original checksumming state */
535  if( Table == CS_EEPROM_TABLE)
536  {
537  CS_AppData.EepromCSState = PreviousState;
538  CS_ResetTablesTblResultEntry(CS_AppData.EepResTablesTblPtr);
539  }
540 
541  if( Table == CS_MEMORY_TABLE)
542  {
543  CS_AppData.MemoryCSState = PreviousState;
544  CS_ResetTablesTblResultEntry(CS_AppData.MemResTablesTblPtr);
545  }
546 
547  if (NumRegionsInTable == 0)
548  {
549  if( Table == CS_EEPROM_TABLE)
550  {
551  strncpy(&TableType[0], "Eeprom", CS_TABLETYPE_NAME_SIZE);
552  }
553  if( Table == CS_MEMORY_TABLE)
554  {
555  strncpy(&TableType[0], "Memory", CS_TABLETYPE_NAME_SIZE);
556  }
557 
558  CFE_EVS_SendEvent (CS_PROCESS_EEPROM_MEMORY_NO_ENTRIES_INF_EID,
559  CFE_EVS_INFORMATION,
560  "CS %s Table: No valid entries in the table",
561  TableType);
562  }
563  return;
564 } /* end of CS_ProcessNewEepromMemoryDefinitionTable () */

568 /* CS processing new definition tables for Tables */

571 void CS_ProcessNewTablesDefinitionTable (CS_Def_Tables_Table_Entry_t * DefinitionTblPtr,
572  CS_Res_Tables_Table_Entry_t * ResultsTblPtr)
573 {
574  CS_Def_Tables_Table_Entry_t * StartOfDefTable = NULL;
575  CS_Def_Tables_Table_Entry_t * DefEntry = NULL;
576  CS_Res_Tables_Table_Entry_t * StartOfResultsTable = NULL;
577  CS_Res_Tables_Table_Entry_t * ResultsEntry = NULL;
578  uint16 Loop = 0;
579  uint16 NumRegionsInTable = 0;
580  uint16 PreviousState = CS_STATE_EMPTY;
581  uint32 AppID = 0;
582  CFE_TBL_Handle_t TableHandle = CFE_TBL_BAD_TABLE_HANDLE;
583  boolean Owned = FALSE;
584  uint16 DefNameIndex = 0;
585  uint16 AppNameIndex = 0;
586  uint16 TableNameIndex = 0;
587  char AppName [OS_MAX_API_NAME] = "\0";
588  char TableAppName [OS_MAX_API_NAME] = "\0";
589  char TableTableName [CFE_TBL_MAX_NAME_LENGTH] = "\0";
590 
591  CFE_PSP_MemCpy(&StartOfResultsTable, ResultsTblPtr, sizeof(StartOfResultsTable));
592  CFE_PSP_MemCpy(&StartOfDefTable, DefinitionTblPtr, sizeof(StartOfDefTable));
593 
594  CFE_ES_GetAppID(&AppID);
595  CFE_ES_GetAppName( AppName, AppID, OS_MAX_API_NAME);
596 
597  /* We don't want to be doing chekcksums while changing the table out */
598  PreviousState = CS_AppData.TablesCSState;
599  CS_AppData.TablesCSState = CS_STATE_DISABLED;
600 
601  /* Assume none of the CS tables are listed in the new Tables table */
602  CS_AppData.EepResTablesTblPtr = NULL;
603  CS_AppData.MemResTablesTblPtr = NULL;
604  CS_AppData.AppResTablesTblPtr = NULL;
605  CS_AppData.TblResTablesTblPtr = NULL;
606 
607  for (Loop = 0; Loop < CS_MAX_NUM_TABLES_TABLE_ENTRIES; Loop++)
608  {
609  DefEntry = & (StartOfDefTable [Loop]);
610 
611  ResultsEntry = & (StartOfResultsTable [Loop]);
612 
613  if (DefEntry -> State != CS_STATE_EMPTY)
614  {
615  /* initialize buffer index values */
616  DefNameIndex = 0;
617  AppNameIndex = 0;
618  TableNameIndex = 0;
619 
620  /* extract application name from CS definition table entry */
621  while ((AppNameIndex < OS_MAX_API_NAME) &&
622  (DefEntry->Name[DefNameIndex] != '\0') && (DefEntry->Name[DefNameIndex] != '.'))
623  {
624  TableAppName[AppNameIndex++] = DefEntry->Name[DefNameIndex++];
625  }
626 
627  /* limit application name length to add string terminator */
628  if (AppNameIndex == OS_MAX_API_NAME)
629  {
630  AppNameIndex = OS_MAX_API_NAME - 1;
631  }
632 
633  /* add string terminator to application name */
634  TableAppName[AppNameIndex] = '\0';
635 
636  /* move entry index to the dot, string terminator, or end of table entry */
637  while ((DefNameIndex < CFE_TBL_MAX_FULL_NAME_LEN) &&
638  (DefEntry->Name[DefNameIndex] != '\0') &&
639  (DefEntry->Name[DefNameIndex] != '.'))
640  {
641  DefNameIndex++;
642  }
643
