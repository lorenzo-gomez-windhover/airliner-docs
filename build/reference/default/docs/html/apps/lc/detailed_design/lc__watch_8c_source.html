31 /* LC_GetHashTableIndex() - convert messageID to hash table index */

33 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
34 
35 uint32 LC_GetHashTableIndex(CFE_SB_MsgId_t MessageID)
36 {
37  /*
38  ** The purpose of a hash function is to take the input value
39  ** and convert it to an index into the hash table. Assume that
40  ** the range of input values is much different (larger) than
41  ** the number of entries in the hash table. Then multiple input
42  ** values must resolve to the same output table index. This is ok
43  ** because each entry in the hash table is a linked list of all
44  ** the inputs with the same hash function result.
45  **
46  ** This particular hash function takes advantage of knowledge
47  ** regarding the format of the input values (cFE MessageID). By
48  ** ignoring the bits that define version number, packet type and
49  ** secondary header (high 5 bits of 16) we are left with the bits
50  ** (mask = 0x7FF) that can identify 2048 unique input telemetry
51  ** packets. Also, by using a fixed hash table size of 256 entries
52  ** and using only the lower 8 bits of the bitmask as the result
53  ** of the hash function, no single hash table entry will have more
54  ** than 8 elements in its linked list.
55  */
56  return((uint32) (MessageID & LC_HASH_TABLE_MASK));
57 
58 } /* End of LC_GetHashTableIndex() */

67 void LC_CreateHashTable(void)
68 {
69  LC_WatchPtList_t *WatchPtLink;
70  CFE_SB_MsgId_t LastMessageID;
71  CFE_SB_MsgId_t MessageID;
72  int32 MessageLinkIndex;
73  int32 WatchPtTblIndex;
74  int32 Result;
75 
76 
77  /* Un-subscribe to any MessageID's referenced in previous Watchpoint Definition Table */
78  for (MessageLinkIndex = 0; MessageLinkIndex < LC_OperData.MessageIDsCount; MessageLinkIndex++)
79  {
80  MessageID = LC_OperData.MessageLinks[MessageLinkIndex].MessageID;
81 
82  if ((Result = CFE_SB_Unsubscribe(MessageID, LC_OperData.CmdPipe)) != CFE_SUCCESS)
83  {
84  CFE_EVS_SendEvent(LC_UNSUB_WP_ERR_EID, CFE_EVS_ERROR,
85  "Error unsubscribing watchpoint: MID=0x%04X, RC=0x%08X",
86  MessageID, (unsigned int)Result);
87  }
88  }
89 
90  /* Initialize hash table structures */
91  CFE_PSP_MemSet(LC_OperData.HashTable, 0, (LC_HASH_TABLE_ENTRIES * sizeof(LC_MessageList_t *)));
92  CFE_PSP_MemSet(LC_OperData.MessageLinks, 0, (LC_MAX_WATCHPOINTS * sizeof(LC_MessageList_t)));
93  CFE_PSP_MemSet(LC_OperData.WatchPtLinks, 0, (LC_MAX_WATCHPOINTS * sizeof(LC_WatchPtList_t)));
94 
95  LC_OperData.MessageIDsCount = 0;
96  LC_OperData.WatchpointCount = 0;
97 
98  LastMessageID = 0xFFFF;
99  WatchPtLink = (LC_WatchPtList_t *) NULL;
100 
101  /* Process each entry in the Watchpoint Definition Table */
102  for (WatchPtTblIndex = 0; WatchPtTblIndex < LC_MAX_WATCHPOINTS; WatchPtTblIndex++)
103  {
104  /* Skip unused watchpoint table entries */
105  if (LC_OperData.WDTPtr[WatchPtTblIndex].DataType != LC_WATCH_NOT_USED)
106  {
107  MessageID = LC_OperData.WDTPtr[WatchPtTblIndex].MessageID;
108 
109  /* Use optimized code path if same MessageID as last watchpoint */
110  if ((LastMessageID == MessageID) && (WatchPtLink != (LC_WatchPtList_t *) NULL))
111  {
112  /* WatchPtLink points to last link in list for this Message ID */
113  WatchPtLink->Next = &LC_OperData.WatchPtLinks[LC_OperData.WatchpointCount++];
114 
115  /* Add new link to end of list, point to new last link */
116  WatchPtLink = WatchPtLink->Next;
117  }
118  else
119  {
120  /* May add message list link and subscribe to MessageID */
121  WatchPtLink = LC_AddWatchpoint(MessageID);
122  }
123 
124  /* Set watchpoint table index for this entry in list */
125  WatchPtLink->WatchIndex = WatchPtTblIndex;
126 
127  LastMessageID = MessageID;
128  }
129  }
130 
131  return;
132 
133 } /* End of LC_CreateHashTable() */

142 LC_WatchPtList_t *LC_AddWatchpoint(CFE_SB_MsgId_t MessageID)
143 {
144  LC_MessageList_t *MessageLink;
145  LC_WatchPtList_t *WatchPtLink;
146  boolean NeedSubscription;
147  int32 HashTableIndex;
148  int32 Result;
149 
150  /* Hash function converts MessageID into hash table index */
151  HashTableIndex = LC_GetHashTableIndex(MessageID);
152 
153  /* Subscribe only for first reference to each MessageID */
154  NeedSubscription = FALSE;
155 
156  /* Each hash table entry is a linked list of MessageID's with same hash result */
157  if (LC_OperData.HashTable[HashTableIndex] == (LC_MessageList_t *) NULL)
158  {
159  /* Get next unused MessageID linked list entry */
160  MessageLink = &LC_OperData.MessageLinks[LC_OperData.MessageIDsCount++];
161 
162  /* Set first (and only) link in this hash table entry linked list */
163  LC_OperData.HashTable[HashTableIndex] = MessageLink;
164 
165  /* Set the MessageID for this link */
166  MessageLink->MessageID = MessageID;
167 
168  /* Subscribe to first link in list */
169  NeedSubscription = TRUE;
170  }
171  else
172  {
173  /* Get start of linked list (all MID's with same hash result) */
174  MessageLink = LC_OperData.HashTable[HashTableIndex];
175 
176  /* Find the link for this MessageID */
177  while (MessageLink->MessageID != MessageID)
178  {
179  if (MessageLink->Next == (LC_MessageList_t *) NULL)
180  {
181  /* Reached end of list without finding MessageID */
182  MessageLink->Next = &LC_OperData.MessageLinks[LC_OperData.MessageIDsCount++];
183  MessageLink = MessageLink->Next;
184 
185  /* Add link with this MessageID (will exit loop) */
186  MessageLink->MessageID = MessageID;
187 
188  /* Subscribe to new link in list */
189  NeedSubscription = TRUE;
190  }
191  else
192  {
193  /* Try the next link in the list */
194  MessageLink = MessageLink->Next;
195  }
196  }
197  }
198 
199  /* Subscribe only once to each MessageID */
200  if (NeedSubscription)
201  {
202  if ((Result = CFE_SB_Subscribe(MessageID, LC_OperData.CmdPipe)) != CFE_SUCCESS)
203  {
204  /* Signal the error, but continue */
205  CFE_EVS_SendEvent(LC_SUB_WP_ERR_EID, CFE_EVS_ERROR,
206  "Error subscribing watchpoint: MID=0x%04X, RC=0x%08X",
207  MessageID, (unsigned int)Result);
208  }
209  }
210 
211  /* MessageLink points to the link for this MessageID */
212  if (MessageLink->WatchPtList == (LC_WatchPtList_t *) NULL)
213  {
214  /* Get next unused watchpoint linked list entry */
215  WatchPtLink = &LC_OperData.WatchPtLinks[LC_OperData.WatchpointCount++];
216 
217  /* Set the start (and only) link in the watchpoint link list */
218  MessageLink->WatchPtList = WatchPtLink;
219  }
220  else
221  {
222  /* Find the end of the watchpoint linked list */
223  WatchPtLink = MessageLink->WatchPtList;
224 
225  while (WatchPtLink->Next != (LC_WatchPtList_t *) NULL)
226  {
227  WatchPtLink = WatchPtLink->Next;
228  }
229 
230  /* Add the new watchpoint link to the end of the list */
231  WatchPtLink->Next = &LC_OperData.WatchPtLinks[LC_OperData.WatchpointCount++];
232  WatchPtLink = WatchPtLink->Next;
233  }
234 
235  /* Return pointer to last link in watchpoint linked list */
236  return(WatchPtLink);
237 
238 } /* End of LC_AddWatchpoint() */

246 void LC_CheckMsgForWPs(CFE_SB_MsgId_t MessageID, CFE_SB_MsgPtr_t MessagePtr)
247 {
248  CFE_TIME_SysTime_t Timestamp;
249  LC_MessageList_t *MessageList;
250  LC_WatchPtList_t *WatchPtList;
251  boolean WatchPtFound = FALSE;
252 
253  /* Do nothing if disabled at the application level */
254  if (LC_AppData.CurrentLCState != LC_STATE_DISABLED)
255  {
256  /* Use message timestamp - if none, use current time */
257  Timestamp = CFE_SB_GetMsgTime(MessagePtr);
258 
259  if ((Timestamp.Seconds == 0) && (Timestamp.Subseconds == 0))
260  {
261  Timestamp = CFE_TIME_GetTime();
262  }
263 
264  /* Performance Log (start time counter) */
265  CFE_ES_PerfLogEntry(LC_WDT_SEARCH_PERF_ID);
266 
267  /* Get start of linked list (all MID's with same hash result) */
268  MessageList = LC_OperData.HashTable[LC_GetHashTableIndex(MessageID)];
269 
270  /* NULL when list is empty or end of list */
271  while (MessageList != (LC_MessageList_t *) NULL)
272  {
273  /* Compare this linked list entry for matching MessageID */
274  if (MessageList->MessageID == MessageID)
275  {
276  /* Stop the search - we found it */
277  break;
278  }
279 
280  /* Max of 8 links per design */
281  MessageList = MessageList->Next;
282  }
283 
284  /* Should be true - else wouldn't subscribe to MessageID */
285  if (MessageList != (LC_MessageList_t *) NULL)
286  {
287  /* Get linked list of WP's that reference MessageID */
288  WatchPtList = MessageList->WatchPtList;
289 
290  /* NULL when list is empty or end of list */
291  while (WatchPtList != (LC_WatchPtList_t *) NULL)
292  {
293  WatchPtFound = TRUE;
294 
295  /* Verify that WP packet offset is within actual packet */
296  if (LC_WPOffsetValid(WatchPtList->WatchIndex, MessagePtr) == TRUE)
297  {
298  LC_ProcessWP(WatchPtList->WatchIndex, MessagePtr, Timestamp);
299  }
300 
301  /* No limit to how many WP's can reference one MessageID */
302  WatchPtList = WatchPtList->Next;
303  }
304  }
305 
306  /* Performance Log (stop time counter) */
307  CFE_ES_PerfLogExit(LC_WDT_SEARCH_PERF_ID);
308 
309  if (WatchPtFound == TRUE)
310  {
311  LC_AppData.MonitoredMsgCount++;
312  }
313  else
314  {
315  /* MessageID with no defined watchpoints */
316  CFE_EVS_SendEvent(LC_MID_INF_EID, CFE_EVS_INFORMATION,
317  "Msg with unreferenced message ID rcvd: ID = 0x%04X", MessageID);
318  }
319  }
320 
321  return;
322 
323 } /* end LC_CheckMsgForWPs */

328 /* Process a single watchpoint */
329 /* */
330 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
331 void LC_ProcessWP(uint16 WatchIndex,
332  CFE_SB_MsgPtr_t MessagePtr,
333  CFE_TIME_SysTime_t Timestamp)
334 {
335  uint8 *WPDataPtr;
336  uint8 PreviousResult;
337  uint8 WPEvalResult;
338  uint32 SizedWPData;
339  uint32 MaskedWPData;
340  uint32 StaleCounter;
341  boolean SizedDataValid;
342 
343  /*
344  ** Setup the pointer and get the massaged data
345  */
346  WPDataPtr = ((uint8 *)MessagePtr) +
347  LC_OperData.WDTPtr[WatchIndex].WatchpointOffset;
348 
349  SizedDataValid = LC_GetSizedWPData(WatchIndex,
350  WPDataPtr,
351  &SizedWPData);
352  if (SizedDataValid == TRUE)
353  {
354  /*
355  ** Get the last evalution result for this watchpoint
356  */
357  PreviousResult = LC_OperData.WRTPtr[WatchIndex].WatchResult;
358 
359  /*
360  ** Apply the defined bitmask for this watchpoint and then
361  ** call the mission defined custom function or do our own
362  ** relational comparison.
363  */
364  MaskedWPData = SizedWPData & LC_OperData.WDTPtr[WatchIndex].BitMask;
365 
366  if (LC_OperData.WDTPtr[WatchIndex].OperatorID == LC_OPER_CUSTOM)
367  {
368  WPEvalResult = LC_CustomFunction(WatchIndex,
369  MaskedWPData,
370  MessagePtr,
371  LC_OperData.WDTPtr[WatchIndex].CustomFuncArgument);
372  }
373  else
374  {
375  WPEvalResult = LC_OperatorCompare(WatchIndex, MaskedWPData);
376  }
377 
378  /*
379  ** Update the watch result
380  */
381  LC_OperData.WRTPtr[WatchIndex].WatchResult = WPEvalResult;
382 
383  /*
384  ** Update the watchpoint statistics based on the evaluation
385  ** result
386  */
387  LC_OperData.WRTPtr[WatchIndex].EvaluationCount++;
388 
389  if (WPEvalResult == LC_WATCH_TRUE)
390  {
391  LC_OperData.WRTPtr[WatchIndex].CumulativeTrueCount++;
392  LC_OperData.WRTPtr[WatchIndex].ConsecutiveTrueCount++;
393  StaleCounter = LC_OperData.WDTPtr[WatchIndex].ResultAgeWhenStale;
394  LC_OperData.WRTPtr[WatchIndex].CountdownToStale = StaleCounter;
395 
396  if ((PreviousResult == LC_WATCH_FALSE) ||
397  (PreviousResult == LC_WATCH_STALE))
398  {
399  LC_OperData.WRTPtr[WatchIndex].LastFalseToTrue.DataType =
400  LC_OperData.WDTPtr[WatchIndex].DataType;
401 
402  LC_OperData.WRTPtr[WatchIndex].FalseToTrueCount++;
403 
404  LC_OperData.WRTPtr[WatchIndex].LastFalseToTrue
405  .Value = MaskedWPData;
406 
407  LC_OperData.WRTPtr[WatchIndex].LastFalseToTrue
408  .Timestamp.Seconds = Timestamp.Seconds;
409 
410  LC_OperData.WRTPtr[WatchIndex].LastFalseToTrue
411  .Timestamp.Subseconds = Timestamp.Subseconds;
412  }
413  }
414  else if (WPEvalResult == LC_WATCH_FALSE)
415  {
416  LC_OperData.WRTPtr[WatchIndex].ConsecutiveTrueCount = 0;
417  StaleCounter = LC_OperData.WDTPtr[WatchIndex].ResultAgeWhenStale;
418  LC_OperData.WRTPtr[WatchIndex].CountdownToStale = StaleCounter;
419 
420  if ((PreviousResult == LC_WATCH_TRUE) ||
421  (PreviousResult == LC_WATCH_STALE))
422  {
423  LC_OperData.WRTPtr[WatchIndex].LastTrueToFalse.DataType =
424  LC_OperData.WDTPtr[WatchIndex].DataType;
425 
426  LC_OperData.WRTPtr[WatchIndex].LastTrueToFalse
427  .Value = MaskedWPData;
428 
429  LC_OperData.WRTPtr[WatchIndex].LastTrueToFalse
430  .Timestamp.Seconds = Timestamp.Seconds;
431 
432  LC_OperData.WRTPtr[WatchIndex].LastTrueToFalse
433  .Timestamp.Subseconds = Timestamp.Subseconds;
434  }
435  }
436  else
437  {
438  /*
439  ** WPEvalResult is STALE or ERROR
440  */
441  LC_OperData.WRTPtr[WatchIndex].CountdownToStale = 0;
442  }
443 
444  } /* end SizedDataValid if */
445 
446  return;
447 
448 } /* end LC_ProcessWP */

452 /* Perform a watchpoint relational comparison */
453 /* */
454 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
455 uint8 LC_OperatorCompare(uint16 WatchIndex,
456  uint32 ProcessedWPData)
457 {
458  uint8 EvalResult;
459  LC_MultiType_t WatchpointValue;
460  LC_MultiType_t ComparisonValue;
461 
462  /*
463  * The "ProcessedWPData" has been already normalized to be
464  * 32 bits wide and in the native CPU byte order. For actual
465  * comparison, it needs to be truncated back down to the same
466  * size as the reference value.
467  */
468  switch (LC_OperData.WDTPtr[WatchIndex].DataType)
469  {
470  case LC_DATA_UBYTE:
471  case LC_DATA_BYTE:
472  WatchpointValue.Unsigned8 = ProcessedWPData & 0xFF;
473  break;
474 
475  case LC_DATA_WORD_BE:
476  case LC_DATA_WORD_LE:
477  case LC_DATA_UWORD_BE:
478  case LC_DATA_UWORD_LE:
479  WatchpointValue.Unsigned16 = ProcessedWPData & 0xFFFF;
480  break;
481 
482  case LC_DATA_DWORD_BE:
483  case LC_DATA_DWORD_LE:
484  case LC_DATA_UDWORD_BE:
485  case LC_DATA_UDWORD_LE:
486  case LC_DATA_FLOAT_BE:
487  case LC_DATA_FLOAT_LE:
488  default:
489  WatchpointValue.Unsigned32 = ProcessedWPData;
490  break;
491  }
492  ComparisonValue = LC_OperData.WDTPtr[WatchIndex].ComparisonValue;
493 
494  /*
495  ** Handle the comparison appropriately depending on the data type
496  ** Any endian difference was handled when the watchpoint
497  ** data was extracted from the SB message
498  */
499  switch (LC_OperData.WDTPtr[WatchIndex].DataType)
500  {
501  /*
502  ** Signed integer types will get sign extended
503  */
504  case LC_DATA_BYTE:
505  EvalResult = LC_SignedCompare(WatchIndex,
506  WatchpointValue.Signed8,
507  ComparisonValue.Signed8);
508  break;
509 
510  case LC_DATA_WORD_BE:
511  case LC_DATA_WORD_LE:
512  EvalResult = LC_SignedCompare(WatchIndex,
513  WatchpointValue.Signed16,
514  ComparisonValue.Signed16);
515  break;
516 
517  case LC_DATA_DWORD_BE:
518  case LC_DATA_DWORD_LE:
519  EvalResult = LC_SignedCompare(WatchIndex,
520  WatchpointValue.Signed32,
521  ComparisonValue.Signed32);
522  break;
523 
524  /*
525  ** Unsigned integer types will get zero extended
526  */
527  case LC_DATA_UBYTE:
528  EvalResult = LC_UnsignedCompare(WatchIndex,
529  WatchpointValue.Unsigned8,
530  ComparisonValue.Unsigned8);
531  break;
532 
533  case LC_DATA_UWORD_BE:
534  case LC_DATA_UWORD_LE:
535  EvalResult = LC_UnsignedCompare(WatchIndex,
536  WatchpointValue.Unsigned16,
537  ComparisonValue.Unsigned16);
538  break;
539 
540  case LC_DATA_UDWORD_BE:
541  case LC_DATA_UDWORD_LE:
542  EvalResult = LC_UnsignedCompare(WatchIndex,
543  WatchpointValue.Unsigned32,
544  ComparisonValue.Unsigned32);
545  break;
546 
547  /*
548  ** Floating point values are handled separately
549  */
550  case LC_DATA_FLOAT_BE:
551  case LC_DATA_FLOAT_LE:
552  EvalResult = LC_FloatCompare(WatchIndex,
553  WatchpointValue,
554  ComparisonValue);
555  break;
556 
557  default:
558  /*
559  ** This should have been caught before now, but we'll
560  ** handle it just in case we ever get here.
561  */
562  CFE_EVS_SendEvent(LC_WP_DATATYPE_ERR_EID, CFE_EVS_ERROR,
563  "WP has undefined data type: WP = %d, DataType = %d",
564  WatchIndex, LC_OperData.WDTPtr[WatchIndex].DataType);
565 
566  EvalResult = LC_WATCH_ERROR;
567  break;
568  }
569 
570  return (EvalResult);
571 
572 } /* end LC_OperatorCompare */

709  ** work and can generate floating point exceptions. By contrast
710  ** comparisons with infinite numbers will behave as they should
711  ** so we don't try to catch those (we would rather they generate

742 
743  case LC_OPER_GT:
744  EvalResult = (WPFloat > CompareFloat) ? LC_WATCH_TRUE : LC_WATCH_FALSE;
745  break;
746 
747  case LC_OPER_GE:
748  EvalResult = (WPFloat >= CompareFloat) ? LC_WATCH_TRUE : LC_WATCH_FALSE;
749  break;
750 
751  default:

753  ** This should have been caught before now, but we'll
754  ** handle it just in case we ever get here.
755  */

761  break;
762 
763  } /* end of switch */

765  } /* end LC_WPIsNAN if */
