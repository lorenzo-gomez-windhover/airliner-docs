48 
49 void LC_AppMain(void)
50 {
51  int32 Status = CFE_SUCCESS;
52  uint32 RunStatus = CFE_ES_APP_RUN;
53 
54  /*
55  ** Performance Log, Start
56  */
57  CFE_ES_PerfLogEntry(LC_APPMAIN_PERF_ID);
58 
59  /*
60  ** Register this application with Executive Services
61  */
62  Status = CFE_ES_RegisterApp();
63 
64  /*
65  ** Perform application specific initialization
66  */
67  if (Status == CFE_SUCCESS)
68  {
69  Status = LC_AppInit();
70  }
71 
72  /*
73  ** Check for start-up error...
74  */
75  if (Status != CFE_SUCCESS)
76  {
77  /*
78  ** Set run status to terminate main loop
79  */
80  RunStatus = CFE_ES_APP_ERROR;
81  }
82 
83  /*
84  ** Application main loop
85  */
86  while(CFE_ES_RunLoop(&RunStatus) == TRUE)
87  {
88  /*
89  ** Performance Log, Stop
90  */
91  CFE_ES_PerfLogExit(LC_APPMAIN_PERF_ID);
92 
93  /*
94  ** Pend on the arrival of the next Software Bus message
95  */
96  Status = CFE_SB_RcvMsg(&LC_OperData.MsgPtr, LC_OperData.CmdPipe, CFE_SB_PEND_FOREVER);
97 
98  /*
99  ** Performance Log, Start
100  */
101  CFE_ES_PerfLogEntry(LC_APPMAIN_PERF_ID);
102 
103  /*
104  ** Process the software bus message
105  */
106  if (Status == CFE_SUCCESS)
107  {
108  Status = LC_AppPipe(LC_OperData.MsgPtr);
109  }
110 
111  /*
112  ** Note: If there were some reason to exit the task
113  ** normally (without error) then we would set
114  ** RunStatus = CFE_ES_APP_EXIT
115  */
116  if (Status != CFE_SUCCESS)
117  {
118  /*
119  ** Set request to terminate main loop
120  */
121  RunStatus = CFE_ES_APP_ERROR;
122  }
123 
124  } /* end CFS_ES_RunLoop while */
125 
126  /*
127  ** Check for "fatal" process error...
128  */
129  if (Status != CFE_SUCCESS)
130  {
131  /*
132  ** Send an event describing the reason for the termination
133  */
134  CFE_EVS_SendEvent(LC_TASK_EXIT_EID, CFE_EVS_CRITICAL,
135  "Task terminating, err = 0x%08X", (unsigned int)Status);
136 
137  /*
138  ** In case cFE Event Services is not working
139  */
140  CFE_ES_WriteToSysLog("LC task terminating, err = 0x%08X\n", (unsigned int)Status);
141  }
142 
143  /*
144  ** Performance Log, Stop
145  */
146  CFE_ES_PerfLogExit(LC_APPMAIN_PERF_ID);
147 
148  /*
149  ** Do not update CDS if inactive or startup was incomplete
150  */
151  if ((LC_OperData.HaveActiveCDS) &&
152  (LC_AppData.CDSSavedOnExit == LC_CDS_SAVED))
153  {
154  LC_UpdateTaskCDS();
155  }
156 
157  /*
158  ** Exit the application
159  */
160  CFE_ES_ExitApp(RunStatus);
161 
162 } /* end LC_AppMain */

334 
335 int32 LC_TableInit(void)
336 {
337  int32 Result;
338 
339  /*
340  ** LC task use of Critical Data Store (CDS)
341  **
342  ** Global application data (LC_AppData)
343  ** Watchpint results dump only table data
344  ** Actionpoint results dump only table data
345  **
346  ** cFE Table Services use of CDS for LC task
347  **
348  ** Watchpint definition loadable table data
349  ** Actionpoint definition loadable table data
350  **
351  ** LC table initialization logic re CDS
352  **
353  ** If LC cannot create all the CDS storage at startup, then LC
354  ** will disable LC use of CDS and continue.
355  **
356  ** If LC cannot register definition tables as critical, then LC
357  ** will disable LC use of CDS and re-register tables as non-critical.
358  **
359  ** If LC cannot register definition and results tables at startup,
360  ** then LC will terminate - table use is a required function.
361  **
362  ** If LC can create all the CDS storage and register definition
363  ** tables as critical, then LC will write to CDS regardless of
364  ** whether LC was able to read from CDS at startup.
365  **
366  ** If LC cannot restore everything from CDS at startup, then LC
367  ** will initialize everything - load default definition tables,
368  ** init results table contents, init global application data.
369  */
370 
371  /* lc_platform_cfg.h */
372  #ifdef LC_SAVE_TO_CDS
373  LC_OperData.HaveActiveCDS = TRUE;
374  #endif
375 
376  /*
377  ** Maintain a detailed record of table initialization results
378  */
379  if (LC_OperData.HaveActiveCDS)
380  {
381  LC_OperData.TableResults |= LC_CDS_ENABLED;
382  }
383 
384  /*
385  ** Create watchpoint and actionpoint result tables
386  */
387  if ((Result = LC_CreateResultTables()) != CFE_SUCCESS)
388  {
389  return(Result);
390  }
391 
392  /*
393  ** If CDS is enabled - create the 3 CDS areas managed by the LC task
394  ** (continue with init, but disable CDS if unable to create all 3)
395  */
396  if (LC_OperData.HaveActiveCDS)
397  {
398  if (LC_CreateTaskCDS() != CFE_SUCCESS)
399  {
400  LC_OperData.HaveActiveCDS = FALSE;
401  }
402  }
403 
404  /*
405  ** Create wp/ap definition tables - critical if CDS enabled
406  */
407  if ((Result = LC_CreateDefinitionTables()) != CFE_SUCCESS)
408  {
409  return(Result);
410  }
411 
412  /*
413  ** CDS still active only if we created 3 CDS areas and 2 critical tables
414  */
415  if (LC_OperData.HaveActiveCDS)
416  {
417  LC_OperData.TableResults |= LC_CDS_CREATED;
418  }
419 
420  /*
421  ** If any CDS area or critical table is not restored - initialize everything.
422  ** (might be due to reset type, CDS disabled or corrupt, table restore error)
423  */
424  if (((LC_OperData.TableResults & LC_WRT_CDS_RESTORED) == LC_WRT_CDS_RESTORED) &&
425  ((LC_OperData.TableResults & LC_ART_CDS_RESTORED) == LC_ART_CDS_RESTORED) &&
426  ((LC_OperData.TableResults & LC_APP_CDS_RESTORED) == LC_APP_CDS_RESTORED) &&
427  ((LC_OperData.TableResults & LC_WDT_TBL_RESTORED) == LC_WDT_TBL_RESTORED) &&
428  ((LC_OperData.TableResults & LC_ADT_TBL_RESTORED) == LC_ADT_TBL_RESTORED))
429  {
430  LC_OperData.TableResults |= LC_CDS_RESTORED;
431 
432  /*
433  ** Get a pointer to the watchpoint definition table data...
434  */
435  Result = CFE_TBL_GetAddress((void *)&LC_OperData.WDTPtr, LC_OperData.WDTHandle);
436 
437  if ((Result != CFE_SUCCESS) && (Result != CFE_TBL_INFO_UPDATED))
438  {
439  CFE_EVS_SendEvent(LC_WDT_GETADDR_ERR_EID, CFE_EVS_ERROR,
440  "Error getting WDT address, RC=0x%08X", (unsigned int)Result);
441  return(Result);
442  }
443 
444  /*
445  ** Get a pointer to the actionpoint definition table data
446  */
447  Result = CFE_TBL_GetAddress((void *)&LC_OperData.ADTPtr, LC_OperData.ADTHandle);
448 
449  if ((Result != CFE_SUCCESS) && (Result != CFE_TBL_INFO_UPDATED))
450  {
451  CFE_EVS_SendEvent(LC_ADT_GETADDR_ERR_EID, CFE_EVS_ERROR,
452  "Error getting ADT address, RC=0x%08X", (unsigned int)Result);
453  return(Result);
454  }
455  }
456  else
457  {
458  if ((Result = LC_LoadDefaultTables()) != CFE_SUCCESS)
459  {
460  return(Result);
461  }
462  }
463 
464  /*
465  ** Create watchpoint hash tables -- also subscribes to watchpoint packets
466  */
467  LC_CreateHashTable();
468 
469  /*
470  ** Display results of CDS initialization (if enabled at startup)
471  */
472  if ((LC_OperData.TableResults & LC_CDS_ENABLED) == LC_CDS_ENABLED)
473  {
474  if ((LC_OperData.TableResults & LC_CDS_RESTORED) == LC_CDS_RESTORED)
475  {
476  CFE_EVS_SendEvent(LC_CDS_RESTORED_INF_EID, CFE_EVS_INFORMATION,
477  "Previous state restored from Critical Data Store");
478  }
479  else if ((LC_OperData.TableResults & LC_CDS_UPDATED) == LC_CDS_UPDATED)
480  {
481  CFE_EVS_SendEvent(LC_CDS_UPDATED_INF_EID, CFE_EVS_INFORMATION,
482  "Default state loaded and written to CDS, activity mask = 0x%08X",
483  (unsigned int)LC_OperData.TableResults);
484  }
485 
486  }
487  else
488  {
489  CFE_EVS_SendEvent(LC_CDS_DISABLED_INF_EID, CFE_EVS_INFORMATION,
490  "LC use of Critical Data Store disabled, activity mask = 0x%08X",
491  (unsigned int)LC_OperData.TableResults);
492  }
493 
494  return(CFE_SUCCESS);
495 
496 } /* LC_TableInit() */

510

512  ** Set "dump only" table option flags
513  */

515

519  */

529 
530  Result = CFE_TBL_GetAddress((void *)&LC_OperData.WRTPtr, LC_OperData.WRTHandle);
531 
532  if (Result != CFE_SUCCESS)
533  {
534  CFE_EVS_SendEvent(LC_WRT_GETADDR_ERR_EID, CFE_EVS_ERROR,
535  "Error getting WRT address, RC=0x%08X", (unsigned int)Result);
536  return(Result);
537  }
538 
539  LC_OperData.TableResults |= LC_WRT_TBL_CREATED;
540 
541  /*

556 
557  if (Result != CFE_SUCCESS)
558  {
559  CFE_EVS_SendEvent(LC_ART_GETADDR_ERR_EID, CFE_EVS_ERROR,
560  "Error getting ART address, RC=0x%08X", (unsigned int)Result);
561  return(Result);
562  }
563 
564  LC_OperData.TableResults |= LC_ART_TBL_CREATED;
565 
566  return(CFE_SUCCESS);
567 
568 } /* LC_CreateResultTables() */

576 
577 int32 LC_CreateDefinitionTables(void)
578 {
579  int32 Result;
580  uint32 DataSize;
581  uint32 OptionFlags;
582 
583  /*
584  ** If CDS is still enabled, try to register the 2 definition tables as critical
585  ** (if error, continue with init - but disable CDS and re-register as non-critical)
586  */
587  if (LC_OperData.HaveActiveCDS)
588  {
589  OptionFlags = CFE_TBL_OPT_DEFAULT | CFE_TBL_OPT_CRITICAL;
590  }
591  else
592  {
593  OptionFlags = CFE_TBL_OPT_DEFAULT;
594  }
595 
596  /*
597  ** Register the Watchpoint Definition Table (WDT)
598  */
599  DataSize = LC_MAX_WATCHPOINTS * sizeof (LC_WDTEntry_t);
600  Result = CFE_TBL_Register(&LC_OperData.WDTHandle, LC_WDT_TABLENAME,
601  DataSize, OptionFlags, LC_ValidateWDT);
602 
603  if ((LC_OperData.HaveActiveCDS) &&
604  ((Result != CFE_TBL_INFO_RECOVERED_TBL) && (Result != CFE_SUCCESS)))
605  {
606  LC_OperData.HaveActiveCDS = FALSE;
607  OptionFlags = CFE_TBL_OPT_DEFAULT;
608 
609  /*
610  ** Re-register the Watchpoint Definition Table (WDT) non-critical
611  */
612  Result = CFE_TBL_Register(&LC_OperData.WDTHandle, LC_WDT_TABLENAME,
613  DataSize, OptionFlags, LC_ValidateWDT);
614  }
615 
616  if (Result == CFE_TBL_INFO_RECOVERED_TBL)
617  {
618  LC_OperData.TableResults |= LC_WDT_CRITICAL_TBL;
619  LC_OperData.TableResults |= LC_WDT_TBL_RESTORED;
620  }
621  else if (Result == CFE_SUCCESS)
622  {
623  if (LC_OperData.HaveActiveCDS)
624  {
625  LC_OperData.TableResults |= LC_WDT_CRITICAL_TBL;
626  }
627  else
628  {
629  LC_OperData.TableResults |= LC_WDT_NOT_CRITICAL;
630  }
631  }
632  else
633  {
634  /*
635  ** Task initialization fails without this table
636  */
637  CFE_EVS_SendEvent(LC_WDT_REGISTER_ERR_EID, CFE_EVS_ERROR,
638  "Error registering WDT, RC=0x%08X", (unsigned int)Result);
639  return(Result);
640  }
641 
642  /*
643  ** Register the Actionpoint Definition Table (ADT)
644  */
645  DataSize = LC_MAX_ACTIONPOINTS * sizeof (LC_ADTEntry_t);
646  Result = CFE_TBL_Register(&LC_OperData.ADTHandle, LC_ADT_TABLENAME,
647  DataSize, OptionFlags, LC_ValidateADT);
648 
649  if ((LC_OperData.HaveActiveCDS) &&
650  ((Result != CFE_TBL_INFO_RECOVERED_TBL) && (Result != CFE_SUCCESS)))
651  {
652  LC_OperData.HaveActiveCDS = FALSE;
653  OptionFlags = CFE_TBL_OPT_DEFAULT;
654 
655  /*
656  ** Re-register the Actionpoint Definition Table (ADT) non-critical
657  */
658  Result = CFE_TBL_Register(&LC_OperData.ADTHandle, LC_ADT_TABLENAME,
659  DataSize, OptionFlags, LC_ValidateADT);
660  }
661 
662  if (Result == CFE_TBL_INFO_RECOVERED_TBL)
663  {
664  LC_OperData.TableResults |= LC_ADT_CRITICAL_TBL;
665  LC_OperData.TableResults |= LC_ADT_TBL_RESTORED;
666  }
667  else if (Result == CFE_SUCCESS)
668  {
669  if (LC_OperData.HaveActiveCDS)
670  {
671  LC_OperData.TableResults |= LC_ADT_CRITICAL_TBL;
672  }
673  else
674  {
675  LC_OperData.TableResults |= LC_ADT_NOT_CRITICAL;
676  }
677  }
678  else
679  {
680  /*
681  ** Task initialization fails without this table
682  */
683  CFE_EVS_SendEvent(LC_ADT_REGISTER_ERR_EID, CFE_EVS_ERROR,
684  "Error registering ADT, RC=0x%08X", (unsigned int)Result);
685  return(Result);
686  }

904  }
