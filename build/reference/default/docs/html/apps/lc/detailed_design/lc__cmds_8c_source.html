34 /* Process a command pipe message */
35 /* */
36 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
37 int32 LC_AppPipe(CFE_SB_MsgPtr_t MessagePtr)
38 {
39  int32 Status = CFE_SUCCESS;
40  CFE_SB_MsgId_t MessageID = 0;
41  uint16 CommandCode = 0;
42 
43  MessageID = CFE_SB_GetMsgId(MessagePtr);
44  switch (MessageID)
45  {
46  /*
47  ** Sample actionpoints request
48  */
49  case LC_SAMPLE_AP_MID:
50  LC_SampleAPReq(MessagePtr);
51  break;
52 
53  /*
54  ** Housekeeping telemetry request
55  ** (only routine that can return a critical error indicator)
56  */
57  case LC_SEND_HK_MID:
58  Status = LC_HousekeepingReq(MessagePtr);
59  break;
60 
61  /*
62  ** LC application commands...
63  */
64  case LC_CMD_MID:
65 
66  CommandCode = CFE_SB_GetCmdCode(MessagePtr);
67  switch (CommandCode)
68  {
69  case LC_NOOP_CC:
70  LC_NoopCmd(MessagePtr);
71  break;
72 
73  case LC_RESET_CC:
74  LC_ResetCmd(MessagePtr);
75  break;
76 
77  case LC_SET_LC_STATE_CC:
78  LC_SetLCStateCmd(MessagePtr);
79  break;
80 
81  case LC_SET_AP_STATE_CC:
82  LC_SetAPStateCmd(MessagePtr);
83  break;
84 
85  case LC_SET_AP_PERMOFF_CC:
86  LC_SetAPPermOffCmd(MessagePtr);
87  break;
88 
89  case LC_RESET_AP_STATS_CC:
90  LC_ResetAPStatsCmd(MessagePtr);
91  break;
92 
93  case LC_RESET_WP_STATS_CC:
94  LC_ResetWPStatsCmd(MessagePtr);
95  break;
96 
97  default:
98  CFE_EVS_SendEvent(LC_CC_ERR_EID, CFE_EVS_ERROR,
99  "Invalid command code: ID = 0x%04X, CC = %d",
100  MessageID, CommandCode);
101 
102  LC_AppData.CmdErrCount++;
103  break;
104 
105  } /* end CommandCode switch */
106  break;
107 
108  /*
109  ** All other message ID's should be monitor
110  ** packets
111  */
112  default:
113  LC_CheckMsgForWPs(MessageID, MessagePtr);
114  break;
115 
116  } /* end MessageID switch */
117 
118  return (Status);
119 
120 } /* End LC_AppPipe */

124 /* Sample Actionpoints Request */
125 /* */
126 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
127 void LC_SampleAPReq(CFE_SB_MsgPtr_t MessagePtr)
128 {
129  LC_SampleAP_t *LC_SampleAP = (LC_SampleAP_t *) MessagePtr;
130  uint16 ExpectedLength = sizeof(LC_SampleAP_t);
131  uint16 WatchIndex;
132  boolean ValidSampleCmd = FALSE;
133 
134  /*
135  ** Verify message packet length
136  */
137  if (LC_VerifyMsgLength(MessagePtr, ExpectedLength))
138  {
139  /*
140  ** Ignore AP sample requests if disabled at the application level
141  */
142  if (LC_AppData.CurrentLCState != LC_STATE_DISABLED)
143  {
144  /*
145  ** Range check the actionpoint array index arguments
146  */
147  if ((LC_SampleAP->StartIndex == LC_ALL_ACTIONPOINTS) &&
148  (LC_SampleAP->EndIndex == LC_ALL_ACTIONPOINTS))
149  {
150  /*
151  ** Allow special "sample all" heritage values
152  */
153  LC_SampleAPs(0, LC_MAX_ACTIONPOINTS - 1);
154  ValidSampleCmd = TRUE;
155  }
156  else if ((LC_SampleAP->StartIndex <= LC_SampleAP->EndIndex) &&
157  (LC_SampleAP->EndIndex < LC_MAX_ACTIONPOINTS))
158  {
159  /*
160  ** Start is less or equal to end, and end is within the array
161  */
162  LC_SampleAPs(LC_SampleAP->StartIndex, LC_SampleAP->EndIndex);
163  ValidSampleCmd = TRUE;
164  }
165  else
166  {
167  /*
168  ** At least one actionpoint array index is out of range
169  */
170  CFE_EVS_SendEvent(LC_APSAMPLE_APNUM_ERR_EID, CFE_EVS_ERROR,
171  "Sample AP error: invalid AP number, start = %d, end = %d",
172  LC_SampleAP->StartIndex, LC_SampleAP->EndIndex);
173  }
174 
175  /*
176  ** Optionally update the age of watchpoint results
177  */
178  if ((LC_SampleAP->UpdateAge != 0) && (ValidSampleCmd))
179  {
180  for (WatchIndex = 0; WatchIndex < LC_MAX_WATCHPOINTS; WatchIndex++)
181  {
182  if (LC_OperData.WRTPtr[WatchIndex].CountdownToStale != 0)
183  {
184  LC_OperData.WRTPtr[WatchIndex].CountdownToStale--;
185 
186  if (LC_OperData.WRTPtr[WatchIndex].CountdownToStale == 0)
187  {
188  LC_OperData.WRTPtr[WatchIndex].WatchResult = LC_WATCH_STALE;
189  }
190  }
191  }
192  }
193  }
194  }
195 
196  return;
197 
198 } /* end LC_SampleAPReq */

200 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
201 /* */
202 /* Housekeeping request */
203 /* */
204 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
205 int32 LC_HousekeepingReq(CFE_SB_MsgPtr_t MessagePtr)
206 {
207  int32 Result;
208  uint16 ExpectedLength = sizeof(LC_NoArgsCmd_t);
209  uint16 TableIndex;
210  uint16 HKIndex;
211  uint8 ByteData;
212 
213  /*
214  ** Verify message packet length
215  */
216  if(LC_VerifyMsgLength(MessagePtr, ExpectedLength))
217  {
218  /*
219  ** Update HK variables
220  */
221  LC_OperData.HkPacket.CmdCount = LC_AppData.CmdCount;
222  LC_OperData.HkPacket.CmdErrCount = LC_AppData.CmdErrCount;
223  LC_OperData.HkPacket.APSampleCount = LC_AppData.APSampleCount;
224  LC_OperData.HkPacket.MonitoredMsgCount = LC_AppData.MonitoredMsgCount;
225  LC_OperData.HkPacket.RTSExecCount = LC_AppData.RTSExecCount;
226  LC_OperData.HkPacket.PassiveRTSExecCount = LC_AppData.PassiveRTSExecCount;
227  LC_OperData.HkPacket.CurrentLCState = LC_AppData.CurrentLCState;
228  LC_OperData.HkPacket.WPsInUse = LC_OperData.WatchpointCount;
229 
230  /*
231  ** Clear out the active actionpoint count, it will get
232  ** recomputed below
233  */
234  LC_OperData.HkPacket.ActiveAPs = 0;
235 
236  /*
237  ** Update packed watch results
238  ** (4 watch results in one 8-bit byte)
239  */
240  for (TableIndex = 0; TableIndex < LC_MAX_WATCHPOINTS; TableIndex += 4)
241  {
242  HKIndex = TableIndex / 4;
243 
244  /*
245  ** Pack in first result
246  */
247  switch (LC_OperData.WRTPtr[TableIndex + 3].WatchResult)
248  {
249  case LC_WATCH_STALE:
250  ByteData = LC_HKWR_STALE << 6;
251  break;
252 
253  case LC_WATCH_FALSE:
254  ByteData = LC_HKWR_FALSE << 6;
255  break;
256 
257  case LC_WATCH_TRUE:
258  ByteData = LC_HKWR_TRUE << 6;
259  break;
260 
261  /*
262  ** We should never get an undefined watch result,
263  ** but we'll set an error result if we do
264  */
265  case LC_WATCH_ERROR:
266  default:
267  ByteData = LC_HKWR_ERROR << 6;
268  break;
269  }
270 
271  /*
272  ** Pack in second result
273  */
274  switch (LC_OperData.WRTPtr[TableIndex + 2].WatchResult)
275  {
276  case LC_WATCH_STALE:
277  ByteData = (ByteData | (LC_HKWR_STALE << 4));
278  break;
279 
280  case LC_WATCH_FALSE:
281  ByteData = (ByteData | (LC_HKWR_FALSE << 4));
282  break;
283 
284  case LC_WATCH_TRUE:
285  ByteData = (ByteData | (LC_HKWR_TRUE << 4));
286  break;
287 
288  case LC_WATCH_ERROR:
289  default:
290  ByteData = (ByteData | (LC_HKWR_ERROR << 4));
291  break;
292  }
293 
294  /*
295  ** Pack in third result
296  */
297  switch (LC_OperData.WRTPtr[TableIndex + 1].WatchResult)
298  {
299  case LC_WATCH_STALE:
300  ByteData = (ByteData | (LC_HKWR_STALE << 2));
301  break;
302 
303  case LC_WATCH_FALSE:
304  ByteData = (ByteData | (LC_HKWR_FALSE << 2));
305  break;
306 
307  case LC_WATCH_TRUE:
308  ByteData = (ByteData | (LC_HKWR_TRUE << 2));
309  break;
310 
311  case LC_WATCH_ERROR:
312  default:
313  ByteData = (ByteData | (LC_HKWR_ERROR << 2));
314  break;
315  }
316 
317  /*
318  ** Pack in fourth and last result
319  */
320  switch (LC_OperData.WRTPtr[TableIndex].WatchResult)
321  {
322  case LC_WATCH_STALE:
323  ByteData = (ByteData | LC_HKWR_STALE);
324  break;
325 
326  case LC_WATCH_FALSE:
327  ByteData = (ByteData | LC_HKWR_FALSE);
328  break;
329 
330  case LC_WATCH_TRUE:
331  ByteData = (ByteData | LC_HKWR_TRUE);
332  break;
333 
334  case LC_WATCH_ERROR:
335  default:
336  ByteData = (ByteData | LC_HKWR_ERROR);
337  break;
338  }
339 
340  /*
341  ** Update houskeeping watch results array
342  */
343  LC_OperData.HkPacket.WPResults[HKIndex] = ByteData;
344 
345  } /* end watch results for loop */
346 
347  /*
348  ** Update packed action results
349  ** (2 action state/result pairs (4 bits each) in one 8-bit byte)
350  */
351  for (TableIndex = 0; TableIndex < LC_MAX_ACTIONPOINTS; TableIndex += 2)
352  {
353  HKIndex = TableIndex / 2;
354 
355  /*
356  ** Pack in first actionpoint, current state
357  */
358  switch (LC_OperData.ARTPtr[TableIndex + 1].CurrentState)
359  {
360  case LC_ACTION_NOT_USED:
361  ByteData = LC_HKAR_STATE_NOT_USED << 6;
362  break;
363 
364  case LC_APSTATE_ACTIVE:
365  ByteData = LC_HKAR_STATE_ACTIVE << 6;
366  LC_OperData.HkPacket.ActiveAPs++;
367  break;
368 
369  case LC_APSTATE_PASSIVE:
370  ByteData = LC_HKAR_STATE_PASSIVE << 6;
371  break;
372 
373  case LC_APSTATE_DISABLED:
374  ByteData = LC_HKAR_STATE_DISABLED << 6;
375  break;
376 
377  /*
378  ** Permanantly disabled actionpoints get reported
379  ** as unused. We should never get an undefined
380  ** action state, but we'll set to not used if we do.
381  */
382  case LC_APSTATE_PERMOFF:
383  default:
384  ByteData = LC_HKAR_STATE_NOT_USED << 6;
385  break;
386  }
387 
388  /*
389  ** Pack in first actionpoint, action result
390  */
391  switch (LC_OperData.ARTPtr[TableIndex + 1].ActionResult)
392  {
393  case LC_ACTION_STALE:
394  ByteData = (ByteData | (LC_HKAR_STALE << 4));
395  break;
396 
397  case LC_ACTION_PASS:
398  ByteData = (ByteData | (LC_HKAR_PASS << 4));
399  break;
400 
401  case LC_ACTION_FAIL:
402  ByteData = (ByteData | (LC_HKAR_FAIL << 4));
403  break;
404 
405  /*
406  ** We should never get an undefined action result,
407  ** but we'll set an error result if we do
408  */
409  case LC_ACTION_ERROR:
410  default:
411  ByteData = (ByteData | (LC_HKAR_ERROR << 4));
412  break;
413  }
414 
415  /*
416  ** Pack in second actionpoint, current state
417  */
418  switch (LC_OperData.ARTPtr[TableIndex].CurrentState)
419  {
420  case LC_ACTION_NOT_USED:
421  ByteData = (ByteData | (LC_HKAR_STATE_NOT_USED << 2));
422  break;
423 
424  case LC_APSTATE_ACTIVE:
425  ByteData = (ByteData | (LC_HKAR_STATE_ACTIVE << 2));
426  LC_OperData.HkPacket.ActiveAPs++;
427  break;
428 
429  case LC_APSTATE_PASSIVE:
430  ByteData = (ByteData | (LC_HKAR_STATE_PASSIVE << 2));
431  break;
432 
433  case LC_APSTATE_DISABLED:
434  ByteData = (ByteData | (LC_HKAR_STATE_DISABLED << 2));
435  break;
436 
437  case LC_APSTATE_PERMOFF:
438  default:
439  ByteData = (ByteData | (LC_HKAR_STATE_NOT_USED << 2));
440  break;
441  }
442 
443  /*
444  ** Pack in second actionpoint, action result
445  */
446  switch (LC_OperData.ARTPtr[TableIndex].ActionResult)
447  {
448  case LC_ACTION_STALE:
449  ByteData = (ByteData | LC_HKAR_STALE);
450  break;
451 
452  case LC_ACTION_PASS:
453  ByteData = (ByteData | LC_HKAR_PASS);
454  break;
455 
456  case LC_ACTION_FAIL:
457  ByteData = (ByteData | LC_HKAR_FAIL);
458  break;
459 
460  case LC_ACTION_ERROR:
461  default:
462  ByteData = (ByteData | LC_HKAR_ERROR);
463  break;
464  }
465 
466  /*
467  ** Update houskeeping action results array
468  */
469  LC_OperData.HkPacket.APResults[HKIndex] = ByteData;
470 
471  } /* end action results for loop */
472 
473  /*
474  ** Timestamp and send housekeeping packet
475  */
476  CFE_SB_TimeStampMsg((CFE_SB_Msg_t *) &LC_OperData.HkPacket);
477  CFE_SB_SendMsg((CFE_SB_Msg_t *) &LC_OperData.HkPacket);
478 
479  } /* end LC_VerifyMsgLength if */
480 
481  /*
482  ** Manage tables - allow cFE to perform dump, update, etc.
483  ** (an error here is fatal - LC must be able to access its tables)
484  */
485  if ((Result = LC_ManageTables()) != CFE_SUCCESS)
486  {
487  return(Result);
488  }
489 
490  /*
491  ** A somewhat arbitrary decision was made to update the CDS only
492  ** as often as we report housekeeping, thus we do it here. One
493  ** alternative was to do the update every time the results tables
494  ** were modified - but that would result in the update occurring
495  ** several times per second. By doing the update now we cut down
496  ** on the update frequency at the cost of the stored data being
497  ** a couple of seconds old when a processor reset does occur.
498  */
499  if (LC_OperData.HaveActiveCDS)
500  {
501  /*
502  ** If CDS is enabled - update the 3 CDS areas managed by LC
503  ** (continue, but disable CDS if unable to update all 3)
504  */
505  if (LC_UpdateTaskCDS() != CFE_SUCCESS)
506  {
507  LC_OperData.HaveActiveCDS = FALSE;
508  }
509  }
510 
511  return(CFE_SUCCESS);
512 
513 } /* end LC_HousekeepingCmd */

631 /* */
632 /* Set actionpoint state command */
633 /* */
634 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
635 void LC_SetAPStateCmd(CFE_SB_MsgPtr_t MessagePtr)
636 {
637  uint16 ExpectedLength = sizeof(LC_SetAPState_t);
638  LC_SetAPState_t *CmdPtr;
639  uint32 TableIndex;
640  uint8 CurrentAPState;
641  boolean ValidState = TRUE;
642  boolean CmdSuccess = FALSE;
643 
644  /*
645  ** Verify message packet length
646  */
647  if(LC_VerifyMsgLength(MessagePtr, ExpectedLength))
648  {
649  CmdPtr = ((LC_SetAPState_t *)MessagePtr);
650 
651  /*
652  ** Do a sanity check on the new actionpoint state
653  ** specified.
654  */
655  switch (CmdPtr -> NewAPState)
656  {
657  case LC_APSTATE_ACTIVE:
658  case LC_APSTATE_PASSIVE:
659  case LC_APSTATE_DISABLED:
660  break;
661 
662  default:
663  ValidState = FALSE;
664  CFE_EVS_SendEvent(LC_APSTATE_NEW_ERR_EID, CFE_EVS_ERROR,
665  "Set AP state error: AP = %d, Invalid new state = %d",
666  CmdPtr -> APNumber, CmdPtr -> NewAPState);
667 
668  LC_AppData.CmdErrCount++;
669  break;
670  }
671 
672  /*
673  ** Do the rest based on the actionpoint ID we were given
674  */
675  if (ValidState == TRUE)
676  {
677  if ((CmdPtr -> APNumber) == LC_ALL_ACTIONPOINTS)
678  {
679  /*
680  ** Set all actionpoints to the new state except those that are not
681  ** used or set permanently off
682  */
683  for (TableIndex = 0; TableIndex < LC_MAX_ACTIONPOINTS; TableIndex++)
684  {
685  CurrentAPState = LC_OperData.ARTPtr[TableIndex].CurrentState;
686 
687  if ((CurrentAPState != LC_ACTION_NOT_USED) &&
688  (CurrentAPState != LC_APSTATE_PERMOFF))
689  {
690  LC_OperData.ARTPtr[TableIndex].CurrentState = CmdPtr -> NewAPState;
691  }
692  }
693 
694  /*
695  ** Set flag that we succeeded
696  */
697  CmdSuccess = TRUE;
698  }
699  else
700  {
701  if ((CmdPtr -> APNumber) < LC_MAX_ACTIONPOINTS)
702  {
703  TableIndex = CmdPtr -> APNumber;
704  CurrentAPState = LC_OperData.ARTPtr[TableIndex].CurrentState;
705 
706  if ((CurrentAPState != LC_ACTION_NOT_USED) &&
707  (CurrentAPState != LC_APSTATE_PERMOFF))
708  {
709  /*
710  ** Update state for single actionpoint specified
711  */
712  LC_OperData.ARTPtr[TableIndex].CurrentState = CmdPtr -> NewAPState;
713 
714  CmdSuccess = TRUE;
715  }
716  else
717  {
718  /*
719  ** Actionpoints that are not used or set permanently
720  ** off can only be changed by a table load
721  */
722  CFE_EVS_SendEvent(LC_APSTATE_CURR_ERR_EID, CFE_EVS_ERROR,
723  "Set AP state error: AP = %d, Invalid current AP state = %d",
724  CmdPtr -> APNumber, CurrentAPState);
725 
726  LC_AppData.CmdErrCount++;
727  }
728  }
729  else
730  {
731  /*
732  ** Actionpoint number is out of range
733  ** (it's zero based, since it's a table index)
