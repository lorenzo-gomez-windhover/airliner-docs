80 /* Initialize event tables. */
81 /* */
82 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
83 int32 RCIN::InitEvent()
84 {
85  int32 iStatus = CFE_SUCCESS;
86  int32 ind = 0;
87  int32 customEventCount = 0;
88 
89  CFE_EVS_BinFilter_t EventTbl[RCIN_MAX_EVENT_FILTERS];
90 
91  /* Initialize the event filter table.
92  * Note: 0 is the CFE_EVS_NO_FILTER mask and event 0 is reserved (not used) */
93  CFE_PSP_MemSet(EventTbl, 0x00, sizeof(EventTbl));
94 
95  EventTbl[ ind].EventID = RCIN_RESERVED_EID;
96  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
97  EventTbl[ ind].EventID = RCIN_INIT_INF_EID;
98  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
99  EventTbl[ ind].EventID = RCIN_CMD_NOOP_EID;
100  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
101  EventTbl[ ind].EventID = RCIN_SUBSCRIBE_ERR_EID;
102  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
103  EventTbl[ ind].EventID = RCIN_PIPE_INIT_ERR_EID;
104  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
105  EventTbl[ ind].EventID = RCIN_RCVMSG_ERR_EID;
106  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
107  EventTbl[ ind].EventID = RCIN_MSGID_ERR_EID;
108  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
109  EventTbl[ ind].EventID = RCIN_CC_ERR_EID;
110  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
111  EventTbl[ ind].EventID = RCIN_MSGLEN_ERR_EID;
112  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
113  EventTbl[ ind].EventID = RCIN_INIT_ERR_EID;
114  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
115  EventTbl[ ind].EventID = RCIN_PUBLISHING_INF_EID;
116  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
117  EventTbl[ ind].EventID = RCIN_NOT_PUBLISHING_ERR_EID;
118  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
119 
120  /* Add custom events to the filter table */
121  customEventCount = RCIN_Custom_Init_EventFilters(ind, EventTbl);
122 
123  if(-1 == customEventCount)
124  {
125  iStatus = CFE_EVS_APP_FILTER_OVERLOAD;
126  (void) CFE_ES_WriteToSysLog("Failed to init custom event filters (0x%08X)\n", (unsigned int)iStatus);
127  goto end_of_function;
128  }
129 
130  /* Register the table with CFE */
131  iStatus = CFE_EVS_Register(EventTbl, (ind + customEventCount), CFE_EVS_BINARY_FILTER);
132  if (iStatus != CFE_SUCCESS)
133  {
134  (void) CFE_ES_WriteToSysLog("RCIN - Failed to register with EVS (0x%08lX)\n", iStatus);
135  }
136 
137 end_of_function:
138  return (iStatus);
139 }

148 {
149  int32 iStatus = CFE_SUCCESS;
150 
151  /* Init schedule pipe and subscribe to wakeup messages */
152  iStatus = CFE_SB_CreatePipe(&SchPipeId, RCIN_SCH_PIPE_DEPTH,
153  RCIN_SCH_PIPE_NAME);
154  if (iStatus == CFE_SUCCESS)
155  {
156  iStatus = CFE_SB_SubscribeEx(RCIN_WAKEUP_MID, SchPipeId, CFE_SB_Default_Qos, RCIN_WAKEUP_MID_MAX_MSG_COUNT);
157  if (iStatus != CFE_SUCCESS)
158  {
159  (void) CFE_EVS_SendEvent(RCIN_SUBSCRIBE_ERR_EID, CFE_EVS_ERROR,
160  "Sch Pipe failed to subscribe to RCIN_WAKEUP_MID. (0x%08lX)",
161  iStatus);
162  goto RCIN_InitPipe_Exit_Tag;
163  }
164 
165  iStatus = CFE_SB_SubscribeEx(RCIN_SEND_HK_MID, SchPipeId, CFE_SB_Default_Qos, RCIN_SEND_HK_MID_MAX_MSG_COUNT);
166  if (iStatus != CFE_SUCCESS)
167  {
168  (void) CFE_EVS_SendEvent(RCIN_SUBSCRIBE_ERR_EID, CFE_EVS_ERROR,
169  "CMD Pipe failed to subscribe to RCIN_SEND_HK_MID. (0x%08X)",
170  (unsigned int)iStatus);
171  goto RCIN_InitPipe_Exit_Tag;
172  }
173  }
174  else
175  {
176  (void) CFE_EVS_SendEvent(RCIN_PIPE_INIT_ERR_EID, CFE_EVS_ERROR,
177  "Failed to create SCH pipe (0x%08lX)",
178  iStatus);
179  goto RCIN_InitPipe_Exit_Tag;
180  }
181 
182  /* Init command pipe and subscribe to command messages */
183  iStatus = CFE_SB_CreatePipe(&CmdPipeId, RCIN_CMD_PIPE_DEPTH,
184  RCIN_CMD_PIPE_NAME);
185  if (iStatus == CFE_SUCCESS)
186  {
187  /* Subscribe to command messages */
188  iStatus = CFE_SB_Subscribe(RCIN_CMD_MID, CmdPipeId);
189 
190  if (iStatus != CFE_SUCCESS)
191  {
192  (void) CFE_EVS_SendEvent(RCIN_SUBSCRIBE_ERR_EID, CFE_EVS_ERROR,
193  "CMD Pipe failed to subscribe to RCIN_CMD_MID. (0x%08lX)",
194  iStatus);
195  goto RCIN_InitPipe_Exit_Tag;
196  }
197  }
198  else
199  {
200  (void) CFE_EVS_SendEvent(RCIN_PIPE_INIT_ERR_EID, CFE_EVS_ERROR,
201  "Failed to create CMD pipe (0x%08lX)", iStatus);
202  goto RCIN_InitPipe_Exit_Tag;
203  }
204 
205 RCIN_InitPipe_Exit_Tag:
206  return (iStatus);
207 }

241 
242  iStatus = InitEvent();

246  goto RCIN_InitApp_Exit_Tag;
247  }
248  else

253  iStatus = InitPipe();

255  {

257  }
258 
259  InitData();
260 
261  returnBool = RCIN_Custom_Init();
262  if (FALSE == returnBool)
263  {
264  iStatus = RCIN_ERROR;
265  goto RCIN_InitApp_Exit_Tag;
266  }
267 
268  HkTlm.State = RCIN_INITIALIZED;
269 
270  /* Register the cleanup callback */
271  iStatus = OS_TaskInstallDeleteHandler(&RCIN_CleanupCallback);
272  if (iStatus != CFE_SUCCESS)
273  {
274  CFE_EVS_SendEvent(RCIN_INIT_ERR_EID, CFE_EVS_ERROR,
275  "Failed to init register cleanup callback (0x%08X)",
276  (unsigned int)iStatus);
277  goto RCIN_InitApp_Exit_Tag;
278  }
279 
280 RCIN_InitApp_Exit_Tag:
281  if (iStatus == CFE_SUCCESS)

283  (void) CFE_EVS_SendEvent(RCIN_INIT_INF_EID, CFE_EVS_INFORMATION,

286  RCIN_MINOR_VERSION,
287  RCIN_REVISION,
288  RCIN_MISSION_REV);
289  }

295  }
296  }
297 
298  return (iStatus);
299 }

304 /* Receive and Process Messages */
305 /* */
306 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
307 int32 RCIN::RcvSchPipeMsg(int32 iBlocking)
308 {
309  int32 iStatus = CFE_SUCCESS;
310  CFE_SB_Msg_t* MsgPtr = NULL;
311  CFE_SB_MsgId_t MsgId;
312 
313  /* Stop Performance Log entry */
314  CFE_ES_PerfLogExit(RCIN_MAIN_TASK_PERF_ID);
315 
316  /* Wait for WakeUp messages from scheduler */
317  iStatus = CFE_SB_RcvMsg(&MsgPtr, SchPipeId, iBlocking);
318 
319  /* Start Performance Log entry */
320  CFE_ES_PerfLogEntry(RCIN_MAIN_TASK_PERF_ID);
321 
322  if (iStatus == CFE_SUCCESS)
323  {
324  MsgId = CFE_SB_GetMsgId(MsgPtr);
325  switch (MsgId)
326  {
327  case RCIN_WAKEUP_MID:
328  {
329  ReadDevice();
330  SendInputRcMsg();
331  break;
332  }
333  case RCIN_SEND_HK_MID:
334  {
335  ProcessCmdPipe();
336  ReportHousekeeping();
337  break;
338  }
339  default:
340  {
341  (void) CFE_EVS_SendEvent(RCIN_MSGID_ERR_EID, CFE_EVS_ERROR,
342  "Recvd invalid SCH msgId (0x%04X)", MsgId);
343  break;
344  }
345  }
346  }
347  else if (iStatus == CFE_SB_NO_MESSAGE)
348  {
349  /* If there's no incoming message, do nothing here,
350  * Note, this section is dead code only if the iBlocking arg
351  * is CFE_SB_PEND_FOREVER. */
352  iStatus = CFE_SUCCESS;
353  }
354  else if (iStatus == CFE_SB_TIME_OUT)
355  {
356  /* If there's no incoming message within a specified time (via the
357  * iBlocking arg, do nothing here.
358  * Note, this section is dead code only if the iBlocking arg
359  * is CFE_SB_PEND_FOREVER. */
360  iStatus = CFE_SUCCESS;
361  }
362  else
363  {
364  (void) CFE_EVS_SendEvent(RCIN_RCVMSG_ERR_EID, CFE_EVS_ERROR,
365  "SCH pipe read error (0x%08lX).", iStatus);
366  }
367 
368  return (iStatus);
369 }

372 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
373 /* */
374 /* Process Incoming Commands */
375 /* */
376 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
377 void RCIN::ProcessCmdPipe()
378 {
379  int32 iStatus = CFE_SUCCESS;
380  CFE_SB_Msg_t* CmdMsgPtr = NULL;
381  CFE_SB_MsgId_t CmdMsgId;
382 
383  /* Process command messages until the pipe is empty */
384  while (1)
385  {
386  iStatus = CFE_SB_RcvMsg(&CmdMsgPtr, CmdPipeId, CFE_SB_POLL);
387  if(iStatus == CFE_SUCCESS)
388  {
389  CmdMsgId = CFE_SB_GetMsgId(CmdMsgPtr);
390  switch (CmdMsgId)
391  {
392  case RCIN_CMD_MID:
393  {
394  ProcessAppCmds(CmdMsgPtr);
395  break;
396  }
397  default:
398  {
399  /* Bump the command error counter for an unknown command.
400  * (This should only occur if it was subscribed to with this
401  * pipe, but not handled in this switch-case.) */
402  HkTlm.usCmdErrCnt++;
403  (void) CFE_EVS_SendEvent(RCIN_MSGID_ERR_EID, CFE_EVS_ERROR,
404  "Recvd invalid CMD msgId (0x%04X)", (unsigned short)CmdMsgId);
405  break;
406  }
407  }
408  }
409  else if (iStatus == CFE_SB_NO_MESSAGE)
410  {
411  break;
412  }
413  else
414  {
415  (void) CFE_EVS_SendEvent(RCIN_RCVMSG_ERR_EID, CFE_EVS_ERROR,
416  "CMD pipe read error (0x%08lX)", iStatus);
417  break;
418  }
419  }
420  return;
421 }

424 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
425 /* */
426 /* Process RCIN Commands */
427 /* */
428 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
429 void RCIN::ProcessAppCmds(CFE_SB_Msg_t* MsgPtr)
430 {
431  uint32 uiCmdCode=0;
432 
433  if (MsgPtr != NULL)
434  {
435  uiCmdCode = CFE_SB_GetCmdCode(MsgPtr);
436  switch (uiCmdCode)
437  {
438  case RCIN_NOOP_CC:
439  {
440  HkTlm.usCmdCnt++;
441  (void) CFE_EVS_SendEvent(RCIN_CMD_NOOP_EID, CFE_EVS_INFORMATION,
442  "Recvd NOOP. Version %d.%d.%d.%d",
443  RCIN_MAJOR_VERSION,
444  RCIN_MINOR_VERSION,
445  RCIN_REVISION,
446  RCIN_MISSION_REV);
447  break;
448  }
449  case RCIN_RESET_CC:
450  {
451  HkTlm.usCmdCnt = 0;
452  HkTlm.usCmdErrCnt = 0;
453  break;
454  }
455  default:
456  {
457  HkTlm.usCmdErrCnt++;
458  (void) CFE_EVS_SendEvent(RCIN_CC_ERR_EID, CFE_EVS_ERROR,
459  "Recvd invalid command code (%u)", (unsigned int)uiCmdCode);
460  break;
461  }
462  }
463  }
464  return;
465 }

526 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
527 /* */
528 /* Read from the Radio Control Receiver */
529 /* */
530 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
531 void RCIN::ReadDevice(void)
532 {
533  boolean returnBool = FALSE;
534 
535  returnBool = RCIN_Custom_Measure(&InputRcMsg);
536  if(FALSE == returnBool)
537  {
538  /* Rollover */
539  if(StrikeCount == UCHAR_MAX)
540  {
541  StrikeCount = 0;
542  }
543  /* Increment strike count */
544  StrikeCount++;
545  /* Measure is returning FALSE set state to not publishing */
546  if(StrikeCount == RCIN_STRIKE_COUNT_THRES)
547  {
548  if (HkTlm.State != RCIN_NOTPUBLISHING)
549  {
550  HkTlm.State = RCIN_NOTPUBLISHING;
551  (void) CFE_EVS_SendEvent(RCIN_NOT_PUBLISHING_ERR_EID, CFE_EVS_ERROR,
552  "RCIN is NOT publishing fresh data");
553  }
554  }
555  }
556  else
557  {
558  StrikeCount = 0;
559  if (HkTlm.State != RCIN_PUBLISHING)
560  {
561  /* Measure is returning TRUE set state to publishing */
562  HkTlm.State = RCIN_PUBLISHING;
563  (void) CFE_EVS_SendEvent(RCIN_PUBLISHING_INF_EID, CFE_EVS_INFORMATION,
564  "RCIN is publishing fresh data");
565  }
566  }
567 
568  InputRcMsg.LastSignal = InputRcMsg.Timestamp;
569  InputRcMsg.Timestamp = PX4LIB_GetPX4TimeUs();
570  return;
571 }

594
