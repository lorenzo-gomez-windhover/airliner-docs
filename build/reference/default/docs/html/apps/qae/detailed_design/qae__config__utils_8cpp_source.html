57 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
58 int32 QAE::InitConfigTbl()
59 {
60  int32 iStatus = 0;
61 
62  /* Register Config table */
63  iStatus = CFE_TBL_Register(&ConfigTblHdl,
64  QAE_CONFIG_TABLENAME,
65  (sizeof(QAE_ConfigTbl_t)),
66  CFE_TBL_OPT_DEFAULT,
67  QAE::ValidateConfigTbl);
68  if (iStatus != CFE_SUCCESS)
69  {
70  /* Note, a critical table could return another nominal code. If this table is
71  * made critical this logic would have to change. */
72  (void) CFE_EVS_SendEvent(QAE_CFGTBL_REG_ERR_EID, CFE_EVS_ERROR,
73  "Failed to register config table (0x%08lX)",
74  iStatus);
75  goto QAE_InitConfigTbl_Exit_Tag;
76  }
77 
78  /* Load Config table file */
79  iStatus = CFE_TBL_Load(ConfigTblHdl,
80  CFE_TBL_SRC_FILE,
81  QAE_CONFIG_TABLE_FILENAME);
82  if (iStatus != CFE_SUCCESS)
83  {
84  /* Note, CFE_SUCCESS is for a successful full table load. If a partial table
85  load is desired then this logic would have to change. */
86  (void) CFE_EVS_SendEvent(QAE_CFGTBL_LOAD_ERR_EID, CFE_EVS_ERROR,
87  "Failed to load Config Table (0x%08lX)",
88  iStatus);
89  goto QAE_InitConfigTbl_Exit_Tag;
90  }
91 
92  iStatus = AcquireConfigPointers();
93 
94 QAE_InitConfigTbl_Exit_Tag:
95  return (iStatus);
96 }

103 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
104 int32 QAE::ValidateConfigTbl(void* ConfigTblPtr)
105 {
106  int32 iStatus = 0;
107  boolean valid_bool = TRUE;
108  QAE_ConfigTbl_t* QAE_ConfigTblPtr = (QAE_ConfigTbl_t*)(ConfigTblPtr);
109 
110  if(!(QAE_ConfigTblPtr->ATT_W_ACC >= ATT_W_ACC_MIN && QAE_ConfigTblPtr->ATT_W_ACC <= ATT_W_ACC_MAX))
111  {
112  valid_bool = FALSE;
113  }
114 
115  if(!(QAE_ConfigTblPtr->ATT_W_MAG >= ATT_W_MAG_MIN && QAE_ConfigTblPtr->ATT_W_MAG <= ATT_W_MAG_MAX))
116  {
117  valid_bool = FALSE;
118  }
119 
120  if(!(QAE_ConfigTblPtr->ATT_W_GYRO_BIAS >= ATT_W_GYRO_BIAS_MIN && QAE_ConfigTblPtr->ATT_W_GYRO_BIAS <= ATT_W_GYRO_BIAS_MAX))
121  {
122  valid_bool = FALSE;
123  }
124 
125  if(!(QAE_ConfigTblPtr->ATT_MAG_DECL >= ATT_MAG_DECL_MIN && QAE_ConfigTblPtr->ATT_MAG_DECL <= ATT_MAG_DECL_MAX))
126  {
127  valid_bool = FALSE;
128  }
129 
130  if(!(QAE_ConfigTblPtr->ATT_MAG_DECL_A == ATT_MAG_DECL_A_MIN || QAE_ConfigTblPtr->ATT_MAG_DECL_A == ATT_MAG_DECL_A_MAX))
131  {
132  valid_bool = FALSE;
133  }
134 
135  if(!(QAE_ConfigTblPtr->ATT_ACC_COMP == ATT_ACC_COMP_MIN || QAE_ConfigTblPtr->ATT_ACC_COMP == ATT_ACC_COMP_MAX))
136  {
137  valid_bool = FALSE;
138  }
139 
140  if(!(QAE_ConfigTblPtr->ATT_BIAS_MAX >= ATT_BIAS_MAX_MIN && QAE_ConfigTblPtr->ATT_BIAS_MAX <= ATT_BIAS_MAX_MAX))
141  {
142  valid_bool = FALSE;
143  }
144 
145 QAE_ValidateConfigTbl_Exit_Tag:
146  if(FALSE == valid_bool)
147  {
148  (void) CFE_EVS_SendEvent(QAE_CFGTBL_VALIDATION_ERR_EID, CFE_EVS_ERROR,
149  "Config params table validation error");
150  iStatus = -1;
151  }
152  return (iStatus);
153 }

158 /* Acquire Config Pointers */
159 /* */
160 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
161 int32 QAE::AcquireConfigPointers(void)
162 {
163  int32 iStatus = CFE_SUCCESS;
164 
165  /*
166  ** Release the table
167  */
168  (void) CFE_TBL_ReleaseAddress(ConfigTblHdl);
169 
170  /*
171  ** Manage the table
172  */
173  iStatus = CFE_TBL_Manage(ConfigTblHdl);
174  if ((iStatus != CFE_SUCCESS) && (iStatus != CFE_TBL_INFO_UPDATED))
175  {
176  (void) CFE_EVS_SendEvent(QAE_CFGTBL_MANAGE_ERR_EID, CFE_EVS_ERROR,
177  "Failed to manage QAE Config table (0x%08lX)",
178  iStatus);
179  goto QAE_AcquireConfigPointers_Exit_Tag;
180  }
181 
182  /*
183  ** Get a pointer to the table
184  */
185  iStatus = CFE_TBL_GetAddress((void**)&ConfigTblPtr, ConfigTblHdl);
186  if (iStatus == CFE_TBL_INFO_UPDATED)
187  {
188  iStatus = CFE_SUCCESS;
189  m_MagDeclination = ConfigTblPtr->ATT_MAG_DECL;
190  }
191  else if(iStatus != CFE_SUCCESS)
192  {
193  ConfigTblPtr = 0;
194  (void) CFE_EVS_SendEvent(QAE_CFGTBL_GETADDR_ERR_EID, CFE_EVS_ERROR,
195  "Failed to get Config table's address (0x%08lX)",
196  iStatus);
197  }
198 
199 QAE_AcquireConfigPointers_Exit_Tag:
200  return (iStatus);
201 }
