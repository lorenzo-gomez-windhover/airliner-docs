55 /* */
56 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
57 int32 VC_InitEvent()
58 {
59  int32 iStatus = CFE_SUCCESS;
60  int32 ind = 0;
61  int32 customEventCount = 0;
62 
63  CFE_EVS_BinFilter_t EventTbl[VC_MAX_EVENT_FILTERS];
64 
65  /* Initialize the event filter table.
66  * Note: 0 is the CFE_EVS_NO_FILTER mask and event 0 is reserved (not used) */
67  CFE_PSP_MemSet(EventTbl, 0x00, sizeof(EventTbl));
68 
69  EventTbl[ ind].EventID = VC_RESERVED_EID;
70  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
71  EventTbl[ ind].EventID = VC_INIT_INF_EID;
72  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
73  EventTbl[ ind].EventID = VC_NOOP_INF_EID;
74  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
75  EventTbl[ ind].EventID = VC_RESET_INF_EID;
76  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
77  EventTbl[ ind].EventID = VC_CMD_INF_EID;
78  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
79  EventTbl[ ind].EventID = VC_DEV_INF_EID;
80  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
81  EventTbl[ ind].EventID = VC_CHA_INF_EID;
82  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
83  EventTbl[ ind].EventID = VC_INIT_ERR_EID;
84  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
85  EventTbl[ ind].EventID = VC_UNINIT_ERR_EID;
86  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
87  EventTbl[ ind].EventID = VC_CMD_ERR_EID;
88  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
89  EventTbl[ ind].EventID = VC_PIPE_ERR_EID;
90  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
91  EventTbl[ ind].EventID = VC_MSGID_ERR_EID;
92  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
93  EventTbl[ ind].EventID = VC_MSGLEN_ERR_EID;
94  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
95 
96  /* Add custom events to the filter table */
97  customEventCount = VC_Custom_Init_EventFilters(ind, EventTbl);
98  if(VC_CUSTOM_ERROR_CODE == customEventCount)
99  {
100  iStatus = CFE_EVS_APP_FILTER_OVERLOAD;
101  (void) CFE_ES_WriteToSysLog("VC - Failed to init custom event filters (0x%08X)\n", (unsigned int)iStatus);
102  goto VC_InitEvent_Exit_Tag;
103  }
104 
105  /* Register the table with CFE */
106  iStatus = CFE_EVS_Register(EventTbl, (ind + customEventCount), CFE_EVS_BINARY_FILTER);
107  if (iStatus != CFE_SUCCESS)
108  {
109  (void) CFE_ES_WriteToSysLog("VC - Failed to register with EVS (0x%08X)\n", (unsigned int)iStatus);
110  }
111 
112 VC_InitEvent_Exit_Tag:
113  return (iStatus);
114 }

261 /* Initialize the application */
262 /* */
263 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
264 int32 VC_InitApp(void)
265 {
266  int32 iStatus = CFE_SUCCESS;
267  int8 hasEvents = 0;
268 
269  iStatus = VC_InitEvent();
270  if (iStatus != CFE_SUCCESS)
271  {
272  (void) CFE_ES_WriteToSysLog("VC - Failed to init events (0x%08X)\n", (unsigned int)iStatus);
273  goto VC_InitApp_Exit_Tag;
274  }
275  else
276  {
277  hasEvents = 1;
278  }
279 
280  iStatus = VC_InitData();
281  if (iStatus != CFE_SUCCESS)
282  {
283  (void) CFE_EVS_SendEvent(VC_INIT_ERR_EID, CFE_EVS_ERROR,
284  "Failed to init data (0x%08X)",
285  (unsigned int)iStatus);
286  goto VC_InitApp_Exit_Tag;
287  }
288 
289  iStatus = VC_InitPipe();
290  if (iStatus != CFE_SUCCESS)
291  {
292  (void) CFE_EVS_SendEvent(VC_INIT_ERR_EID, CFE_EVS_ERROR,
293  "Failed to init pipes (0x%08X)",
294  (unsigned int)iStatus);
295  goto VC_InitApp_Exit_Tag;
296  }
297 
298  /* Initialize data transmission resources */
299  if (FALSE == VC_Transmit_Init())
300  {
301  /* Data transmit sink failed to initialize for now error out */
302  (void) CFE_EVS_SendEvent(VC_INIT_ERR_EID, CFE_EVS_ERROR,"VC_Transmit_Init failed");
303  iStatus = -1;
304  goto VC_InitApp_Exit_Tag;
305  }
306 
307  /* Initialize devices */
308  if (FALSE == VC_Devices_Init())
309  {
310  /* Device failed to initialize for now error out */
311  (void) CFE_EVS_SendEvent(VC_INIT_ERR_EID, CFE_EVS_ERROR,"VC_Devices_Init failed");
312  iStatus = -1;
313  goto VC_InitApp_Exit_Tag;
314  }
315  else
316  {
317  /* Initialization was successful set state */
318  VC_AppData.AppState = VC_INITIALIZED;
319  }
320 
321  /* Start streaming */
322  if (FALSE == VC_Devices_Start())
323  {
324  /* Start streaming failed, raise event but for now don't error out */
325  (void) CFE_EVS_SendEvent(VC_INIT_ERR_EID, CFE_EVS_ERROR,"VC_Devices_Start failed in InitApp");
326  }
327  else
328  {
329  /* Streaming is successful set state */
330  VC_AppData.AppState = VC_STREAMING;
331  }
332 
333  /* Register the cleanup callback */
334  iStatus = OS_TaskInstallDeleteHandler(&VC_CleanupCallback);
335  if (iStatus != CFE_SUCCESS)
336  {
337  (void) CFE_EVS_SendEvent(VC_INIT_ERR_EID, CFE_EVS_ERROR,
338  "Failed to init register cleanup callback (0x%08X)",
339  (unsigned int)iStatus);
340  goto VC_InitApp_Exit_Tag;
341  }
342 
343 VC_InitApp_Exit_Tag:
344  if (iStatus == CFE_SUCCESS)
345  {
346  /* Application startup event message */
347  (void) CFE_EVS_SendEvent(VC_INIT_INF_EID, CFE_EVS_INFORMATION,
348  "Initialized. Version %d.%d.%d.%d",
349  VC_MAJOR_VERSION,
350  VC_MINOR_VERSION,
351  VC_REVISION,
352  VC_MISSION_REV);
353  }
354  else
355  {
356  if (hasEvents == 1)
357  {
358  (void) CFE_EVS_SendEvent(VC_INIT_ERR_EID, CFE_EVS_ERROR, "Application failed to initialize");
359  }
360  else
361  {
362  (void) CFE_ES_WriteToSysLog("VC - Application failed to initialize\n");
363  }
364  }
365  return (iStatus);
366 } /* End of VC_AppInit() */

370 /* Receive and Process Messages */
371 /* */
372 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
373 int32 VC_RcvMsg(int32 iBlocking)
374 {
375  int32 iStatus = CFE_SUCCESS;
376  CFE_SB_Msg_t* MsgPtr = NULL;
377  CFE_SB_MsgId_t MsgId;
378 
379  /* Stop Performance Log entry */
380  CFE_ES_PerfLogExit(VC_MAIN_TASK_PERF_ID);
381 
382  /* Wait for WakeUp messages from scheduler */
383  iStatus = CFE_SB_RcvMsg(&MsgPtr, VC_AppData.SchPipeId, iBlocking);
384 
385  /* Start Performance Log entry */
386  CFE_ES_PerfLogEntry(VC_MAIN_TASK_PERF_ID);
387 
388  if (iStatus == CFE_SUCCESS)
389  {
390  MsgId = CFE_SB_GetMsgId(MsgPtr);
391  switch (MsgId)
392  {
393  case VC_PROCESS_CMDS_MID:
394  {
395  VC_ProcessNewCmds();
396  break;
397  }
398  case VC_SEND_HK_MID:
399  {
400  VC_ReportHousekeeping(MsgPtr);
401  break;
402  }
403  default:
404  {
405  (void) CFE_EVS_SendEvent(VC_MSGID_ERR_EID, CFE_EVS_ERROR,
406  "Recvd invalid SCH msgId (0x%04X)", (unsigned short)MsgId);
407  break;
408  }
409  }
410  }
411  else if (iStatus == CFE_SB_NO_MESSAGE)
412  {
413  /* If there's no incoming message, do nothing here.
414  * Note, this section is dead code only if the iBlocking arg
415  * is CFE_SB_PEND_FOREVER. */
416  iStatus = CFE_SUCCESS;
417  }
418  else if (iStatus == CFE_SB_TIME_OUT)
419  {
420  /* If there's no incoming message within a specified time (via the
421  * iBlocking arg, do nothing here.
422  * Note, this section is dead code only if the iBlocking arg
423  * is CFE_SB_PEND_FOREVER. */
424  iStatus = CFE_SUCCESS;
425  }
426  else
427  {
428  (void) CFE_EVS_SendEvent(VC_PIPE_ERR_EID, CFE_EVS_ERROR,
429  "SB pipe read error (0x%08X), app will exit", (unsigned int)iStatus);
430  VC_AppData.uiRunStatus= CFE_ES_APP_ERROR;
431  }
432 
433  return (iStatus);
434 }

437 /* */
438 /* Process Incoming Commands */
439 /* */
440 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
441 void VC_ProcessNewCmds()
442 {
443  int iStatus = CFE_SUCCESS;
444  CFE_SB_Msg_t* CmdMsgPtr=NULL;
445  CFE_SB_MsgId_t CmdMsgId;
446 
447  /* Process command messages till the pipe is empty */
448  while (1)
449  {
450  iStatus = CFE_SB_RcvMsg(&CmdMsgPtr, VC_AppData.CmdPipeId, CFE_SB_POLL);
451  if(iStatus == CFE_SUCCESS)
452  {
453  CmdMsgId = CFE_SB_GetMsgId(CmdMsgPtr);
454  switch (CmdMsgId)
455  {
456  case VC_CMD_MID:
457  {
458  VC_ProcessNewAppCmds(CmdMsgPtr);
459  break;
460  }
461  default:
462  {
463  /* Bump the command error counter for an unknown command.
464  * (This should only occur if it was subscribed to with this
465  * pipe, but not handled in this switch-case.) */
466  VC_AppData.HkTlm.usCmdErrCnt++;
467  (void) CFE_EVS_SendEvent(VC_MSGID_ERR_EID, CFE_EVS_ERROR,
468  "Recvd invalid CMD msgId (0x%04X)", (unsigned short)CmdMsgId);
469  break;
470  }
471  }
472  }
473  else if (iStatus == CFE_SB_NO_MESSAGE)
474  {
475  break;
476  }
477  else
478  {
479  (void) CFE_EVS_SendEvent(VC_PIPE_ERR_EID, CFE_EVS_ERROR,
480  "CMD pipe read error (0x%08X)", (unsigned int)iStatus);
481  VC_AppData.uiRunStatus = CFE_ES_APP_ERROR;
482  break;
483  }
484  }
485  return;
486 }

521 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
522 /* */
523 /* VC application entry point and main process loop */
524 /* */
525 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
526 void VC_AppMain(void)
527 {
528  /* Setup the RunStatus variable */
529  VC_AppData.uiRunStatus = CFE_ES_APP_RUN;
530 
531  /* Register the application with Executive Services */
532  int32 iStatus = CFE_ES_RegisterApp();
533  if (iStatus != CFE_SUCCESS)
534  {
535  (void) CFE_ES_WriteToSysLog("VC - Failed to register the app (0x%08X)\n", (unsigned int)iStatus);
536  }
537 
538  /* Create the first Performance Log entry */
539  CFE_ES_PerfLogEntry(VC_MAIN_TASK_PERF_ID);
540 
541  /* Perform application initializations */
542  if (iStatus == CFE_SUCCESS)
543  {
544  iStatus = VC_InitApp();
545  }
546 
547  if (iStatus == CFE_SUCCESS)
548  {
549  /* Do not perform performance monitoring on startup sync */
550  CFE_ES_PerfLogExit(VC_MAIN_TASK_PERF_ID);
551  CFE_ES_WaitForStartupSync(VC_STARTUP_TIMEOUT_MSEC);
552  CFE_ES_PerfLogEntry(VC_MAIN_TASK_PERF_ID);
553  }
554  else
555  {
556  VC_AppData.uiRunStatus = CFE_ES_APP_ERROR;
557  }
558 
559  /* Application main loop */
560  while(CFE_ES_RunLoop(&VC_AppData.uiRunStatus) == TRUE)
561  {
562  int32 iStatus = VC_RcvMsg(VC_SCH_PIPE_PEND_TIME);
563  if (iStatus != CFE_SUCCESS)
564  {
565  CFE_EVS_SendEvent(VC_PIPE_ERR_EID, CFE_EVS_ERROR, \
566  "SB Pipe Read Error, VC App will exit. RC = 0x%08X", (unsigned int)iStatus);
567 
568  VC_AppData.uiRunStatus = CFE_ES_APP_ERROR;
569  }
570 
571  } /* end CFS_ES_RunLoop while */
572 
573  /* Pre-normal exit cleanup */
574  VC_CleanupExit();
575 
576  /* Stop Performance Log entry */
577  CFE_ES_PerfLogExit(VC_MAIN_TASK_PERF_ID);
578 
579  /* Exit the application */
580  CFE_ES_ExitApp(VC_AppData.uiRunStatus);
581 } /* end VC_AppMain */
