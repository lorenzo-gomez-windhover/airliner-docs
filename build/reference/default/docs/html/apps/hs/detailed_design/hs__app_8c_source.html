107 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
108 /* */
109 /* HS application entry point and main process loop */
110 /* */
111 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
112 void HS_AppMain(void)
113 {
114  int32 Status = CFE_SUCCESS;
115  uint32 RunStatus = CFE_ES_APP_RUN;
116 
117  /*
118  ** Performance Log, Start
119  */
120  CFE_ES_PerfLogEntry(HS_APPMAIN_PERF_ID);
121 
122  /*
123  ** Register this application with Executive Services
124  */
125  Status = CFE_ES_RegisterApp();
126 
127  /*
128  ** Perform application specific initialization
129  */
130  if (Status == CFE_SUCCESS)
131  {
132  Status = HS_AppInit();
133  }
134 
135  /*
136  ** If no errors were detected during initialization, then wait for everyone to start
137  */
138  if (Status == CFE_SUCCESS)
139  {
140  CFE_ES_WaitForStartupSync(HS_STARTUP_SYNC_TIMEOUT);
141 
142  /*
143  ** Enable and set the watchdog timer
144  */
145  CFE_PSP_WatchdogSet(HS_WATCHDOG_TIMEOUT_VALUE);
146  CFE_PSP_WatchdogService();
147  CFE_PSP_WatchdogEnable();
148  CFE_PSP_WatchdogService();
149 
150  /*
151  ** Subscribe to Event Messages
152  */
153  if (HS_AppData.CurrentEventMonState == HS_STATE_ENABLED)
154  {
155  Status = CFE_SB_SubscribeEx(CFE_EVS_EVENT_MSG_MID,
156  HS_AppData.EventPipe,
157  CFE_SB_Default_Qos,
158  HS_EVENT_PIPE_DEPTH);
159  if (Status != CFE_SUCCESS)
160  {
161  CFE_EVS_SendEvent(HS_SUB_EVS_ERR_EID, CFE_EVS_ERROR,
162  "Error Subscribing to Events,RC=0x%08X",(unsigned int)Status);
163  }
164  }
165  }
166 
167  if (Status != CFE_SUCCESS)
168  {
169  /*
170  ** Set run status to terminate main loop
171  */
172  RunStatus = CFE_ES_APP_ERROR;
173  }
174 
175  /*
176  ** Application main loop
177  */
178  while(CFE_ES_RunLoop(&RunStatus) == TRUE)
179  {
180  /*
181  ** Performance Log, Stop
182  */
183  CFE_ES_PerfLogExit(HS_APPMAIN_PERF_ID);
184 
185  /*
186  ** Task Delay for a configured timeout
187  */
188 #if HS_POST_PROCESSING_DELAY != 0
189  OS_TaskDelay(HS_POST_PROCESSING_DELAY);
190 #endif
191 
192  /*
193  ** Task Delay for a configured timeout
194  */
195  Status = CFE_SB_RcvMsg(&HS_AppData.MsgPtr, HS_AppData.WakeupPipe, HS_WAKEUP_TIMEOUT);
196 
197  /*
198  ** Performance Log, Start
199  */
200  CFE_ES_PerfLogEntry(HS_APPMAIN_PERF_ID);
201 
202  /*
203  ** Process the software bus message
204  */
205  if ((Status == CFE_SUCCESS) ||
206  (Status == CFE_SB_NO_MESSAGE) ||
207  (Status == CFE_SB_TIME_OUT))
208  {
209  Status = HS_ProcessMain();
210  }
211 
212  /*
213  ** Note: If there were some reason to exit the task
214  ** normally (without error) then we would set
215  ** RunStatus = CFE_ES_APP_EXIT
216  */
217  if (Status != CFE_SUCCESS)
218  {
219  /*
220  ** Set request to terminate main loop
221  */
222  RunStatus = CFE_ES_APP_ERROR;
223  }
224 
225  } /* end CFS_ES_RunLoop while */
226 
227  /*
228  ** Check for "fatal" process error...
229  */
230  if (Status != CFE_SUCCESS)
231  {
232  /*
233  ** Send an event describing the reason for the termination
234  */
235  CFE_EVS_SendEvent(HS_APP_EXIT_EID, CFE_EVS_CRITICAL,
236  "Application Terminating, err = 0x%08X", (unsigned int)Status);
237 
238  /*
239  ** In case cFE Event Services is not working
240  */
241  CFE_ES_WriteToSysLog("HS App: Application Terminating, ERR = 0x%08X\n", (unsigned int)Status);
242  }
243 
244  HS_CustomCleanup();
245 
246  /*
247  ** Performance Log, Stop
248  */
249  CFE_ES_PerfLogExit(HS_APPMAIN_PERF_ID);
250 
251  /*
252  ** Exit the application
253  */
254  CFE_ES_ExitApp(RunStatus);
255 
256 } /* end HS_AppMain */

283 #else
284  HS_AppData.ExeCountState = HS_STATE_DISABLED;

299  /*
300  ** Register for event services...
301  */
302  Status = CFE_EVS_Register (NULL, 0, CFE_EVS_BINARY_FILTER);
303  if (Status != CFE_SUCCESS)

306  return (Status);
307  }
308 
309 
310  /*
311  ** Create Critical Data Store
312  */
313  Status = CFE_ES_RegisterCDS(&HS_AppData.MyCDSHandle,
314  sizeof(HS_CDSData_t),
315  HS_CDSNAME);
316 
317  if (Status == CFE_ES_CDS_ALREADY_EXISTS)
318  {
319  /*
320  ** Critical Data Store already existed, we need to get a
321  ** copy of its current contents to see if we can use it
322  */
323  Status = CFE_ES_RestoreFromCDS(&HS_AppData.CDSData, HS_AppData.MyCDSHandle);
324 
325  if (Status == CFE_SUCCESS)
326  {
327  if((HS_AppData.CDSData.ResetsPerformed != (uint16) ~HS_AppData.CDSData.ResetsPerformedNot) ||
328  (HS_AppData.CDSData.MaxResets != (uint16) ~HS_AppData.CDSData.MaxResetsNot))
329  {
330  /*
331  ** Report error restoring data
332  */
333  CFE_EVS_SendEvent(HS_CDS_CORRUPT_ERR_EID, CFE_EVS_ERROR,
334  "Data in CDS was corrupt, initializing resets data");
335  /*
336  ** If data was corrupt, initialize data
337  */
338  HS_SetCDSData(0, HS_MAX_RESTART_ACTIONS);
339  }
340 
341  }
342  else
343  {
344  /*
345  ** Report error restoring data
346  */
347  CFE_EVS_SendEvent(HS_CDS_RESTORE_ERR_EID, CFE_EVS_ERROR,
348  "Failed to restore data from CDS (Err=0x%08x), initializing resets data", (unsigned int)Status);
349  /*
350  ** If data could not be retrieved, initialize data
351  */
352  HS_SetCDSData(0, HS_MAX_RESTART_ACTIONS);
353  }
354 
355  Status = CFE_SUCCESS;
356 
357  }
358  else if (Status == CFE_SUCCESS)
359  {
360  /*
361  ** If CDS did not previously exist, initialize data
362  */
363  HS_SetCDSData(0, HS_MAX_RESTART_ACTIONS);
364  }
365  else
366  {
367  /*
368  ** Disable saving to CDS
369  */
370  HS_AppData.CDSState = HS_STATE_DISABLED;
371 
372  /*
373  ** Initialize values anyway (they will not be saved)
374  */
375  HS_SetCDSData(0, HS_MAX_RESTART_ACTIONS);
376  }
377 
378  /*
379  ** Set up the HS Software Bus
380  */
381  Status = HS_SbInit();
382  if(Status != CFE_SUCCESS)
383  {
384  return (Status);
385  }
386 
387  /*
388  ** Register The HS Tables
389  */
390  Status = HS_TblInit();
391  if(Status != CFE_SUCCESS)
392  {
393  return (Status);
394  }
395 
396  /*
397  ** Perform custom initialization (for cpu utilization monitoring)
398  */
399  Status = HS_CustomInit();
400 
401  /*
402  ** Application initialization event
403  */
404  CFE_EVS_SendEvent (HS_INIT_EID, CFE_EVS_INFORMATION,
405  "HS Initialized. Version %d.%d.%d.%d",
406  HS_MAJOR_VERSION,
407  HS_MINOR_VERSION,
408  HS_REVISION,
409  HS_MISSION_REV);
410 
411  return CFE_SUCCESS;
412 
413 } /* end HS_AppInit */

730  */
731  Status = CFE_SUCCESS;
732  }
733 
734  /*

736  */
737  while (Status == CFE_SUCCESS)
738  {
739  /*

742  Status = CFE_SB_RcvMsg(&HS_AppData.MsgPtr, HS_AppData.CmdPipe, CFE_SB_POLL);
