71 /* Process a command pipe message */
72 /* */
73 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
74 void HS_AppPipe(CFE_SB_MsgPtr_t MessagePtr)
75 {
76  CFE_SB_MsgId_t MessageID = 0;
77  uint16 CommandCode = 0;
78 
79  MessageID = CFE_SB_GetMsgId(MessagePtr);
80  switch (MessageID)
81  {
82 
83  /*
84  ** Housekeeping telemetry request
85  */
86  case HS_SEND_HK_MID:
87  HS_HousekeepingReq(MessagePtr);
88  break;
89 
90  /*
91  ** HS application commands...
92  */
93  case HS_CMD_MID:
94 
95  CommandCode = CFE_SB_GetCmdCode(MessagePtr);
96  switch (CommandCode)
97  {
98  case HS_NOOP_CC:
99  HS_NoopCmd(MessagePtr);
100  break;
101 
102  case HS_RESET_CC:
103  HS_ResetCmd(MessagePtr);
104  break;
105 
106  case HS_ENABLE_APPMON_CC:
107  HS_EnableAppMonCmd(MessagePtr);
108  break;
109 
110  case HS_DISABLE_APPMON_CC:
111  HS_DisableAppMonCmd(MessagePtr);
112  break;
113 
114  case HS_ENABLE_EVENTMON_CC:
115  HS_EnableEventMonCmd(MessagePtr);
116  break;
117 
118  case HS_DISABLE_EVENTMON_CC:
119  HS_DisableEventMonCmd(MessagePtr);
120  break;
121 
122  case HS_ENABLE_ALIVENESS_CC:
123  HS_EnableAlivenessCmd(MessagePtr);
124  break;
125 
126  case HS_DISABLE_ALIVENESS_CC:
127  HS_DisableAlivenessCmd(MessagePtr);
128  break;
129 
130  case HS_RESET_RESETS_PERFORMED_CC:
131  HS_ResetResetsPerformedCmd(MessagePtr);
132  break;
133 
134  case HS_SET_MAX_RESETS_CC:
135  HS_SetMaxResetsCmd(MessagePtr);
136  break;
137 
138  case HS_ENABLE_CPUHOG_CC:
139  HS_EnableCPUHogCmd(MessagePtr);
140  break;
141 
142  case HS_DISABLE_CPUHOG_CC:
143  HS_DisableCPUHogCmd(MessagePtr);
144  break;
145 
146  default:
147  if (HS_CustomCommands(MessagePtr) != CFE_SUCCESS)
148  {
149  CFE_EVS_SendEvent(HS_CC_ERR_EID, CFE_EVS_ERROR,
150  "Invalid command code: ID = 0x%04X, CC = %d",
151  MessageID, CommandCode);
152 
153  HS_AppData.CmdErrCount++;
154  }
155  break;
156 
157  } /* end CommandCode switch */
158  break;
159 
160  /*
161  ** Unrecognized Message ID
162  */
163  default:
164  HS_AppData.CmdErrCount++;
165  CFE_EVS_SendEvent(HS_MID_ERR_EID, CFE_EVS_ERROR,
166  "Invalid command pipe message ID: 0x%04X", MessageID);
167  break;
168 
169  } /* end MessageID switch */
170 
171  return;
172 
173 } /* End HS_AppPipe */

175 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
176 /* */
177 /* Housekeeping request */
178 /* */
179 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
180 void HS_HousekeepingReq(CFE_SB_MsgPtr_t MessagePtr)
181 {
182  uint16 ExpectedLength = sizeof(HS_NoArgsCmd_t);
183  uint32 AppId;
184 #if HS_MAX_EXEC_CNT_SLOTS != 0
185  uint32 ExeCount;
186  uint32 TaskId;
187  CFE_ES_TaskInfo_t TaskInfo;
188 #endif
189  int32 Status;
190  uint32 TableIndex;
191 
192  /*
193  ** Verify message packet length
194  */
195  if(HS_VerifyMsgLength(MessagePtr, ExpectedLength))
196  {
197  /*
198  ** Update HK variables
199  */
200  HS_AppData.HkPacket.CmdCount = (uint8) HS_AppData.CmdCount;
201  HS_AppData.HkPacket.CmdErrCount = (uint8) HS_AppData.CmdErrCount;
202  HS_AppData.HkPacket.CurrentAppMonState = HS_AppData.CurrentAppMonState;
203  HS_AppData.HkPacket.CurrentEventMonState = HS_AppData.CurrentEventMonState;
204  HS_AppData.HkPacket.CurrentAlivenessState = HS_AppData.CurrentAlivenessState;
205  HS_AppData.HkPacket.CurrentCPUHogState = HS_AppData.CurrentCPUHogState;
206  HS_AppData.HkPacket.ResetsPerformed = HS_AppData.CDSData.ResetsPerformed;
207  HS_AppData.HkPacket.MaxResets = HS_AppData.CDSData.MaxResets;
208  HS_AppData.HkPacket.EventsMonitoredCount = HS_AppData.EventsMonitoredCount;
209  HS_AppData.HkPacket.MsgActExec = HS_AppData.MsgActExec;
210 
211  /*
212  ** Calculate the current number of invalid event monitor entries
213  */
214  HS_AppData.HkPacket.InvalidEventMonCount = 0;
215 
216  for(TableIndex = 0; TableIndex < HS_MAX_MONITORED_EVENTS; TableIndex++)
217  {
218  if(HS_AppData.EMTablePtr[TableIndex].ActionType != HS_EMT_ACT_NOACT)
219  {
220  Status = CFE_ES_GetAppIDByName(&AppId, HS_AppData.EMTablePtr[TableIndex].AppName);
221 
222  if (Status == CFE_ES_ERR_APPNAME)
223  {
224  HS_AppData.HkPacket.InvalidEventMonCount++;
225 
226  }
227  }
228  }
229 
230  /*
231  ** Build the HK status flags byte
232  */
233  HS_AppData.HkPacket.StatusFlags = 0;
234 #if HS_MAX_EXEC_CNT_SLOTS != 0
235  if(HS_AppData.ExeCountState == HS_STATE_ENABLED)
236  {
237  HS_AppData.HkPacket.StatusFlags |= HS_LOADED_XCT;
238  }
239 #endif
240  if(HS_AppData.MsgActsState == HS_STATE_ENABLED)
241  {
242  HS_AppData.HkPacket.StatusFlags |= HS_LOADED_MAT;
243  }
244  if(HS_AppData.AppMonLoaded == HS_STATE_ENABLED)
245  {
246  HS_AppData.HkPacket.StatusFlags |= HS_LOADED_AMT;
247  }
248  if(HS_AppData.EventMonLoaded == HS_STATE_ENABLED)
249  {
250  HS_AppData.HkPacket.StatusFlags |= HS_LOADED_EMT;
251  }
252  if(HS_AppData.CDSState == HS_STATE_ENABLED)
253  {
254  HS_AppData.HkPacket.StatusFlags |= HS_CDS_IN_USE;
255  }
256 
257  /*
258  ** Update the AppMon Enables
259  */
260  for(TableIndex = 0; TableIndex <= ((HS_MAX_MONITORED_APPS -1) / HS_BITS_PER_APPMON_ENABLE); TableIndex++)
261  {
262  HS_AppData.HkPacket.AppMonEnables[TableIndex] = HS_AppData.AppMonEnables[TableIndex];
263  }
264 
265 
266  HS_AppData.HkPacket.UtilCpuAvg = HS_AppData.UtilCpuAvg;
267  HS_AppData.HkPacket.UtilCpuPeak = HS_AppData.UtilCpuPeak;
268 
269 #if HS_MAX_EXEC_CNT_SLOTS != 0
270  /*
271  ** Add the execution counters
272  */
273  for(TableIndex = 0; TableIndex < HS_MAX_EXEC_CNT_SLOTS; TableIndex++)
274  {
275 
276  ExeCount = HS_INVALID_EXECOUNT;
277 
278  if((HS_AppData.ExeCountState == HS_STATE_ENABLED) &&
279  ((HS_AppData.XCTablePtr[TableIndex].ResourceType == HS_XCT_TYPE_APP_MAIN) ||
280  (HS_AppData.XCTablePtr[TableIndex].ResourceType == HS_XCT_TYPE_APP_CHILD)))
281  {
282 
283  Status = OS_TaskGetIdByName(&TaskId, HS_AppData.XCTablePtr[TableIndex].ResourceName);
284 
285  if (Status == OS_SUCCESS)
286  {
287  Status = CFE_ES_GetTaskInfo(&TaskInfo, TaskId);
288  if (Status == CFE_SUCCESS)
289  {
290  ExeCount = TaskInfo.ExecutionCounter;
291  }
292 
293  }
294 
295  }
296  else if((HS_AppData.ExeCountState == HS_STATE_ENABLED) &&
297  ((HS_AppData.XCTablePtr[TableIndex].ResourceType == HS_XCT_TYPE_DEVICE) ||
298  (HS_AppData.XCTablePtr[TableIndex].ResourceType == HS_XCT_TYPE_ISR)))
299  {
300 
301  Status = CFE_ES_GetGenCounterIDByName(&TaskId, HS_AppData.XCTablePtr[TableIndex].ResourceName);
302 
303  if (Status == CFE_SUCCESS)
304  {
305  CFE_ES_GetGenCount(TaskId, &ExeCount);
306  }
307 
308  }
309  HS_AppData.HkPacket.ExeCounts[TableIndex] = ExeCount;
310 
311  }
312 
313 #endif
314 
315  /*
316  ** Timestamp and send housekeeping packet
317  */
318  CFE_SB_TimeStampMsg((CFE_SB_Msg_t *) &HS_AppData.HkPacket);
319  CFE_SB_SendMsg((CFE_SB_Msg_t *) &HS_AppData.HkPacket);
320 
321  } /* end HS_VerifyMsgLength if */
322 
323  return;
324 
325 } /* end HS_HousekeepingCmd */
