80 /* Monitor Applications */
81 /* */
82 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
83 void HS_MonitorApplications(void)
84 {
85  CFE_ES_AppInfo_t AppInfo;
86  uint32 AppId;
87  int32 Status;
88  uint32 TableIndex;
89  uint16 ActionType;
90  uint32 MsgActsIndex;
91 
92  for(TableIndex = 0; TableIndex < HS_MAX_MONITORED_APPS; TableIndex++)
93  {
94 
95  ActionType = HS_AppData.AMTablePtr[TableIndex].ActionType;
96 
97  /*
98  ** Check this App if it has an action, and hasn't already expired
99  */
100  if((ActionType != HS_AMT_ACT_NOACT) &&
101  (HS_AppData.AppMonCheckInCountdown[TableIndex] != 0))
102  {
103  Status = CFE_ES_GetAppIDByName(&AppId, HS_AppData.AMTablePtr[TableIndex].AppName);
104 
105  if (Status == CFE_SUCCESS)
106  {
107  Status = CFE_ES_GetAppInfo(&AppInfo, AppId);
108  }
109  else if(HS_AppData.AppMonCheckInCountdown[TableIndex] == HS_AppData.AMTablePtr[TableIndex].CycleCount)
110  {
111  /*
112  ** Only send an event the first time the App fails to resolve
113  */
114  CFE_EVS_SendEvent(HS_APPMON_APPNAME_ERR_EID, CFE_EVS_ERROR,
115  "App Monitor App Name not found: APP:(%s)",
116  HS_AppData.AMTablePtr[TableIndex].AppName);
117  }
118 
119  /*
120  ** Failure to get an execution counter is not considered an automatic failure (or eventworthy)
121  */
122  if((Status == CFE_SUCCESS) &&
123  (HS_AppData.AppMonLastExeCount[TableIndex] != AppInfo.ExecutionCounter))
124  {
125  /*
126  ** Set the current count, and reset the timeout
127  */
128  HS_AppData.AppMonCheckInCountdown[TableIndex] = HS_AppData.AMTablePtr[TableIndex].CycleCount;
129  HS_AppData.AppMonLastExeCount[TableIndex] = AppInfo.ExecutionCounter;
130  }
131  else
132  {
133  HS_AppData.AppMonCheckInCountdown[TableIndex]--;
134 
135  /*
136  ** Take Action once the counter reaches zero
137  */
138  if(HS_AppData.AppMonCheckInCountdown[TableIndex] == 0)
139  {
140 
141  /*
142  ** Unset the enabled bit flag
143  */
144  CFE_CLR(HS_AppData.AppMonEnables[TableIndex / HS_BITS_PER_APPMON_ENABLE],
145  (TableIndex % HS_BITS_PER_APPMON_ENABLE));
146  switch (ActionType)
147  {
148 
149  case HS_AMT_ACT_PROC_RESET:
150  CFE_EVS_SendEvent(HS_APPMON_PROC_ERR_EID, CFE_EVS_ERROR,
151  "App Monitor Failure: APP:(%s): Action: Processor Reset",
152  HS_AppData.AMTablePtr[TableIndex].AppName);
153 
154  /*
155  ** Perform a reset if we can
156  */
157  if(HS_AppData.CDSData.ResetsPerformed < HS_AppData.CDSData.MaxResets)
158  {
159  HS_SetCDSData((HS_AppData.CDSData.ResetsPerformed + 1), HS_AppData.CDSData.MaxResets);
160 
161  OS_TaskDelay(HS_RESET_TASK_DELAY);
162  CFE_ES_WriteToSysLog("HS App: App Monitor Failure: APP:(%s): Action: Processor Reset\n",
163  HS_AppData.AMTablePtr[TableIndex].AppName);
164  HS_AppData.ServiceWatchdogFlag = HS_STATE_DISABLED;
165  CFE_ES_ResetCFE(CFE_ES_PROCESSOR_RESET);
166  }
167  else
168  {
169  CFE_EVS_SendEvent(HS_RESET_LIMIT_ERR_EID, CFE_EVS_ERROR,
170  "Processor Reset Action Limit Reached: No Reset Performed");
171  }
172 
173  break;
174 
175  case HS_AMT_ACT_APP_RESTART:
176  CFE_EVS_SendEvent(HS_APPMON_RESTART_ERR_EID, CFE_EVS_ERROR,
177  "App Monitor Failure: APP:(%s) Action: Restart Application",
178  HS_AppData.AMTablePtr[TableIndex].AppName);
179  /*
180  ** Attempt to restart the App if we resolved the AppId
181  */
182  if (Status == CFE_SUCCESS)
183  {
184  Status = CFE_ES_RestartApp(AppId);
185  }
186 
187  /*
188  ** Report an error; either no valid AppId, or RestartApp failed
189  */
190  if (Status != CFE_SUCCESS)
191  {
192  CFE_EVS_SendEvent(HS_APPMON_NOT_RESTARTED_ERR_EID, CFE_EVS_ERROR,
193  "Call to Restart App Failed: APP:(%s) ERR: 0x%08X",
194  HS_AppData.AMTablePtr[TableIndex].AppName, (unsigned int)Status);
195  }
196 
197  break;
198 
199  case HS_AMT_ACT_EVENT:
200  CFE_EVS_SendEvent(HS_APPMON_FAIL_ERR_EID, CFE_EVS_ERROR,
201  "App Monitor Failure: APP:(%s): Action: Event Only",
202  HS_AppData.AMTablePtr[TableIndex].AppName);
203  break;
204 
205  /*
206  ** Also the case for Message Action types
207  */
208  case HS_AMT_ACT_NOACT:
209  default:
210  /*
211  ** Check to see if this is a Message Action Type
212  */
213  if((HS_AppData.MsgActsState == HS_STATE_ENABLED) &&
214  (ActionType > HS_AMT_ACT_LAST_NONMSG) &&
215  (ActionType <= (HS_AMT_ACT_LAST_NONMSG + HS_MAX_MSG_ACT_TYPES)))
216  {
217  MsgActsIndex = ActionType - HS_AMT_ACT_LAST_NONMSG - 1;
218 
219  /*
220  ** Send the message if off cooldown and not disabled
221  */
222  if((HS_AppData.MsgActCooldown[MsgActsIndex] == 0) &&
223  (HS_AppData.MATablePtr[MsgActsIndex].EnableState != HS_MAT_STATE_DISABLED))
224  {
225  CFE_SB_SendMsg((CFE_SB_Msg_t *) HS_AppData.MATablePtr[MsgActsIndex].Message);
226  HS_AppData.MsgActExec++;
227  HS_AppData.MsgActCooldown[MsgActsIndex] = HS_AppData.MATablePtr[MsgActsIndex].Cooldown;
228  if(HS_AppData.MATablePtr[MsgActsIndex].EnableState != HS_MAT_STATE_NOEVENT)
229  {
230  CFE_EVS_SendEvent(HS_APPMON_MSGACTS_ERR_EID, CFE_EVS_ERROR,
231  "App Monitor Failure: APP:(%s): Action: Message Action Index: %d",
232  HS_AppData.AMTablePtr[TableIndex].AppName, (int)MsgActsIndex);
233 
234  }
235 
236  }
237 
238  }
239 
240  /* Otherwise, Take No Action */
241  break;
242  } /* end switch */
243 
244  } /* end (HS_AppData.AppMonCheckInCountdown[TableIndex] == 0) if */
245 
246  } /* end "failed to update counter" else */
247 
248  } /* end (HS_AppData.AppMonCheckInCountdown[TableIndex] != 0) if */
249 
250  } /* end for loop */
251 
252  return;
253 
254 } /* end HS_MonitorApplications */

271 
272  for(TableIndex = 0; TableIndex < HS_MAX_MONITORED_EVENTS; TableIndex++)
273  {
274  ActionType = HS_AppData.EMTablePtr[TableIndex].ActionType;
275 
276  /*
277  ** Check this Event Monitor if it has an action, and the event IDs match
278  */
279  if ((ActionType != HS_EMT_ACT_NOACT) &&
280  (HS_AppData.EMTablePtr[TableIndex].EventID == EventPtr->Payload.PacketID.EventID))
281  {
282  if ( strncmp(HS_AppData.EMTablePtr[TableIndex].AppName, EventPtr->Payload.PacketID.AppName, OS_MAX_API_NAME) == 0 )
283  {
284 
285  /*
286  ** Perform the action if the strings also match
287  */
288  switch (ActionType)
289  {
290 
291  case HS_EMT_ACT_PROC_RESET:
292  CFE_EVS_SendEvent(HS_EVENTMON_PROC_ERR_EID, CFE_EVS_ERROR,
293  "Event Monitor: APP:(%s) EID:(%d): Action: Processor Reset",
294  HS_AppData.EMTablePtr[TableIndex].AppName,
295  HS_AppData.EMTablePtr[TableIndex].EventID);
296 
297  /*
298  ** Perform a reset if we can
299  */

303 
304  OS_TaskDelay(HS_RESET_TASK_DELAY);
305  CFE_ES_WriteToSysLog("HS App: Event Monitor: APP:(%s) EID:(%d): Action: Processor Reset\n",
306  HS_AppData.EMTablePtr[TableIndex].AppName,
307  (int)HS_AppData.EMTablePtr[TableIndex].EventID);
308  HS_AppData.ServiceWatchdogFlag = HS_STATE_DISABLED;
309  CFE_ES_ResetCFE(CFE_ES_PROCESSOR_RESET);
310  }
311  else
312  {
313  CFE_EVS_SendEvent(HS_RESET_LIMIT_ERR_EID, CFE_EVS_ERROR,
314  "Processor Reset Action Limit Reached: No Reset Performed");
315  }
316 
317 
318  break;
319 
320  case HS_EMT_ACT_APP_RESTART:
321  /*
322  ** Check to see if the App is still there, and try to restart if it is
323  */
324  Status = CFE_ES_GetAppIDByName(&AppId, HS_AppData.EMTablePtr[TableIndex].AppName);
325  if (Status == CFE_SUCCESS)
326  {
327  CFE_EVS_SendEvent(HS_EVENTMON_RESTART_ERR_EID, CFE_EVS_ERROR,
328  "Event Monitor: APP:(%s) EID:(%d): Action: Restart Application",
329  HS_AppData.EMTablePtr[TableIndex].AppName,
330  HS_AppData.EMTablePtr[TableIndex].EventID);
331  Status = CFE_ES_RestartApp(AppId);
332  }
333 
334  if (Status != CFE_SUCCESS)
335  {
336  CFE_EVS_SendEvent(HS_EVENTMON_NOT_RESTARTED_ERR_EID, CFE_EVS_ERROR,
337  "Call to Restart App Failed: APP:(%s) ERR: 0x%08X",
338  HS_AppData.EMTablePtr[TableIndex].AppName, (unsigned int)Status);
339  }
340 
341  break;
342 
343 
344  case HS_EMT_ACT_APP_DELETE:
345  /*
346  ** Check to see if the App is still there, and try to delete if it is
347  */
348  Status = CFE_ES_GetAppIDByName(&AppId, HS_AppData.EMTablePtr[TableIndex].AppName);
349  if (Status == CFE_SUCCESS)
350  {
351  CFE_EVS_SendEvent(HS_EVENTMON_DELETE_ERR_EID, CFE_EVS_ERROR,
352  "Event Monitor: APP:(%s) EID:(%d): Action: Delete Application",
353  HS_AppData.EMTablePtr[TableIndex].AppName,
354  HS_AppData.EMTablePtr[TableIndex].EventID);
355  Status = CFE_ES_DeleteApp(AppId);
356  }
357 
358  if (Status != CFE_SUCCESS)
359  {
360  CFE_EVS_SendEvent(HS_EVENTMON_NOT_DELETED_ERR_EID, CFE_EVS_ERROR,
361  "Call to Delete App Failed: APP:(%s) ERR: 0x%08X",
362  HS_AppData.EMTablePtr[TableIndex].AppName, (unsigned int)Status);
363  }
364 
365  break;
366 
367  /*
368  ** Also the case for Message Action types
369  */
370  case HS_EMT_ACT_NOACT:
371  default:
372  /*
373  ** Check to see if this is a Message Action Type
374  */
375  if((HS_AppData.MsgActsState == HS_STATE_ENABLED) &&
376  (ActionType > HS_EMT_ACT_LAST_NONMSG) &&
377  (ActionType <= (HS_EMT_ACT_LAST_NONMSG + HS_MAX_MSG_ACT_TYPES)))
378  {
379  MsgActsIndex = ActionType - HS_EMT_ACT_LAST_NONMSG - 1;
380 
381  /*
382  ** Send the message if off cooldown and not disabled
383  */
384  if((HS_AppData.MsgActCooldown[MsgActsIndex] == 0) &&
385  (HS_AppData.MATablePtr[MsgActsIndex].EnableState != HS_MAT_STATE_DISABLED))
386  {
387  CFE_SB_SendMsg((CFE_SB_Msg_t *) HS_AppData.MATablePtr[MsgActsIndex].Message);
388  HS_AppData.MsgActExec++;
389  HS_AppData.MsgActCooldown[MsgActsIndex] = HS_AppData.MATablePtr[MsgActsIndex].Cooldown;
390  if(HS_AppData.MATablePtr[MsgActsIndex].EnableState != HS_MAT_STATE_NOEVENT)
391  {
392  CFE_EVS_SendEvent(HS_EVENTMON_MSGACTS_ERR_EID, CFE_EVS_ERROR,
393  "Event Monitor: APP:(%s) EID:(%d): Action: Message Action Index: %d",
394  HS_AppData.EMTablePtr[TableIndex].AppName,
395  HS_AppData.EMTablePtr[TableIndex].EventID, (int)MsgActsIndex);
396  }
397 
398  }
399  }
400 
401  /* Otherwise, Take No Action */
402  break;
403  } /* end switch */
404 
405  } /* end AppName comparison */
406 
407  } /* end EventID comparison */
408 
409  } /* end for loop */
410 
411  return;
412 
413 } /* end HS_MonitorEvent */

417 /* Monitor CPU Utilization and Hogging */
418 /* */
419 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
420 void HS_MonitorUtilization(void)
421 {
422  int32 CurrentUtil;
423  uint32 UtilIndex;
424  uint32 CombinedUtil = 0;
425  uint32 PeakUtil = 0;
426  uint32 ThisUtilIndex = HS_AppData.CurrentCPUUtilIndex;
427 
428  HS_AppData.CurrentCPUUtilIndex++;
429 
430  if(HS_AppData.CurrentCPUUtilIndex >= HS_UTIL_PEAK_NUM_INTERVAL)
431  {
432  HS_AppData.CurrentCPUUtilIndex = 0;
433  }
434 
435  CurrentUtil = HS_CustomGetUtil();
436 
437  if (CurrentUtil > HS_UTIL_PER_INTERVAL_TOTAL)
438  {
439  CurrentUtil = HS_UTIL_PER_INTERVAL_TOTAL;
440  }
441  else if (CurrentUtil < 0)
442  {
443  CurrentUtil = 0;
444  }
445 
446 
447  if ((CurrentUtil >= HS_UTIL_PER_INTERVAL_HOGGING) &&
448  (HS_AppData.CurrentCPUHogState == HS_STATE_ENABLED))
449  {
450  HS_AppData.CurrentCPUHoggingTime++;
451 
452  if (HS_AppData.CurrentCPUHoggingTime == HS_AppData.MaxCPUHoggingTime)
453  {
454  CFE_EVS_SendEvent(HS_CPUMON_HOGGING_ERR_EID, CFE_EVS_ERROR, "CPU Hogging Detected");
455  CFE_ES_WriteToSysLog("HS App: CPU Hogging Detected\n");
456  }
457  }
458  else
459  {
460  HS_AppData.CurrentCPUHoggingTime = 0;
461  }
462 
463  HS_AppData.UtilizationTracker[ThisUtilIndex] = CurrentUtil;
464 
465  for(UtilIndex = 0; UtilIndex < HS_UTIL_PEAK_NUM_INTERVAL; UtilIndex++)
466  {
467  if (HS_AppData.UtilizationTracker[UtilIndex] > PeakUtil)
468  {
469  PeakUtil = HS_AppData.UtilizationTracker[UtilIndex];
470  }
471 
472  if (ThisUtilIndex >= HS_UTIL_AVERAGE_NUM_INTERVAL)
473  {
474  if ((UtilIndex > (ThisUtilIndex - HS_UTIL_AVERAGE_NUM_INTERVAL)) &&
475  (UtilIndex <= ThisUtilIndex))
476  {
477  CombinedUtil += HS_AppData.UtilizationTracker[UtilIndex];
478  }
479  }
480  else
481  {
482  if (UtilIndex <= ThisUtilIndex)
483  {
484  CombinedUtil += HS_AppData.UtilizationTracker[UtilIndex];
485  }
486  else if (UtilIndex > (HS_UTIL_PEAK_NUM_INTERVAL - (HS_UTIL_AVERAGE_NUM_INTERVAL - ThisUtilIndex)))
487  {
488  CombinedUtil += HS_AppData.UtilizationTracker[UtilIndex];
489  }
490 
491  }
492 
493  }
494 
495  HS_AppData.UtilCpuAvg = (CombinedUtil / HS_UTIL_AVERAGE_NUM_INTERVAL);
496  HS_AppData.UtilCpuPeak = PeakUtil;
497 
498  return;
499 
500 } /* end HS_MonitorUtilization */

504 /* Validate the Application Monitor Table */
505 /* */
506 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
507 int32 HS_ValidateAMTable(void *TableData)
508 {
509  HS_AMTEntry_t *TableArray = (HS_AMTEntry_t *) TableData;
510 
511  int32 TableResult = CFE_SUCCESS;
512  uint32 TableIndex;
513  int32 EntryResult;
514 
515  uint16 ActionType;
516  uint16 CycleCount;
517  uint16 NullTerm;
518 
519  uint32 GoodCount = 0;
520  uint32 BadCount = 0;
521  uint32 UnusedCount = 0;
522  char BadName[OS_MAX_API_NAME] = "";
523 
524  for (TableIndex = 0; TableIndex < HS_MAX_MONITORED_APPS; TableIndex++ )
525  {
526 
527  ActionType = TableArray[TableIndex].ActionType;
528  CycleCount = TableArray[TableIndex].CycleCount;
529  NullTerm = TableArray[TableIndex].NullTerm;
530  EntryResult = HS_AMTVAL_NO_ERR;
531 
532  if ((CycleCount == 0) ||
533  (ActionType == HS_AMT_ACT_NOACT))
534  {
535  /*
536  ** Unused table entry
537  */
538  UnusedCount++;
539  }
540  else if (NullTerm != 0)
541  {
542  /*
543  ** Null Terminator Safety Buffer is not Null
544  */
545  EntryResult = HS_AMTVAL_ERR_NUL;
546  BadCount++;
547  }
548  else if(ActionType > (HS_AMT_ACT_LAST_NONMSG + HS_MAX_MSG_ACT_TYPES))
549  {
550  /*
551  ** Action Type is not valid
552  */
553  EntryResult = HS_AMTVAL_ERR_ACT;
554  BadCount++;
555  }
556  else
557  {
558  /*
559  ** Otherwise, this entry is good
560  */
561  GoodCount++;
562  }
563  /*
564  ** Generate detailed event for "first" error
565  */
566  if ((EntryResult != HS_AMTVAL_NO_ERR) && (TableResult == CFE_SUCCESS))
567  {
568  strncpy(BadName,TableArray[TableIndex].AppName,OS_MAX_API_NAME);
569  BadName[OS_MAX_API_NAME-1] = '\0';
570  CFE_EVS_SendEvent(HS_AMTVAL_ERR_EID, CFE_EVS_ERROR,
571  "AppMon verify err: Entry = %d, Err = %d, Action = %d, App = %s",
572  (int)TableIndex, (int)EntryResult, ActionType, BadName );
573  TableResult = EntryResult;
574  }
575 
576  }
577 
578  /*
579  ** Generate informational event with error totals
580  */
581  CFE_EVS_SendEvent(HS_AMTVAL_INF_EID, CFE_EVS_INFORMATION,
582  "AppMon verify results: good = %d, bad = %d, unused = %d",
583  (int)GoodCount, (int)BadCount, (int)UnusedCount);
584 
585  return(TableResult);
586 
587 } /* end HS_ValidateAMTable */
