54 int32 FM_ChildInit(void)
55 {
56  int32 TaskTextLen = 64;
57  char TaskText[TaskTextLen];
58  int32 Result;
59 
60 
61  /* Create counting semaphore (given by parent to wake-up child) */
62  Result = OS_CountSemCreate(&FM_GlobalData.ChildSemaphore, FM_CHILD_SEM_NAME, 0, 0);
63  if (Result != CFE_SUCCESS)
64  {
65  strncpy(TaskText, "create semaphore failed", TaskTextLen);
66  }
67  else
68  {
69  /* Create mutex semaphore (protect access to ChildQueueCount) */
70  Result = OS_MutSemCreate(&FM_GlobalData.ChildQueueCountSem, FM_QUEUE_SEM_NAME, 0);
71 
72  if (Result != CFE_SUCCESS)
73  {
74  strncpy(TaskText, "create queue count semaphore failed", TaskTextLen);
75  }
76  else
77  {
78  /* Create child task (low priority command handler) */
79  Result = CFE_ES_CreateChildTask(&FM_GlobalData.ChildTaskID,
80  FM_CHILD_TASK_NAME,
81  FM_ChildTask, 0,
82  FM_CHILD_TASK_STACK_SIZE,
83  FM_CHILD_TASK_PRIORITY,
84  FM_CHILD_TASK_FLAGS);
85  if (Result != CFE_SUCCESS)
86  {
87  strncpy(TaskText, "create task failed", TaskTextLen);
88  }
89  }
90  }
91 
92  if (Result != CFE_SUCCESS)
93  {
94  CFE_EVS_SendEvent(FM_CHILD_INIT_ERR_EID, CFE_EVS_ERROR,
95  "Child Task initialization error: %s: result = %d",
96  TaskText, (int)Result);
97  }
98 
99  return(Result);
100 
101 } /* End of FM_ChildInit() */

110 void FM_ChildTask(void)
111 {
112  char *TaskText = "Child Task";
113  int32 Result;
114 
115  /*
116  ** The child task runs until the parent dies (normal end) or
117  ** until it encounters a fatal error (semaphore error, etc.)...
118  */
119  Result = CFE_ES_RegisterChildTask();
120 
121  if (Result != CFE_SUCCESS)
122  {
123  CFE_EVS_SendEvent(FM_CHILD_INIT_ERR_EID, CFE_EVS_ERROR,
124  "%s initialization error: register child failed: result = %d",
125  TaskText, (int)Result);
126  }
127  else
128  {
129  CFE_EVS_SendEvent(FM_CHILD_INIT_EID, CFE_EVS_INFORMATION,
130  "%s initialization complete", TaskText);
131 
132  /* Child task process loop */
133  FM_ChildLoop();
134  }
135 
136  /* Stop the parent from invoking the child task */
137  FM_GlobalData.ChildSemaphore = FM_CHILD_SEM_INVALID;
138 
139  /* This call allows cFE to clean-up system resources */
140  CFE_ES_ExitChildTask();
141 
142  return;
143 
144 } /* End of FM_ChildTask() */

148 /* */
149 /* FM child task -- main process loop */
150 /* */
151 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
152 
153 void FM_ChildLoop(void)
154 {
155  char *TaskText = "Child Task termination error: ";
156  int32 Result = CFE_SUCCESS;
157 
158  while (Result == CFE_SUCCESS)
159  {
160  /* Pend on the "handshake" semaphore */
161  Result = OS_CountSemTake(FM_GlobalData.ChildSemaphore);
162 
163  /* Mark the period when this task is active */
164  CFE_ES_PerfLogEntry(FM_CHILD_TASK_PERF_ID);
165 
166  if (Result == CFE_SUCCESS)
167  {
168  /* Make sure the parent/child handshake is not broken */
169  if (FM_GlobalData.ChildQueueCount == 0)
170  {
171  FM_GlobalData.ChildCmdErrCounter++;
172  CFE_EVS_SendEvent(FM_CHILD_TERM_ERR_EID, CFE_EVS_ERROR,
173  "%s empty queue", TaskText);
174 
175  /* Set result that will terminate child task run loop */
176  Result = CFE_OS_ERROR;
177  }
178  else if (FM_GlobalData.ChildReadIndex >= FM_CHILD_QUEUE_DEPTH)
179  {
180  FM_GlobalData.ChildCmdErrCounter++;
181  CFE_EVS_SendEvent(FM_CHILD_TERM_ERR_EID, CFE_EVS_ERROR,
182  "%s invalid queue index: index = %d",
183  TaskText, (int)FM_GlobalData.ChildReadIndex);
184 
185  /* Set result that will terminate child task run loop */
186  Result = CFE_OS_ERROR;
187  }
188  else
189  {
190  /* Invoke the child task command handler */
191  FM_ChildProcess();
192  }
193  }
194  else
195  {
196  CFE_EVS_SendEvent(FM_CHILD_TERM_ERR_EID, CFE_EVS_ERROR,
197  "%s semaphore take failed: result = %d",
198  TaskText, (int)Result);
199  }
200 
201  CFE_ES_PerfLogExit(FM_CHILD_TASK_PERF_ID);
202  }
203 
204  return;
205 
206 } /* End of FM_ChildLoop() */

216 {

228

233  case FM_RENAME_CC:
234  FM_ChildRenameCmd(CmdArgs);

238  FM_ChildDeleteCmd(CmdArgs);
239  break;

242  FM_ChildDeleteAllCmd(CmdArgs);
243  break;
244 
245  case FM_DECOMPRESS_CC:
246  FM_ChildDecompressCmd(CmdArgs);
247  break;
248 
249  case FM_CONCAT_CC:
250  FM_ChildConcatCmd(CmdArgs);
251  break;
252 
253  case FM_CREATE_DIR_CC:
254  FM_ChildCreateDirCmd(CmdArgs);
255  break;
256 
257  case FM_DELETE_DIR_CC:
258  FM_ChildDeleteDirCmd(CmdArgs);
259  break;
260 
261  case FM_GET_FILE_INFO_CC:
262  FM_ChildFileInfoCmd(CmdArgs);
263  break;
264 
265  case FM_GET_DIR_FILE_CC:
266  FM_ChildDirListFileCmd(CmdArgs);
267  break;
268 
269  case FM_GET_DIR_PKT_CC:
270  FM_ChildDirListPktCmd(CmdArgs);
271  break;
272 
273  case FM_DELETE_INT_CC:
274  FM_ChildDeleteCmd(CmdArgs);
275  break;
276 
277  case FM_SET_FILE_PERM_CC:
278  FM_ChildSetPermissionsCmd(CmdArgs);
279  break;
280 
281  default:
282  FM_GlobalData.ChildCmdErrCounter++;
283  CFE_EVS_SendEvent(FM_CHILD_EXE_ERR_EID, CFE_EVS_ERROR,
284  "%s execution error: invalid command code: cc = %d",
285  TaskText, (int)CmdArgs->CommandCode);
286  break;
287  }
288 
289  /* Update the handshake queue read index */
290  FM_GlobalData.ChildReadIndex++;
291 
292  if (FM_GlobalData.ChildReadIndex >= FM_CHILD_QUEUE_DEPTH)
293  {
294  FM_GlobalData.ChildReadIndex = 0;
295  }
296 
297  /* Prevent parent/child updating queue counter at same time */
298  OS_MutSemTake(FM_GlobalData.ChildQueueCountSem);
299  FM_GlobalData.ChildQueueCount--;
300  OS_MutSemGive(FM_GlobalData.ChildQueueCountSem);
301 
302  return;
303 
304 } /* End of FM_ChildProcess() */

491 
492 void FM_ChildDeleteAllCmd(FM_ChildQueueEntry_t *CmdArgs)
493 {
494  char *CmdText = "Delete All Files";
495  char *Directory;
496  char *DirWithSep;
497  os_dirp_t DirPtr;
498  os_dirent_t *DirEntry;
499  int32 OS_Status;
500  uint32 FilenameState;
501  uint32 NameLength;
502  uint32 DeleteCount = 0;
503  uint32 FilesNotDeletedCount = 0;
504  uint32 DirectoriesSkippedCount = 0;
505  char Filename[OS_MAX_PATH_LEN];
506 
507  /* Report current child task activity */
508  FM_GlobalData.ChildCurrentCC = CmdArgs->CommandCode;
509 
510  /*
511  ** Command argument useage for this command:
512  **
513  ** CmdArgs->CommandCode = FM_DELETE_ALL_CC
514  ** CmdArgs->Source1 = directory name
515  ** CmdArgs->Source2 = directory name plus separator
516  */
517  Directory = CmdArgs->Source1;
518  DirWithSep = CmdArgs->Source2;
519 
520  /* Open directory so that we can read from it */
521  DirPtr = OS_opendir(Directory);
522 
523  if (DirPtr == NULL)
524  {
525  FM_GlobalData.ChildCmdErrCounter++;
526 
527  /* Send command failure event (error) */
528  CFE_EVS_SendEvent(FM_DELETE_ALL_OS_ERR_EID, CFE_EVS_ERROR,
529  "%s error: OS_opendir failed: dir = %s",
530  CmdText, Directory);
531  }
532  else
533  {
534  /* Read each directory entry and delete the files */
535  while ((DirEntry = OS_readdir(DirPtr)) != NULL )
536  {
537  /*
538  ** Ignore the "." and ".." directory entries
539  */
540  if ((strcmp(OS_DIRENTRY_NAME(*DirEntry), FM_THIS_DIRECTORY) != 0) &&
541  (strcmp(OS_DIRENTRY_NAME(*DirEntry), FM_PARENT_DIRECTORY) != 0))
542  {
543  /* Construct full path filename */
544  NameLength = strlen(DirWithSep) + strlen(OS_DIRENTRY_NAME(*DirEntry));
545 
546  if (NameLength >= OS_MAX_PATH_LEN)
547  {
548  FilesNotDeletedCount++;
549  }
550  else
551  {
552  /* Note: Directory name already has trailing "/" appended */
553  strcpy(Filename, DirWithSep);
554  strcat(Filename, OS_DIRENTRY_NAME(*DirEntry));
555 
556  /* What kind of directory entry is this? */
557  FilenameState = FM_GetFilenameState(Filename, OS_MAX_PATH_LEN, FALSE);
558 
559  if (FilenameState == FM_NAME_IS_INVALID)
560  {
561  FilesNotDeletedCount++;
562  }
563  else if (FilenameState == FM_NAME_IS_NOT_IN_USE)
564  {
565  /* This result is very unlikely - the */
566  /* name existed a moment ago when */
567  /* the directory entry was read but */
568  /* now the call to OS_stat() failed */
569  /* implying that the entry is gone */
570  FilesNotDeletedCount++;
571  }
572  else if (FilenameState == FM_NAME_IS_DIRECTORY)
573  {
574  DirectoriesSkippedCount++;
575  }
576  else if (FilenameState == FM_NAME_IS_FILE_OPEN)
577  {
578  FilesNotDeletedCount++;
579  }
580  else if (FilenameState == FM_NAME_IS_FILE_CLOSED)
581  {
582  if ((OS_Status = OS_remove(Filename)) == OS_SUCCESS)
583  {
584  /*
585  ** After deleting the file, rewind the directory
586  ** to keep the file system from getting confused
587  */
588  OS_rewinddir(DirPtr);
589 
590  /* Increment delete count */
591  DeleteCount++;
592  }
593  else
594  {
595  FilesNotDeletedCount++;
596  }
597  }
598  }
599  } /* end if "." or ".." directory entries */
600  } /* End while OS_readdir */
601 
602  OS_closedir(DirPtr);
603 
604  /* Send command completion event (info) */
605  CFE_EVS_SendEvent(FM_DELETE_ALL_CMD_EID, CFE_EVS_DEBUG,
606  "%s command: deleted %d files: dir = %s",
607  CmdText, (int)DeleteCount, Directory);
608  FM_GlobalData.ChildCmdCounter++;
609 
610  if ( FilesNotDeletedCount > 0 )
611  {
612  /* If errors occured, report generic event(s) */
613  CFE_EVS_SendEvent(FM_DELETE_ALL_WARNING_EID, CFE_EVS_INFORMATION,
614  "%s command: one or more files could not be deleted. Files may be open : dir = %s",
615  CmdText, Directory);
616  FM_GlobalData.ChildCmdWarnCounter++;
617  }
618 
619  if ( DirectoriesSkippedCount > 0 )
620  {
621  /* If errors occured, report generic event(s) */
622  CFE_EVS_SendEvent(FM_DELETE_ALL_WARNING_EID, CFE_EVS_INFORMATION,
623  "%s command: one or more directories skipped : dir = %s",
624  CmdText, Directory);
625  FM_GlobalData.ChildCmdWarnCounter++;
626  }
627 
628  } /* end if DirPtr == NULL */
629 
630  /* Report previous child task activity */
631  FM_GlobalData.ChildPreviousCC = CmdArgs->CommandCode;
632  FM_GlobalData.ChildCurrentCC = 0;
633 
634  return;
635 
636 } /* End of FM_ChildDeleteAllCmd() */
