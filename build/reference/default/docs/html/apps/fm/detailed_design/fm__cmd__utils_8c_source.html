95  uint32 OpenFilesCount = 0;
96  int32 FDTableIndex;
97  OS_FDTableEntry FDTableEntry;
98  CFE_ES_TaskInfo_t TaskInfo;
99 
100  /* Get system info for each file descriptor table entry */
101  for (FDTableIndex = 0; FDTableIndex < OS_MAX_NUM_OPEN_FILES; FDTableIndex++)
102  {
103  /* If FD table entry is valid - then file is open */
104  if (OS_FDGetInfo(FDTableIndex, &FDTableEntry) == OS_FS_SUCCESS)
105  {
106  /* Getting the list of filenames is optional */
107  if (OpenFilesData != (FM_OpenFilesEntry_t *) NULL)
108  {
109  /* FDTableEntry.Path has logical filename saved when file was opened */
110  strcpy(OpenFilesData[OpenFilesCount].LogicalName, FDTableEntry.Path);

113  CFE_PSP_MemSet(&TaskInfo, 0, sizeof(CFE_ES_TaskInfo_t));

118  }

136 uint32 FM_GetFilenameState(char *Filename, uint32 BufferSize, boolean FileInfoCmd)
137 {
138  OS_FDTableEntry FDTableEntry = {0};
139  os_fstat_t FileStatus = {0};
140  uint32 FilenameState = FM_NAME_IS_INVALID;
141  boolean FilenameIsValid = FALSE;
142  int32 StringLength = 0;
143  uint32 i;
144 
145 
146  /* Search Filename for a string terminator */
147  for (StringLength = 0; StringLength < BufferSize; StringLength++)
148  {
149  if (Filename[StringLength] == '\0')
150  {
151  break;
152  }
153  }
154 
155  /* Verify that Filename is not empty and has a terminator */
156  if ((StringLength > 0) && (StringLength < BufferSize))
157  {
158  /* Verify that the string characters are OK for a filename */
159  FilenameIsValid = CFS_IsValidFilename(Filename, StringLength);
160  }
161 
162  /* If Filename is valid, then determine its state */
163  if (FilenameIsValid)
164  {
165  /* Check to see if Filename is in use */
166  if (OS_stat(Filename, &FileStatus) == OS_SUCCESS)
167  {
168  /* Filename is in use, is it also a directory? */
169 #ifdef OS_FILESTAT_ISDIR
170  if (OS_FILESTAT_ISDIR(FileStatus))
171 #else
172  if (S_ISDIR(FileStatus.st_mode))
173 #endif
174  {
175  /* Filename is a directory */
176  FilenameState = FM_NAME_IS_DIRECTORY;
177  }
178  else
179  {
180  /* Filename is a file, but is it open? */
181  FilenameState = FM_NAME_IS_FILE_CLOSED;
182 
183  /* Search for Filename in current list of open files */
184  for (i = 0; i < OS_MAX_NUM_OPEN_FILES; i++)
185  {
186  /* Get system info for each file descriptor table entry */
187  /* If the FD table entry is valid - then the file is open */
188  if (OS_FDGetInfo(i, &FDTableEntry) == OS_FS_SUCCESS)
189  {
190  if (strcmp(FDTableEntry.Path, Filename) == 0)
191  {
192  FilenameState = FM_NAME_IS_FILE_OPEN;
193  break;
194  }
195  }
196  }
197  }
198 
199  /* Save the last modify time and file size for File Info commands */
200  if (FileInfoCmd)
201  {
202 #ifdef OS_FILESTAT_TIME
203  FM_GlobalData.FileStatTime = OS_FILESTAT_TIME(FileStatus);
204 #else
205  FM_GlobalData.FileStatTime = FileStatus.st_mtime;
206 #endif
207 #ifdef OS_FILESTAT_SIZE
208  FM_GlobalData.FileStatSize = OS_FILESTAT_SIZE(FileStatus);
209 #else
210  FM_GlobalData.FileStatSize = FileStatus.st_size;
211 #endif
212  }
213  }
214  else
215  {
216  /* Cannot get file stat - therefore does not exist */
217  FilenameState = FM_NAME_IS_NOT_IN_USE;
218 
219  /* Save the last modify time and file size for File Info commands */
220  if (FileInfoCmd)
221  {
222  FM_GlobalData.FileStatSize = 0;
223  FM_GlobalData.FileStatTime = 0;
224  }
225  }
226  }
227 
228  return(FilenameState);
229 
230 } /* End FM_GetFilenameState */

524 boolean FM_VerifyChildTask(uint32 EventID, char *CmdText)
525 {
526  boolean Result = FALSE;
527 
528  /* Copy of child queue count that child task cannot change */
529  uint8 LocalQueueCount = FM_GlobalData.ChildQueueCount;
530 
531  /* Verify child task is active and queue interface is healthy */
532  if (FM_GlobalData.ChildSemaphore == FM_CHILD_SEM_INVALID)
533  {
534  CFE_EVS_SendEvent(EventID, CFE_EVS_ERROR,
535  "%s error: child task is disabled", CmdText);
536 
537  /* Child task disabled - cannot add another command */
538  Result = FALSE;
539  }
540  else if (LocalQueueCount == FM_CHILD_QUEUE_DEPTH)
541  {
542  CFE_EVS_SendEvent(EventID, CFE_EVS_ERROR,
543  "%s error: child task queue is full", CmdText);
544 
545  /* Queue full - cannot add another command */
546  Result = FALSE;
547  }
548  else if ((LocalQueueCount > FM_CHILD_QUEUE_DEPTH) ||
549  (FM_GlobalData.ChildWriteIndex >= FM_CHILD_QUEUE_DEPTH))
550  {
551  CFE_EVS_SendEvent(EventID, CFE_EVS_ERROR,
552  "%s error: child task interface is broken: count = %d, index = %d",
553  CmdText, LocalQueueCount, FM_GlobalData.ChildWriteIndex);
554 
555  /* Queue broken - cannot add another command */
556  Result = FALSE;
557  }
558  else
559  {
560  CFE_PSP_MemSet(&FM_GlobalData.ChildQueue[FM_GlobalData.ChildWriteIndex],
561  0, sizeof(FM_ChildQueueEntry_t));
562 
563  /* OK to add another command to the queue */
564  Result = TRUE;
565  }
566 
567  return(Result);
568 
569 } /* End FM_VerifyChildTask */

577 
578 void FM_InvokeChildTask(void)
579 {
580 
581  /* Update callers queue index */
582  FM_GlobalData.ChildWriteIndex++;
583 
584  if (FM_GlobalData.ChildWriteIndex >= FM_CHILD_QUEUE_DEPTH)
585  {
586  FM_GlobalData.ChildWriteIndex = 0;
587  }
588 
589  /* Prevent parent/child updating queue counter at same time */
590  OS_MutSemTake(FM_GlobalData.ChildQueueCountSem);
591  FM_GlobalData.ChildQueueCount++;
592  OS_MutSemGive(FM_GlobalData.ChildQueueCountSem);
593 
594  /* Does the child task still have a semaphore? */
595  if (FM_GlobalData.ChildSemaphore != FM_CHILD_SEM_INVALID)
596  {
597  /* Signal child task to call command handler */
598  OS_CountSemGive(FM_GlobalData.ChildSemaphore);
599  }
600 
601  return;
602 
603 } /* End of FM_InvokeChildTask */
