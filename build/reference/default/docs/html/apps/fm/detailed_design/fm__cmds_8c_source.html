111  char *CmdText = "Copy File";

113 
114  /* Verify command packet length */
115  CommandResult = FM_IsValidCmdPktLength(MessagePtr, sizeof(FM_CopyFileCmd_t),
116  FM_COPY_PKT_ERR_EID, CmdText);
117 
118  /* Verify that overwrite argument is valid */
119  if (CommandResult == TRUE)
120  {
121  CommandResult = FM_VerifyOverwrite(CmdPtr->Overwrite,
122  FM_COPY_OVR_ERR_EID, CmdText);
123  }
124 
125  /* Verify that source file exists and is not a directory */
126  if (CommandResult == TRUE)
127  {
128  CommandResult = FM_VerifyFileExists(CmdPtr->Source, sizeof(CmdPtr->Source),
129  FM_COPY_SRC_ERR_EID, CmdText);
130  }
131 
132  /* Verify target filename per the overwrite argument */
133  if (CommandResult == TRUE)
134  {
135  if (CmdPtr->Overwrite == 0)
136  {
137  CommandResult = FM_VerifyFileNoExist(CmdPtr->Target, sizeof(CmdPtr->Target),
138  FM_COPY_TGT_ERR_EID, CmdText);
139  }
140  else
141  {
142  CommandResult = FM_VerifyFileNotOpen(CmdPtr->Target, sizeof(CmdPtr->Target),
143  FM_COPY_TGT_ERR_EID, CmdText);
144  }
145  }
146 
147  /* Check for lower priority child task availability */
148  if (CommandResult == TRUE)
149  {
150  CommandResult = FM_VerifyChildTask(FM_COPY_CHILD_ERR_EID, CmdText);
151  }
152 
153  /* Prepare command for child task execution */
154  if (CommandResult == TRUE)
155  {
156  CmdArgs = &FM_GlobalData.ChildQueue[FM_GlobalData.ChildWriteIndex];
157 
158  /* Set handshake queue command args */
159  CmdArgs->CommandCode = FM_COPY_CC;
160  strcpy(CmdArgs->Source1, CmdPtr->Source);
161  strcpy(CmdArgs->Target, CmdPtr->Target);
162 
163  /* Invoke lower priority child task */
164  FM_InvokeChildTask();
165  }
166 
167  return(CommandResult);
168 
169 } /* End of FM_CopyFileCmd() */
