78 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
79 int32 MS5611::InitEvent()
80 {
81  int32 iStatus = CFE_SUCCESS;
82  int32 ind = 0;
83  int32 customEventCount = 0;
84 
85  CFE_EVS_BinFilter_t EventTbl[CFE_EVS_MAX_EVENT_FILTERS];
86 
87  /* Initialize the event filter table.
88  * Note: 0 is the CFE_EVS_NO_FILTER mask and event 0 is reserved (not used) */
89  CFE_PSP_MemSet(EventTbl, 0x00, sizeof(EventTbl));
90 
91  /* CFE_EVS_MAX_EVENT_FILTERS limits the number of filters per app. */
92  /* Add platform independent events to filter */
93  EventTbl[ ind].EventID = MS5611_RESERVED_EID;
94  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
95  EventTbl[ ind].EventID = MS5611_READ_ERR_EID;
96  EventTbl[ind++].Mask = CFE_EVS_FIRST_16_STOP;
97  EventTbl[ ind].EventID = MS5611_INIT_INF_EID;
98  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
99  EventTbl[ ind].EventID = MS5611_CMD_NOOP_EID;
100  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
101  EventTbl[ ind].EventID = MS5611_SUBSCRIBE_ERR_EID;
102  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
103  EventTbl[ ind].EventID = MS5611_PIPE_INIT_ERR_EID;
104  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
105  EventTbl[ ind].EventID = MS5611_CFGTBL_MANAGE_ERR_EID;
106  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
107  EventTbl[ ind].EventID = MS5611_CFGTBL_REG_ERR_EID;
108  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
109  EventTbl[ ind].EventID = MS5611_CFGTBL_GETADDR_ERR_EID;
110  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
111  EventTbl[ ind].EventID = MS5611_RCVMSG_ERR_EID;
112  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
113  EventTbl[ ind].EventID = MS5611_MSGID_ERR_EID;
114  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
115  EventTbl[ ind].EventID = MS5611_CC_ERR_EID;
116  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
117  EventTbl[ ind].EventID = MS5611_MSGLEN_ERR_EID;
118  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
119  EventTbl[ ind].EventID = MPU9250_CFGTBL_REG_ERR_EID;
120  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
121  EventTbl[ ind].EventID = MS5611_UNINIT_ERR_EID;
122  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
123  EventTbl[ ind].EventID = MS5611_CFGTBL_LOAD_ERR_EID;
124  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
125  EventTbl[ ind].EventID = MS5611_INIT_ERR_EID;
126  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
127  EventTbl[ ind].EventID = MS5611_READ_ERR_EID;
128  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
129  EventTbl[ ind].EventID = MS5611_CFGTBL_VALIDATION_ERR_EID;
130  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
131 
132  /* Add custom events to the filter table */
133  customEventCount = MS5611_Custom_Init_EventFilters(ind, EventTbl);
134 
135  if(MS5611_ERROR == customEventCount)
136  {
137  iStatus = CFE_EVS_APP_FILTER_OVERLOAD;
138  (void) CFE_ES_WriteToSysLog("Failed to init custom event filters (0x%08X)\n", (unsigned int)iStatus);
139  goto end_of_function;
140  }
141 
142  /* Register the table with CFE */
143  iStatus = CFE_EVS_Register(EventTbl, (ind + customEventCount), CFE_EVS_BINARY_FILTER);
144  if (iStatus != CFE_SUCCESS)
145  {
146  (void) CFE_ES_WriteToSysLog("MS5611 - Failed to register with EVS (0x%08lX)\n", iStatus);
147  }
148 
149 end_of_function:
150 
151  return (iStatus);
152 }

157 /* */
158 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
159 int32 MS5611::InitPipe()
160 {
161  int32 iStatus = CFE_SUCCESS;
162 
163  /* Init schedule pipe and subscribe to wakeup messages */
164  iStatus = CFE_SB_CreatePipe(&SchPipeId,
165  MS5611_SCH_PIPE_DEPTH,
166  MS5611_SCH_PIPE_NAME);
167  if (iStatus == CFE_SUCCESS)
168  {
169  iStatus = CFE_SB_SubscribeEx(MS5611_MEASURE_MID, SchPipeId,
170  CFE_SB_Default_Qos, MS5611_MEASURE_MID_MAX_MSG_COUNT);
171  if (iStatus != CFE_SUCCESS)
172  {
173  (void) CFE_EVS_SendEvent(MS5611_SUBSCRIBE_ERR_EID, CFE_EVS_ERROR,
174  "Sch Pipe failed to subscribe to MS5611_MEASURE_MID. (0x%08lX)",
175  iStatus);
176  goto MS5611_InitPipe_Exit_Tag;
177  }
178 
179  iStatus = CFE_SB_SubscribeEx(MS5611_SEND_HK_MID, SchPipeId,
180  CFE_SB_Default_Qos, MS5611_SEND_HK_MID_MAX_MSG_COUNT);
181  if (iStatus != CFE_SUCCESS)
182  {
183  (void) CFE_EVS_SendEvent(MS5611_SUBSCRIBE_ERR_EID, CFE_EVS_ERROR,
184  "CMD Pipe failed to subscribe to MS5611_SEND_HK_MID. (0x%08X)",
185  (unsigned int)iStatus);
186  goto MS5611_InitPipe_Exit_Tag;
187  }
188  }
189  else
190  {
191  (void) CFE_EVS_SendEvent(MS5611_PIPE_INIT_ERR_EID, CFE_EVS_ERROR,
192  "Failed to create SCH pipe (0x%08lX)",
193  iStatus);
194  goto MS5611_InitPipe_Exit_Tag;
195  }
196 
197  /* Init command pipe and subscribe to command messages */
198  iStatus = CFE_SB_CreatePipe(&CmdPipeId,
199  MS5611_CMD_PIPE_DEPTH,
200  MS5611_CMD_PIPE_NAME);
201  if (iStatus == CFE_SUCCESS)
202  {
203  /* Subscribe to command messages */
204  iStatus = CFE_SB_Subscribe(MS5611_CMD_MID, CmdPipeId);
205 
206  if (iStatus != CFE_SUCCESS)
207  {
208  (void) CFE_EVS_SendEvent(MS5611_SUBSCRIBE_ERR_EID, CFE_EVS_ERROR,
209  "CMD Pipe failed to subscribe to MS5611_CMD_MID. (0x%08lX)",
210  iStatus);
211  goto MS5611_InitPipe_Exit_Tag;
212  }
213  }
214  else
215  {
216  (void) CFE_EVS_SendEvent(MS5611_PIPE_INIT_ERR_EID, CFE_EVS_ERROR,
217  "Failed to create CMD pipe (0x%08lX)",
218  iStatus);
219  goto MS5611_InitPipe_Exit_Tag;
220  }
221 
222 MS5611_InitPipe_Exit_Tag:
223  return (iStatus);
224 }

249 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
250 /* */
251 /* MS5611 initialization */
252 /* */
253 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
254 int32 MS5611::InitApp()
255 {
256  int32 iStatus = CFE_SUCCESS;
257  int8 hasEvents = 0;
258  osalbool returnBool = TRUE;
259  int32 i = 0;
260 
261  iStatus = InitEvent();
262  if (iStatus != CFE_SUCCESS)
263  {
264  (void) CFE_ES_WriteToSysLog("MS5611 - Failed to init events (0x%08lX)\n", iStatus);
265  goto MS5611_InitApp_Exit_Tag;
266  }
267  else
268  {
269  hasEvents = 1;
270  }
271 
272  iStatus = InitPipe();
273  if (iStatus != CFE_SUCCESS)
274  {
275  goto MS5611_InitApp_Exit_Tag;
276  }
277 
278  iStatus = InitConfigTbl();
279  if (iStatus != CFE_SUCCESS)
280  {
281  goto MS5611_InitApp_Exit_Tag;
282  }
283 
284  /* Init data include copy params from config table */
285  InitData();
286 
287  returnBool = MS5611_Custom_Init();
288  if (FALSE == returnBool)
289  {
290  iStatus = MS5611_ERROR;
291  goto MS5611_InitApp_Exit_Tag;
292  }
293 
294  /* Get calibration coefficients from device PROM */
295  for(i = 0; i < MS5611_COEF_SIZE; ++i)
296  {
297  returnBool = MS5611_ReadPROM(i, &MS5611_Coefficients[i]);
298  /* Copy to diagnostic message */
299  Diag.Coefficients[i] = MS5611_Coefficients[i];
300  if (FALSE == returnBool)
301  {
302  iStatus = MS5611_ERROR;
303  (void) CFE_EVS_SendEvent(MS5611_INIT_ERR_EID, CFE_EVS_ERROR,
304  "Failed read from device PROM");
305  goto MS5611_InitApp_Exit_Tag;
306  }
307  }
308 
309  /* Validate the CRC code */
310  returnBool = ValidateCRC();
311  if (FALSE == returnBool)
312  {
313  (void) CFE_EVS_SendEvent(MS5611_INIT_ERR_EID, CFE_EVS_ERROR,
314  "Failed CRC check");
315  iStatus = MS5611_ERROR;
316  goto MS5611_InitApp_Exit_Tag;
317  }
318 
319  /* Register the cleanup callback */
320  iStatus = OS_TaskInstallDeleteHandler(&MS5611_CleanupCallback);
321  if (iStatus != CFE_SUCCESS)
322  {
323  (void) CFE_EVS_SendEvent(MS5611_INIT_ERR_EID, CFE_EVS_ERROR,
324  "Failed to init register cleanup callback (0x%08X)",
325  (unsigned int)iStatus);
326  goto MS5611_InitApp_Exit_Tag;
327  }
328 
329  /* Set application state to initialized */
330  HkTlm.State = MS5611_INITIALIZED;
331 
332 MS5611_InitApp_Exit_Tag:
333  if (iStatus == CFE_SUCCESS)
334  {
335  (void) CFE_EVS_SendEvent(MS5611_INIT_INF_EID, CFE_EVS_INFORMATION,
336  "Initialized. Version %d.%d.%d.%d",
337  MS5611_MAJOR_VERSION,
338  MS5611_MINOR_VERSION,
339  MS5611_REVISION,
340  MS5611_MISSION_REV);
341  }
342  else
343  {
344  if (hasEvents == 1)
345  {
346  (void) CFE_ES_WriteToSysLog("MS5611 - Application failed to initialize\n");
347  }
348  }
349 
350  return (iStatus);
351 }

355 /* Receive and Process Messages */
356 /* */
357 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
358 int32 MS5611::RcvSchPipeMsg(int32 iBlocking)
359 {
360  int32 iStatus = CFE_SUCCESS;
361  CFE_SB_Msg_t* MsgPtr = NULL;
362  CFE_SB_MsgId_t MsgId = 0;
363 
364  /* Stop Performance Log entry */
365  CFE_ES_PerfLogExit(MS5611_MAIN_TASK_PERF_ID);
366 
367  /* Wait for WakeUp messages from scheduler */
368  iStatus = CFE_SB_RcvMsg(&MsgPtr, SchPipeId, iBlocking);
369 
370  /* Start Performance Log entry */
371  CFE_ES_PerfLogEntry(MS5611_MAIN_TASK_PERF_ID);
372 
373  if (iStatus == CFE_SUCCESS)
374  {
375  MsgId = CFE_SB_GetMsgId(MsgPtr);
376  switch (MsgId)
377  {
378  case MS5611_SEND_HK_MID:
379  {
380  ProcessCmdPipe();
381  ReportHousekeeping();
382  break;
383  }
384  case MS5611_MEASURE_MID:
385  {
386  ReadDevice();
387  break;
388  }
389  default:
390  {
391  (void) CFE_EVS_SendEvent(MS5611_MSGID_ERR_EID, CFE_EVS_ERROR,
392  "Recvd invalid SCH msgId (0x%04X)", MsgId);
393  break;
394  }
395  }
396  }
397  else if (iStatus == CFE_SB_NO_MESSAGE)
398  {
399  /* If there's no incoming message, do nothing.
400  * Note, this section is dead code only if the iBlocking arg
401  * is CFE_SB_PEND_FOREVER. */
402  iStatus = CFE_SUCCESS;
403  }
404  else if (iStatus == CFE_SB_TIME_OUT)
405  {
406  /* If there's no incoming message within a specified time (via the
407  * iBlocking arg, do nothing.
408  * Note, this section is dead code only if the iBlocking arg
409  * is CFE_SB_PEND_FOREVER. */
410  iStatus = CFE_SUCCESS;
411  }
412  else
413  {
414  (void) CFE_EVS_SendEvent(MS5611_RCVMSG_ERR_EID, CFE_EVS_ERROR,
415  "SCH pipe read error (0x%08lX).", iStatus);
416  }
417 
418  return (iStatus);
419 }

421 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
422 /* */
423 /* Process Incoming Commands */
424 /* */
425 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
426 void MS5611::ProcessCmdPipe()
427 {
428  int32 iStatus = CFE_SUCCESS;
429  CFE_SB_Msg_t* CmdMsgPtr = NULL;
430  CFE_SB_MsgId_t CmdMsgId = 0;
431 
432  /* Process command messages until the pipe is empty */
433  while (1)
434  {
435  iStatus = CFE_SB_RcvMsg(&CmdMsgPtr, CmdPipeId, CFE_SB_POLL);
436  if(iStatus == CFE_SUCCESS)
437  {
438  CmdMsgId = CFE_SB_GetMsgId(CmdMsgPtr);
439  switch (CmdMsgId)
440  {
441  case MS5611_CMD_MID:
442  {
443  ProcessAppCmds(CmdMsgPtr);
444  break;
445  }
446 
447  default:
448  {
449  /* Bump the command error counter for an unknown command.
450  * (This should only occur if it was subscribed to with this
451  * pipe, but not handled in this switch-case.) */
452  HkTlm.usCmdErrCnt++;
453  (void) CFE_EVS_SendEvent(MS5611_MSGID_ERR_EID, CFE_EVS_ERROR,
454  "Recvd invalid CMD msgId (0x%04X)", (unsigned short)CmdMsgId);
455  break;
456  }
457  }
458  }
459  else if (iStatus == CFE_SB_NO_MESSAGE)
460  {
461  break;
462  }
463  else
464  {
465  (void) CFE_EVS_SendEvent(MS5611_RCVMSG_ERR_EID, CFE_EVS_ERROR,
466  "CMD pipe read error (0x%08lX)", iStatus);
467  break;
468  }
469  }
470  return;
471 }

473 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
474 /* */
475 /* Process MS5611 Commands */
476 /* */
477 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
478 void MS5611::ProcessAppCmds(CFE_SB_Msg_t* MsgPtr)
479 {
480  uint32 uiCmdCode = 0;
481 
482  if (MsgPtr != NULL)
483  {
484  uint16 ExpectedLength = sizeof(MS5611_NoArgCmd_t);
485  /* Length verification is then the same for all three commands */
486  if (VerifyCmdLength(MsgPtr, ExpectedLength))
487  {
488  uiCmdCode = CFE_SB_GetCmdCode(MsgPtr);
489  switch (uiCmdCode)
490  {
491  case MS5611_NOOP_CC:
492  {
493  HkTlm.usCmdCnt++;
494  (void) CFE_EVS_SendEvent(MS5611_CMD_NOOP_EID, CFE_EVS_INFORMATION,
495  "Recvd NOOP. Version %d.%d.%d.%d",
496  MS5611_MAJOR_VERSION,
497  MS5611_MINOR_VERSION,
498  MS5611_REVISION,
499  MS5611_MISSION_REV);
500  break;
501  }
502  case MS5611_RESET_CC:
503  {
504  HkTlm.usCmdCnt = 0;
505  HkTlm.usCmdErrCnt = 0;
506  break;
507  }
508  case MS5611_SEND_DIAG_TLM_CC:
509  {
510  HkTlm.usCmdCnt++;
511  ReportDiagnostic();
512  break;
513  }
514  default:
515  {
516  HkTlm.usCmdErrCnt++;
517  (void) CFE_EVS_SendEvent(MS5611_CC_ERR_EID, CFE_EVS_ERROR,
518  "Recvd invalid command code (%u)", (unsigned int)uiCmdCode);
519  break;
520  }
521  }
522  }
523  }
524  return;
525 }

553 /* Verify Command Length */
554 /* */
555 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
556 osalbool MS5611::VerifyCmdLength(CFE_SB_Msg_t* MsgPtr,
557  uint16 usExpectedLen)
558 {
559  osalbool bResult = TRUE;
560  uint16 usMsgLen = 0;
561 
562  if (MsgPtr != NULL)
563  {
564  usMsgLen = CFE_SB_GetTotalMsgLength(MsgPtr);
565 
566  if (usExpectedLen != usMsgLen)
567  {
568  bResult = FALSE;
569  CFE_SB_MsgId_t MsgId = CFE_SB_GetMsgId(MsgPtr);
570  uint16 usCmdCode = CFE_SB_GetCmdCode(MsgPtr);
571 
572  (void) CFE_EVS_SendEvent(MS5611_MSGLEN_ERR_EID, CFE_EVS_ERROR,
573  "Rcvd invalid msgLen: msgId=0x%08X, cmdCode=%d, "
574  "msgLen=%d, expectedLen=%d",
575  MsgId, usCmdCode, usMsgLen, usExpectedLen);
576  HkTlm.usCmdErrCnt++;
577  }
578  }
579 
580  return (bResult);
581 }

652 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
653 /* */
654 /* Get a raw pressure and temperature measurement from the device */
655 /* */
656 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
657 osalbool MS5611::GetMeasurement(int32 *Pressure, int32 *Temperature)
658 {
659  /* ADC value of the pressure conversion */
660  uint32 D1 = 0;
661  /* difference between actual and measured temp */
662  int32 dT = 0;
663  /* offset at actual temperature */
664  int64 OFF = 0;
665  /* sensitivity at actual temperature */
666  int64 SENS = 0;
667  int32 T2 = 0;
668  int64 OFF2 = 0;
669  int64 SENS2 = 0;
670  int64 TEMP = 0;
671  int32 TempValidate = 0;
672  int32 PressValidate = 0;
673  osalbool returnBool = TRUE;
674 
675  returnBool = MS5611_D1Conversion();
676  if (FALSE == returnBool)
677  {
678  (void) CFE_EVS_SendEvent(MS5611_READ_ERR_EID, CFE_EVS_ERROR,
679  "Get measurement D1 conversion failed");
680  returnBool = FALSE;
681  goto end_of_function;
682  }
683 
684  returnBool = MS5611_ReadADCResult(&D1);
685  if (FALSE == returnBool)
686  {
687  (void) CFE_EVS_SendEvent(MS5611_READ_ERR_EID, CFE_EVS_ERROR,
688  "Read ADC result D1 failed");
689  returnBool = FALSE;
690  goto end_of_function;
691  }
692 
693  /* Measurement ratio pressure per temperature measurements */
694  if(0 == Diag.MeasureCount % MS5611_PRESS_TEMP_MEAS_RATIO)
695  {
696  returnBool = MS5611_D2Conversion();
697  if (FALSE == returnBool)
698  {
699  (void) CFE_EVS_SendEvent(MS5611_READ_ERR_EID, CFE_EVS_ERROR,
700  "Get measurement D2 conversion failed");
701  returnBool = FALSE;
702  goto end_of_function;
703  }
704 
705  returnBool = MS5611_ReadADCResult(&D2);
706  if (FALSE == returnBool)
707  {
708  (void) CFE_EVS_SendEvent(MS5611_READ_ERR_EID, CFE_EVS_ERROR,
709  "Read ADC result D2 failed");
710  returnBool = FALSE;
711  goto end_of_function;
712  }
713  }
714 
715  if(0xFF == Diag.MeasureCount)
716  {
717  /* Roll over */
718  Diag.MeasureCount = 0;
719  }
720  /* Increment the measure count */
721  Diag.MeasureCount++;
722 
723  /*
724  * The following equations are from the MS5611-01BA03 spec sheet which can
725  * be found on the architecture page of this app on Hangar at:
726  * https://hangar.windhoverlabs.com/wiki/x/EwCaB
727  */
728 
729  /* D2 - C5 * 2^8 */
730  dT = (int32)D2 - ((int32)MS5611_Coefficients[5] << 8);
731 
732  /* The following two equations must be solved with 64-bit integers (long long int)
733  * or overflow could occur. */
734 
735  /* C2 * 2^16 + (C4 * dT )/ 2^7 */
736  OFF = ((int64)MS5611_Coefficients[2] << 16) + (((int64)MS5611_Coefficients[4] * dT) >> 7);
737  /* C1 * 2^15 + (C3 * dT )/ 2^8 */
738  SENS = ((int64)MS5611_Coefficients[1] << 15) + (((int64)MS5611_Coefficients[3] * dT) >> 8);
739  /* 2000 + dT * C6 / 2^23 */
740  TempValidate = 2000 + (int32)(((int64)dT * MS5611_Coefficients[6]) >> 23);
741 
742  /* Validate temperature */
743  if(TempValidate > MS5611_TEMP_MIN && TempValidate < MS5611_TEMP_MAX)
744  {
745  *Temperature = TempValidate;
746  }
747  else
748  {
749  (void) CFE_EVS_SendEvent(MS5611_READ_ERR_EID, CFE_EVS_ERROR,
750  "Temperature out of range value = %ld", TempValidate);
751  returnBool = FALSE;
752  goto end_of_function;
753  }
754  /* Second order temperature compensation */
755  if(*Temperature < 2000)
756  {
757  T2 = (dT * dT) >> 31;
758  TEMP = (int64)*Temperature - 2000;
759  OFF2 = (5 * TEMP * TEMP) >> 1;
760  SENS2 = (5 * TEMP * TEMP) >> 2;
761 
762  if(*Temperature < -1500)
763  {
764  TEMP = *Temperature + 1500;
765  OFF2 += 7 * TEMP * TEMP;
766  SENS2 += (11 * TEMP * TEMP) >> 1;
767  }
768 
769  *Temperature -= T2;
770  OFF -= OFF2;
771  SENS -= SENS2;
772  }
773  /* (D1 * SENS / 2^21 - OFF) / 2^15*/
774  PressValidate = ((((D1 * SENS) >> 21) - OFF ) >> 15);
775 
776  /* Validate pressure */
777  if(PressValidate > MS5611_PRESS_MIN && PressValidate < MS5611_PRESS_MAX)
778  {
779  *Pressure = PressValidate;
780  }
781  else
782  {
783  (void) CFE_EVS_SendEvent(MS5611_READ_ERR_EID, CFE_EVS_ERROR,
784  "Pressure out of range value = %ld", PressValidate);
785  returnBool = FALSE;
786  goto end_of_function;
787  }
788 
789  /* Update diagnostic message */
790  Diag.RawPressure = *Pressure;
791  Diag.RawTemperature = *Temperature;
792 
793 end_of_function:
794  return (returnBool);
795 }

797 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
798 /* */
799 /* Read from the device and convert to altitude */
800 /* */
801 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
802 void MS5611::ReadDevice(void)
803 {
804  int32 pressure = 0;
805  int32 temperature = 0;
806  osalbool returnBool = TRUE;
807 
808  returnBool = GetMeasurement(&pressure, &temperature);
809  if (FALSE == returnBool)
810  {
811  (void) CFE_EVS_SendEvent(MS5611_READ_ERR_EID, CFE_EVS_ERROR,
812  "Read failure, altitude not updated");
813  goto end_of_function;
814  }
815  else
816  {
817  /* Stamp time */
818  SensorBaro.Timestamp = PX4LIB_GetPX4TimeUs();
819  /* Convert to Celsius */
820  SensorBaro.Temperature = temperature / 100.0f;
821  /* convert to millibar */
822  SensorBaro.Pressure = pressure / 100.0f;
823 
824  /* tropospheric properties (0-11km) for standard atmosphere */
825  /* temperature at base height in Kelvin */
826  const double T1 = 15.0 + 273.15;
827  /* temperature gradient in degrees per metre */
828  const double a = -6.5 / 1000;
829  /* gravity constant in m/s/s */
830  const double g = 9.80665;
831  /* ideal gas constant in J/kg/K */
832  const double R = 287.05;
833  /* measured pressure in kPa */
834  double p = static_cast<double>(pressure) / 1000.0;
835 
836  /* current pressure at MSL in kPa double p1 = 101325 / 1000.0 */
837 
838  /*
839  * This equation is from the Derivation Relating Altitude to Air
840  * Pressure paper (Figure 8) which can be found at:
841  *
842  * https://hangar.windhoverlabs.com/wiki/x/EwCaB
843  *
844  * Solve:
845  *
846  * / -(aR / g) \
847  * | (p / p1) . T1 | - T1
848  * \ /
849  * h = ------------------------------- + h1
850  * a
851  *
852  * Note: h1 is the base height of an atmospheric region or layer and
853  * is not needed at low altitudes.
854  */
855  SensorBaro.Altitude = (((pow((p / ConfigTblPtr->p1), (-(a * R) / g))) * T1) - T1) / a;
856 
857  /* Update diagnostic message */
858  Diag.Pressure = SensorBaro.Pressure;
859  Diag.Temperature = SensorBaro.Temperature;
860  Diag.Altitude = SensorBaro.Altitude;
861 
862  /* Send the SensorBaro message */
863  CFE_SB_TimeStampMsg((CFE_SB_Msg_t*)&SensorBaro);
864  CFE_SB_SendMsg((CFE_SB_Msg_t*)&SensorBaro);
865  }
866 
867 end_of_function:
868  return;
869 }

871 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
872 /* */
873 /* Calculate a CRC4 code */
874 /* */
875 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
876 uint8 MS5611::CRC4(uint16 n_prom[])
877 {
878  uint32 cnt = 0;
879  /* crc remainder */
880  uint16 n_rem = 0;
881  /* original crc value */
882  uint16 crc_read = 0;
883  uint8 n_bit = 0;
884  n_rem = 0x00;
885  /* save the original CRC */
886  crc_read = n_prom[7];
887  /* replace the crc byte with 0 */
888  n_prom[7] = (0xFF00 & (n_prom[7]));
889  /* operation is performed on bytes */
890  for (cnt = 0; cnt < 16; ++cnt)
891  {
892  /* choose LSB or MSB */
893  if (cnt%2 == 1)
894  {
895  n_rem ^= (uint16) ((n_prom[cnt>>1]) & 0x00FF);
896  }
897  else
898  {
899  n_rem ^= (uint16) (n_prom[cnt>>1]>>8);
900  }
901  for (n_bit = 8; n_bit > 0; n_bit--)
902  {
903  if (n_rem & (0x8000))
904  {
905  n_rem = (n_rem << 1) ^ 0x3000;
906  }
907  else
908  {
909  n_rem = (n_rem << 1);
910  }
911  }
912  }
913  /* the final 4-bit remainder is the CRC code */
914  n_rem = (0x000F & (n_rem >> 12));
915  /* Restore the crc_read to its original value */
916  n_prom[7] = crc_read;
917  return (n_rem ^ 0x00);
918 }

937

941  }
942  else
943  {

945  "CRC check failed PROM = %u: CRC = %u",
946  (unsigned int)promCRC, (unsigned int) returnedCRC);
947  }
948  return (returnBool);
