348 /* Receive and Process Messages */
349 /* */
350 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
351 int32 NAV::RcvSchPipeMsg(int32 iBlocking)
352 {
353  int32 iStatus = CFE_SUCCESS;
354  CFE_SB_Msg_t* MsgPtr = NULL;
355  CFE_SB_MsgId_t MsgId;
356 
357  /* Stop Performance Log entry */
358  CFE_ES_PerfLogExit(NAV_MAIN_TASK_PERF_ID);
359 
360  /* Wait for WakeUp messages from scheduler */
361  iStatus = CFE_SB_RcvMsg(&MsgPtr, SchPipeId, iBlocking);
362 
363  /* Start Performance Log entry */
364  CFE_ES_PerfLogEntry(NAV_MAIN_TASK_PERF_ID);
365 
366  if (iStatus == CFE_SUCCESS)
367  {
368  MsgId = CFE_SB_GetMsgId(MsgPtr);
369  switch (MsgId)
370  {
371  case NAV_WAKEUP_MID:
372  /* Check global position availability */
373  if (!CVT.VehicleGlobalPosition.Timestamp == 0)
374  {
375  Execute();
376  }
377 
378  break;
379 
380  case NAV_SEND_HK_MID:
381  ProcessCmdPipe();
382  ReportHousekeeping();
383  break;
384 
385  case PX4_HOME_POSITION_MID:
386  CFE_PSP_MemCpy(&CVT.HomePositionMsg, MsgPtr, sizeof(CVT.HomePositionMsg));
387  break;
388 
389  case PX4_MISSION_MID:
390  CFE_PSP_MemCpy(&CVT.MissionMsg, MsgPtr, sizeof(CVT.MissionMsg));
391  break;
392 
393  case PX4_VEHICLE_GLOBAL_POSITION_MID:
394  CFE_PSP_MemCpy(&CVT.VehicleGlobalPosition, MsgPtr,
395  sizeof(CVT.VehicleGlobalPosition));
396  break;
397 
398  case PX4_VEHICLE_STATUS_MID:
399  CFE_PSP_MemCpy(&CVT.VehicleStatusMsg, MsgPtr, sizeof(CVT.VehicleStatusMsg));
400  break;
401 
402  case PX4_VEHICLE_LAND_DETECTED_MID:
403  CFE_PSP_MemCpy(&CVT.VehicleLandDetectedMsg, MsgPtr,
404  sizeof(CVT.VehicleLandDetectedMsg));
405  break;
406 
407  case PX4_VEHICLE_LOCAL_POSITION_MID:
408  CFE_PSP_MemCpy(&CVT.VehicleLocalPositionMsg, MsgPtr,
409  sizeof(CVT.VehicleLocalPositionMsg));
410  break;
411 
412  case PX4_VEHICLE_COMMAND_MID:
413  NewCommandArrived = TRUE;
414  CFE_PSP_MemCpy(&CVT.VehicleCommandMsg, MsgPtr,
415  sizeof(CVT.VehicleCommandMsg));
416  break;
417 
418  default:
419  (void) CFE_EVS_SendEvent(NAV_MSGID_ERR_EID, CFE_EVS_ERROR,
420  "Recvd invalid SCH msgId (0x%04X)", MsgId);
421  break;
422  }
423  }
424  else if (iStatus == CFE_SB_NO_MESSAGE)
425  {
426  iStatus = CFE_SUCCESS;
427  }
428  else if (iStatus == CFE_SB_TIME_OUT)
429  {
430  iStatus = CFE_SUCCESS;
431  }
432  else
433  {
434  (void) CFE_EVS_SendEvent(NAV_RCVMSG_ERR_EID, CFE_EVS_ERROR,
435  "SCH pipe read error (0x%08lX).", iStatus);
436  }
437 
438  return (iStatus);
439 }

443 /* Process Incoming Commands */
444 /* */
445 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
446 void NAV::ProcessCmdPipe()
447 {
448  int32 iStatus = CFE_SUCCESS;
449  CFE_SB_Msg_t* CmdMsgPtr = NULL;
450  CFE_SB_MsgId_t CmdMsgId;
451 
452  /* Process command messages until the pipe is empty */
453  while (1)
454  {
455  iStatus = CFE_SB_RcvMsg(&CmdMsgPtr, CmdPipeId, CFE_SB_POLL);
456  if (iStatus == CFE_SUCCESS)
457  {
458  CmdMsgId = CFE_SB_GetMsgId(CmdMsgPtr);
459  switch (CmdMsgId)
460  {
461  case NAV_CMD_MID:
462  ProcessAppCmds(CmdMsgPtr);
463  break;
464 
465  default:
466  /* Bump the command error counter for an unknown command.
467  * (This should only occur if it was subscribed to with this
468  * pipe, but not handled in this switch-case.) */
469  HkTlm.usCmdErrCnt++;
470  (void) CFE_EVS_SendEvent(NAV_MSGID_ERR_EID, CFE_EVS_ERROR,
471  "Recvd invalid CMD msgId (0x%04X)",
472  (unsigned short) CmdMsgId);
473  break;
474  }
475  }
476  else if (iStatus == CFE_SB_NO_MESSAGE)
477  {
478  break;
479  }
480  else
481  {
482  (void) CFE_EVS_SendEvent(NAV_RCVMSG_ERR_EID, CFE_EVS_ERROR,
483  "CMD pipe read error (0x%08lX)", iStatus);
484  break;
485  }
486  }
487 }

495 void NAV::ProcessAppCmds(CFE_SB_Msg_t* MsgPtr)
496 {
497  uint32 uiCmdCode = 0;
498 
499  if (MsgPtr != NULL)
500  {
501  uiCmdCode = CFE_SB_GetCmdCode(MsgPtr);
502  switch (uiCmdCode)
503  {
504  case NAV_NOOP_CC:
505  HkTlm.usCmdCnt++;
506  (void) CFE_EVS_SendEvent(NAV_CMD_NOOP_EID, CFE_EVS_INFORMATION,
507  "Recvd NOOP. Version %d.%d.%d.%d",
508  NAV_MAJOR_VERSION,
509  NAV_MINOR_VERSION,
510  NAV_REVISION,
511  NAV_MISSION_REV);
512  break;
513 
514  case NAV_RESET_CC:
515  HkTlm.usCmdCnt = 0;
516  HkTlm.usCmdErrCnt = 0;
517  break;
518 
519  default:
520  HkTlm.usCmdErrCnt++;
521  (void) CFE_EVS_SendEvent(NAV_CC_ERR_EID, CFE_EVS_ERROR,
522  "Recvd invalid command code (%u)",
523  (unsigned int) uiCmdCode);
524  break;
525  }
526  }
527 }

565 osalbool NAV::VerifyCmdLength(CFE_SB_Msg_t* MsgPtr, uint16 usExpectedLen)
566 {
567  osalbool bResult = TRUE;
568  uint16 usMsgLen = 0;
569 
570  if (MsgPtr != NULL)

573 
574  if (usExpectedLen != usMsgLen)
575  {

581  "Rcvd invalid msgLen: msgId=0x%08X, cmdCode=%d, "
582  "msgLen=%d, expectedLen=%d", MsgId, usCmdCode,
583  usMsgLen, usExpectedLen);
584  HkTlm.usCmdErrCnt++;
585  }
586  }
587

930  }
931 
932  return 0;

935 void NAV::Takeoff()
936 {
937  float AbsoluteAltitude = 0.0f;
938  float MinPitch = 0.0f;
939  float MinAbsoluteAltitude = 0.0f;
940 
941  /* Use home position if valid */
942  osalbool HomePosValidFlag = HomePositionValid();
943  if (HomePosValidFlag)
944  {
945  MinAbsoluteAltitude = CVT.VehicleGlobalPosition.Alt
946  + ConfigTblPtr->NAV_MIS_TAKEOFF_ALT;
947  }
948  else
949  {
950  MinAbsoluteAltitude = ConfigTblPtr->NAV_MIS_TAKEOFF_ALT;
951  }
952 
953  /* Use altitude if it is already set. If home position is invalid use
954  * MinAbsoluteAltitude */
955  if (TakeoffTripletMsg.Current.Valid &&
956  isfinite(TakeoffTripletMsg.Current.Alt) && HomePosValidFlag)
957  {
958  AbsoluteAltitude = TakeoffTripletMsg.Current.Alt;
959 
960  /* If set altitude is less than minimum clearance raise it to that level and throw a warning */
961  if (AbsoluteAltitude < MinAbsoluteAltitude)
962  {
963  AbsoluteAltitude = MinAbsoluteAltitude;
964 
965  if (AbsoluteAltitude < MinAbsoluteAltitude - 0.1f)
966  {
967  (void) CFE_EVS_SendEvent(NAV_LOW_CLEARANCE_ALT_EID,
968  CFE_EVS_INFORMATION,
969  "Using minimum takeoff altitude : %.2f meters",
970  AbsoluteAltitude);
971  }
972  }
973  }
974  else
975  {
976  /* Raise to minimum clearance */
977  AbsoluteAltitude = MinAbsoluteAltitude;
978  (void) CFE_EVS_SendEvent(NAV_LOW_CLEARANCE_ALT_EID, CFE_EVS_INFORMATION,
979  "Using minimum takeoff altitude: %.2f meters",
980  AbsoluteAltitude);
981  }
982 
983  /* If new altitude is lower than current altitude, don't go down */
984  if (AbsoluteAltitude < CVT.VehicleGlobalPosition.Alt)
985  {
986  AbsoluteAltitude = CVT.VehicleGlobalPosition.Alt;
987  (void) CFE_EVS_SendEvent(NAV_HIGH_ALT_EID, CFE_EVS_INFORMATION,
988  "Set altitude already higher than take off altitude");
989  }
990 
991  /* Set mission item to takeoff */
992  MissionItem.NavCmd = PX4_VehicleCmd_t::PX4_VEHICLE_CMD_NAV_TAKEOFF;
993 
994  /* Use current position */
995  MissionItem.Lat = CVT.VehicleGlobalPosition.Lat;
996  MissionItem.Lon = CVT.VehicleGlobalPosition.Lon;
997  MissionItem.Yaw = CVT.VehicleGlobalPosition.Yaw;
998  MissionItem.Altitude = AbsoluteAltitude;
999  MissionItem.AltitudeIsRelative = FALSE;
1000  MissionItem.LoiterRadius = ConfigTblPtr->NAV_LOITER_RAD;
1001  MissionItem.PitchMin = MinPitch;
1002  MissionItem.AutoContinue = FALSE;
1003  MissionItem.Origin = NAV_Origin_t::ORIGIN_ONBOARD;
1004 
1005  /* Set mission result message */
1006  MissionResultMsg.Reached = FALSE;
1007  MissionResultMsg.Finished = FALSE;
1008 
1009  /* Record mission update event in osalbool */
1010  MissionResultUpdated = TRUE;
1011 
1012  /* Reset mission item reached */
1013  WaypointPositionReached = FALSE;
1014  WaypointYawReached = FALSE;
1015  TimeFirstInsideOrbit = 0;
1016  TimeWpReached = 0;
1017 
1018  /* Convert mission item to current set point */
1019  ConvertMissionItemToCurrentSetpoint(&PositionSetpointTripletMsg.Current,
1020  &MissionItem);
1021  PositionSetpointTripletMsg.Previous.Valid = FALSE;
1022  PositionSetpointTripletMsg.Next.Valid = FALSE;
1023 
1024  /* Set YawValid to true */
1025  PositionSetpointTripletMsg.Current.YawValid = TRUE;
1026 
1027  if (TakeoffTripletMsg.Current.Valid)
1028  {
1029  if (isfinite(TakeoffTripletMsg.Current.Yaw))
1030  {
1031  PositionSetpointTripletMsg.Current.Yaw =
1032  TakeoffTripletMsg.Current.Yaw;
1033  }
1034 
1035  if (isfinite(TakeoffTripletMsg.Current.Lat) &&
1036  isfinite(TakeoffTripletMsg.Current.Lon))
1037  {
1038  PositionSetpointTripletMsg.Current.Lat =
1039  TakeoffTripletMsg.Current.Lat;
1040  PositionSetpointTripletMsg.Current.Lon =
1041  TakeoffTripletMsg.Current.Lon;
1042  }
1043 
1044  /* Mark this as done */
1045  CFE_PSP_MemSet(&TakeoffTripletMsg, 0, sizeof(TakeoffTripletMsg));
1046  }
1047 
1048  CanLoiterAtSetpoint = TRUE;
1049  PositionSetpointTripletUpdated = TRUE;
1050 }
