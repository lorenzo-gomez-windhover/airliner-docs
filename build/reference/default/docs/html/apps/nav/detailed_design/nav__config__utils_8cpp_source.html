49 /* */
50 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
51 int32 NAV::InitConfigTbl()
52 {
53  int32 iStatus = CFE_SUCCESS;
54 
55  /* Register Config table */
56  iStatus = CFE_TBL_Register(&ConfigTblHdl,
57  NAV_CONFIG_TABLENAME,
58  (sizeof(NAV_ConfigTbl_t)),
59  CFE_TBL_OPT_DEFAULT,
60  NAV::ValidateConfigTbl);
61  if (iStatus != CFE_SUCCESS)
62  {
63  /* Note, a critical table could return another nominal code. If this table is
64  * made critical this logic would have to change. */
65  (void) CFE_EVS_SendEvent(NAV_CFGTBL_REG_ERR_EID, CFE_EVS_ERROR,
66  "Failed to register config table (0x%08lX)",
67  iStatus);
68  goto NAV_InitConfigTbl_Exit_Tag;
69  }
70 
71  /* Load Config table file */
72  iStatus = CFE_TBL_Load(ConfigTblHdl,
73  CFE_TBL_SRC_FILE,
74  NAV_CONFIG_TABLE_FILENAME);
75  if (iStatus != CFE_SUCCESS)
76  {
77  /* Note, CFE_SUCCESS is for a successful full table load. If a partial table
78  load is desired then this logic would have to change. */
79  (void) CFE_EVS_SendEvent(NAV_CFGTBL_LOAD_ERR_EID, CFE_EVS_ERROR,
80  "Failed to load Config Table (0x%08lX)",
81  iStatus);
82  goto NAV_InitConfigTbl_Exit_Tag;
83  }
84 
85  iStatus = AcquireConfigPointers();
86 
87 NAV_InitConfigTbl_Exit_Tag:
88  return (iStatus);
89 }

185 /* */
186 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
187 int32 NAV::AcquireConfigPointers(void)
188 {
189  int32 iStatus = CFE_SUCCESS;
190 
191  /*
192  ** Release the table
193  */
194  (void) CFE_TBL_ReleaseAddress(ConfigTblHdl);
195 
196  /*
197  ** Manage the table
198  */
199  iStatus = CFE_TBL_Manage(ConfigTblHdl);
200  if ((iStatus != CFE_SUCCESS) && (iStatus != CFE_TBL_INFO_UPDATED))
201  {
202  (void) CFE_EVS_SendEvent(NAV_CFGTBL_MANAGE_ERR_EID, CFE_EVS_ERROR,
203  "Failed to manage NAV Config table (0x%08lX)",
204  iStatus);
205  goto NAV_AcquireConfigPointers_Exit_Tag;
206  }
207 
208  /*
209  ** Get a pointer to the table
210  */
211  iStatus = CFE_TBL_GetAddress((void**)&ConfigTblPtr, ConfigTblHdl);
212  if (iStatus == CFE_TBL_INFO_UPDATED)
213  {
214  iStatus = CFE_SUCCESS;
215  }
216  else if(iStatus != CFE_SUCCESS)
217  {
218  ConfigTblPtr = 0;
219  (void) CFE_EVS_SendEvent(NAV_CFGTBL_GETADDR_ERR_EID, CFE_EVS_ERROR,
220  "Failed to get Config table's address (0x%08lX)",
221  iStatus);
222  }
223 
224 NAV_AcquireConfigPointers_Exit_Tag:
225  return iStatus;
226 }
