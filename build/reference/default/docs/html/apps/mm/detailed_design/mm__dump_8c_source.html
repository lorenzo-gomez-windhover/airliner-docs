84 void MM_PeekCmd(CFE_SB_MsgPtr_t MessagePtr)
85 {
86  boolean Valid = TRUE;
87  MM_PeekCmd_t *CmdPtr;
88  uint32 SrcAddress;
89  uint16 ExpectedLength = sizeof(MM_PeekCmd_t);
90 
91  /* Verify command packet length */
92  if(MM_VerifyCmdLength(MessagePtr, ExpectedLength))
93  {
94  CmdPtr = ((MM_PeekCmd_t *)MessagePtr);
95 
96  /* Resolve the symbolic address in command message */
97  Valid = CFS_ResolveSymAddr(&(CmdPtr->SrcSymAddress), &SrcAddress);
98 
99  if(Valid == TRUE)
100  {
101  /* Run necessary checks on command parameters */
102  Valid = MM_VerifyPeekPokeParams(SrcAddress, CmdPtr->MemType, CmdPtr->DataSize);
103 
104  /* Check the specified memory type and call the appropriate routine */
105  if(Valid == TRUE)
106  {
107  /*
108  ** We use this single peek routine for all memory types
109  ** (including the optional ones)
110  */
111  MM_PeekMem(CmdPtr, SrcAddress);
112  }
113 
114  } /* end CFS_ResolveSymAddr if */
115  else
116  {
117  MM_AppData.ErrCounter++;
118  CFE_EVS_SendEvent(MM_SYMNAME_ERR_EID, CFE_EVS_ERROR,
119  "Symbolic address can't be resolved: Name = '%s'",
120  CmdPtr->SrcSymAddress.SymName);
121  }
122 
123  } /* end MM_VerifyCmdLength if */
124 
125  return;
126 
127 } /* end MM_PeekCmd */

131 /* Read 8,16, or 32 bits of data from any given input address */
132 /* */
133 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
134 void MM_PeekMem (MM_PeekCmd_t *CmdPtr,
135  uint32 SrcAddress)
136 {
137  boolean ValidPeek = TRUE;
138  uint8 ByteValue = 0;
139  uint16 WordValue = 0;
140  uint32 DWordValue = 0;
141  int32 PSP_Status = 0;
142  uint32 BytesProcessed = 0;
143  uint32 DataValue = 0;
144 
145  /*
146  ** Read the requested number of bytes and report in an event message
147  */
148  switch(CmdPtr->DataSize)
149  {
150  case MM_BYTE_BIT_WIDTH:
151 
152  PSP_Status = CFE_PSP_MemRead8(SrcAddress, &ByteValue);
153  if (PSP_Status == CFE_PSP_SUCCESS)
154  {
155  DataValue = (uint32) ByteValue;
156  BytesProcessed = sizeof (uint8);
157  CFE_EVS_SendEvent(MM_PEEK_BYTE_INF_EID, CFE_EVS_INFORMATION,
158  "Peek Command: Addr = 0x%08X Size = 8 bits Data = 0x%02X",
159  (unsigned int)SrcAddress, ByteValue);
160  }
161  else
162  {
163  ValidPeek = FALSE;
164  MM_AppData.ErrCounter++;
165  CFE_EVS_SendEvent(MM_PSP_READ_ERR_EID, CFE_EVS_ERROR,
166  "PSP read memory error: RC=0x%08X, Address=0x%08X, MemType=MEM8",
167  (unsigned int)PSP_Status, (unsigned int)SrcAddress);
168  }
169  break;
170 
171  case MM_WORD_BIT_WIDTH:
172 
173  PSP_Status = CFE_PSP_MemRead16(SrcAddress, &WordValue);
174  if (PSP_Status == CFE_PSP_SUCCESS)
175  {
176  DataValue = (uint32) WordValue;
177  BytesProcessed = sizeof (uint16);
178  CFE_EVS_SendEvent(MM_PEEK_WORD_INF_EID, CFE_EVS_INFORMATION,
179  "Peek Command: Addr = 0x%08X Size = 16 bits Data = 0x%04X",
180  (unsigned int)SrcAddress, (unsigned short)DataValue);
181  }
182  else
183  {
184  ValidPeek = FALSE;
185  MM_AppData.ErrCounter++;
186  CFE_EVS_SendEvent(MM_PSP_READ_ERR_EID, CFE_EVS_ERROR,
187  "PSP read memory error: RC=0x%08X, Address=0x%08X, MemType=MEM16",
188  (unsigned int)PSP_Status, (unsigned int)SrcAddress);
189  }
190  break;
191 
192  case MM_DWORD_BIT_WIDTH:
193 
194  PSP_Status = CFE_PSP_MemRead32(SrcAddress, &DWordValue);
195  if (PSP_Status == CFE_PSP_SUCCESS)
196  {
197  DataValue = DWordValue;
198  BytesProcessed = sizeof(uint32);
199  CFE_EVS_SendEvent(MM_PEEK_DWORD_INF_EID, CFE_EVS_INFORMATION,
200  "Peek Command: Addr = 0x%08X Size = 32 bits Data = 0x%08X",
201  (unsigned int)SrcAddress, (unsigned int)DataValue);
202  }
203  else
204  {
205  ValidPeek = FALSE;
206  MM_AppData.ErrCounter++;
207  CFE_EVS_SendEvent(MM_PSP_READ_ERR_EID, CFE_EVS_ERROR,
208  "PSP read memory error: RC=0x%08X, Address=0x%08X, MemType=MEM32",
209  (unsigned int)PSP_Status, (unsigned int)SrcAddress);
210  }
211  break;
212 
213  /*
214  ** We don't need a default case, a bad DataSize will get caught
215  ** in the MM_VerifyPeekPokeParams function and we won't get here
216  */
217  default:
218  break;
219  }
220 
221  if (ValidPeek)
222  {
223  MM_AppData.CmdCounter++;
224  MM_AppData.LastAction = MM_PEEK;
225  MM_AppData.MemType = CmdPtr->MemType;
226  MM_AppData.Address = SrcAddress;
227  MM_AppData.BytesProcessed = BytesProcessed;
228  MM_AppData.DataValue = DataValue;
229  }
230 
231  return;
232 
233 } /* end MM_PeekMem */

237 /* Dump memory to file comand */
238 /* */
239 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
240 void MM_DumpMemToFileCmd(CFE_SB_MsgPtr_t MessagePtr)
241 {
242  boolean Valid = TRUE;
243  int32 OS_Status = OS_SUCCESS;
244  int32 FileHandle;
245  uint32 SrcAddress = 0;
246  MM_DumpMemToFileCmd_t *CmdPtr;
247  CFE_FS_Header_t CFEFileHeader;
248  MM_LoadDumpFileHeader_t MMFileHeader;
249  uint16 ExpectedLength = sizeof(MM_DumpMemToFileCmd_t);
250 
251  /* Verify command packet length */
252  if(MM_VerifyCmdLength(MessagePtr, ExpectedLength))
253  {
254  CmdPtr = ((MM_DumpMemToFileCmd_t *)MessagePtr);
255 
256  /*
257  ** NUL terminate the very end of the file name string array as a
258  ** safety measure
259  */
260  CmdPtr->FileName[OS_MAX_PATH_LEN - 1] = '\0';
261 
262  /* Verify filename doesn't have any illegal characters */
263  Valid = CFS_IsValidFilename(CmdPtr->FileName, strlen(CmdPtr->FileName));
264  if(Valid == TRUE)
265  {
266  /* Resolve the symbolic address in command message */
267  Valid = CFS_ResolveSymAddr(&(CmdPtr->SrcSymAddress), &SrcAddress);
268 
269  if(Valid == TRUE)
270  {
271  /* Run necessary checks on command parameters */
272  Valid = MM_VerifyFileDumpParams(SrcAddress, CmdPtr->MemType, CmdPtr->NumOfBytes);
273 
274  if(Valid == TRUE)
275  {
276  /*
277  ** Initialize the cFE primary file header structure
278  */
279  CFE_PSP_MemSet(&CFEFileHeader, 0, sizeof(CFE_FS_Header_t));
280  CFEFileHeader.SubType = MM_CFE_HDR_SUBTYPE;
281  strcpy(&CFEFileHeader.Description[0], MM_CFE_HDR_DESCRIPTION);
282 
283  /*
284  ** Initialize the MM secondary file header structure
285  */
286  CFE_PSP_MemSet(&MMFileHeader, 0, sizeof(MM_LoadDumpFileHeader_t));
287  MMFileHeader.SymAddress.SymName[0] = MM_CLEAR_SYMNAME;
288 
289  /*
290  ** Copy command data to file secondary header
291  */
292  MMFileHeader.SymAddress.Offset = SrcAddress;
293  MMFileHeader.MemType = CmdPtr->MemType;
294  MMFileHeader.NumOfBytes = CmdPtr->NumOfBytes;
295 
296  /*
297  ** Create and open dump file
298  */
299  if((FileHandle = OS_creat(CmdPtr->FileName, OS_READ_WRITE)) >= 0)
300  {
301  /* Write the file headers */
302  Valid = MM_WriteFileHeaders(CmdPtr->FileName, FileHandle, &CFEFileHeader, &MMFileHeader);
303  if(Valid == TRUE)
304  {
305  switch(MMFileHeader.MemType)
306  {
307  case MM_RAM:
308  case MM_EEPROM:
309  Valid = MM_DumpMemToFile(FileHandle, CmdPtr->FileName, &MMFileHeader);
310  break;
311 
312 #if (MM_OPT_CODE_MEM32_MEMTYPE == TRUE)
313  case MM_MEM32:
314  Valid = MM_DumpMem32ToFile(FileHandle, CmdPtr->FileName, &MMFileHeader);
315  break;
316 #endif /* MM_OPT_CODE_MEM32_MEMTYPE */
317 
318 #if (MM_OPT_CODE_MEM16_MEMTYPE == TRUE)
319  case MM_MEM16:
320  Valid = MM_DumpMem16ToFile(FileHandle, CmdPtr->FileName, &MMFileHeader);
321  break;
322 #endif /* MM_OPT_CODE_MEM16_MEMTYPE */
323 
324 #if (MM_OPT_CODE_MEM8_MEMTYPE == TRUE)
325  case MM_MEM8:
326  Valid = MM_DumpMem8ToFile(FileHandle, CmdPtr->FileName, &MMFileHeader);
327  break;
328 #endif /* MM_OPT_CODE_MEM8_MEMTYPE */
329  default:
330  /* This branch will never be executed. MMFileHeader.MemType will always
331  * be valid value for this switch statement it is verified via
332  * MM_VerifyFileDumpParams */
333  break;
334  }
335 
336  if(Valid == TRUE)
337  {
338  /*
339  ** Compute CRC of dumped data
340  */
341  OS_lseek(FileHandle, (sizeof(CFE_FS_Header_t) + sizeof(MM_LoadDumpFileHeader_t)), OS_SEEK_SET);
342 
343  OS_Status = CFS_ComputeCRCFromFile(FileHandle, &MMFileHeader.Crc, MM_DUMP_FILE_CRC_TYPE);
344  if(OS_Status == OS_SUCCESS)
345  {
346  /*
347  ** Rewrite the file headers. The subfunctions will take care of moving
348  ** the file pointer to the beginning of the file so we don't need to do it
349  ** here.
350  */
351  Valid = MM_WriteFileHeaders(CmdPtr->FileName, FileHandle, &CFEFileHeader, &MMFileHeader);
352 
353  } /* end CFS_ComputeCRCFromFile if */
354  else
355  {
356  MM_AppData.ErrCounter++;
357  CFE_EVS_SendEvent(MM_CFS_COMPUTECRCFROMFILE_ERR_EID, CFE_EVS_ERROR,
358  "CFS_ComputeCRCFromFile error received: RC = 0x%08X File = '%s'", (unsigned int)OS_Status,
359  CmdPtr->FileName);
360  }
361 
362  } /* end Valid == TRUE if */
363 
364  if(Valid == TRUE)
365  {
366  MM_AppData.CmdCounter++;
367  CFE_EVS_SendEvent(MM_DMP_MEM_FILE_INF_EID, CFE_EVS_INFORMATION,
368  "Dump Memory To File Command: Dumped %d bytes from address 0x%08X to file '%s'",
369  (int)MM_AppData.BytesProcessed, (unsigned int)SrcAddress, CmdPtr->FileName);
370  /*
371  ** Update last action statistics
372  */
373  MM_AppData.LastAction = MM_DUMP_TO_FILE;
374  strncpy(MM_AppData.FileName, CmdPtr->FileName, OS_MAX_PATH_LEN);
375  MM_AppData.MemType = CmdPtr->MemType;
376  MM_AppData.Address = SrcAddress;
377  MM_AppData.BytesProcessed = CmdPtr->NumOfBytes;
378  }
379 
380  } /* end MM_WriteFileHeaders if */
381 
382  /*
383  ** Don't need an 'else' here. MM_WriteFileHeaders will increment
384  ** the error counter and generate an event message if needed.
385  */
386 
387  /* Close dump file */
388  if((OS_Status = OS_close(FileHandle)) != OS_SUCCESS)
389  {
390  MM_AppData.ErrCounter++;
391  CFE_EVS_SendEvent(MM_OS_CLOSE_ERR_EID, CFE_EVS_ERROR,
392  "OS_close error received: RC = 0x%08X File = '%s'",
393  (unsigned int)OS_Status, CmdPtr->FileName);
394  }
395 
396  } /* end OS_creat if */
397  else
398  {
399  MM_AppData.ErrCounter++;
400  CFE_EVS_SendEvent(MM_OS_CREAT_ERR_EID, CFE_EVS_ERROR,
401  "OS_creat error received: RC = 0x%08X File = '%s'",
402  (unsigned int)FileHandle, CmdPtr->FileName);
403  }
404 
405  } /* end MM_VerifyFileDumpParams if */
406 
407  } /* end CFS_ResolveSymAddr if */
408  else
409  {
410  MM_AppData.ErrCounter++;
411  CFE_EVS_SendEvent(MM_SYMNAME_ERR_EID, CFE_EVS_ERROR,
412  "Symbolic address can't be resolved: Name = '%s'",
413  CmdPtr->SrcSymAddress.SymName);
414  }
415  } /* end IsValidFilename if */
416  else
417  {
418  MM_AppData.ErrCounter++;
419  CFE_EVS_SendEvent(MM_CMD_FNAME_ERR_EID, CFE_EVS_ERROR,
420  "Command specified filename invalid: Name = '%s'", CmdPtr->FileName);
421  }
422 
423  } /* end MM_VerifyCmdLength if */
424 
425  return;
426 
427 } /* end MM_DumpMemoryToFileCmd */

429 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
430 /* */
431 /* Dump the requested number of bytes from memory to a file */
432 /* */
433 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
434 boolean MM_DumpMemToFile(uint32 FileHandle,
435  char *FileName,
436  MM_LoadDumpFileHeader_t *FileHeader)
437 {
438  boolean ValidDump = FALSE;
439  int32 OS_Status;
440  int32 PSP_Status;
441  uint32 BytesRemaining = FileHeader->NumOfBytes;
442  uint32 BytesProcessed = 0;
443  uint32 SegmentSize = MM_MAX_DUMP_DATA_SEG;
444  uint8 *SourcePtr = (uint8 *)FileHeader->SymAddress.Offset;
445  uint8 *ioBuffer = (uint8 *) &MM_AppData.DumpBuffer[0];
446 
447  while (BytesRemaining != 0)
448  {
449  if (BytesRemaining < MM_MAX_DUMP_DATA_SEG)
450  {
451  SegmentSize = BytesRemaining;
452  }
453 
454  PSP_Status = CFE_PSP_MemCpy(ioBuffer, SourcePtr, SegmentSize);
455  if (PSP_Status == CFE_PSP_SUCCESS)
456  {
457  OS_Status = OS_write(FileHandle, ioBuffer, SegmentSize);
458  if (OS_Status == SegmentSize)
459  {
460  SourcePtr += SegmentSize;
461  BytesRemaining -= SegmentSize;
462  BytesProcessed += SegmentSize;
463 
464  /* Prevent CPU hogging between dump segments */
465  if (BytesRemaining != 0)
466  {
467  MM_SegmentBreak();
468  }
469  }
470  else
471  {
472  BytesRemaining = 0;
473  MM_AppData.ErrCounter++;
474  CFE_EVS_SendEvent(MM_OS_WRITE_EXP_ERR_EID, CFE_EVS_ERROR,
475  "OS_write error received: RC = 0x%08X, Expected = %d, File = '%s'",
476  (unsigned int)OS_Status, (int)SegmentSize, FileName);
477  }
478  }
479  else
480  {
481  BytesRemaining = 0;
482  MM_AppData.ErrCounter++;
483  CFE_EVS_SendEvent(MM_PSP_COPY_ERR_EID, CFE_EVS_ERROR,
484  "PSP copy memory error: RC=0x%08X, Src=0x%08X, Tgt=0x%08X, Size=0x%08X",
485  (unsigned int)PSP_Status, (unsigned int) SourcePtr, (unsigned int)ioBuffer, (unsigned int)SegmentSize);
486  }
487  }
488 
489  /* Update last action statistics */
490  if (BytesProcessed == FileHeader->NumOfBytes)
491  {
492  ValidDump = TRUE;
493  MM_AppData.LastAction = MM_DUMP_TO_FILE;
494  MM_AppData.MemType = FileHeader->MemType;
495  MM_AppData.Address = FileHeader->SymAddress.Offset;
496  MM_AppData.BytesProcessed = BytesProcessed;
497  strncpy(MM_AppData.FileName, FileName, OS_MAX_PATH_LEN);
498  }
499 
500  return(ValidDump);
501 
502 } /* end MM_DumpMemToFile */

705 /* Dump memory in event message command */
706 /* */
707 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
708 void MM_DumpInEventCmd(CFE_SB_MsgPtr_t MessagePtr)
709 {
710  boolean Valid = TRUE;
711  MM_DumpInEventCmd_t *CmdPtr;
712  uint32 i;
713  uint32 SrcAddress;
714  uint16 ExpectedLength = sizeof(MM_DumpInEventCmd_t);
715  uint8 *BytePtr;
716  char TempString[MM_DUMPINEVENT_TEMP_CHARS];
717 static char EventString[CFE_EVS_MAX_MESSAGE_LENGTH];
718 
719  /*
720  ** Allocate a dump buffer. It's declared this way to ensure it stays
721  ** longword aligned since MM_MAX_DUMP_INEVENT_BYTES can be adjusted
722  ** by changing the maximum event message string size.
723  */
724  uint32 DumpBuffer[(MM_MAX_DUMP_INEVENT_BYTES + 3)/4];
725 
726  /* Verify command packet length */
727  if(MM_VerifyCmdLength(MessagePtr, ExpectedLength))
728  {
729  CmdPtr = ((MM_DumpInEventCmd_t *)MessagePtr);
730 
731  /* Resolve the symbolic source address in the command message */
732  Valid = CFS_ResolveSymAddr(&(CmdPtr->SrcSymAddress), &SrcAddress);
733 
734  if(Valid == TRUE)
735  {
736  /* Run necessary checks on command parameters */
737  Valid = MM_VerifyDumpInEventParams(SrcAddress, CmdPtr->MemType, CmdPtr->NumOfBytes);
738 
739  if(Valid == TRUE)
740  {
741  /* Fill a local data buffer with the dump words */
742  Valid = MM_FillDumpInEventBuffer(SrcAddress, CmdPtr, (uint8 *)DumpBuffer);
743 
744  if(Valid == TRUE)
745  {
746  /*
747  ** Prepare event message string header
748  ** 13 characters, not counting NUL terminator
749  */
750  strcpy(EventString, "Memory Dump: ");
751 
752  /*
753  ** Build dump data string
754  ** Each byte of data requires 5 characters of string space
755  ** Note this really only allows up to ~15 bytes using default config
756  */
757  BytePtr = (uint8 *)DumpBuffer;
758  for (i = 0; i < CmdPtr->NumOfBytes && i < MM_MAX_DUMP_INEVENT_BYTES; i++)
759  {
760  sprintf(TempString, "0x%02X ", *BytePtr);
761  strcat(EventString, TempString);
762  BytePtr++;
763  }
764 
765  /*
766  ** Append tail
767  ** This adds up to 33 characters including the NUL terminator
768  */
769  sprintf(TempString, "from address: 0x%08lX", (unsigned long)SrcAddress);
770  strcat(EventString, TempString);
771 
772  /* Send it out */
773  CFE_EVS_SendEvent(MM_DUMP_INEVENT_INF_EID, CFE_EVS_INFORMATION, "%s", EventString);
774  /* Update telemetry */
775  MM_AppData.LastAction = MM_DUMP_INEVENT;
776  MM_AppData.MemType = CmdPtr->MemType;
777  MM_AppData.Address = SrcAddress;
778  MM_AppData.BytesProcessed = CmdPtr->NumOfBytes;
779  MM_AppData.CmdCounter++;
780  } /* end MM_FillDumpInEventBuffer if */
781  } /* end MM_VerifyDumpInEventParams if */
782  } /* end CFS_ResolveSymAddr if */
783  else
784  {
785  MM_AppData.ErrCounter++;
786  CFE_EVS_SendEvent(MM_SYMNAME_ERR_EID, CFE_EVS_ERROR,
787  "Symbolic address can't be resolved: Name = '%s'",
788  CmdPtr->SrcSymAddress.SymName);
789  }
790 
791  } /* end MM_VerifyCmdLength if */
792 
793  return;
794 
795 } /* end MM_DumpWordsInEventCmd */
