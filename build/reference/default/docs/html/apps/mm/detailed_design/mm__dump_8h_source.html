51 /*
52 ** This macro defines the maximum number of bytes that can be dumped
53 ** in an event message string based upon the setting of the
54 ** CFE_EVS_MAX_MESSAGE_LENGTH configuration parameter.
55 **
56 ** The event message format is:
57 ** Message head "Memory Dump: " 13 characters
58 ** Message body "0xFF " 5 characters per dump byte
59 ** Message tail "from address: 0xFFFFFFFF" 33 characters including NUL on 64-bit system
60 */
64 #define MM_MAX_DUMP_INEVENT_BYTES ((CFE_EVS_MAX_MESSAGE_LENGTH - (13 + 33)) / 5)
65 
67 /*
68 ** This macro defines the size of the scratch buffer used to build
69 ** the dump in event message string. Set it to the size of the
70 ** largest piece shown above including room for a NUL terminator.
71 */
75 #define MM_DUMPINEVENT_TEMP_CHARS 36
