81 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
82 /* */
83 /* MM application entry point and main process loop */
84 /* */
85 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
86 void MM_AppMain(void)
87 {
88  int32 Status = CFE_SUCCESS;
89 
90  /*
91  ** Register the application with Executive Services
92  */
93  CFE_ES_RegisterApp();
94 
95  /*
96  ** Create the first Performance Log entry
97  */
98  CFE_ES_PerfLogEntry(MM_APPMAIN_PERF_ID);
99 
100  /*
101  ** Perform application specific initialization
102  */
103  Status = MM_AppInit();
104  if (Status != CFE_SUCCESS)
105  {
106  MM_AppData.RunStatus = CFE_ES_APP_ERROR;
107  }
108 
109  /*
110  ** Application main loop
111  */
112  while(CFE_ES_RunLoop(&MM_AppData.RunStatus) == TRUE)
113  {
114  /*
115  ** Performance Log exit stamp
116  */
117  CFE_ES_PerfLogExit(MM_APPMAIN_PERF_ID);
118 
119  /*
120  ** Pend on the arrival of the next Software Bus message
121  */
122  Status = CFE_SB_RcvMsg(&MM_AppData.MsgPtr, MM_AppData.CmdPipe, CFE_SB_PEND_FOREVER);
123 
124  /*
125  ** Performance Log entry stamp
126  */
127  CFE_ES_PerfLogEntry(MM_APPMAIN_PERF_ID);
128 
129  /*
130  ** Check the return status from the software bus
131  */
132  if (Status == CFE_SUCCESS)
133  {
134  /* Process Software Bus message */
135  MM_AppPipe(MM_AppData.MsgPtr);
136  }
137  else
138  {
139  /*
140  ** Exit on pipe read error
141  */
142  CFE_EVS_SendEvent(MM_PIPE_ERR_EID, CFE_EVS_ERROR,
143  "SB Pipe Read Error, App will exit. RC = 0x%08X", (unsigned int)Status);
144 
145  MM_AppData.RunStatus = CFE_ES_APP_ERROR;
146 
147  }
148  } /* end CFS_ES_RunLoop while */
149 
150  /*
151  ** Performance Log exit stamp
152  */
153  CFE_ES_PerfLogExit(MM_APPMAIN_PERF_ID);
154 
155  /*
156  ** Exit the application
157  */
158  CFE_ES_ExitApp(MM_AppData.RunStatus);
159 
160 } /* end MM_AppMain */

167 int32 MM_AppInit(void)
168 {
169  int32 Status = CFE_SUCCESS;
170 
171  /*
172  ** MM doesn't use the critical data store and
173  ** doesn't need to identify power on vs. processor resets.
174  ** If this changes add it here as shown in the qq_app.c template
175  */
176 
177  /*
178  ** Setup the RunStatus variable
179  */
180  MM_AppData.RunStatus = CFE_ES_APP_RUN;
181 
182  /*
183  ** Initialize application command execution counters
184  */
185  MM_AppData.CmdCounter = 0;
186  MM_AppData.ErrCounter = 0;
187 
188  /*
189  ** Initialize application configuration data
190  */
191  strcpy(MM_AppData.PipeName, "MM_CMD_PIPE");
192  MM_AppData.PipeDepth = MM_CMD_PIPE_DEPTH;
193 
194  MM_AppData.LimitHK = MM_HK_LIMIT;
195  MM_AppData.LimitCmd = MM_CMD_LIMIT;
196 
197  /*
198  ** Register for event services
199  */
200  Status = CFE_EVS_Register(NULL, 0, CFE_EVS_BINARY_FILTER);
201

203  {

205  return (Status);
206  }
207 
208  /*
209  ** Initialize the local housekeeping telemetry packet (clear user data area)
210  */
211  CFE_SB_InitMsg(&MM_AppData.HkPacket, MM_HK_TLM_MID, sizeof(MM_HkPacket_t), TRUE);
212

215  */
216  Status = CFE_SB_CreatePipe(&MM_AppData.CmdPipe, MM_AppData.PipeDepth, MM_AppData.PipeName);
217  if (Status != CFE_SUCCESS)
218  {
219  CFE_ES_WriteToSysLog("MM App: Error Creating SB Pipe, RC = 0x%08X\n", (unsigned int)Status);
220  return (Status);
221  }
222 
223  /*
224  ** Subscribe to Housekeeping request commands
225  */
226  Status = CFE_SB_Subscribe(MM_SEND_HK_MID, MM_AppData.CmdPipe);
227  if (Status != CFE_SUCCESS)
228  {
229  CFE_ES_WriteToSysLog("MM App: Error Subscribing to HK Request, RC = 0x%08X\n", (unsigned int)Status);
230  return (Status);
231  }
232 
233  /*
234  ** Subscribe to MM ground command packets
235  */
236  Status = CFE_SB_Subscribe(MM_CMD_MID, MM_AppData.CmdPipe);
237  if (Status != CFE_SUCCESS)
238  {
239  CFE_ES_WriteToSysLog("MM App: Error Subscribing to MM Command, RC = 0x%08X\n", (unsigned int)Status);
240  return (Status);
241  }
242 
243  /*
244  ** MM doesn't use tables. If this changes add table registration
245  ** and initialization here as shown in the qq_app.c template
246  */
247 
248  /*
249  ** MM doesn't use the critical data store. If this changes add CDS
250  ** creation here as shown in the qq_app.c template
251  */
252 
253  /*
254  ** Initialize MM housekeeping information
255  */
256  MM_ResetHk();
257 
258  /*
259  ** Application startup event message
260  */
261  CFE_EVS_SendEvent(MM_INIT_INF_EID, CFE_EVS_INFORMATION,
262  "MM Initialized. Version %d.%d.%d.%d",
263  MM_MAJOR_VERSION,
264  MM_MINOR_VERSION,
265  MM_REVISION,
266  MM_MISSION_REV);
267 
268  return(CFE_SUCCESS);
269 
270 } /* end MM_AppInit */

272 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
273 /* */
274 /* Process a command pipe message */
275 /* */
276 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
277 void MM_AppPipe(CFE_SB_MsgPtr_t msg)
278 {
279  CFE_SB_MsgId_t MessageID = 0;
280  uint16 CommandCode = 0;
281 
282  MessageID = CFE_SB_GetMsgId(msg);
283  switch (MessageID)
284  {
285  /*
286  ** Housekeeping telemetry request
287  */
288  case MM_SEND_HK_MID:
289  MM_HousekeepingCmd(msg);
290  break;
291 
292  /*
293  ** MM ground commands
294  */
295  case MM_CMD_MID:
296  MM_ResetHk(); /* Clear all "Last Action" data */
297  CommandCode = CFE_SB_GetCmdCode(msg);
298  switch (CommandCode)
299  {
300  case MM_NOOP_CC:
301  MM_NoopCmd(msg);
302  break;
303 
304  case MM_RESET_CC:
305  MM_ResetCmd(msg);
306  break;
307 
308  case MM_PEEK_CC:
309  MM_PeekCmd(msg);
310  break;
311 
312  case MM_POKE_CC:
313  MM_PokeCmd(msg);
314  break;
315 
316  case MM_LOAD_MEM_WID_CC:
317  MM_LoadMemWIDCmd(msg);
318  break;
319 
320  case MM_LOAD_MEM_FROM_FILE_CC:
321  MM_LoadMemFromFileCmd(msg);
322  break;
323 
324  case MM_DUMP_MEM_TO_FILE_CC:
325  MM_DumpMemToFileCmd(msg);
326  break;
327 
328  case MM_DUMP_IN_EVENT_CC:
329  MM_DumpInEventCmd(msg);
330  break;
331 
332  case MM_FILL_MEM_CC:
333  MM_FillMemCmd(msg);
334  break;
335 
336  case MM_LOOKUP_SYM_CC:
337  MM_LookupSymbolCmd(msg);
338  break;
339 
340  case MM_SYMTBL_TO_FILE_CC:
341  MM_SymTblToFileCmd(msg);
342  break;
343 
344  case MM_ENABLE_EEPROM_WRITE_CC:
345  MM_EepromWriteEnaCmd(msg);
346  break;
347 
348  case MM_DISABLE_EEPROM_WRITE_CC:
349  MM_EepromWriteDisCmd(msg);
350  break;
351 
352  default:
353  MM_AppData.ErrCounter++;
354  CFE_EVS_SendEvent(MM_CC1_ERR_EID, CFE_EVS_ERROR,
355  "Invalid ground command code: ID = 0x%X, CC = %d",
356  MessageID, CommandCode);
357  break;
358  }
359  break;
360 
361  /*
362  ** Unrecognized Message ID
363  */
364  default:
365  MM_AppData.ErrCounter++;
366  CFE_EVS_SendEvent(MM_MID_ERR_EID, CFE_EVS_ERROR,
367  "Invalid command pipe message ID: 0x%X", MessageID);
368  break;
369 
370  } /* end switch */
371 
372  return;
373 
374 } /* End MM_AppPipe */

476 /* Lookup symbol name command */
477 /* */
478 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
479 void MM_LookupSymbolCmd(CFE_SB_MsgPtr_t msg)
480 {
481  int32 OS_Status = OS_ERROR; /* Set to error instead of success since we explicitly test for success */
482  uint32 ResolvedAddr = 0;
483  MM_LookupSymCmd_t *CmdPtr = NULL;
484  uint16 ExpectedLength = sizeof(MM_LookupSymCmd_t);
485 
486  /*
487  ** Verify command packet length
488  */
489  if(MM_VerifyCmdLength(msg, ExpectedLength))
490  {
491  CmdPtr = ((MM_LookupSymCmd_t *) msg);
492 
493  /*
494  ** NUL terminate the very end of the symbol name string as a
495  ** safety measure
496  */
497  CmdPtr->SymName[OS_MAX_SYM_LEN - 1] = '\0';
498 
499  /*
500  ** Check if the symbol name string is a nul string
501  */
502  if(strlen(CmdPtr->SymName) == 0)
503  {
504  MM_AppData.ErrCounter++;
505  CFE_EVS_SendEvent(MM_SYMNAME_NUL_ERR_EID, CFE_EVS_ERROR,
506  "NUL (empty) string specified as symbol name");
507  }
508  else
509  {
510  /*
511  ** If symbol name is not an empty string look it up using the OSAL API
512  */
513  OS_Status = OS_SymbolLookup(&ResolvedAddr, CmdPtr->SymName);
514  if (OS_Status == OS_SUCCESS)
515  {
516  /* Update telemetry */
517  MM_AppData.LastAction = MM_SYM_LOOKUP;
518  MM_AppData.Address = ResolvedAddr;
519  MM_AppData.CmdCounter++;
520 
521  CFE_EVS_SendEvent(MM_SYM_LOOKUP_INF_EID, CFE_EVS_INFORMATION,
522  "Symbol Lookup Command: Name = '%s' Addr = 0x%08X",
523  CmdPtr->SymName, (unsigned int)ResolvedAddr);
524  }
525  else
526  {
527  MM_AppData.ErrCounter++;
528  CFE_EVS_SendEvent(MM_SYMNAME_ERR_EID, CFE_EVS_ERROR,
529  "Symbolic address can't be resolved: Name = '%s'",
530  CmdPtr->SymName);
531  }
532 
533  } /* end strlen(CmdPtr->SymName) == 0 else */
534 
535  } /* end MM_VerifyCmdLength if */
536 
537  return;
538 
539 } /* end MM_LookupSymbolCmd */
