78 /* Load memory from a file using only 32 bit wide writes */
79 /* */
80 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
81 boolean MM_LoadMem32FromFile(uint32 FileHandle,
82  char *FileName,
83  MM_LoadDumpFileHeader_t *FileHeader,
84  uint32 DestAddress)
85 {
86  uint32 i = 0;
87  int32 ReadLength = 0;
88  int32 PSP_Status = CFE_PSP_SUCCESS;
89  int32 BytesProcessed = 0;
90  int32 BytesRemaining = FileHeader->NumOfBytes;
91  uint32 *DataPointer32 = (uint32 *) DestAddress;
92  uint32 *ioBuffer32 = (uint32 *) &MM_AppData.LoadBuffer[0];
93  uint32 SegmentSize = MM_MAX_LOAD_DATA_SEG;
94  boolean Valid = FALSE;
95 
96  while (BytesRemaining != 0)
97  {
98  if (BytesRemaining < MM_MAX_LOAD_DATA_SEG)
99  {
100  SegmentSize = BytesRemaining;
101  }
102 
103  /* Read file data into i/o buffer */
104  if ((ReadLength = OS_read(FileHandle, ioBuffer32, SegmentSize)) != SegmentSize)
105  {
106  MM_AppData.ErrCounter++;
107  BytesRemaining = 0;
108  CFE_EVS_SendEvent(MM_OS_READ_ERR_EID, CFE_EVS_ERROR,
109  "OS_read error received: RC = 0x%08X Expected = %d File = '%s'",
110  (unsigned int)ReadLength, (int)SegmentSize, FileName);
111  }
112  else
113  {
114  /* Load memory from i/o buffer using 32 bit wide writes */
115  for (i = 0; i < (SegmentSize / sizeof(uint32)); i++)
116  {
117  PSP_Status = CFE_PSP_MemWrite32((uint32) DataPointer32, ioBuffer32[i]);
118  if (PSP_Status == CFE_PSP_SUCCESS)
119  {
120  DataPointer32++;
121  }
122  else
123  {
124  /* CFE_PSP_MemWrite32 error */
125  BytesRemaining = 0;
126  MM_AppData.ErrCounter++;
127  CFE_EVS_SendEvent(MM_PSP_WRITE_ERR_EID, CFE_EVS_ERROR,
128  "PSP write memory error: RC=0x%08X, Address=0x%08X, MemType=MEM32",
129  (unsigned int)PSP_Status, (unsigned int)DataPointer32);
130  /* Stop load segment loop */
131  break;
132  }
133  }
134 
135  if (PSP_Status == CFE_PSP_SUCCESS)
136  {
137  BytesProcessed += SegmentSize;
138  BytesRemaining -= SegmentSize;
139 
140  /* Prevent CPU hogging between load segments */
141  if (BytesRemaining != 0)
142  {
143  MM_SegmentBreak();
144  }
145  }
146  }
147  }
148 
149  /* Update last action statistics */
150  if (BytesProcessed == FileHeader->NumOfBytes)
151  {
152  Valid = TRUE;
153  MM_AppData.LastAction = MM_LOAD_FROM_FILE;
154  MM_AppData.MemType = MM_MEM32;
155  MM_AppData.Address = DestAddress;
156  MM_AppData.BytesProcessed = BytesProcessed;
157  strncpy(MM_AppData.FileName, FileName, OS_MAX_PATH_LEN);
158  }
159 
160  return(Valid);
161 
162 } /* end MM_LoadMem32FromFile */

166 /* */
167 /* Dump the requested number of bytes from memory to a file using */
168 /* only 32 bit wide reads */
169 /* */
170 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
171 boolean MM_DumpMem32ToFile(uint32 FileHandle,
172  char *FileName,
173  MM_LoadDumpFileHeader_t *FileHeader)
174 {
175  boolean Valid = TRUE;
176  int32 OS_Status = OS_ERROR;
177  int32 PSP_Status = CFE_PSP_SUCCESS;
178  uint32 i = 0;
179  uint32 BytesProcessed = 0;
180  uint32 BytesRemaining = FileHeader->NumOfBytes;
181  uint32 *DataPointer32 = (uint32 *) FileHeader->SymAddress.Offset;
182  uint32 *ioBuffer32 = (uint32 *) &MM_AppData.DumpBuffer[0];
183  uint32 SegmentSize = MM_MAX_DUMP_DATA_SEG;
184 
185  while (BytesRemaining != 0)
186  {
187  if (BytesRemaining < MM_MAX_DUMP_DATA_SEG)
188  {
189  SegmentSize = BytesRemaining;
190  }
191 
192  /* Load RAM data into i/o buffer */
193  for (i = 0; i < SegmentSize; i++)
194  {
195  PSP_Status = CFE_PSP_MemRead32((uint32) DataPointer32, &ioBuffer32[i]);
196  if (PSP_Status == CFE_PSP_SUCCESS)
197  {
198  DataPointer32++;
199  }
200  else
201  {
202  /* CFE_PSP_MemRead32 error */
203  Valid = FALSE;
204  BytesRemaining = 0;
205  MM_AppData.ErrCounter++;
206  CFE_EVS_SendEvent(MM_PSP_READ_ERR_EID, CFE_EVS_ERROR,
207  "PSP read memory error: RC=0x%08X, Src=0x%08X, Tgt=0x%08X, Type=MEM32",
208  (unsigned int)PSP_Status, (unsigned int)DataPointer32, (unsigned int)&ioBuffer32[i]);
209  /* Stop load i/o buffer loop */
210  break;
211  }
212  }
213 
214  /* Check for error loading i/o buffer */
215  if (PSP_Status == CFE_PSP_SUCCESS)
216  {
217  /* Write i/o buffer contents to file */
218  if ((OS_Status = OS_write(FileHandle, ioBuffer32, SegmentSize)) == SegmentSize)
219  {
220  /* Update process counters */
221  BytesRemaining -= SegmentSize;
222  BytesProcessed += SegmentSize;
223 
224  /* Prevent CPU hogging between dump segments */
225  if (BytesRemaining != 0)
226  {
227  MM_SegmentBreak();
228  }
229  }
230  else
231  {
232  /* OS_write error */
233  Valid = FALSE;
234  BytesRemaining = 0;
235  MM_AppData.ErrCounter++;
236  CFE_EVS_SendEvent(MM_OS_WRITE_EXP_ERR_EID, CFE_EVS_ERROR,
237  "OS_write error received: RC = 0x%08X Expected = %d File = '%s'",
238  (unsigned int)OS_Status, (int)SegmentSize, FileName);
239  }
240  }
241  }
242 
243  if (Valid)
244  {
245  /* Update last action statistics */
246  MM_AppData.LastAction = MM_DUMP_TO_FILE;
247  MM_AppData.MemType = MM_MEM32;
248  MM_AppData.Address = FileHeader->SymAddress.Offset;
249  strncpy(MM_AppData.FileName, FileName, OS_MAX_PATH_LEN);
250  MM_AppData.BytesProcessed = BytesProcessed;
251  }
252 
253  return(Valid);
254 
255 } /* end MM_DumpMem32ToFile */

258 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
259 /* */
260 /* Fill memory with the command specified fill pattern using only */
261 /* 32 bit wide writes */
262 /* */
263 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
264 void MM_FillMem32(uint32 DestAddress,
265  MM_FillMemCmd_t *CmdPtr)
266 {
267  uint32 i = 0;
268  int32 PSP_Status = CFE_PSP_SUCCESS;
269  uint32 BytesProcessed = 0;
270  uint32 BytesRemaining = CmdPtr->NumOfBytes;
271  uint32 FillPattern32 = CmdPtr->FillPattern;
272  uint32 *DataPointer32 = (uint32 *) DestAddress;
273  uint32 SegmentSize = MM_MAX_FILL_DATA_SEG;
274 
275  while (BytesRemaining != 0)
276  {
277  /* Set size of next segment */
278  if (BytesRemaining < MM_MAX_FILL_DATA_SEG)
279  {
280  SegmentSize = BytesRemaining;
281  }
282 
283  /* Fill next segment */
284  for (i = 0; i < (SegmentSize / sizeof(uint32)); i++)
285  {
286  PSP_Status = CFE_PSP_MemWrite32((uint32) DataPointer32, FillPattern32);
287  if (PSP_Status == CFE_PSP_SUCCESS)
288  {
289  DataPointer32++;
290  }
291  else
292  {
293  /* CFE_PSP_MemWrite32 error */
294  BytesRemaining = 0;
295  MM_AppData.ErrCounter++;
296  CFE_EVS_SendEvent(MM_PSP_WRITE_ERR_EID, CFE_EVS_ERROR,
297  "PSP write memory error: RC=0x%08X, Address=0x%08X, MemType=MEM32",
298  (unsigned int)PSP_Status, (unsigned int)DataPointer32);
299  /* Stop fill segment loop */
300  break;
301  }
302  }
303 
304  if (PSP_Status == CFE_PSP_SUCCESS)
305  {
306  /* Update process counters */
307  BytesRemaining -= SegmentSize;
308  BytesProcessed += SegmentSize;
309 
310  /* Prevent CPU hogging between fill segments */
311  if (BytesRemaining != 0)
312  {
313  MM_SegmentBreak();
314  }
315  }
316  }
317 
318  /* Update last action statistics */
319  if (BytesProcessed == CmdPtr->NumOfBytes)
320  {
321  MM_AppData.LastAction = MM_FILL;
322  MM_AppData.MemType = MM_MEM32;
323  MM_AppData.Address = DestAddress;
324  MM_AppData.DataValue = FillPattern32;
325  MM_AppData.BytesProcessed = BytesProcessed;
326  }
327 
328  return;
329 
330 }/* End MM_FillMem32 */
