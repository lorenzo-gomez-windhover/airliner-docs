108 int32 ULR::InitEvent()
109 {
110  int32 iStatus=CFE_SUCCESS;
111  int32 ind = 0;
112 
113  CFE_EVS_BinFilter_t EventTbl[ULR_MAX_EVENT_FILTERS];
114 
115  /* Initialize the event filter table.
116  * Note: 0 is the CFE_EVS_NO_FILTER mask and event 0 is reserved (not used) */
117  CFE_PSP_MemSet(EventTbl, 0x00, sizeof(EventTbl));
118 
119  EventTbl[ ind].EventID = ULR_RESERVED_EID;
120  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
121  EventTbl[ ind].EventID = ULR_INIT_INF_EID;
122  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
123  EventTbl[ ind].EventID = ULR_CMD_NOOP_EID;
124  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
125  EventTbl[ ind].EventID = ULR_SUBSCRIBE_ERR_EID;
126  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
127  EventTbl[ ind].EventID = ULR_PIPE_INIT_ERR_EID;
128  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
129  EventTbl[ ind].EventID = ULR_RCVMSG_ERR_EID;
130  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
131  EventTbl[ ind].EventID = ULR_MSGID_ERR_EID;
132  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
133  EventTbl[ ind].EventID = ULR_CC_ERR_EID;
134  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
135  EventTbl[ ind].EventID = ULR_MSGLEN_ERR_EID;
136  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
137  EventTbl[ ind].EventID = ULR_CUSTOM_INIT_ERR_EID;
138  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
139  EventTbl[ ind].EventID = ULR_UNEXPECTED_DATA_BEFORE_VERSION_ID;
140  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
141  EventTbl[ ind].EventID = ULR_UNEXPECTED_DATA_BEFORE_HEADER;
142  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
143  EventTbl[ ind].EventID = ULR_INVALID_CHECKSUM;
144  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
145  EventTbl[ ind].EventID = ULR_INVALID_PARSER_STATE;
146  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
147  EventTbl[ ind].EventID = ULR_LISTENER_CREATE_CHDTASK_ERR_EID;
148  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;

166 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
167 int32 ULR::InitPipe()
168 {
169  int32 iStatus=CFE_SUCCESS;
170 
171  /* Init schedule pipe and subscribe to wakeup messages */
172  iStatus = CFE_SB_CreatePipe(&SchPipeId,
173  ULR_SCH_PIPE_DEPTH,
174  ULR_SCH_PIPE_NAME);
175  if (iStatus == CFE_SUCCESS)
176  {
177  iStatus = CFE_SB_SubscribeEx(ULR_MEASURE_MID, SchPipeId, CFE_SB_Default_Qos, ULR_MEASURE_MID_MAX_MSG_COUNT);
178  if (iStatus != CFE_SUCCESS)
179  {
180  (void) CFE_EVS_SendEvent(ULR_SUBSCRIBE_ERR_EID, CFE_EVS_ERROR,
181  "Sch Pipe failed to subscribe to ULR_MEASURE_MID. (0x%08lX)",
182  iStatus);
183  goto ULR_InitPipe_Exit_Tag;
184  }
185 
186  iStatus = CFE_SB_SubscribeEx(ULR_SEND_HK_MID, SchPipeId, CFE_SB_Default_Qos, ULR_SEND_HK_MID_MAX_MSG_COUNT);
187  if (iStatus != CFE_SUCCESS)
188  {
189  (void) CFE_EVS_SendEvent(ULR_SUBSCRIBE_ERR_EID, CFE_EVS_ERROR,
190  "CMD Pipe failed to subscribe to ULR_SEND_HK_MID. (0x%08X)",
191  (unsigned int)iStatus);
192  goto ULR_InitPipe_Exit_Tag;
193  }
194  }
195  else
196  {
197  (void) CFE_EVS_SendEvent(ULR_PIPE_INIT_ERR_EID, CFE_EVS_ERROR,
198  "Failed to create SCH pipe (0x%08lX)",
199  iStatus);
200  goto ULR_InitPipe_Exit_Tag;
201  }
202 
203  /* Init command pipe and subscribe to command messages */
204  iStatus = CFE_SB_CreatePipe(&CmdPipeId,
205  ULR_CMD_PIPE_DEPTH,
206  ULR_CMD_PIPE_NAME);
207  if (iStatus == CFE_SUCCESS)
208  {
209  /* Subscribe to command messages */
210  iStatus = CFE_SB_Subscribe(ULR_CMD_MID, CmdPipeId);
211 
212  if (iStatus != CFE_SUCCESS)
213  {
214  (void) CFE_EVS_SendEvent(ULR_SUBSCRIBE_ERR_EID, CFE_EVS_ERROR,
215  "CMD Pipe failed to subscribe to ULR_CMD_MID. (0x%08lX)",
216  iStatus);
217  goto ULR_InitPipe_Exit_Tag;
218  }
219  }
220  else
221  {
222  (void) CFE_EVS_SendEvent(ULR_PIPE_INIT_ERR_EID, CFE_EVS_ERROR,
223  "Failed to create CMD pipe (0x%08lX)",
224  iStatus);
225  goto ULR_InitPipe_Exit_Tag;
226  }
227 
228 ULR_InitPipe_Exit_Tag:
229  return iStatus;
230 }

337 /* Receive and Process Messages */
338 /* */
339 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
340 int32 ULR::RcvSchPipeMsg(int32 iBlocking)
341 {
342  int32 iStatus=CFE_SUCCESS;
343  CFE_SB_Msg_t* MsgPtr=NULL;
344  CFE_SB_MsgId_t MsgId;
345 
346  /* Stop Performance Log entry */
347  CFE_ES_PerfLogExit(ULR_MAIN_TASK_PERF_ID);
348 
349  /* Wait for WakeUp messages from scheduler */
350  iStatus = CFE_SB_RcvMsg(&MsgPtr, SchPipeId, iBlocking);
351 
352  /* Start Performance Log entry */
353  CFE_ES_PerfLogEntry(ULR_MAIN_TASK_PERF_ID);
354 
355  if (iStatus == CFE_SUCCESS)
356  {
357  MsgId = CFE_SB_GetMsgId(MsgPtr);
358  switch (MsgId)
359  {
360  case ULR_MEASURE_MID:
361  {
362  /* When rate monotonic call ReportDistance() here */
363  break;
364  }
365 
366  case ULR_SEND_HK_MID:
367  {
368  ProcessCmdPipe();
369  ReportHousekeeping();
370  break;
371  }
372 
373  default:
374  {
375  (void) CFE_EVS_SendEvent(ULR_MSGID_ERR_EID, CFE_EVS_ERROR,
376  "Recvd invalid SCH msgId (0x%04X)", MsgId);
377  break;
378  }
379  }
380  }
381  else if (iStatus == CFE_SB_NO_MESSAGE)
382  {
383  iStatus = CFE_SUCCESS;
384  }
385  else if (iStatus == CFE_SB_TIME_OUT)
386  {
387  iStatus = CFE_SUCCESS;
388  }
389  else
390  {
391  (void) CFE_EVS_SendEvent(ULR_RCVMSG_ERR_EID, CFE_EVS_ERROR,
392  "SCH pipe read error (0x%08lX).", iStatus);
393  }
394 
395  return iStatus;
396 }

400 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
401 /* */
402 /* Process Incoming Commands */
403 /* */
404 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
405 void ULR::ProcessCmdPipe()
406 {
407  int32 iStatus = CFE_SUCCESS;
408  CFE_SB_Msg_t* CmdMsgPtr=NULL;
409  CFE_SB_MsgId_t CmdMsgId;
410 
411  /* Process command messages until the pipe is empty */
412  while (1)
413  {
414  iStatus = CFE_SB_RcvMsg(&CmdMsgPtr, CmdPipeId, CFE_SB_POLL);
415  if(iStatus == CFE_SUCCESS)
416  {
417  CmdMsgId = CFE_SB_GetMsgId(CmdMsgPtr);
418  switch (CmdMsgId)
419  {
420  case ULR_CMD_MID:
421  {
422  ProcessAppCmds(CmdMsgPtr);
423  break;
424  }
425 
426  default:
427  {
428  /* Bump the command error counter for an unknown command.
429  * (This should only occur if it was subscribed to with this
430  * pipe, but not handled in this switch-case.) */
431  HkTlm.usCmdErrCnt++;
432  (void) CFE_EVS_SendEvent(ULR_MSGID_ERR_EID, CFE_EVS_ERROR,
433  "Recvd invalid CMD msgId (0x%04X)", (unsigned short)CmdMsgId);
434  break;
435  }
436  }
437  }
438  else if (iStatus == CFE_SB_NO_MESSAGE)
439  {
440  break;
441  }
442  else
443  {
444  (void) CFE_EVS_SendEvent(ULR_RCVMSG_ERR_EID, CFE_EVS_ERROR,
445  "CMD pipe read error (0x%08lX)", iStatus);
446  break;
447  }
448  }
449 }

453 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
454 /* */
455 /* Process ULR Commands */
456 /* */
457 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
458 void ULR::ProcessAppCmds(CFE_SB_Msg_t* MsgPtr)
459 {
460  uint32 uiCmdCode=0;
461 
462  if (MsgPtr != NULL)
463  {
464  uiCmdCode = CFE_SB_GetCmdCode(MsgPtr);
465  switch (uiCmdCode)
466  {
467  case ULR_NOOP_CC:
468  {
469  HkTlm.usCmdCnt++;
470  (void) CFE_EVS_SendEvent(ULR_CMD_NOOP_EID, CFE_EVS_INFORMATION,
471  "Recvd NOOP. Version %d.%d.%d.%d",
472  ULR_MAJOR_VERSION,
473  ULR_MINOR_VERSION,
474  ULR_REVISION,
475  ULR_MISSION_REV);
476  break;
477  }
478 
479  case ULR_RESET_CC:
480  {
481  HkTlm.usCmdCnt = 0;
482  HkTlm.usCmdErrCnt = 0;
483  break;
484  }
485 
486  default:
487  {
488  HkTlm.usCmdErrCnt++;
489  (void) CFE_EVS_SendEvent(ULR_CC_ERR_EID, CFE_EVS_ERROR,
490  "Recvd invalid command code (%u)", (unsigned int)uiCmdCode);
491  break;
492  }
493  }
494  }
495 }

547

561  MsgId, usCmdCode, usMsgLen, usExpectedLen);
562  HkTlm.usCmdErrCnt++;
563  }
564  }
565 
566  return bResult;

709 /* */
710 /* Listener task main function */
711 /* */
712 void ULR::ListenerTaskMain(void)
713 {
714 
715  CFE_ES_RegisterChildTask();
716 
717  while(ChildContinueExec())
718  {
719  uint8 buf[ULR_BUF_LEN];
720  uint32 size = ULR_BUF_LEN;
721 
722  int32 iStatus = ReadDevice(buf, &size);
723  if(iStatus == CFE_SUCCESS)
724  {
725  for(uint32 i = 0; i < size; i++)
726  {
727  switch(ParserState)
728  {
729  case ULR_PARSER_STATE_UNINITIALIZED:
730  {
731  if(buf[i] == ULR_HEADER_SYNC_SYMBOL)
732  {
733  ParserState = ULR_PARSER_STATE_WAITING_FOR_VERSION_ID;
734  }
735  break;
736  }
737 
738  case ULR_PARSER_STATE_WAITING_FOR_HEADER:
739  {
740  if(buf[i] == ULR_HEADER_SYNC_SYMBOL)
741  {
742  ParserState = ULR_PARSER_STATE_WAITING_FOR_VERSION_ID;
743  }
744  else
745  {
746  (void) CFE_EVS_SendEvent(ULR_UNEXPECTED_DATA_BEFORE_HEADER, CFE_EVS_ERROR,
747  "Received 0x%02x before header.", buf[i]);
748  ParserState = ULR_PARSER_STATE_UNINITIALIZED;
749  }
750  break;
751  }
752 
753  case ULR_PARSER_STATE_WAITING_FOR_VERSION_ID:
754  {
755  if(buf[i] == 0x01)
756  {
757  UartMessage.VersionID = buf[i];
758  ParserState = ULR_PARSER_STATE_WAITING_FOR_ALT_BYTE_1;
759  }
760  else
761  {
762  (void) CFE_EVS_SendEvent(ULR_UNEXPECTED_DATA_BEFORE_VERSION_ID, CFE_EVS_ERROR,
763  "Received 0x%02x before version ID.", buf[i]);
764  ParserState = ULR_PARSER_STATE_UNINITIALIZED;
765  }
766  break;
767  }
768 
769  case ULR_PARSER_STATE_WAITING_FOR_ALT_BYTE_1:
770  {
771  UartMessage.AltitudeL = buf[i];
772  ParserState = ULR_PARSER_STATE_WAITING_FOR_ALT_BYTE_2;
773  break;
774  }
775 
776  case ULR_PARSER_STATE_WAITING_FOR_ALT_BYTE_2:
777  {
778  UartMessage.AltitudeH = buf[i];
779  ParserState = ULR_PARSER_STATE_WAITING_FOR_SNR;
780  break;
781  }
782 
783  case ULR_PARSER_STATE_WAITING_FOR_SNR:
784  {
785  UartMessage.SNR = buf[i];
786  ParserState = ULR_PARSER_STATE_WAITING_FOR_CHECKSUM;
787  break;
788  }
789 
790  case ULR_PARSER_STATE_WAITING_FOR_CHECKSUM:
791  {
792  UartMessage.Checksum = buf[i];
793  if(IsChecksumOk())
794  {
795  OS_MutSemTake(Mutex);
796  DistanceSensor.Timestamp = PX4LIB_GetPX4TimeUs();
797  DistanceSensor.MinDistance = ULR_MIN_DISTANCE;
798  DistanceSensor.MaxDistance = ULR_MAX_DISTANCE;
799  DistanceSensor.CurrentDistance = HeightFilter.apply( ((UartMessage.AltitudeH << 8) + UartMessage.AltitudeL) / 100.0f);
800  DistanceSensor.Covariance = ULR_SENS_VARIANCE;
801  DistanceSensor.Type = PX4_DISTANCE_SENSOR_RADAR;
802  DistanceSensor.ID = 0;
803  DistanceSensor.Orientation = (PX4_SensorOrientation_t)0;
804  OS_MutSemGive(Mutex);
805 
806  ReportDistance();
807  }
808  ParserState = ULR_PARSER_STATE_WAITING_FOR_HEADER;
809  break;
810  }
811 
812  default:
813  {
814  (void) CFE_EVS_SendEvent(ULR_INVALID_PARSER_STATE, CFE_EVS_ERROR,
815  "Parser in invalid state.");
816  ParserState = ULR_PARSER_STATE_UNINITIALIZED;
817  break;
818  }
819  }
820  }
821  }
822  }
823 
824  CFE_ES_ExitChildTask();
825 }
