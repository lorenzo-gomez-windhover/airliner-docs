73 /* \brief Data Storage packet processor
74 **
75 ** \par Description
76 ** This function searches for a packet filter table entry that
77 ** matches the input argument Message ID. If no matching packet
78 ** filter table entry is found, the packet referenced via the
79 ** Message Pointer is discarded (filtered). When a matching
80 ** packet table entry is found, each of the multiple filters
81 ** defined for the packet are tested by applying the common
82 ** CFS filter algorithm to the table defined filter parameters.
83 ** Packets that are passed by any filter continue with the
84 ** write process. Next step: prepare to write data to a file.
85 **
86 ** \par Assumptions, External Events, and Notes:
87 ** (none)
88 **
89 ** \param [in] Message ID (extracted from message pointer)
90 ** \param [in] Software Bus message pointer (#CFE_SB_MsgPtr_t)
91 **
92 ** \sa #DS_PacketEntry_t, #DS_FilterParms_t, #DS_DestFileEntry_t
93 */
94 void DS_FileStorePacket(CFE_SB_MsgId_t MessageID, CFE_SB_MsgPtr_t MessagePtr);

97 /*******************************************************************/
98 /* \brief Prepare to write to a data storage destination file
99 **
100 ** \par Description
101 ** This function is called for packets that have passed the
102 ** common CFS filter algorithm. The function first queries the
103 ** packet length and determines whether writing the packet to
104 ** an existing data storage file would exceed the max file size
105 ** limit. If so, the existing destination file header is updated
106 ** and the file is closed. Then the function determines whether
107 ** a new file needs to be opened and if so, creates the file.
108 ** Files may be closed due to size or by command, and files may
109 ** have not yet been created because this is the first packet
110 ** destined for that file. Next step: write data to file.
111 **
112 ** \par Assumptions, External Events, and Notes:
113 ** (none)
114 **
115 ** \param [in] Destination file index
116 ** \param [in] Software Bus message pointer (#CFE_SB_MsgPtr_t)
117 **
118 ** \sa #DS_AppFileStatus_t, #DS_DestFileEntry_t
119 */
120 void DS_FileSetupWrite(int32 FileIndex, CFE_SB_MsgPtr_t MessagePtr);

238 /* \brief Construct the sequence portion of a filename
239 **
240 ** \par Description
241 ** The filename sequence string will be constructed to contain
242 ** either the current date and time, or a file sequence count

250 ** (none)
251 **
252 ** \param [in] Pointer to buffer for sequence portion of filename
253 ** \param [in] Filename type (date and time vs sequence count)
254 ** \param [in] Sequence counter (used only if type is sequence)
255 **
256 ** \sa #DS_DestFileEntry_t
