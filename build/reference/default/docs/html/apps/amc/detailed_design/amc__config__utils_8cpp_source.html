49 /* */
50 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
51 int32 AMC::InitConfigTbl(void)
52 {
53  int32 iStatus=0;
54 
55  /* Register Config table */
56  iStatus = CFE_TBL_Register(&PwmConfigTblHdl,
57  AMC_PWM_CONFIG_TABLENAME,
58  (sizeof(AMC_PwmConfigTbl_t)),
59  CFE_TBL_OPT_DEFAULT,
60  AMC::ValidatePwmCfgTbl);
61  if (iStatus != CFE_SUCCESS)
62  {
63  /* Note, a critical table could return another nominal code. If this
64  * table is made critical this logic would have to change.
65  */
66  CFE_EVS_SendEvent(AMC_PWM_CFGTBL_REG_ERR_EID, CFE_EVS_ERROR,
67  "Failed to register PWM table (0x%08X)",
68  (unsigned int)iStatus);
69  goto AMC_InitConfigTbl_Exit_Tag;
70  }
71 
72  /* Register Mixer table */
73  iStatus = CFE_TBL_Register(&MixerConfigTblHdl,
74  AMC_MIXER_CONFIG_TABLENAME,
75  (sizeof(MultirotorMixer_ConfigTable_t)),
76  CFE_TBL_OPT_DEFAULT,
77  AMC::ValidateMixerCfgTbl);
78  if (iStatus != CFE_SUCCESS)
79  {
80  /* Note, a critical table could return another nominal code. If this
81  * table is made critical this logic would have to change.
82  */
83  CFE_EVS_SendEvent(AMC_MIXER_CFGTBL_REG_ERR_EID, CFE_EVS_ERROR,
84  "Failed to register Mixer table (0x%08X)",
85  (unsigned int)iStatus);
86  goto AMC_InitConfigTbl_Exit_Tag;
87  }
88 
89  /* Load Config table file */
90  iStatus = CFE_TBL_Load(PwmConfigTblHdl,
91  CFE_TBL_SRC_FILE,
92  AMC_PWM_CONFIG_TABLE_FILENAME);
93  if (iStatus != CFE_SUCCESS)
94  {
95  /* Note, CFE_SUCCESS is for a successful full table load. If a
96  * partial table load is desired then this logic would have to
97  * change.
98  */
99  CFE_EVS_SendEvent(AMC_PWM_CFGTBL_LOAD_ERR_EID, CFE_EVS_ERROR,
100  "Failed to load PWM Config Table (0x%08X)",
101  (unsigned int)iStatus);
102  goto AMC_InitConfigTbl_Exit_Tag;
103  }
104 
105  /* Load Config table file */
106  iStatus = CFE_TBL_Load(MixerConfigTblHdl,
107  CFE_TBL_SRC_FILE,
108  AMC_MIXER_CONFIG_TABLE_FILENAME);
109  if (iStatus != CFE_SUCCESS)
110  {
111  /* Note, CFE_SUCCESS is for a successful full table load. If a
112  * partial table load is desired then this logic would have to
113  * change.
114  */
115  CFE_EVS_SendEvent(AMC_MIXER_CFGTBL_LOAD_ERR_EID, CFE_EVS_ERROR,
116  "Failed to load Mixer Config Table (0x%08X)",
117  (unsigned int)iStatus);
118  goto AMC_InitConfigTbl_Exit_Tag;
119  }
120 
121  iStatus = AcquireConfigPointers();
122 
123 AMC_InitConfigTbl_Exit_Tag:
124  return iStatus;
125 }

132 int32 AMC::ValidatePwmCfgTbl(void* ConfigTblPtr)
133 {
134  uint32 i = 0;
135  int32 iStatus=0;
136  AMC_PwmConfigTbl_t* AMC_PwmConfigTblPtr =
137  (AMC_PwmConfigTbl_t*)(ConfigTblPtr);
138 
139  if(AMC_PwmConfigTblPtr->PwmMin < AMC_PwmConfigTblPtr->PwmDisarmed)
140  {
141  CFE_EVS_SendEvent(AMC_PWM_CFGTBL_MIN_LT_DISARMED_ERR_EID,
142  CFE_EVS_ERROR,
143  "PWM Tbl Vldt: Min (%u) less than Disarmed (%u) speed.",
144  (unsigned int)AMC_PwmConfigTblPtr->PwmMin,
145  (unsigned int)AMC_PwmConfigTblPtr->PwmDisarmed);
146  iStatus = -1;
147  goto AMC_ValidatePwmCfgTbl_Exit_Tag;
148  }
149 
150  if(AMC_PwmConfigTblPtr->PwmMax < AMC_PwmConfigTblPtr->PwmMin)
151  {
152  CFE_EVS_SendEvent(AMC_PWM_CFGTBL_MAX_LT_MIN_ERR_EID, CFE_EVS_ERROR,
153  "PWM Tbl Vldt: Max (%u) less than Min (%u) speed.",
154  (unsigned int)AMC_PwmConfigTblPtr->PwmMax,
155  (unsigned int)AMC_PwmConfigTblPtr->PwmMin);
156  iStatus = -1;
157  goto AMC_ValidatePwmCfgTbl_Exit_Tag;
158  }
159 
160 AMC_ValidatePwmCfgTbl_Exit_Tag:
161  return iStatus;
162 }

166 /* Validate Mixer Configuration Table */
167 /* */
168 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
169 int32 AMC::ValidateMixerCfgTbl(void* ConfigTblPtr)
170 {
171  uint32 i = 0;
172  int32 iStatus=0;
173  MultirotorMixer_ConfigTable_t* TblPtr =
174  (MultirotorMixer_ConfigTable_t*)(ConfigTblPtr);
175  boolean rollScaleNotFinite = FALSE;
176  boolean pitchScaleNotFinite = FALSE;
177  boolean yawScaleNotFinite = FALSE;
178  boolean idleSpeedNotFinite = FALSE;
179  boolean deltaOutMaxNotFinite = FALSE;
180  boolean invalidGeometry = FALSE;
181  boolean invalidRotorCount = FALSE;
182  boolean outScaleNotFinite = FALSE;
183 
184  rollScaleNotFinite = !isfinite(TblPtr->RollScale);
185  pitchScaleNotFinite = !isfinite(TblPtr->PitchScale);
186  yawScaleNotFinite = !isfinite(TblPtr->YawScale);
187  idleSpeedNotFinite = !isfinite(TblPtr->IdleSpeed);
188  deltaOutMaxNotFinite = !isfinite(TblPtr->DeltaOutMax);
189  if(TblPtr->Geometry >= MIXER_MAX_GEOMETRY)
190  {
191  invalidGeometry = TRUE;
192  }
193 
194  if((TblPtr->RotorCount < 1) ||
195  (TblPtr->RotorCount < AMC_MAX_MOTOR_OUTPUTS))
196  {
197  invalidRotorCount = TRUE;
198  }
199 
200  if(rollScaleNotFinite || pitchScaleNotFinite || yawScaleNotFinite
201  || idleSpeedNotFinite || invalidGeometry )
202  {
203  CFE_EVS_SendEvent(AMC_MIXER_INVALID_MIXER_ERR_EID, CFE_EVS_ERROR,
204  "Mixer cfg tbl invld. R[%u] P[%u] Y[%u] I[%u] G[%u] \
205  D[%u] RC[%u]",
206  rollScaleNotFinite, pitchScaleNotFinite, yawScaleNotFinite,
207  idleSpeedNotFinite, invalidGeometry, deltaOutMaxNotFinite,
208  invalidRotorCount);
209  iStatus = -1;
210  goto AMC_ValidateMixerCfgTbl_Exit_Tag;
211  }
212 
213  for(i = 0; i < TblPtr->RotorCount; ++i)
214  {
215  rollScaleNotFinite = !isfinite(TblPtr->RotorConfig[i].RollScale);
216  pitchScaleNotFinite = !isfinite(TblPtr->RotorConfig[i].PitchScale);
217  yawScaleNotFinite = !isfinite(TblPtr->RotorConfig[i].YawScale);
218  outScaleNotFinite = !isfinite(TblPtr->RotorConfig[i].OutScale);
219 
220  if(rollScaleNotFinite || pitchScaleNotFinite || yawScaleNotFinite
221  || outScaleNotFinite )
222  {
223  CFE_EVS_SendEvent(AMC_MIXER_INVLD_ROTOR_CONFIG_ERR_EID,
224  CFE_EVS_ERROR, "Mixer cfg tbl invld rotor[%u] config. \
225  R[%u] P[%u] Y[%u] O[%u]",
226  (unsigned int)i, rollScaleNotFinite, pitchScaleNotFinite,
227  yawScaleNotFinite,
228  outScaleNotFinite);
229  iStatus = -1;
230  goto AMC_ValidateMixerCfgTbl_Exit_Tag;
231  }
232  }
233 
234  for(i = TblPtr->RotorCount; i < AMC_MAX_MOTOR_OUTPUTS; ++i)
235  {
236  boolean unusedRollScaleNotZero = FALSE;
237  boolean unusedPitchScaleNotZero = FALSE;
238  boolean unusedYawScaleNotZero = FALSE;
239  boolean unusedOutScaleNotZero = FALSE;
240 
241  if((uint32)TblPtr->RotorConfig[i].RollScale != 0)
242  {
243  unusedRollScaleNotZero = TRUE;
244  }
245 
246  if((uint32)TblPtr->RotorConfig[i].PitchScale != 0)
247  {
248  unusedPitchScaleNotZero = TRUE;
249  }
250 
251  if((uint32)TblPtr->RotorConfig[i].YawScale != 0)
252  {
253  unusedYawScaleNotZero = TRUE;
254  }
255 
256  if((uint32)TblPtr->RotorConfig[i].OutScale != 0)
257  {
258  unusedOutScaleNotZero = TRUE;
259  }
260 
261  if(unusedRollScaleNotZero || unusedPitchScaleNotZero ||
262  unusedYawScaleNotZero || unusedOutScaleNotZero)
263  {
264  CFE_EVS_SendEvent(AMC_MIXER_TBLVAL_NONZER_DATA_ERR_EID,
265  CFE_EVS_ERROR,
266  "Mixer cfg tbl contains unused nonzero data.");
267  iStatus = -1;
268  goto AMC_ValidateMixerCfgTbl_Exit_Tag;
269  }
270  }
271 
272 AMC_ValidateMixerCfgTbl_Exit_Tag:
273  return iStatus;
274 }

278 /* Acquire Config Pointers */
279 /* */
280 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
281 int32 AMC::AcquireConfigPointers(void)
282 {
283  int32 iStatus = CFE_SUCCESS;
284 
285  /*
286  ** Release the table
287  */
288  /* TODO: This return value can indicate success, error, or that the info
289  * has been updated. We ignore this return value in favor of checking
290  * CFE_TBL_Manage(), but be sure this is the behavior you want.
291  */
292  CFE_TBL_ReleaseAddress(PwmConfigTblHdl);
293 
294  CFE_TBL_ReleaseAddress(MixerConfigTblHdl);
295 
296  /*
297  ** Manage the table
298  */
299  iStatus = CFE_TBL_Manage(PwmConfigTblHdl);
300  if ((iStatus != CFE_SUCCESS) && (iStatus != CFE_TBL_INFO_UPDATED))
301  {
302  CFE_EVS_SendEvent(AMC_PWM_CFGTBL_MANAGE_ERR_EID, CFE_EVS_ERROR,
303  "Failed to manage PWM Config table (0x%08X)",
304  (unsigned int)iStatus);
305  goto AMC_AcquireConfigPointers_Exit_Tag;
306  }
307 
308  iStatus = CFE_TBL_Manage(MixerConfigTblHdl);
309  if ((iStatus != CFE_SUCCESS) && (iStatus != CFE_TBL_INFO_UPDATED))
310  {
311  CFE_EVS_SendEvent(AMC_MIXER_CFGTBL_MANAGE_ERR_EID, CFE_EVS_ERROR,
312  "Failed to manage Mixer Config table (0x%08X)",
313  (unsigned int)iStatus);
314  goto AMC_AcquireConfigPointers_Exit_Tag;
315  }
316 
317  /*
318  ** Get a pointer to the table
319  */
320  iStatus = CFE_TBL_GetAddress((void**)&PwmConfigTblPtr,
321  PwmConfigTblHdl);
322  if (iStatus == CFE_TBL_INFO_UPDATED)
323  {
324  iStatus = CFE_SUCCESS;
325  }
326  else if(iStatus != CFE_SUCCESS)
327  {
328  PwmConfigTblPtr = 0;
329  CFE_EVS_SendEvent(AMC_PWM_CFGTBL_GETADDR_ERR_EID, CFE_EVS_ERROR,
330  "Failed to get PWM Config table's address (0x%08X)",
331  (unsigned int)iStatus);
332  }
333 
334  iStatus = CFE_TBL_GetAddress((void**)&MixerConfigTblPtr,
335  MixerConfigTblHdl);
336  if (iStatus == CFE_TBL_INFO_UPDATED)
337  {
338  iStatus = CFE_SUCCESS;
339  }
340  else if(iStatus != CFE_SUCCESS)
341  {
342  MixerConfigTblPtr = 0;
343  CFE_EVS_SendEvent(AMC_MIXER_CFGTBL_GETADDR_ERR_EID, CFE_EVS_ERROR,
344  "Failed to get Mixer Config table's address (0x%08X)",
345  (unsigned int)iStatus);
346  }
347 
348 AMC_AcquireConfigPointers_Exit_Tag:
349  return iStatus;
350 
351 }
