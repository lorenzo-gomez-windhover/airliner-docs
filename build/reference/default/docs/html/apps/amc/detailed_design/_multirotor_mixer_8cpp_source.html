85 uint32
86 MultirotorMixer::mix(float *outputs, uint32 space, uint16 *status_reg)
87 {
88  /* Summary of mixing strategy:
89  1) mix roll, pitch and thrust without yaw.
90  2) if some outputs violate range [0,1] then try to shift all outputs to minimize violation ->
91  increase or decrease total thrust (boost). The total increase or decrease of thrust is limited
92  (max_thrust_diff). If after the shift some outputs still violate the bounds then scale roll & pitch.
93  In case there is violation at the lower and upper bound then try to shift such that violation is equal
94  on both sides.
95  3) mix in yaw and scale if it leads to limit violation.
96  4) scale all outputs to range [idle_speed,1]
97  */
98 
99  float roll = Contrain(get_control(0, 0) * m_ConfigTablePtr->RollScale, -1.0f, 1.0f);
100  float pitch = Contrain(get_control(0, 1) * m_ConfigTablePtr->PitchScale, -1.0f, 1.0f);
101  float yaw = Contrain(get_control(0, 2) * m_ConfigTablePtr->YawScale, -1.0f, 1.0f);
102  float thrust = Contrain(get_control(0, 3) * 1.0f, 0.0f, 1.0f);
103  float min_out = 1.0f;
104  float max_out = 0.0f;
105 
106  // clean out class variable used to capture saturation
107  m_SaturationStatus.value = 0;
108 
109  // thrust boost parameters
110  float thrust_increase_factor = 1.5f;
111  float thrust_decrease_factor = 0.6f;
112 
113  /* perform initial mix pass yielding unbounded outputs, ignore yaw */
114  for (uint32 i = 0; i < m_ConfigTablePtr->RotorCount; i++) {
115  float out = roll * m_ConfigTablePtr->RotorConfig[i].RollScale +
116  pitch * m_ConfigTablePtr->RotorConfig[i].PitchScale +
117  thrust;
118 
119  out *= m_ConfigTablePtr->RotorConfig[i].OutScale;
120 
121  /* calculate min and max output values */
122  if (out < min_out) {
123  min_out = out;
124  }
125 
126  if (out > max_out) {
127  max_out = out;
128  }
129 
130  outputs[i] = out;
131  }
132 
133  float boost = 0.0f; // value added to demanded thrust (can also be negative)
134  float roll_pitch_scale = 1.0f; // scale for demanded roll and pitch
135 
136  if (min_out < 0.0f && max_out < 1.0f && -min_out <= 1.0f - max_out) {
137  float max_thrust_diff = thrust * thrust_increase_factor - thrust;
138 
139  if (max_thrust_diff >= -min_out) {
140  boost = -min_out;
141 
142  } else {
143  boost = max_thrust_diff;
144  roll_pitch_scale = (thrust + boost) / (thrust - min_out);
145  }
146 
147  } else if (max_out > 1.0f && min_out > 0.0f && min_out >= max_out - 1.0f) {
148  float max_thrust_diff = thrust - thrust_decrease_factor * thrust;
149 
150  if (max_thrust_diff >= max_out - 1.0f) {
151  boost = -(max_out - 1.0f);
152 
153  } else {
154  boost = -max_thrust_diff;
155  roll_pitch_scale = (1 - (thrust + boost)) / (max_out - thrust);
156  }
157 
158  } else if (min_out < 0.0f && max_out < 1.0f && -min_out > 1.0f - max_out) {
159  float max_thrust_diff = thrust * thrust_increase_factor - thrust;
160  boost = math::constrain(-min_out - (1.0f - max_out) / 2.0f, 0.0f, max_thrust_diff);
161  roll_pitch_scale = (thrust + boost) / (thrust - min_out);
162 
163  } else if (max_out > 1.0f && min_out > 0.0f && min_out < max_out - 1.0f) {
164  float max_thrust_diff = thrust - thrust_decrease_factor * thrust;
165  boost = math::constrain(-(max_out - 1.0f - min_out) / 2.0f, -max_thrust_diff, 0.0f);
166  roll_pitch_scale = (1 - (thrust + boost)) / (max_out - thrust);
167 
168  } else if (min_out < 0.0f && max_out > 1.0f) {
169  boost = math::constrain(-(max_out - 1.0f + min_out) / 2.0f, thrust_decrease_factor * thrust - thrust,
170  thrust_increase_factor * thrust - thrust);
171  roll_pitch_scale = (thrust + boost) / (thrust - min_out);
172  }
173 
174  // capture saturation
175  if (min_out < 0.0f) {
176  m_SaturationStatus.flags.motor_neg = true;
177  }
178 
179  if (max_out > 1.0f) {
180  m_SaturationStatus.flags.motor_pos = true;
181  }
182 
183  // Thrust reduction is used to reduce the collective thrust if we hit
184  // the upper throttle limit
185  float thrust_reduction = 0.0f;
186 
187  // mix again but now with thrust boost, scale roll/pitch and also add yaw
188  for (uint32 i = 0; i < m_ConfigTablePtr->RotorCount; i++) {
189  float out = (roll * m_ConfigTablePtr->RotorConfig[i].RollScale +
190  pitch * m_ConfigTablePtr->RotorConfig[i].PitchScale) * roll_pitch_scale +
191  yaw * m_ConfigTablePtr->RotorConfig[i].YawScale +
192  thrust + boost;
193 
194  out *= m_ConfigTablePtr->RotorConfig[i].OutScale;
195 
196  // scale yaw if it violates limits. inform about yaw limit reached
197  if (out < 0.0f) {
198  if (fabsf(m_ConfigTablePtr->RotorConfig[i].YawScale) <= FLT_EPSILON) {
199  yaw = 0.0f;
200 
201  } else {
202  yaw = -((roll * m_ConfigTablePtr->RotorConfig[i].RollScale + pitch * m_ConfigTablePtr->RotorConfig[i].PitchScale) *
203  roll_pitch_scale + thrust + boost) / m_ConfigTablePtr->RotorConfig[i].YawScale;
204  }
205 
206  } else if (out > 1.0f) {
207  // allow to reduce thrust to get some yaw response
208  float prop_reduction = fminf(0.15f, out - 1.0f);
209  // keep the maximum requested reduction
210  thrust_reduction = fmaxf(thrust_reduction, prop_reduction);
211 
212  if (fabsf(m_ConfigTablePtr->RotorConfig[i].YawScale) <= FLT_EPSILON) {
213  yaw = 0.0f;
214 
215  } else {
216  yaw = (1.0f - ((roll * m_ConfigTablePtr->RotorConfig[i].RollScale + pitch * m_ConfigTablePtr->RotorConfig[i].PitchScale) *
217  roll_pitch_scale + (thrust - thrust_reduction) + boost)) / m_ConfigTablePtr->RotorConfig[i].YawScale;
218  }
219  }
220  }
221 
222  // Apply collective thrust reduction, the maximum for one prop
223  thrust -= thrust_reduction;
224 
225  // add yaw and scale outputs to range idle_speed...1
226  for (uint32 i = 0; i < m_ConfigTablePtr->RotorCount; i++) {
227  /* shift to output range here to avoid runtime calculation */
228  float idle_speed_fixed = -1.0f + m_ConfigTablePtr->IdleSpeed * 2.0f;
229 
230  outputs[i] = (roll * m_ConfigTablePtr->RotorConfig[i].RollScale +
231  pitch * m_ConfigTablePtr->RotorConfig[i].PitchScale) * roll_pitch_scale +
232  yaw * m_ConfigTablePtr->RotorConfig[i].YawScale +
233  thrust + boost;
234 
235  /*
236  implement simple model for static relationship between applied motor pwm and motor thrust
237  model: thrust = (1 - _thrust_factor) * PWM + _thrust_factor * PWM^2
238  this model assumes normalized input / output in the range [0,1] so this is the right place
239  to do it as at this stage the outputs are in that range.
240  */
241  if (m_ThrustFactor > 0.0f) {
242  outputs[i] = -(1.0f - m_ThrustFactor) / (2.0f * m_ThrustFactor) + sqrtf((1.0f - m_ThrustFactor) *
243  (1.0f - m_ThrustFactor) / (4.0f * m_ThrustFactor * m_ThrustFactor) + (outputs[i] < 0.0f ? 0.0f : outputs[i] /
244  m_ThrustFactor));
245  }
246 
247  outputs[i] = math::constrain(idle_speed_fixed + (outputs[i] * (1.0f - idle_speed_fixed)), idle_speed_fixed, 1.0f);
248 
249  }
250 
251  /* slew rate limiting and saturation checking */
252  for (uint32 i = 0; i < m_ConfigTablePtr->RotorCount; i++) {
253  /* shift to output range here to avoid runtime calculation */
254  float idle_speed_fixed = -1.0f + m_ConfigTablePtr->IdleSpeed * 2.0f;
255  bool clipping_high = false;
256  bool clipping_low = false;
257 
258  // check for saturation against static limits
259  if (outputs[i] > 0.99f) {
260  clipping_high = true;
261 
262  } else if (outputs[i] < idle_speed_fixed + 0.01f) {
263  clipping_low = true;
264 
265  }
266 
267  // check for saturation against slew rate limits
268  if (m_ConfigTablePtr->DeltaOutMax > 0.0f) {
269  float delta_out = outputs[i] - m_OutputsPrev[i];
270 
271  if (delta_out > m_ConfigTablePtr->DeltaOutMax) {
272  outputs[i] = m_OutputsPrev[i] + m_ConfigTablePtr->DeltaOutMax;
273  clipping_high = true;
274 
275  } else if (delta_out < -m_ConfigTablePtr->DeltaOutMax) {
276  outputs[i] = m_OutputsPrev[i] - m_ConfigTablePtr->DeltaOutMax;
277  clipping_low = true;
278 
279  }
280  }
281 
282  m_OutputsPrev[i] = outputs[i];
283 
284  // update the saturation status report
285  update_saturation_status(i, clipping_high, clipping_low);
286 
287  }
288 
289  // this will force the caller of the mixer to always supply new slew rate values, otherwise no slew rate limiting will happen
290  m_ConfigTablePtr->DeltaOutMax = 0.0f;
291 
292  // Notify saturation status
293  if (status_reg != nullptr) {
294  (*status_reg) = m_SaturationStatus.value;
295  }
296 
297  return m_ConfigTablePtr->RotorCount;
298 }

300 /*
301  * This function update the control saturation status report using hte following inputs:
302  *
303  * index: 0 based index identifying the motor that is saturating
304  * clipping_high: true if the motor demand is being limited in the positive direction
305  * clipping_low: true if the motor demand is being limited in the negative direction
306 */
307 void
308 MultirotorMixer::update_saturation_status(uint32 index, bool clipping_high, bool clipping_low)
309 {
310  // The motor is saturated at the upper limit
311  // check which control axes and which directions are contributing
312  if (clipping_high) {
313  if (m_ConfigTablePtr->RotorConfig[index].RollScale > 0.0f) {
314  // A positive change in roll will increase saturation
315  m_SaturationStatus.flags.roll_pos = true;
316 
317  } else if (m_ConfigTablePtr->RotorConfig[index].RollScale < 0.0f) {
318  // A negative change in roll will increase saturation
319  m_SaturationStatus.flags.roll_neg = true;
320 
321  }
322 
323  // check if the pitch input is saturating
324  if (m_ConfigTablePtr->RotorConfig[index].PitchScale > 0.0f) {
325  // A positive change in pitch will increase saturation
326  m_SaturationStatus.flags.pitch_pos = true;
327 
328  } else if (m_ConfigTablePtr->RotorConfig[index].PitchScale < 0.0f) {
329  // A negative change in pitch will increase saturation
330  m_SaturationStatus.flags.pitch_neg = true;
331 
332  }
333 
334  // check if the yaw input is saturating
335  if (m_ConfigTablePtr->RotorConfig[index].YawScale > 0.0f) {
336  // A positive change in yaw will increase saturation
337  m_SaturationStatus.flags.yaw_pos = true;
338 
339  } else if (m_ConfigTablePtr->RotorConfig[index].YawScale < 0.0f) {
340  // A negative change in yaw will increase saturation
341  m_SaturationStatus.flags.yaw_neg = true;
342 
343  }
344 
345  // A positive change in thrust will increase saturation
346  m_SaturationStatus.flags.thrust_pos = true;
347 
348  }
349 
350  // The motor is saturated at the lower limit
351  // check which control axes and which directions are contributing
352  if (clipping_low) {
353  // check if the roll input is saturating
354  if (m_ConfigTablePtr->RotorConfig[index].RollScale > 0.0f) {
355  // A negative change in roll will increase saturation
356  m_SaturationStatus.flags.roll_neg = true;
357 
358  } else if (m_ConfigTablePtr->RotorConfig[index].RollScale < 0.0f) {
359  // A positive change in roll will increase saturation
360  m_SaturationStatus.flags.roll_pos = true;
361 
362  }
363

366  // A negative change in pitch will increase saturation
367  m_SaturationStatus.flags.pitch_neg = true;
368 
369  } else if (m_ConfigTablePtr->RotorConfig[index].PitchScale < 0.0f) {
370  // A positive change in pitch will increase saturation
371  m_SaturationStatus.flags.pitch_pos = true;
372 
373  }
374

376  if (m_ConfigTablePtr->RotorConfig[index].YawScale > 0.0f) {
377  // A negative change in yaw will increase saturation
378  m_SaturationStatus.flags.yaw_neg = true;
379 
380  } else if (m_ConfigTablePtr->RotorConfig[index].YawScale < 0.0f) {
381  // A positive change in yaw will increase saturation
382  m_SaturationStatus.flags.yaw_pos = true;
383 
384  }
385 
386  // A negative change in thrust will increase saturation
