67  const unsigned num_channels, const uint16 reverse_mask,

69  const uint16 *max_pwm, const float *output, uint16 *effective_pwm,
70  PwmLimit_Data_t *limit)
71 {
72  /* first evaluate state changes */
73  switch (limit->state)
74  {
75  case PWM_LIMIT_STATE_INIT:
76  {
77  if (armed)
78  {
79  uint32 sec = 0;
80  uint32 ms = 0;
81  uint64 now = 0;
82  uint64 delta = 0;
83 
84  /* set arming time for the first call */
85  if (limit->time_armed == 0)
86  {
87  /* reset arming time, used for ramp timing */
88  limit->time_armed = PX4LIB_GetPX4TimeUs();
89  }
90 
91  /* reset arming time, used for ramp timing */
92  now = PX4LIB_GetPX4TimeUs();
93 
94  delta = now - limit->time_armed;
95 
96  if (delta >= INIT_TIME_US)
97  {
98  limit->state = PWM_LIMIT_STATE_OFF;
99  }
100  }
101 
102  break;
103  }
104 
105  case PWM_LIMIT_STATE_OFF:
106  {
107  if (armed)
108  {
109  /* reset arming time, used for ramp timing */
110  limit->time_armed = PX4LIB_GetPX4TimeUs();
111 
112  limit->state = PWM_LIMIT_STATE_RAMP;
113  }
114 
115  break;
116  }
117 
118  case PWM_LIMIT_STATE_RAMP:
119  {
120  if (!armed)
121  {
122  limit->state = PWM_LIMIT_STATE_OFF;
123  }
124  else if (hrt_elapsed_time(&limit->time_armed) >= RAMP_TIME_US)
125  {
126  limit->state = PWM_LIMIT_STATE_ON;
127  }
128 
129  break;
130  }
131 
132  case PWM_LIMIT_STATE_ON:
133  {
134  if (!armed)
135  {
136  limit->state = PWM_LIMIT_STATE_OFF;
137  }
138 
139  break;
140  }
141 
142  default:
143  {
144  break;
145  }
146  }
147 
148  /* if the system is pre-armed, the limit state is temporarily on,
149  * as some outputs are valid and the non-valid outputs have been
150  * set to NaN. This is not stored in the state machine though,
151  * as the throttle channels need to go through the ramp at
152  * regular arming time.
153  */
154 
155  unsigned local_limit_state = limit->state;
156 
157  if (pre_armed)
158  {
159  local_limit_state = PWM_LIMIT_STATE_ON;
160  }
161 
162  unsigned progress;
163 
164  /* then set effective_pwm based on state */
165  switch (local_limit_state)
166  {
167  case PWM_LIMIT_STATE_OFF:
168  /* Fallthru. */
169  case PWM_LIMIT_STATE_INIT:
170  {
171  uint32 i = 0;
172  for (i = 0; i < num_channels; ++i)
173  {
174  effective_pwm[i] = disarmed_pwm[i];
175  }
176 
177  break;
178  }
179 
180  case PWM_LIMIT_STATE_RAMP:
181  {
182  uint32 i = 0;
183  hrt_abstime diff = hrt_elapsed_time(&limit->time_armed);
184 
185  progress = diff * PROGRESS_INT_SCALING / RAMP_TIME_US;
186 
187  if (progress > PROGRESS_INT_SCALING)
188  {
189  progress = PROGRESS_INT_SCALING;
190  }
191 
192  for (i = 0; i < num_channels; ++i)
193  {
194  float control_value = output[i];
195  uint16 ramp_min_pwm;
196 
197  /* check for invalid / disabled channels */
198  if (!isfinite(control_value))
199  {
200  effective_pwm[i] = disarmed_pwm[i];
201  continue;
202  }
203 
204  /* if a disarmed pwm value was set, blend between disarmed and min */
205  if (disarmed_pwm[i] > 0)
206  {
207  uint32 disarmed_min_diff;
208  /* safeguard against overflows */
209  uint32 disarmed = disarmed_pwm[i];
210 
211  if (disarmed > min_pwm[i])
212  {
213  disarmed = min_pwm[i];
214  }
215 
216  disarmed_min_diff = min_pwm[i] - disarmed;
217  ramp_min_pwm = disarmed
218  + (disarmed_min_diff * progress) / PROGRESS_INT_SCALING;
219 
220  }
221  else
222  {
223  /* no disarmed pwm value set, choose min pwm */
224  ramp_min_pwm = min_pwm[i];
225  }
226 
227  if (reverse_mask & (1 << i))
228  {
229  control_value = -1.0f * control_value;
230  }
231 
232  effective_pwm[i] = control_value * (max_pwm[i] - ramp_min_pwm) / 2
233  + (max_pwm[i] + ramp_min_pwm) / 2;
234 
235  /* last line of defense against invalid inputs */
236  if (effective_pwm[i] < ramp_min_pwm)
237  {
238  effective_pwm[i] = ramp_min_pwm;
239  }
240  else if (effective_pwm[i] > max_pwm[i])
241  {
242  effective_pwm[i] = max_pwm[i];
243  }
244  }
245  break;
246  }
247 
248  case PWM_LIMIT_STATE_ON:
249  {
250  uint32 i = 0;
251  for (i = 0; i < num_channels; ++i)
252  {
253  float control_value = output[i];
254 
255  /* check for invalid / disabled channels */
256  if (!isfinite(control_value))
257  {
258  effective_pwm[i] = disarmed_pwm[i];
259  continue;
260  }
261 
262  if (reverse_mask & (1 << i))
263  {
264  control_value = -1.0f * control_value;
265  }
266 
267  effective_pwm[i] = control_value * (max_pwm[i] - min_pwm[i]) / 2
268  + (max_pwm[i] + min_pwm[i]) / 2;
269 
270  /* last line of defense against invalid inputs */
271  if (effective_pwm[i] < min_pwm[i])
272  {
273  effective_pwm[i] = min_pwm[i];
274  }
275  else if (effective_pwm[i] > max_pwm[i])
276  {
277  effective_pwm[i] = max_pwm[i];
278  }
279  }
280 
281  break;
282  }
283 
284  default:
285  {
286  break;
287  }
288  }
289 }
