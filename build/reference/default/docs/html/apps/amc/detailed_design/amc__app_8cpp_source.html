448  /* Initialize all internal data. */

529 /* Receive and Process Messages */
530 /* */
531 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
532 int32 AMC::RcvSchPipeMsg(int32 iBlocking)
533 {
534  int32 iStatus=CFE_SUCCESS;
535  CFE_SB_Msg_t* MsgPtr=NULL;
536  CFE_SB_MsgId_t MsgId;
537 
538  /* Stop Performance Log entry */
539  CFE_ES_PerfLogExit(AMC_MAIN_TASK_PERF_ID);
540 
541  /* Wait for WakeUp messages from scheduler */
542  iStatus = CFE_SB_RcvMsg(&MsgPtr, SchPipeId, iBlocking);
543 
544  /* Start Performance Log entry */
545  CFE_ES_PerfLogEntry(AMC_MAIN_TASK_PERF_ID);
546 
547  if (iStatus == CFE_SUCCESS)
548  {
549  MsgId = CFE_SB_GetMsgId(MsgPtr);
550  switch (MsgId)
551  {
552  case AMC_UPDATE_MOTORS_MID:
553  {
554  ProcessDataPipe();
555  break;
556  }
557 
558  case AMC_SEND_HK_MID:
559  {
560  ReportHousekeeping();
561  // TODO: Move these somewhere more appropriate later
562  ProcessParamPipe();
563  ProcessCmdPipe();
564  break;
565  }
566 
567  default:
568  {
569  CFE_EVS_SendEvent(AMC_MSGID_ERR_EID, CFE_EVS_ERROR,
570  "Recvd invalid SCH msgId (0x%04X)",
571  (unsigned short)MsgId);
572  }
573  }
574  }
575  else if (iStatus == CFE_SB_TIME_OUT)
576  {
577  /* TODO: If there's no incoming message within a specified time (via the
578  * iBlocking arg, you can do something here, or nothing.
579  * Note, this section is dead code only if the iBlocking arg
580  * is CFE_SB_PEND_FOREVER. */
581  UpdateMotors();
582  iStatus = CFE_SUCCESS;
583  }
584  else
585  {
586  CFE_EVS_SendEvent(AMC_RCVMSG_ERR_EID, CFE_EVS_ERROR,
587  "SCH pipe read error (0x%08X).", (unsigned int)iStatus);
588  }
589 
590  return iStatus;
591 }

594 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
595 /* */
596 /* Process Incoming Commands */
597 /* */
598 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
599 void AMC::ProcessCmdPipe(void)
600 {
601  int32 iStatus = CFE_SUCCESS;
602  CFE_SB_Msg_t* CmdMsgPtr=NULL;
603  CFE_SB_MsgId_t CmdMsgId;
604  bool contProcessing = true;
605 
606  /* Process command messages until the pipe is empty */
607  while (contProcessing)
608  {
609  iStatus = CFE_SB_RcvMsg(&CmdMsgPtr, CmdPipeId, CFE_SB_POLL);
610  if(iStatus == CFE_SUCCESS)
611  {
612  /* We did receive a message. Process it. */
613  CmdMsgId = CFE_SB_GetMsgId(CmdMsgPtr);
614  switch (CmdMsgId)
615  {
616  case AMC_CMD_MID:
617  {
618  /* We did receive a command. Process it. */
619  ProcessAppCmds(CmdMsgPtr);
620  break;
621  }
622 
623  default:
624  {
625  /* Bump the command error counter for an unknown command.
626  * (This should only occur if it was subscribed to with
627  * this pipe, but not handled in this switch-case.) */
628  HkTlm.usCmdErrCnt++;
629  CFE_EVS_SendEvent(AMC_MSGID_ERR_EID, CFE_EVS_ERROR,
630  "Recvd invalid CMD msgId (0x%04X)",
631  (unsigned short)CmdMsgId);
632  break;
633  }
634  }
635  }
636  else if (iStatus == CFE_SB_NO_MESSAGE)
637  {
638  /* The command pipe is empty. Break the function and continue
639  * on. */
640  contProcessing = false;
641  }
642  else
643  {
