49 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
50 int32 MAC::InitConfigTbl()
51 {
52  int32 iStatus = 0;
53 
54  /* Register Param table */
55  iStatus = CFE_TBL_Register(&ParamTblHdl,
56  MAC_PARAM_TABLENAME,
57  (sizeof(MAC_ParamTbl_t)),
58  CFE_TBL_OPT_DEFAULT,
59  MAC::ValidateParamTbl);
60  if (iStatus != CFE_SUCCESS)
61  {
62  /* Note, a critical table could return another nominal code. If this table is
63  * made critical this logic would have to change. */
64  (void) CFE_EVS_SendEvent(MAC_INIT_ERR_EID, CFE_EVS_ERROR,
65  "Failed to register param table (0x%08X)",
66  (unsigned int)iStatus);
67  goto MAC_InitConfigTbl_Exit_Tag;
68  }
69 
70  /* Load Config table file */
71  iStatus = CFE_TBL_Load(ParamTblHdl,
72  CFE_TBL_SRC_FILE,
73  MAC_PARAM_TABLE_FILENAME);
74  if (iStatus != CFE_SUCCESS)
75  {
76  /* Note, CFE_SUCCESS is for a successful full table load. If a partial table
77  load is desired then this logic would have to change. */
78  (void) CFE_EVS_SendEvent(MAC_INIT_ERR_EID, CFE_EVS_ERROR,
79  "Failed to load Config Table (0x%08X)",
80  (unsigned int)iStatus);
81  goto MAC_InitConfigTbl_Exit_Tag;
82  }
83 
84  iStatus = AcquireConfigPointers();
85 
86 MAC_InitConfigTbl_Exit_Tag:
87  return (iStatus);
88 }

428 /* Acquire Config Pointers */
429 /* */
430 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
431 int32 MAC::AcquireConfigPointers(void)
432 {
433  int32 iStatus = CFE_SUCCESS;
434 
435  /*
436  ** Release the table
437  */
438  (void) CFE_TBL_ReleaseAddress(ParamTblHdl);
439 
440  /*
441  ** Manage the table
442  */
443  iStatus = CFE_TBL_Manage(ParamTblHdl);
444  if ((iStatus != CFE_SUCCESS) && (iStatus != CFE_TBL_INFO_UPDATED))
445  {
446  (void) CFE_EVS_SendEvent(MAC_CONFIG_TABLE_ERR_EID, CFE_EVS_ERROR,
447  "Failed to manage PWM Config table (0x%08X)",
448  (unsigned int)iStatus);
449  goto MAC_AcquireConfigPointers_Exit_Tag;
450  }
451 
452  /*
453  ** Get a pointer to the table
454  */
455  iStatus = CFE_TBL_GetAddress((void**)&ParamTblPtr,
456  ParamTblHdl);
457  if (iStatus == CFE_TBL_INFO_UPDATED)
458  {
459  UpdateParams();
460  iStatus = CFE_SUCCESS;
461  }
462  else if(iStatus != CFE_SUCCESS)
463  {
464  ParamTblPtr = 0;
465  (void) CFE_EVS_SendEvent(MAC_CONFIG_TABLE_ERR_EID, CFE_EVS_ERROR,
466  "Failed to get Param table's address (0x%08X)",
467  (unsigned int)iStatus);
468  }
469 
470 MAC_AcquireConfigPointers_Exit_Tag:
471  return (iStatus);
472 
473 }
