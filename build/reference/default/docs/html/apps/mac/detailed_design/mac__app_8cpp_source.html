87 {
88  int32 iStatus = CFE_SUCCESS;
89  uint32 ind = 0;
90 
91  /* Initialize the event filter table.
92  * Note: 0 is the CFE_EVS_NO_FILTER mask and event 0 is reserved (not used) */
93  CFE_PSP_MemSet((void*)EventTbl, 0x00, sizeof(EventTbl));
94 
95  EventTbl[ ind].EventID = MAC_RESERVED_EID;
96  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
97 
98  EventTbl[ ind].EventID = MAC_INF_EID;
99  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
100 
101  EventTbl[ ind].EventID = MAC_CONFIG_TABLE_ERR_EID;
102  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
103 
104  EventTbl[ ind].EventID = MAC_PIPE_ERR_EID;
105  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
106 
107  EventTbl[ ind].EventID = MAC_MSGID_ERR_EID;
108  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
109 
110  EventTbl[ ind].EventID = MAC_MSGLEN_ERR_EID;
111  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
112 
113  EventTbl[ ind].EventID = MAC_CMD_INF_EID;
114  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;
115 
116  EventTbl[ ind].EventID = MAC_INIT_INF_EID;
117  EventTbl[ind++].Mask = CFE_EVS_NO_FILTER;

452

455

457  {

467

469  {

474  break;
475  }
476 
477  default:
478  {
479  (void) CFE_EVS_SendEvent(MAC_MSGID_ERR_EID, CFE_EVS_ERROR,

481  break;
482  }
483  }
484  }
485  else if (iStatus == CFE_SB_NO_MESSAGE)
486  {
487  iStatus = CFE_SUCCESS;
488  }
489  else if (iStatus == CFE_SB_TIME_OUT)
490  {
491  iStatus = CFE_SUCCESS;
492  }

495  (void) CFE_EVS_SendEvent(MAC_PIPE_ERR_EID, CFE_EVS_ERROR,
496  "SB pipe read error (0x%08X), app will exit", (unsigned int)iStatus);
497  uiRunStatus= CFE_ES_APP_ERROR;
498  }
499

942 void MAC::ControlAttitude(float dt)
943 {
944  int32 i = 0;
945  m_ThrustSp = CVT.VAttSp.Thrust;
946 
947  /* Construct attitude setpoint rotation matrix */
948  math::Quaternion q_sp(
949  CVT.VAttSp.Q_D[0],
950  CVT.VAttSp.Q_D[1],
951  CVT.VAttSp.Q_D[2],
952  CVT.VAttSp.Q_D[3]);
953 
954  math::Matrix3F3 R_sp = q_sp.RotationMatrix();
955 
956  /* Get current rotation matrix from control state quaternions */
957  math::Quaternion q_att(CVT.ControlState.Q[0], CVT.ControlState.Q[1], CVT.ControlState.Q[2], CVT.ControlState.Q[3]);
958  math::Matrix3F3 R = q_att.RotationMatrix();
959 
960  /* All input data is ready, run controller itself */
961 
962  /* Try to move thrust vector shortest way, because yaw response is slower than roll/pitch */
963  math::Vector3F R_z(R[0][2], R[1][2], R[2][2]);
964  math::Vector3F R_sp_z(R_sp[0][2], R_sp[1][2], R_sp[2][2]);
965 
966  /* Axis and sin(angle) of desired rotation */
967  math::Vector3F e_R = R.Transpose() * (R_z % R_sp_z);
968 
969  /* Calculate angle error */
970  float e_R_z_sin = e_R.Length();
971  float e_R_z_cos = R_z * R_sp_z;
972 
973  /* Calculate weight for yaw control */
974  float yaw_w = R_sp[2][2] * R_sp[2][2];
975 
976  /* Calculate rotation matrix after roll/pitch only rotation */
977  math::Matrix3F3 R_rp;
978 
979  if (e_R_z_sin > 0.0f)
980  {
981  /* Get axis-angle representation */
982  float e_R_z_angle = atan2f(e_R_z_sin, e_R_z_cos);
983  math::Vector3F e_R_z_axis = e_R / e_R_z_sin;
984 
985  e_R = e_R_z_axis * e_R_z_angle;
986 
987  /* Cross product matrix for e_R_axis */
988  math::Matrix3F3 e_R_cp;
989  e_R_cp.Zero();
990  e_R_cp[0][1] = -e_R_z_axis[2];
991  e_R_cp[0][2] = e_R_z_axis[1];
992  e_R_cp[1][0] = e_R_z_axis[2];
993  e_R_cp[1][2] = -e_R_z_axis[0];
994  e_R_cp[2][0] = -e_R_z_axis[1];
995  e_R_cp[2][1] = e_R_z_axis[0];
996 
997  /* Rotation matrix for roll/pitch only rotation */
998  R_rp = R * (math::Matrix3F3::Identity() + e_R_cp * e_R_z_sin + e_R_cp * e_R_cp * (1.0f - e_R_z_cos));
999 
1000  }
1001  else
1002  {
1003  /* Zero roll/pitch rotation */
1004  R_rp = R;
1005  }
1006 
1007  /* R_rp and R_sp has the same Z axis, calculate yaw error */
1008  math::Vector3F R_sp_x(R_sp[0][0], R_sp[1][0], R_sp[2][0]);
1009  math::Vector3F R_rp_x(R_rp[0][0], R_rp[1][0], R_rp[2][0]);
1010  e_R[2] = atan2f((R_rp_x % R_sp_x) * R_sp_z, R_rp_x * R_sp_x) * yaw_w;
1011 
1012  if (e_R_z_cos < 0.0f)
1013  {
1014  /* For large thrust vector rotations use another rotation method:
1015  * calculate angle and axis for R -> R_sp rotation directly */
1016  math::Quaternion q_error(R.Transpose() * R_sp);
1017  math::Vector3F e_R_d = q_error[0] >= 0.0f ? q_error.Imaginary() * 2.0f : -q_error.Imaginary() * 2.0f;
1018 
1019  /* Use fusion of Z axis based rotation and direct rotation */
1020  float direct_w = e_R_z_cos * e_R_z_cos * yaw_w;
1021  e_R = e_R * (1.0f - direct_w) + e_R_d * direct_w;
1022  }
1023 
1024  /* Calculate angular rates setpoint */
1025  m_AngularRatesSetpoint = m_Params.att_p.EMult(e_R);
1026 
1027  /* Limit rates */
1028  for (i = 0; i < 3; ++i)
1029  {
1030  if ((CVT.VControlMode.ControlVelocityEnabled || CVT.VControlMode.ControlAutoEnabled) &&
1031  !CVT.VControlMode.ControlManualEnabled)
1032  {
1033  m_AngularRatesSetpoint.Constrain(i, -m_Params.auto_rate_max[i], m_Params.auto_rate_max[i]);
1034  }
1035  else
1036  {
1037  m_AngularRatesSetpoint.Constrain(i, -m_Params.mc_rate_max[i], m_Params.mc_rate_max[i]);
1038  }
1039  }
1040 
1041  /* Feed forward yaw setpoint rate */
1042  m_AngularRatesSetpoint[2] += CVT.VAttSp.YawSpMoveRate * yaw_w;
1043 }

1045 void MAC::ControlAttitudeRates(float dt)
1046 {
1047  /* Rotation matrix for the orientation that the board is mounted */
1048  math::Matrix3F3 boardRotation;
1049  uint32 i = 0;
1050 
1051  /* Reset integral if disarmed */
1052  if(!CVT.Armed.Armed || !CVT.VehicleStatus.IsRotaryWing)
1053  {
1054  m_AngularRatesIntegralError.Zero();
1055  }
1056 
1057  /* Get transformation matrix from sensor/board to body frame */
1058  boardRotation = boardRotation.RotationMatrix((math::Matrix3F3::Rotation_t)ParamTblPtr->SENS_BOARD_ROT);
1059 
1060  /* Fine tune the rotation */
1061  math::Matrix3F3 boardRotationOffset;
1062  boardRotationOffset = math::Matrix3F3::FromEuler(M_DEG_TO_RAD_F * ParamTblPtr->SENS_BOARD_X_OFF,
1063  M_DEG_TO_RAD_F * ParamTblPtr->SENS_BOARD_Y_OFF,
1064  M_DEG_TO_RAD_F * ParamTblPtr->SENS_BOARD_Z_OFF);
1065  boardRotation = boardRotationOffset * boardRotation;
1066 
1067  /* Get the raw gyro data and correct for thermal errors */
1068  math::Vector3F rates;
1069 
1070  if (m_SelectedGyro == 0)
1071  {
1072  rates[0] = (CVT.SensorGyro.X - CVT.SensorCorrection.gyro_offset_0[0]) * CVT.SensorCorrection.gyro_scale_0[0];
1073  rates[1] = (CVT.SensorGyro.Y - CVT.SensorCorrection.gyro_offset_0[1]) * CVT.SensorCorrection.gyro_scale_0[1];
1074  rates[2] = (CVT.SensorGyro.Z - CVT.SensorCorrection.gyro_offset_0[2]) * CVT.SensorCorrection.gyro_scale_0[2];
1075  }
1076  else if (m_SelectedGyro == 1)
1077  {
1078  rates[0] = (CVT.SensorGyro.X - CVT.SensorCorrection.gyro_offset_1[0]) * CVT.SensorCorrection.gyro_scale_1[0];
1079  rates[1] = (CVT.SensorGyro.Y - CVT.SensorCorrection.gyro_offset_1[1]) * CVT.SensorCorrection.gyro_scale_1[1];
1080  rates[2] = (CVT.SensorGyro.Z - CVT.SensorCorrection.gyro_offset_1[2]) * CVT.SensorCorrection.gyro_scale_1[2];
1081  }
1082  else if (m_SelectedGyro == 2)
1083  {
1084  rates[0] = (CVT.SensorGyro.X - CVT.SensorCorrection.gyro_offset_2[0]) * CVT.SensorCorrection.gyro_scale_2[0];
1085  rates[1] = (CVT.SensorGyro.Y - CVT.SensorCorrection.gyro_offset_2[1]) * CVT.SensorCorrection.gyro_scale_2[1];
1086  rates[2] = (CVT.SensorGyro.Z - CVT.SensorCorrection.gyro_offset_2[2]) * CVT.SensorCorrection.gyro_scale_2[2];
1087  }
1088  else
1089  {
1090  rates[0] = CVT.SensorGyro.X;
