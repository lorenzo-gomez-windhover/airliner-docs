53 int32 ParamsConsumer::InitParams(PRMLIB_ParamRegistrations_t &Params)
54 {
55  int32 iStatus;
56 
57  /* Init param pipe and subscribe to param messages */
58  iStatus = CFE_SB_CreatePipe(&ParamPipeId,
59  Params.PipeDepth,
60  Params.PipeName);
61  if (iStatus != CFE_SUCCESS)
62  {
63  /* We failed to create the pipe for param messages. Immediately jump
64  * to the end of the function to abort initialization and return
65  * the error code.
66  */
67  goto InitParams_Exit_Tag;
68  }
69 
70  /* Subscribe to the PRMLIB_PARAM_UPDATED_MID message to receive
71  * parameter requests. */
72  iStatus = CFE_SB_SubscribeEx(PRMLIB_PARAM_UPDATED_MID, ParamPipeId,
73  CFE_SB_Default_Qos, 1);
74  if (iStatus != CFE_SUCCESS)
75  {
76  /* We failed to create the pipe for param messages. Immediately jump
77  * to the end of the function to abort initialization and return the
78  * error code.
79  */
80  goto InitParams_Exit_Tag;
81  }
82 
83  /* Loop through all parameters just passed to the function. */
84  for(uint32 i = 0; i < PRMLIB_MAX_REGISTRATIONS; ++i)
85  {
86  /* Check to see if the name is blank. If the name is blank, we will
87  * assume we've reached the end of the list and just terminate the
88  * loop.
89  */
90  if(Params.Registrations[i].Name[0] == 0)
91  {
92  /* The name is blank. Terminate the loop. */
93  break;
94  }
95 
96  iStatus = PRMLIB_ParamRegister(Params.Registrations[i].Name,
97  Params.Registrations[i].Value, Params.Registrations[i].Type);
98  if(iStatus != CFE_SUCCESS)
99  {
100  break;
101  }
102 
103  CFE_PSP_MemCpy((void*)ParamRegistration[i].Name, (void*)Params.Registrations[i].Name, sizeof(ParamRegistration[i].Name));
104  CFE_PSP_MemCpy(&ParamRegistration[i].Type, &Params.Registrations[i].Type, sizeof(ParamRegistration[i].Type));
105  CFE_PSP_MemCpy(&ParamRegistration[i].Value, &Params.Registrations[i].Value, sizeof(ParamRegistration[i].Value));
106  }
107 
108 InitParams_Exit_Tag:
109 
110  return iStatus;
111 }

219 int32 ParamsConsumer::CheckParams(void)
220 {
221  int32 iStatus = CFE_SUCCESS;
222  CFE_SB_Msg_t* MsgPtr = 0;
223  CFE_SB_MsgId_t MsgId = 0;
224  osalbool contProcessing = TRUE;
225  PRMLIB_ParamRegistration_t paramDataOut[PRMLIB_MAX_REGISTRATIONS];
226  uint32 paramsOut = 0;
227 
228  /* Process param messages until the pipe is empty */
229  while (contProcessing)
230  {
231  iStatus = CFE_SB_RcvMsg(&MsgPtr, ParamPipeId, CFE_SB_POLL);
232  if(iStatus == CFE_SUCCESS)
233  {
234  /* We did receive a message. Process it. */
235  MsgId = CFE_SB_GetMsgId(MsgPtr);
236  switch (MsgId)
237  {
238  case PRMLIB_PARAM_UPDATED_MID:
239  {
240  /* We did receive a parameter request. Process it. */
241  PRMLIB_UpdatedParamMsg_t *paramMsg = (PRMLIB_UpdatedParamMsg_t*)MsgPtr;
242 
243  PRMLIB_ParamRegistration_t *paramRegistration = GetParamRegistration(paramMsg->name);
244  if(paramRegistration != 0)
245  {
246  osalbool isValid;
247  uint8 newValue[8];
248  int8 val_int8 = 0;
249  uint16 val_uint16 = 0;
250  int16 val_int16 = 0;
251  uint32 val_uint32 = 0;

254

256  if(iStatus != CFE_SUCCESS)
257  {
258  contProcessing = FALSE;
259  }
260 
261  switch(paramRegistration->Type)
262  {
263  case TYPE_UINT8:
264  {
265  uint8 typedValue = 0;
266  CFE_PSP_MemCpy(&typedValue, newValue, sizeof(typedValue));
267  isValid = onParamValidate(paramMsg->name, typedValue);
268  if(isValid == TRUE)
269  {
270  isValid = onParamValidate(paramRegistration->Value, typedValue);
271  if(isValid == TRUE)
272  {
273  CFE_PSP_MemCpy(paramRegistration->Value, &typedValue, sizeof(typedValue));
274  }
275  }
276  break;
277  }

279  case TYPE_INT8:

363  isValid = onParamValidate(paramMsg->name, typedValue);

370  }

372  break;
373  }
374 
375  default:
376  {
377  isValid = FALSE;
378  break;
379  }
380  }
381 
382  if(isValid == TRUE)
383  {
384  CFE_PSP_MemCpy(&paramDataOut[paramsOut], paramRegistration, sizeof(paramDataOut[paramsOut]));
385  paramsOut++;
386  }
387  }
388  break;

390 
391  default:
392  {
393  contProcessing = FALSE;
394  iStatus = -1;
395  break;
396  }
397  }
398  }
399  else if (iStatus == CFE_SB_NO_MESSAGE)
400  {
401  /* The pipe is empty. Break the loop and continue on. */
402  contProcessing = FALSE;
403  }
404  else
405  {
406  /* Something failed. Quit the loop. */
407  iStatus = -2;
408  contProcessing = FALSE;
409  }
410  }
411 
412  if(paramsOut > 0)
413  {
414  onParamsChange(paramDataOut, paramsOut);
415  }
416 
417  return iStatus;
418 }
