117 int32 PRMLIB_AddParam(PRMLIB_ParamData_t param)
118 {
119  int32 Status = -1;
120  int i = 0;
121 
122  /* Lock the mutex */
123  OS_MutSemTake(PRMLIB_AppData.ParamTblMutex);
124 
125  /* Iterate over table to find first empty index */
126  for(i = 0; i < PRMLIB_PARAM_TBL_MAX_ENTRY; ++i)
127  {
128  if (PRMLIB_AppData.ParamTbl[i].enabled == 0)
129  {
130  /* Update parameter message with current table index values */
131  PRMLIB_AppData.ParamTbl[i].enabled = 1;
132  memcpy(PRMLIB_AppData.ParamTbl[i].param_data.name, param.name,
133  sizeof(param.name)); //need to clear string?
134  PRMLIB_AppData.ParamTbl[i].param_data.type = param.type;
135  PRMLIB_SetParamValue(&PRMLIB_AppData.ParamTbl[i].param_data, &param.value);
136  Status = CFE_SUCCESS;
137  break;
138  }
139  }
140 
141  /* Increment total parameter count */
142  PRMLIB_AppData.ParamCount++;
143 
144  /* Unlock the mutex */
145  OS_MutSemGive(PRMLIB_AppData.ParamTblMutex);
146 
147  return Status;
148 }

159  int i = 0;
160

163

175  *ParamIndex = idx;

181  }
182  }

185

583 void PRMLIB_GetParamValue(PRMLIB_ParamData_t param, void* val)
584 {
585  uint8 val_uint8 = 0;
586  int8 val_int8 = 0;
587  uint16 val_uint16 = 0;
588  int16 val_int16 = 0;
589  uint32 val_uint32 = 0;
590  int32 val_int32 = 0;
591  float val_float = 0.0;
592 
593  switch(param.type)
594  {
595  case TYPE_UINT8:
596  memcpy(&val_uint8, param.value, sizeof(uint8));
597  memcpy(val, &val_uint8, sizeof(uint8));
598  break;
599  case TYPE_INT8:
600  memcpy(&val_int8, param.value, sizeof(int8));
601  memcpy(val, &val_int8, sizeof(int8));
602  break;
603  case TYPE_UINT16:
604  memcpy(&val_uint16, param.value, sizeof(uint16));
605  memcpy(val, &val_uint16, sizeof(uint16));
606  break;
607  case TYPE_INT16:
608  memcpy(&val_int16, param.value, sizeof(int16));
609  memcpy(val, &val_int16, sizeof(int16));
610  break;
611  case TYPE_UINT32:
612  memcpy(&val_uint32, param.value, sizeof(uint32));
613  memcpy(val, &val_uint32, sizeof(uint32));
614  break;
615  case TYPE_INT32:
616  memcpy(&val_int32, param.value, sizeof(int32));
617  memcpy(val, &val_int32, sizeof(int32));
618  break;
619  case TYPE_REAL32:
620  memcpy(&val_float, param.value, sizeof(float));
621  memcpy(val, &val_float, sizeof(float));
