237 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
238 int32 LGC::RcvSchPipeMsg(int32 iBlocking)
239 {
240  int32 iStatus = CFE_SUCCESS;
241  CFE_SB_Msg_t* MsgPtr = NULL;
242  CFE_SB_MsgId_t MsgId = 0;
243 
244  /* Stop Performance Log entry */
245  CFE_ES_PerfLogExit(LGC_MAIN_TASK_PERF_ID);
246 
247  /* Wait for WakeUp messages from scheduler */
248  iStatus = CFE_SB_RcvMsg(&MsgPtr, SchPipeId, iBlocking);
249 
250  /* Start Performance Log entry */
251  CFE_ES_PerfLogEntry(LGC_MAIN_TASK_PERF_ID);
252 
253  if (iStatus == CFE_SUCCESS)
254  {
255  MsgId = CFE_SB_GetMsgId(MsgPtr);
256  switch (MsgId)
257  {
258  case LGC_WAKEUP_MID:
259  {
260  CheckSwitchPosition();
261  break;
262  }
263  case LGC_SEND_HK_MID:
264  {
265  ProcessCmdPipe();
266  ReportHousekeeping();
267  break;
268  }
269  case PX4_MANUAL_CONTROL_SETPOINT_MID:
270  {
271  memcpy(&CVT.m_ManualControlSetpointMsg, MsgPtr, sizeof(CVT.m_ManualControlSetpointMsg));
272  break;
273  }
274  case PX4_VEHICLE_STATUS_MID:
275  {
276  memcpy(&CVT.m_VehicleStatusMsg, MsgPtr, sizeof(CVT.m_VehicleStatusMsg));
277  break;
278  }
279  default:
280  {
281  (void) CFE_EVS_SendEvent(LGC_MSGID_ERR_EID, CFE_EVS_ERROR,
282  "Recvd invalid SCH msgId (0x%04X)", MsgId);
283  }
284  }
285  }
286  else if (iStatus == CFE_SB_NO_MESSAGE)
287  {
288  /* If there's no incoming message, do nothing here.
289  * Note, this section is dead code only if the iBlocking arg
290  * is CFE_SB_PEND_FOREVER. */
291  iStatus = CFE_SUCCESS;
292  }
293  else if (iStatus == CFE_SB_TIME_OUT)
294  {
295  /* If there's no incoming message within a specified time (via the
296  * iBlocking arg, you can do nothing here.
297  * Note, this section is dead code only if the iBlocking arg
298  * is CFE_SB_PEND_FOREVER. */
299  iStatus = CFE_SUCCESS;
300  }
301  else
302  {
303  (void) CFE_EVS_SendEvent(LGC_RCVMSG_ERR_EID, CFE_EVS_ERROR,
304  "SCH pipe read error (0x%08lX).", iStatus);
305  }
306 
307  return (iStatus);
308 }

315 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
316 void LGC::ProcessCmdPipe()
317 {
318  int32 iStatus = CFE_SUCCESS;
319  CFE_SB_Msg_t* CmdMsgPtr = NULL;
320  CFE_SB_MsgId_t CmdMsgId = 0;
321 
322  /* Process command messages until the pipe is empty */
323  while (1)
324  {
325  iStatus = CFE_SB_RcvMsg(&CmdMsgPtr, CmdPipeId, CFE_SB_POLL);
326  if(iStatus == CFE_SUCCESS)
327  {
328  CmdMsgId = CFE_SB_GetMsgId(CmdMsgPtr);
329  switch (CmdMsgId)
330  {
331  case LGC_CMD_MID:
332  {
333  ProcessAppCmds(CmdMsgPtr);
334  break;
335  }
336  default:
337  {
338  /* Bump the command error counter for an unknown command.
339  * (This should only occur if it was subscribed to with this
340  * pipe, but not handled in this switch-case.) */
341  HkTlm.usCmdErrCnt++;
342  (void) CFE_EVS_SendEvent(LGC_MSGID_ERR_EID, CFE_EVS_ERROR,
343  "Recvd invalid CMD msgId (0x%04X)", (unsigned short)CmdMsgId);
344  break;
345  }
346  }
347  }
348  else if (iStatus == CFE_SB_NO_MESSAGE)
349  {
350  break;
351  }
352  else
353  {
354  (void) CFE_EVS_SendEvent(LGC_RCVMSG_ERR_EID, CFE_EVS_ERROR,
355  "CMD pipe read error (0x%08lX)", iStatus);
356  break;
357  }
358  }
359  return;
360 }

370 {
371  uint32 uiCmdCode = 0;
372 
373  if (MsgPtr != NULL)
374  {
375  uiCmdCode = CFE_SB_GetCmdCode(MsgPtr);
376  switch (uiCmdCode)

379  {
380  HkTlm.usCmdCnt++;
381  (void) CFE_EVS_SendEvent(LGC_CMD_NOOP_EID, CFE_EVS_INFORMATION,
382  "Recvd NOOP. Version %d.%d.%d.%d",
383  LGC_MAJOR_VERSION,
384  LGC_MINOR_VERSION,
385  LGC_REVISION,
386  LGC_MISSION_REV);
387  break;
388  }
389  case LGC_RESET_CC:
390  {
391  HkTlm.usCmdCnt = 0;
392  HkTlm.usCmdErrCnt = 0;
393  break;
394  }
395  default:
396  {
397  HkTlm.usCmdErrCnt++;
398  (void) CFE_EVS_SendEvent(LGC_CC_ERR_EID, CFE_EVS_ERROR,
399  "Recvd invalid command code (%u)", (unsigned int)uiCmdCode);
400  break;
401  }
402  }
403  }
404  return;
405 }

469 void LGC::AppMain()
470 {
471  /* Register the application with Executive Services */
472  uiRunStatus = CFE_ES_APP_RUN;
473 
474  int32 iStatus = CFE_ES_RegisterApp();
475  if (iStatus != CFE_SUCCESS)
476  {
477  (void) CFE_ES_WriteToSysLog("LGC - Failed to register the app (0x%08lX)\n", iStatus);
478  }
479 
480  /* Start Performance Log entry */
481  CFE_ES_PerfLogEntry(LGC_MAIN_TASK_PERF_ID);
482 
483  /* Perform application initializations */
484  if (iStatus == CFE_SUCCESS)
485  {
486  iStatus = InitApp();
487  }
488 
489  if (iStatus == CFE_SUCCESS)
490  {
491  /* Do not perform performance monitoring on startup sync */
492  CFE_ES_PerfLogExit(LGC_MAIN_TASK_PERF_ID);
493  CFE_ES_WaitForStartupSync(LGC_STARTUP_TIMEOUT_MSEC);
494  CFE_ES_PerfLogEntry(LGC_MAIN_TASK_PERF_ID);
495  }
496  else
497  {
498  uiRunStatus = CFE_ES_APP_ERROR;
499  }
500 
501  /* Application main loop */
502  while (CFE_ES_RunLoop(&uiRunStatus) == TRUE)
503  {
504  RcvSchPipeMsg(LGC_SCH_PIPE_PEND_TIME);
505 
506  iStatus = AcquireConfigPointers();

508  {

510  uiRunStatus = CFE_ES_APP_ERROR;
511  }
512  }
513 
514  /* Stop Performance Log entry */
515  CFE_ES_PerfLogExit(LGC_MAIN_TASK_PERF_ID);
516

518  CFE_ES_ExitApp(uiRunStatus);
