43 /* Run the Classifier algorithm */
44 /* */
45 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
46 void TO_Classifier_Run(TO_ChannelData_t *channel)
47 {
48  int32 status = CFE_SUCCESS;
49  CFE_SB_MsgPtr_t DataMsgPtr = NULL;
50  uint32 ii;
51  TO_PriorityQueue_t *pqueue = NULL;
52  TO_MessageFlow_t *msgFlow = NULL;
53  uint32 msgFlowIndex = 0;
54  uint32 pQueueIndex = 0;
55  CFE_SB_MsgId_t DataMsgID;
56  uint32 totalMsgLength = 0;
57 
58  /* Process telemetry messages till the pipe is empty, or until we hit the
59  * maximum number of messages we want to process in this frame. */
60  for (ii = 0; ii < TO_MAX_MSGS_OUT_PER_FRAME; ++ii)
61  {
62  status = CFE_SB_RcvMsg(&DataMsgPtr, channel->DataPipeId, CFE_SB_POLL);
63 
64  if (status != CFE_SUCCESS)
65  {
66  if (status != CFE_SB_NO_MESSAGE)
67  {
68  (void) CFE_EVS_SendEvent(TO_PIPE_READ_ERR_EID,
69  CFE_EVS_ERROR,
70  "Data pipe read error (0x%08X) on channel %d",
71  (unsigned int)status,
72  (unsigned int)channel->channelIdx);
73  }
74  return;
75  }
76 
77  if (DataMsgPtr != NULL)
78  {
79  DataMsgID = CFE_SB_GetMsgId(DataMsgPtr);
80 
81  totalMsgLength = CFE_SB_GetTotalMsgLength(DataMsgPtr);
82  if (totalMsgLength > TO_MAX_MSG_LENGTH)
83  {
84  (void) CFE_EVS_SendEvent(TO_TLM_MSG_LEN_ERR_EID, CFE_EVS_ERROR,
85  "Message too long (size = %lu > max = %d) for msgId = (0x%04X) on channel (%u)",
86  totalMsgLength,
87  TO_MAX_MSG_LENGTH,
88  (unsigned short)DataMsgID,
89  (unsigned short)channel->channelIdx);
90  (void) OS_MutSemTake(TO_AppData.MutexID);
91  TO_AppData.HkTlm.usTotalMsgDropped++;
92  (void) OS_MutSemGive(TO_AppData.MutexID);
93 
94  channel->DropMsgCount++;
95  continue;
96  }
97 
98  /* Get the first Message Flow object. If this returns null, the
99  * Message ID is not in the table at all so we shouldn't have
100  * received this message. Raise an event.
101  */
102  msgFlow = TO_MessageFlow_GetObject(channel, DataMsgID, &msgFlowIndex);
103  if (NULL == msgFlow)
104  {
105  (void) CFE_EVS_SendEvent(TO_MF_MSG_ID_ERR_EID,
106  CFE_EVS_ERROR,
107  "Classifier Recvd invalid msgId (0x%04X) or message flow was removed on channel (%u)",
108  (unsigned short)DataMsgID,
109  (unsigned short)channel->channelIdx);
110 
111  (void) OS_MutSemTake(TO_AppData.MutexID);
112  TO_AppData.HkTlm.usTotalMsgDropped++;
113  (void) OS_MutSemGive(TO_AppData.MutexID);
114 
115  channel->DropMsgCount++;
116  continue;
117  }
118 
119  /* Get the Priority Queue assigned to this Message Flow. */
120  pqueue = TO_MessageFlow_GetPQueue(channel, msgFlow, &pQueueIndex);
121  if (pqueue != NULL)
122  {
123  /* Queue the message. The else portion will handle all cases where the message
124  * was not queued for the following reasons: configuration table pointer was not
125  * available, queue is full, or memory full error.
126  */
127  status = TO_PriorityQueue_QueueMsg(channel, DataMsgPtr, pQueueIndex);
128 
129  if (CFE_SUCCESS == status)
130  {
131  /* The message was queued. Increment counters. */
132  channel->DumpTbl.MessageFlow[msgFlowIndex].QueuedMsgCnt++;
133 
134  channel->QueuedMsgCount++;
135  }
136  /* The call to TO_PriorityQueue_QueueMsg may generate the following errors:
137  * TO_PRIORITY_QUEUE_FULL_ERR (OS_QUEUE_FULL), TO_MEMORY_FULL_ERR, CFE_ES_ERR_MEM_HANDLE,
138  * OS_ERR_INVALID_ID, OS_INVALID_POINTER, OS_ERROR are all handled by this else clause */
139  else
140  {
141  /* Queue is full. Increment counters and drop the message. */
142  channel->DumpTbl.MessageFlow[msgFlowIndex].DroppedMsgCnt++;
143 
144  (void) OS_MutSemTake(TO_AppData.MutexID);
145  TO_AppData.HkTlm.usTotalMsgDropped++;
146  (void) OS_MutSemGive(TO_AppData.MutexID);
147 
148  channel->DropMsgCount++;
149 
150  (void) CFE_EVS_SendEvent(TO_MSG_DROP_FROM_FLOW_DBG_EID,
151  CFE_EVS_DEBUG,
152  "PQ full (PQ %u, channel %u). Error code (%d) Dropped message 0x%04x",
153  (unsigned int)pQueueIndex,
154  (unsigned int)channel->channelIdx,
155  (int)status,
156  (unsigned int)DataMsgID);
157  }
158  }
159  }
160  }
161 }
