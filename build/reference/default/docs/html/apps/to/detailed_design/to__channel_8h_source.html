66 /**
67 ** \brief Channel States.
68 *
69 * \par TO channels can have three states:
70 *
71 * TO_CHANNEL_UNKNOWN
72 * TO_CHANNEL_OPENED
73 * TO_CHANNEL_CLOSED
74 *
75 * \note All channels that are opened will never be commanded to close.
76 * The only time where a channel is closed is during clean-up.
77 */
78 typedef enum
79 {
80 
81  /** \brief Channel is unknown.
82  *
83  * \par
84  * This state means the channel was either never initialized or failed to initialized
85  * and is in an unusable state (no data flow).
86  */
87  TO_CHANNEL_UNKNOWN = 0,
88 
89  /** \brief Channel closed state.
90  *
91  * \par
92  * This state means that the channel is not operational (no data flow).
93  */
94  TO_CHANNEL_CLOSED = 1,
95 
96  /** \brief Channel opened state.
97  *
98  * \par
99  * This state means the channel is operational (data can flow).
100  */
101  TO_CHANNEL_OPENED = 2,
102 
103  /** \brief This is a count of Channel states and should not be used. */
104  TO_CHANNEL_STATE_COUNT
105 
106 } TO_ChannelState_t OS_ALIGN(4);
107 
108 /*Make sure uint8 can store enum*/

234 /************************************************************************/
235 /** \brief Configures and initializes a single channel.
236 **
237 ** \par Description
238 ** Initializes channel specific data and structures. Sets up channel pipe
239 ** as well as initializing the cf throttling semaphore for each channel.
240 **
241 ** \par Assumptions, External Events, and Notes:
242 ** Semaphore name must be unique among channels; initialization will fail if two
243 ** channels attempt to use the same semaphore.
244 ** The pointers for ChannelName, ConfigTableName, ConfigTableFileName,
245 ** DumpTableName, and CfCntSemName are not checked for NULL since those
246 ** are passed in as arguments that have been macro defined in to_platform_cfg.h
247 ** and are checked in a verify header to ensure the macros exist.
248 **
249 ** Only the index is range-checked and no other argument checking is performed.
250 **
251 ** \param[in] index
252 ** Index of the channel, cannot be greater than TO_MAX_CHANNELS
253 **
254 ** \param[in] ChannelName
255 ** Name of channel, will be prefixed with TO_ .
256 ** NOTE: Since each channel has a child task thread with the same name,
257 ** and in RTEMS threads must have names of 4 chars, this param should be a single char.
258 **
259 ** The 4-char limit is an RTEMS limit. The RTEMS OSAL only takes the first 4 chars of
260 ** the name per RTEMS limits for the underlying RTEMS message queue but still
261 ** preserves the full name given to it in the OSAL
262 ** The 1-char limit helps with keeping the name short enough to view it via
263 ** RTEMS reports
264 **
265 ** \param[in] ConfigTableName
266 ** Name of the config table for the channel as referred to by TBL services
267 **
268 ** \param[in] ConfigTableFileName
269 ** File path of the channel config table
270 **
271 ** \param[in] DumpTableName
272 ** Name of the dump table for the channel as referred to by TBL services
273 **
274 ** \param[in] CfCntSemMax
275 ** The initial maximum value of the CF throttling semaphore for this channel.
276 **
277 ** \param[in] CfCntSemName
278 ** Name of the CF throttling semaphore, must be unique among each channel.
279 **
280 ** \return
281 ** CFE_SUCCESS if successful. #TO_CHANNEL_OUT_OF_RANGE_ERR or OSAL error if unsuccessful.
282 **
283 *************************************************************************/
284 int32 TO_Channel_OpenChannel(const uint32 index, const char *ChannelName,
285  const char *ConfigTableName, const char *ConfigTableFileName,
286  TO_ChannelTbl_t *BackupTblPtr, const char *DumpTableName,
287  const uint32 CfCntSemMax, const char *CfCntSemName);
