42 /* Buildup a channel output queue */
43 /* */
44 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
45 int32 TO_OutputQueue_Buildup(TO_ChannelData_t* channel)
46 {
47  int32 status = OS_SUCCESS;
48 
49  /*
50  * Now that we have the new table, create all the new resources we need
51  * starting with queues for the channels and priority queues.
52  */
53  char QueueName[OS_MAX_API_NAME];
54  snprintf(QueueName, OS_MAX_API_NAME, "TO_%s_OUT", (const char*)&channel->ChannelName[0]);
55 
56  status = OS_QueueCreate(&channel->OutputQueue.OSALQueueID,
57  QueueName,
58  TO_OUTPUT_QUEUE_DEPTH,
59  sizeof(CFE_SB_Msg_t*),
60  0);
61 
62  if (status != OS_SUCCESS)
63  {
64  (void) CFE_EVS_SendEvent(TO_CONFIG_TABLE_ERR_EID,
65  CFE_EVS_ERROR,
66  "Failed to create '%s' output channel queue for channel %d. err=%ld",
67  channel->ChannelName, channel->channelIdx, status);
68 
69  channel->OutputQueue.OSALQueueID = OS_MAX_QUEUES;
70  }
71 
72  return status;
73 }

77 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
78 /* */
79 /* Teardown channel output queue in preparation for a */
80 /* reconfiguration. */
81 /* */
82 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
83 int32 TO_OutputQueue_Teardown(TO_ChannelData_t *channel)
84 {
85  int32 status = CFE_SUCCESS;
86  int32 putStatus = 0;
87  void *buffer = NULL;
88  uint32 nBytesCopied = 0;
89 
90  if (channel->OutputQueue.OSALQueueID != OS_MAX_QUEUES)
91  {
92  while (OS_SUCCESS == status)
93  {
94  status = OS_QueueGet(
95  channel->OutputQueue.OSALQueueID,
96  &buffer, sizeof(buffer), &nBytesCopied, OS_CHECK);
97  if (OS_SUCCESS == status)
98  {
99  putStatus = CFE_ES_PutPoolBuf(channel->MemPoolHandle, (uint32*)buffer);
100  if (putStatus < 0)
101  {
102  /* Failed to return memory back to memory pool. Not much we can do but report it
103  * and keep processing the queue until its empty.
104  */
105  (void) CFE_EVS_SendEvent(TO_PUT_POOL_ERR_EID,
106  CFE_EVS_ERROR,
107  "Failed to return message back to memory pool on output queue teardown, channel %d. (%d)",
108  (unsigned int)channel->channelIdx,
109  (int)putStatus);
110  } else {
111  /* Since status is positive, it is safe to cast */
112  channel->MemInUse -= (uint32)putStatus;
113  }
114  }
115 
116  }
117 
118  if (status != OS_QUEUE_EMPTY)
119  {
120  /* The OSAL failed to pop all the messages off. Not much we can do. Report it and keep going. */
121  (void) CFE_EVS_SendEvent(TO_OSQUEUE_GET_ERROR_EID,
122  CFE_EVS_ERROR,
123  "Failed to pop all messages from channel '%s' (channel id = %d) output queue. (%d)",
124  channel->ChannelName,
125  (unsigned int)channel->channelIdx,
126  (int)status);
127  }
128  }
129 
130  status = TO_OutputChannel_CustomTeardown(channel->channelIdx);
131 
132  return status;
133 }

152 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
153 /* */
154 /* Push a message on the output channel queue. */
155 /* */
156 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
157 int32 TO_OutputQueue_QueueMsg(TO_ChannelData_t *channel, CFE_SB_MsgPtr_t MsgPtr)
158 {
159  int32 status = 0;
160 
161  status = OS_QueuePut(channel->OutputQueue.OSALQueueID, &MsgPtr, sizeof(MsgPtr), 0);
162  if (OS_QUEUE_FULL == status)
163  {
164  /* This is not supposed to happen since we only queue when the channel
165  * is not full, but trap it anyway. Since the queue is full and the
166  * message will not be sent, deallocate the memory allocated as it
167  * won't be needed
168  */
169  status = CFE_ES_PutPoolBuf(channel->MemPoolHandle, (uint32 *)MsgPtr);
170  if (status < 0)
171  {
172  (void) CFE_EVS_SendEvent(TO_PUT_POOL_ERR_EID,
173  CFE_EVS_ERROR,
174  "PutPoolBuf: channel=%d, error=%i",
175  (unsigned int)channel->channelIdx,
176  (int)status);
177 
178  return status;
179  }
180 
181  /* Since status is positive, it is safe to cast */
182  channel->MemInUse -= (uint32)status;
183  }
184  else if (status != CFE_SUCCESS)
185  {
186  (void) CFE_EVS_SendEvent(TO_OSQUEUE_PUT_ERROR_EID,
187  CFE_EVS_ERROR,
188  "OS_QueuePut failed: channel=%d size=%u error=%i",
189  (unsigned int)channel->channelIdx,
190  sizeof(MsgPtr),
191  (int)status);
192  }
193 
194  return status;
195 }

203 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
204 osalbool TO_OutputChannel_Query(uint16 ChannelIdx)
205 {
206  TO_ChannelData_t *channel = NULL;
207 
208  /* First, check if the channel index is valid. */
209  if (ChannelIdx >= TO_MAX_CHANNELS)
210  {
211  (void) CFE_EVS_SendEvent(TO_OUT_CH_INFO_ERR_EID,
212  CFE_EVS_ERROR,
213  "Invalid channel index (index = %d, max = %d).",
214  (unsigned int)ChannelIdx,
215  TO_MAX_CHANNELS);
216 
217  return FALSE;
218  }
219  channel = &TO_AppData.ChannelData[ChannelIdx];
220 
221  /* Next, see if the channel is open. */
222  if (channel->State != TO_CHANNEL_OPENED)
223  {
224  (void) CFE_EVS_SendEvent(TO_OUT_CH_INFO_ERR_EID,
225  CFE_EVS_ERROR,
226  "Channel %d not open.",
227  (unsigned int)ChannelIdx);
228 
229  return FALSE;
230  }
231 
232  (void) CFE_EVS_SendEvent(TO_OUT_CH_INFO_EID,
233  CFE_EVS_INFORMATION,
234  "CHANNEL=%d S=%d ML=%i SC=%u CQC=%d HWM=%d",
235  (unsigned int)ChannelIdx,
236  (unsigned int)channel->State,
237  TO_OUTPUT_QUEUE_DEPTH,
238  (unsigned int)channel->OutputQueue.SentCount,
239  (unsigned int)channel->OutputQueue.CurrentlyQueuedCnt,
240  (unsigned int)channel->OutputQueue.HighwaterMark);
241  return TRUE;
242 }
