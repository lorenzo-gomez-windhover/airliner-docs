65 /** \brief Definition for a single telemetry message flow entry */
66 typedef struct
67 {
68  /**
69  * \brief The Message ID is used to select the message from the cFE Software Bus (SB) to send
70  * through the TO channel managed by the table.
71  *
72  * \par
73  * Units: none.
74  *
75  * \par
76  * Limits: must be defined in cfe_mission_cfg.h
77  */
78  CFE_SB_MsgId_t MsgId;
79 
80  /**
81  * \brief Maximum number of messages in a Message Flow.
82  *
83  * \par
84  * Units: Number of messages that can be in a Message Flow.
85  *
86  * \par
87  * Limits: The mininum value is 1. The maximum value depends on available memory and is
88  * an OPS driven parameter. The sum of all the msglimits in a priority queue
89  * should not be greater than the msglimit of that priority queue.
90  */
91  uint16 MsgLimit;
92 
93  /**
94  * \brief The Priority Queue ID is an index which selects the priority queue the message will be
95  * placed into.
96  *
97  * \par
98  * Units: none.
99  *
100  * \par
101  * Limits: If the Priority Queue ID does not match a priority queue then the table is invalid
102  */
103  uint16 PQueueID;
