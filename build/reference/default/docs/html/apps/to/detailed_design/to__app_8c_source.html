93 /* Initialize event tables */
94 /* */
95 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
96 int32 TO_InitEvent(void)
97 {
98  int32 regStatus = CFE_SUCCESS;
99  int32 customStatus = CFE_SUCCESS;
100  int32 ind = 0;
101  uint32 channelID = 0;
102 
103  /* Initialize the event filter table.
104  * Note: 0 is the CFE_EVS_NO_FILTER mask and event 0 is reserved (not used) */
105  (void) CFE_PSP_MemSet((void*)TO_AppData.EventTbl, 0x00, sizeof(TO_AppData.EventTbl));
106 
107  /* Choose the events you want to filter. CFE_EVS_MAX_EVENT_FILTERS
108  * limits the number of filters per app. An explicit CFE_EVS_NO_FILTER
109  * (the default) has been provided as an example. */
110 
111  TO_AppData.EventTbl[ ind].EventID = TO_MSG_ID_ERR_EID;
112  TO_AppData.EventTbl[ind++].Mask = CFE_EVS_FIRST_4_STOP;
113 
114  TO_AppData.EventTbl[ ind].EventID = TO_PIPE_READ_ERR_EID;
115  TO_AppData.EventTbl[ind++].Mask = CFE_EVS_FIRST_4_STOP;
116 
117  TO_AppData.EventTbl[ ind].EventID = TO_MSG_LEN_ERR_EID;
118  TO_AppData.EventTbl[ind++].Mask = CFE_EVS_FIRST_4_STOP;
119 
120  TO_AppData.EventTbl[ ind].EventID = TO_TLM_MSG_LEN_ERR_EID;
121  TO_AppData.EventTbl[ind++].Mask = CFE_EVS_FIRST_4_STOP;
122 
123  TO_AppData.EventTbl[ ind].EventID = TO_SB_RECEIVE_ERR_EID;
124  TO_AppData.EventTbl[ind++].Mask = CFE_EVS_FIRST_4_STOP;
125 
126  for (channelID = 0; channelID < TO_MAX_CHANNELS; channelID++)
127  {
128  TO_AppData.EventTbl[ ind].EventID = TO_MSG_ID_ERR_EID;
129  TO_AppData.EventTbl[ind++].Mask = CFE_EVS_FIRST_4_STOP;
130 
131  TO_AppData.EventTbl[ ind].EventID = TO_PUT_POOL_ERR_EID;
132  TO_AppData.EventTbl[ind++].Mask = CFE_EVS_FIRST_4_STOP;
133 
134  TO_AppData.EventTbl[ ind].EventID = TO_PIPE_READ_ERR_EID;
135  TO_AppData.EventTbl[ind++].Mask = CFE_EVS_FIRST_4_STOP;
136 
137  TO_AppData.EventTbl[ ind].EventID = TO_TLM_MSG_LEN_ERR_EID;
138  TO_AppData.EventTbl[ind++].Mask = CFE_EVS_FIRST_4_STOP;
139 
140  TO_AppData.EventTbl[ ind].EventID = TO_MF_MSG_ID_ERR_EID;
141  TO_AppData.EventTbl[ind++].Mask = CFE_EVS_FIRST_ONE_STOP;
142  }
143 
144  customStatus = TO_Custom_InitEvent(&ind);
145 
146  if(TO_CUSTOM_INITEVENT_BAD_IND_ERR == customStatus)
147  {
148  (void) CFE_ES_WriteToSysLog("TO - Custom Init Event error: event table index out of range. Custom EID filters not set.");
149  }
150  else if(TO_CUSTOM_INITEVENT_INDEX_OCCUPIED == customStatus)
151  {
152  (void) CFE_ES_WriteToSysLog("TO - Custom Init Event error: event table occupied at starting index. Custom EID filters not set.");
153  }
154  else if(TO_CUSTOM_INITEVENT_TOO_MANY_FILTERS == customStatus)
155  {
156  (void) CFE_ES_WriteToSysLog("TO - Custom Init Event error: Filter limit reached. Some custom EID filters not set.");
157  }
158  else if(CFE_SUCCESS == customStatus)
159  {
160  (void) CFE_ES_WriteToSysLog("TO - Custom Init Event: Successfully set custom EID filters.");
161  }
162  else
163  {
164  (void) CFE_ES_WriteToSysLog("TO - Custom Init Event: Unknown Error.");
165  }
166 
167  /* Register the table with CFE */
168  regStatus = CFE_EVS_Register(TO_AppData.EventTbl,
169  (uint16)ind, CFE_EVS_BINARY_FILTER);
170  if (regStatus != CFE_SUCCESS)
171  {
172  (void) CFE_ES_WriteToSysLog("TO - Failed to register with EVS (0x%08X)\n", (unsigned int)regStatus);
173  }
174  else
175  {
176  /* If there were custom event init errors, send an EVS message*/
177  switch (customStatus)
178  {
179  case TO_CUSTOM_INITEVENT_BAD_IND_ERR:
180  (void) CFE_EVS_SendEvent(TO_EVT_CUSTOM_FILTERS_ERR_EID,
181  CFE_EVS_ERROR,
182  "Custom Init Event - Event table index out of range. Custom EID filters not set.");
183  break;
184 
185  case TO_CUSTOM_INITEVENT_INDEX_OCCUPIED:
186  (void) CFE_EVS_SendEvent(TO_EVT_CUSTOM_FILTERS_ERR_EID,
187  CFE_EVS_ERROR,
188  "Custom Init Event - Event table occupied at starting index. Custom EID filters not set.");
189  break;
190 
191  case TO_CUSTOM_INITEVENT_TOO_MANY_FILTERS:
192  (void) CFE_EVS_SendEvent(TO_EVT_CUSTOM_FILTERS_ERR_EID,
193  CFE_EVS_ERROR,
194  "Custom Init Event - Filter limit reached. Some custom EID filters not set.");
195  break;
196 
197  case CFE_SUCCESS:
198  break;
199 
200  default:
201  (void) CFE_EVS_SendEvent(TO_EVT_CUSTOM_FILTERS_ERR_EID,
202  CFE_EVS_ERROR,
203  "Custom Init Event - Unknown error.");
204  break;
205  }
206  }
207  return (regStatus);
208 }

217 {
218  int32 status = CFE_SUCCESS;
219 
220  /* Init schedule pipe and subscribe to wakeup messages */
221  status = CFE_SB_CreatePipe(&TO_AppData.SchPipeId,
222  TO_SCH_PIPE_DEPTH,
223  TO_SCH_PIPE_NAME);
224  if (status == CFE_SUCCESS)
225  {
226  status = CFE_SB_SubscribeEx(TO_SEND_TLM_MID, TO_AppData.SchPipeId, CFE_SB_Default_Qos, TO_SCH_PIPE_WAKEUP_RESERVED);
227  if (status != CFE_SUCCESS)
228  {
229  (void) CFE_EVS_SendEvent(TO_SUBSCRIBE_ERR_EID,

232  (unsigned int)status);
233  goto TO_InitPipe_Exit_Tag;
234  }
235 
236  status = CFE_SB_SubscribeEx(TO_SEND_HK_MID, TO_AppData.SchPipeId, CFE_SB_Default_Qos, TO_SCH_PIPE_SEND_HK_RESERVED);
237  if (status != CFE_SUCCESS)
238  {
239  (void) CFE_EVS_SendEvent(TO_SUBSCRIBE_ERR_EID,
240  CFE_EVS_ERROR,
241  "Sch Pipe failed to subscribe to TO_SEND_HK_MID. (0x%08X)",
242  (unsigned int)status);
243  goto TO_InitPipe_Exit_Tag;
244  }
245 
246  }
247  else
248  {
249  (void) CFE_EVS_SendEvent(TO_INIT_SCHPIPE_ERR_EID,
250  CFE_EVS_ERROR,
251  "Failed to create SCH pipe (0x%08X)",
252  (unsigned int)status);
253  goto TO_InitPipe_Exit_Tag;
254  }
255 
256  /* Init command pipe and subscribe to command messages */
257  status = CFE_SB_CreatePipe(&TO_AppData.CmdPipeId,
258  TO_CMD_PIPE_DEPTH,
259  TO_CMD_PIPE_NAME);
260  if (status == CFE_SUCCESS)
261  {
262  /* Subscribe to command messages */
263  status = CFE_SB_Subscribe(TO_CMD_MID, TO_AppData.CmdPipeId);
264  if (status != CFE_SUCCESS)
265  {
266  (void) CFE_EVS_SendEvent(TO_SUBSCRIBE_ERR_EID,
267  CFE_EVS_ERROR,
268  "CMD Pipe failed to subscribe to TO_CMD_MID. (0x%08X)",
269  (unsigned int)status);
270  }
271  }
272  else
273  {
274  (void) CFE_EVS_SendEvent(TO_INIT_CMDPIPE_ERR_EID,
275  CFE_EVS_ERROR,
276  "Failed to create CMD pipe (0x%08X)",
277  (unsigned int)status);
278  }
279 
280 TO_InitPipe_Exit_Tag:
281  return (status);
282 }

418 /* */
419 /* Receive and Process Messages */
420 /* */
421 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
422 int32 TO_RcvMsg(int32 blocking)
423 {
424  int32 status=CFE_SUCCESS;
425  CFE_SB_Msg_t *msgPtr=NULL;
426  CFE_SB_MsgId_t MsgId;
427 
428  CFE_ES_PerfLogExit(TO_MAIN_TASK_PERF_ID);
429 
430  /* Wait for WakeUp messages from scheduler */
431  status = CFE_SB_RcvMsg(&msgPtr, TO_AppData.SchPipeId, blocking);
432 
433  /* Start Performance Log entry */
434  CFE_ES_PerfLogEntry(TO_MAIN_TASK_PERF_ID);
435 
436  if (CFE_SUCCESS == status)
437  {
438  MsgId = CFE_SB_GetMsgId(msgPtr);
439  switch (MsgId)
440  {
441  case TO_SEND_TLM_MID:
442  TO_ProcessNewCmds();
443  TO_Channel_ProcessTelemetryAll();
444  break;
445 
446  case TO_SEND_HK_MID:
447  TO_ReportHousekeeping();
448  break;
449 
450  default:
451  (void) CFE_EVS_SendEvent(TO_MSG_ID_ERR_EID,
452  CFE_EVS_ERROR,
453  "Recvd invalid SCH msgId (0x%04X)", (unsigned short)MsgId);
454  }
455  }
456  else if (CFE_SB_NO_MESSAGE == status)
457  {
458  /* If there's no incoming message, you can do something here, or
459  * nothing. Note, this section is dead code only if the blocking arg
460  * is CFE_SB_PEND_FOREVER. */
461  status = CFE_SUCCESS;
462  }
463  else if (CFE_SB_TIME_OUT == status)
464  {
465  /* If there's no incoming message within a specified time (via the
466  * blocking arg, you can do something here, or nothing.
467  * Note, this section is dead code only if the blocking arg
468  * is CFE_SB_PEND_FOREVER. */
469  status = CFE_SUCCESS;
470  }
471  else
472  {
473  (void) CFE_EVS_SendEvent(TO_PIPE_READ_ERR_EID,
474  CFE_EVS_ERROR,
475  "SB pipe read error (0x%08X)",
476  (unsigned int)status);
477  }
478 
479  TO_ManageAllAppTables();
480 
481  return (status);
482 }

488 /* Process Incoming Commands */
489 /* */
490 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
491 void TO_ProcessNewCmds(void)
492 {
493  int32 status = CFE_SUCCESS;
494  CFE_SB_Msg_t *cmdMsgPtr=NULL;
495  CFE_SB_MsgId_t cmdMsgId;
496  uint32 i;
497 
498  /* Process command messages till the pipe is empty */
499  for (i = 0; i < TO_CMD_PIPE_DEPTH; ++i)
500  {
501  status = CFE_SB_RcvMsg(&cmdMsgPtr, TO_AppData.CmdPipeId, CFE_SB_POLL);
502  if (status == CFE_SUCCESS)
503  {
504  cmdMsgId = CFE_SB_GetMsgId(cmdMsgPtr);
505  switch (cmdMsgId)
506  {
507  case TO_CMD_MID:
508  TO_ProcessNewAppCmds(cmdMsgPtr);
509  break;
510 
511  default:
512  /* Bump the command error counter for an unknown command.
513  * (This should only occur if it was subscribed to with this
514  * pipe, but not handled in this switch-case.) */
515  TO_AppData.HkTlm.usCmdErrCnt++;
516  (void) CFE_EVS_SendEvent(TO_MSG_ID_ERR_EID,
517  CFE_EVS_ERROR,
518  "Recvd invalid CMD msgId (0x%04X)",
519  (unsigned short)cmdMsgId);
520  break;
521  }
522  }
523  else if (CFE_SB_NO_MESSAGE == status)
524  {
525  break;
526  }
527  else
528  {
529  (void) CFE_EVS_SendEvent(TO_PIPE_READ_ERR_EID,
530  CFE_EVS_ERROR,
531  "CMD pipe read error (0x%08X)", (unsigned int)status);
532  }
533  }
534 }

537 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
538 /* */
539 /* Process TO Commands */
540 /* */
541 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
542 void TO_ProcessNewAppCmds(CFE_SB_Msg_t *msgPtr)
543 {
544  uint32 cmdCode = 0;
545  osalbool sizeOk = FALSE;
546  osalbool isCmdOk = FALSE;
547 
548  if (msgPtr != NULL)
549  {
550  cmdCode = CFE_SB_GetCmdCode(msgPtr);
551  switch (cmdCode)
552  {
553  case TO_NOOP_CC:
554  {
555  sizeOk = TO_VerifyCmdLength(msgPtr, sizeof(TO_NoArgCmd_t));
556 
557  if (TRUE == sizeOk)
558  {
559  (void) OS_MutSemTake(TO_AppData.MutexID);
560  TO_AppData.HkTlm.usCmdCnt++;
561  (void) OS_MutSemGive(TO_AppData.MutexID);
562  (void) CFE_EVS_SendEvent(TO_CMD_NOOP_INF_EID, CFE_EVS_INFORMATION,
563  "Executed NOOP cmd (%u), Version %d.%d.%d.%d",
564  (unsigned int)cmdCode,
565  TO_MAJOR_VERSION,
566  TO_MINOR_VERSION,
567  TO_REVISION,
568  TO_MISSION_REV);
569  TO_PrintCustomVersion();
570  }
571  break;
572  }
573 
574  case TO_RESET_CC:
575  {
576  sizeOk = TO_VerifyCmdLength(msgPtr, sizeof(TO_NoArgCmd_t));
577 
578  if (TRUE == sizeOk)
579  {
580  (void) OS_MutSemTake(TO_AppData.MutexID);
581  TO_AppData.HkTlm.usCmdCnt = 0;
582  TO_AppData.HkTlm.usCmdErrCnt = 0;
583  TO_AppData.HkTlm.usTotalMsgDropped = 0;
584  TO_AppData.HkTlm.SentBytes = 0;
585  (void) OS_MutSemGive(TO_AppData.MutexID);
586 
587  TO_Channel_ResetCountsAll();
588 
589  (void) CFE_EVS_SendEvent(TO_CMD_RESET_INF_EID, CFE_EVS_INFORMATION,
590  "Executed RESET cmd (%u)", (unsigned int)cmdCode);
591  }
592  break;
593  }
594 
595  case TO_ADD_MESSAGE_FLOW_CC:
596  {
597  sizeOk = TO_VerifyCmdLength(msgPtr, sizeof(TO_AddMessageFlowCmd_t));
598 
599  if (TRUE == sizeOk)
600  {
601  TO_AddMessageFlowCmd_t *cmd = (TO_AddMessageFlowCmd_t*)msgPtr;
602 
603  isCmdOk = TO_MessageFlow_Add(cmd->ChannelIdx, cmd->MsgID, cmd->MsgLimit, cmd->PQueueIdx);
604 
605  if (FALSE == isCmdOk)
606  {
607  (void) OS_MutSemTake(TO_AppData.MutexID);
608  TO_AppData.HkTlm.usCmdErrCnt++;
609  (void) OS_MutSemGive(TO_AppData.MutexID);
610  }
611  else
612  {
613  (void) OS_MutSemTake(TO_AppData.MutexID);
614  TO_AppData.HkTlm.usCmdCnt++;
