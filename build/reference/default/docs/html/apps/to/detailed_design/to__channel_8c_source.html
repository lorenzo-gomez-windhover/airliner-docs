126 int32 TO_Channel_OpenChannel(const uint32 index, const char *ChannelName,
127  const char *ConfigTableName, const char *ConfigTableFileName,
128  TO_ChannelTbl_t *BackupTblPtr, const char *DumpTableName,
129  const uint32 CfCntSemMax, const char *CfCntSemName)
130 {
131  int32 status = TO_CHANNEL_OPEN_ERR;
132  char pipeName[OS_MAX_API_NAME];
133  TO_ChannelData_t *channel = NULL;
134 
135  if (index >= TO_MAX_CHANNELS)
136  {
137  status = TO_CHANNEL_OUT_OF_RANGE_ERR;
138  (void) CFE_EVS_SendEvent(TO_INIT_APP_ERR_EID,
139  CFE_EVS_ERROR,
140  "Invalid channel index of %u.", (unsigned int)index);
141  return status;
142  }
143 
144  channel = &TO_AppData.ChannelData[index];
145 
146  if(channel->State == TO_CHANNEL_CLOSED)
147  {
148  TO_Channel_LockByRef(channel);
149 
150  snprintf(pipeName, OS_MAX_API_NAME, "TO_%s", ChannelName);
151 
152  strncpy(channel->ChannelName,
153  ChannelName, sizeof(channel->ChannelName));
154  channel->ChannelName[sizeof(channel->ChannelName) - 1] = '\0';
155 
156  strncpy(channel->ConfigTableName,
157  ConfigTableName, sizeof(channel->ConfigTableName));
158  channel->ConfigTableName[sizeof(channel->ConfigTableName) - 1] = '\0';
159 
160  strncpy(channel->ConfigTableFileName,
161  ConfigTableFileName, sizeof(channel->ConfigTableFileName));
162  channel->ConfigTableFileName[sizeof(channel->ConfigTableFileName) - 1] = '\0';
163 
164  channel->BackupTblPtr = BackupTblPtr;
165 
166  strncpy(channel->DumpTableName,
167  DumpTableName, sizeof(channel->DumpTableName));
168  channel->DumpTableName[sizeof(channel->DumpTableName) - 1] = '\0';
169 
170  /* Set up CF Throttling semaphore */
171  channel->CfCntSemMax = CfCntSemMax;
172  strncpy(channel->CfCntSemName, CfCntSemName,
173  sizeof(channel->CfCntSemName));
174  channel->CfCntSemName[OS_MAX_API_NAME - 1] = '\0';
175 
176  /* Initialize CF Counting Sem to CfCntSemMax */
177  /* NOTE: This will fail if two channels attempt to use the same
178  * CF channel. */
179  status = OS_CountSemCreate(&channel->CfCntSemId,
180  channel->CfCntSemName,
181  channel->CfCntSemMax, 0);
182  if (status != OS_SUCCESS)
183  {
184  CFE_EVS_SendEvent(TO_INIT_APP_ERR_EID,
185  CFE_EVS_ERROR,
186  "Failed to create counting semaphore "
187  "for CF channel semaphore:%s for TO channel(%u):%s . "
188  "(OSAL Error:%d)",
189  channel->CfCntSemName, (unsigned int)index, channel->ChannelName, (int)status);
190 
191  TO_Channel_UnlockByRef(channel);
192  return status;
193  }
194 
195  status = TO_OutputQueue_Buildup(channel);
196  if (status != OS_SUCCESS)
197  {
198  /* This is a critical error for this channel. No sense in continuing. Destroy
199  * the counting semaphore we just created, and return the error back to the caller.
200  */
201  OS_CountSemDelete(channel->CfCntSemId);
202 
203  channel->State = TO_CHANNEL_CLOSED;
204 
205  TO_Channel_UnlockByRef(channel);
206  return status;
207  }
208 
209  /* Init data pipe and subscribe to messages on the data pipe */
210  status = CFE_SB_CreatePipe(&channel->DataPipeId,
211  TO_DATA_PIPE_DEPTH,
212  pipeName);
213  if (status != CFE_SUCCESS)
214  {
215  /* This is a critical error for this channel. No sense in continuing. Destroy
216  * the counting semaphore and call the Output queue teardown function before
217  * returning the error back to the caller.
218  */
219  (void) OS_CountSemDelete(channel->CfCntSemId);
220  (void) TO_OutputQueue_Teardown(channel);
221 
222  (void) CFE_EVS_SendEvent(TO_INIT_DATAPIPE_ERR_EID,
223  CFE_EVS_ERROR,
224  "Failed to create channel (%u) '%s' pipe (0x%08X)",
225  (unsigned int)index,
226  ChannelName,
227  (unsigned int)status);
228 
229  channel->State = TO_CHANNEL_CLOSED;
230 
231  TO_Channel_UnlockByRef(channel);
232  return status;
233  }
234 
235  /* Set channel state: required state to complete table initialization */
236  channel->State = TO_CHANNEL_OPENED;
237 
238  status = TO_InitTables(channel);
239  if (status != CFE_SUCCESS)
240  {
241  /* This is a critical error for this channel. No sense in continuing. Destroy
242  * the counting semaphore, call the Output queue teardown function, and delete
243  * the pipe before returning the error back to the caller.
244  */
245  (void) OS_CountSemDelete(channel->CfCntSemId);
246  (void) TO_OutputQueue_Teardown(channel);
247  (void) CFE_SB_DeletePipe(channel->DataPipeId);
248 
249  CFE_EVS_SendEvent(TO_INIT_CONFIG_ERR_EID,
250  CFE_EVS_ERROR,
251  "Failed to init config tables (0x%08X) for channel %u",
252  (unsigned int)status,
253  (unsigned int)index);
254 
255  channel->State = TO_CHANNEL_CLOSED;
256 
257  TO_Channel_UnlockByRef(channel);
258  return status;
259  }
260 
261  /* Initialize the memory pool for the priority queues and output channel
262  * queues.
263  */
264  status = CFE_ES_PoolCreateEx(&channel->MemPoolHandle,
265  channel->MemPoolBuffer,
266  TO_NUM_BYTES_IN_MEM_POOL,
267  TO_MAX_MEMPOOL_BLK_SIZES,
268  &TO_MemPoolDefSize[0],
269  CFE_ES_USE_MUTEX);
270  if (status != CFE_SUCCESS)
271  {
272  /* This is a critical error for this channel. No sense in continuing. Destroy
273  * the counting semaphore, call the Output queue teardown function, and delete
274  * the pipe before returning the error back to the caller. Don't need to do
275  * anything with the tables. They will just be reinitialized the next time
276  * we try opening the channel.
277  */
278  (void) OS_CountSemDelete(channel->CfCntSemId);
279  (void) TO_OutputQueue_Teardown(channel);
280  (void) CFE_SB_DeletePipe(channel->DataPipeId);
281 
282  (void) CFE_EVS_SendEvent(TO_CR_POOL_ERR_EID,
283  CFE_EVS_ERROR,
284  "Error creating memory pool (0x%08X) for channel %u",(unsigned int)status,
285  (unsigned int)index);
286 
287  channel->State = TO_CHANNEL_CLOSED;
288 
289  TO_Channel_UnlockByRef(channel);
290 
291  return status;
292  }
293 
294  TO_Channel_UnlockByRef(channel);
295  }
296  return status;
297 }

565 osalbool TO_Channel_SBPipe_Dequeue_All(uint16 index)
566 {
567  int32 status = CFE_SUCCESS;
568  int32 i;
569  CFE_SB_MsgPtr_t DataMsgPtr = NULL;
570  TO_ChannelData_t *channel = NULL;
571 
572  if (index >= TO_MAX_CHANNELS)
573  {
574  (void) CFE_EVS_SendEvent(TO_FLUSH_INVALID_CHIDX_ERR_EID, CFE_EVS_ERROR,
575  "ChannelID %u out of range in SB pipe dequeue all.", (unsigned int)index);
576  return FALSE;
577  }
578 
579  channel = &TO_AppData.ChannelData[index];
580 
581  /* Dequeue until empty */
582  for (i = 0; i < TO_DATA_PIPE_DEPTH; ++i)
583  {
584  status = CFE_SB_RcvMsg(&DataMsgPtr, channel->DataPipeId, CFE_SB_POLL);
585  if (CFE_SUCCESS == status)
586  {
587  continue;
588  }
589  else if (CFE_SB_NO_MESSAGE == status)
590  {
591  /* Break early if empty */
592  break;
593  }
594  else
595  {
596  (void) CFE_EVS_SendEvent(TO_PIPE_READ_ERR_EID,
597  CFE_EVS_ERROR,
598  "Data pipe read error on SBPipe dequeue all (0x%08X) on channel %d",
599  (unsigned int)status,
600  index);
601  return FALSE;
602  }
603  }
604 
605  return TRUE;
