41 /* Build up all the message flows */
42 /* */
43 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
44 int32 TO_MessageFlow_Buildup(TO_ChannelData_t *channel)
45 {
46  uint32 i;
47  int32 status = CFE_SUCCESS;
48 
49  if (NULL == channel)
50  {
51  return TO_MESSAGE_FLOW_BAD_ARG_ERR;
52  }
53 
54  if (NULL == channel->ConfigTblPtr)
55  {
56  return TO_MESSAGE_FLOW_NO_TABLE_ERR;
57  }
58 
59  /*
60  * Create message flows by subscribing to messages.
61  */
62  for (i = 0; i < TO_MAX_MESSAGE_FLOWS; ++i)
63  {
64  if (channel->ConfigTblPtr->MessageFlow[i].MsgId != 0)
65  {
66  /* Subscribe to message. */
67  status = CFE_SB_SubscribeEx(channel->ConfigTblPtr->MessageFlow[i].MsgId, channel->DataPipeId,
68  CFE_SB_Default_Qos, channel->ConfigTblPtr->MessageFlow[i].MsgLimit);
69  if (status != CFE_SUCCESS)
70  {
71  /* We failed to subscribe to a message. However, lets just keep going so we can maybe
72  * subscribe to at least some of the messages. Report and keep going.
73  */
74  (void) CFE_EVS_SendEvent(TO_SUBSCRIBE_ERR_EID,
75  CFE_EVS_ERROR,
76  "Message flow failed to subscribe to (0x%08X) on channel %d. (%ld)",
77  channel->ConfigTblPtr->MessageFlow[i].MsgId,
78  channel->channelIdx,
79  status);
80  }
81  }
82  }
83  return CFE_SUCCESS;
84 }

93 int32 TO_MessageFlow_TeardownAll(TO_ChannelData_t *channel)
94 {
95  uint32 i;
96  int32 status = CFE_SUCCESS;
97 
98  if (NULL == channel)
99  {
100  return TO_MESSAGE_FLOW_BAD_ARG_ERR;
101  }
102 
103  if (NULL == channel->ConfigTblPtr)
104  {
105  return TO_MESSAGE_FLOW_NO_TABLE_ERR;
106  }
107 
108  for (i = 0; i < TO_MAX_MESSAGE_FLOWS; ++i)
109  {
110  if (channel->ConfigTblPtr->MessageFlow[i].MsgId != 0)

119  * Will not return the failure back.

122  CFE_EVS_ERROR,
123  "Message flow failed to unsubscribe from 0x%04x on channel %d. (%ld)",
124  channel->ConfigTblPtr->MessageFlow[i].MsgId,
125  channel->channelIdx,
126  status);
127  }
128  }
129  }
130  return CFE_SUCCESS;
131 }

244 /* */
245 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
246 osalbool TO_MessageFlow_Add(
247  uint16 ChannelIdx,
248  CFE_SB_MsgId_t MsgID,
249  uint16 MsgLimit,
250  uint16 PQueueIdx)
251 {
252  uint32 i;
253  uint32 j;
254  TO_MessageFlow_t *msgFlow = NULL;
255  osalbool added = FALSE;
256  int32 status = CFE_SUCCESS;
257  uint8 queueIdx = 0;
258  uint32 decimationFactorsSum = 0;
259 
260  TO_ChannelData_t *channel = NULL;
261 
262  /* First, check if the channel index is valid. */
263  if (ChannelIdx >= TO_MAX_CHANNELS)
264  {
265  (void) CFE_EVS_SendEvent(TO_CMD_ADD_MSG_FLOW_ERR_EID,
266  CFE_EVS_ERROR,
267  "Invalid channel (channel = %d, but max = %d).",
268  ChannelIdx, TO_MAX_CHANNELS);
269 
270  return FALSE;
271  }
272 
273  channel = &TO_AppData.ChannelData[ChannelIdx];
274 
275  if (NULL == channel->ConfigTblPtr)
276  {
277  (void) CFE_EVS_SendEvent(TO_MSG_FLOW_MISSING_TBL_ERR_EID,
278  CFE_EVS_ERROR,
279  "Channel missing valid table data (channel = %d).",
280  ChannelIdx);
281 
282  return FALSE;
283  }
284 
285  TO_Channel_LockByRef(channel);
286 
287  /* Next, see if the channel is open. */
288  if (channel->State != TO_CHANNEL_OPENED)
289  {
290  (void) CFE_EVS_SendEvent(TO_CMD_ADD_MSG_FLOW_ERR_EID,
291  CFE_EVS_ERROR,
292  "Channel (%d) not open.", ChannelIdx);
293 
294  TO_Channel_UnlockByRef(channel);
295  return FALSE;
296  }
297 
298  /* Next, see if there is a flow with this message ID. */
299  msgFlow = TO_MessageFlow_GetObject(channel, MsgID, NULL);
300 
301  if (NULL != msgFlow)
302  {
303  /* A message flow already exists. Reject the request. */
304  (void) CFE_EVS_SendEvent(TO_CMD_ADD_MSG_FLOW_ERR_EID,
305  CFE_EVS_ERROR,
306  "Message flow is already defined for channel %d.",
307  ChannelIdx);
308 
309  TO_Channel_UnlockByRef(channel);
310  return FALSE;
311  }
312 
313  /* Now that we know a message flow definition doesn't already exist, find
314  * an unused entry so we can add one. Unused entries have the MsgID set
315  * to 0.
316  */
317  for (i = 0; i < TO_MAX_MESSAGE_FLOWS; ++i)
318  {
319  if (channel->ConfigTblPtr->MessageFlow[i].MsgId == 0)
320  {
321  /* We found an unused entry. Set the message flow definition
322  * here. First, make sure the requested priority queue is valid.
323  */
324  if (TO_PriorityQueue_IsValid(channel, PQueueIdx) == FALSE)
325  {
326  /* This is an invalid priority queue. */
327  (void) CFE_EVS_SendEvent(TO_CMD_ADD_MSG_FLOW_ERR_EID,
328  CFE_EVS_ERROR,
329  "Priority Queue Index %u is invalid for channel %d.",
330  PQueueIdx, ChannelIdx);
331 
332  TO_Channel_UnlockByRef(channel);
333  return FALSE;
334  }
335 
336  /* Now subscribe to the message to ensure the message ID is
337  * valid.
338  */
339  status = CFE_SB_SubscribeEx(MsgID, channel->DataPipeId,
340  CFE_SB_Default_Qos, MsgLimit);
341  if (status != CFE_SUCCESS)
342  {
343  (void) CFE_EVS_SendEvent(TO_CMD_ADD_MSG_FLOW_ERR_EID,
344  CFE_EVS_ERROR,
345  "Message flow failed to subscribe to (0x%08X) on channel %d. (%ld)",
346  MsgID, ChannelIdx, status);
347 
348  TO_Channel_UnlockByRef(channel);
349  return FALSE;
350  }
351 
352  /* Now that the message was successfully subscribed to, set the
353  * message flow definition.
354  */
355  channel->ConfigTblPtr->MessageFlow[i].MsgId = MsgID;
356  channel->ConfigTblPtr->MessageFlow[i].MsgLimit = MsgLimit;
357  channel->ConfigTblPtr->MessageFlow[i].PQueueID = PQueueIdx;
358  channel->DumpTbl.MessageFlow[i].DroppedMsgCnt = 0;
359  channel->DumpTbl.MessageFlow[i].QueuedMsgCnt = 0;
360  channel->DumpTbl.MessageFlow[i].SBMsgCnt = 0;
361 
362  CFE_TBL_Modified(channel->ConfigTblHdl);
363 
364  added = TRUE;
365  break;
366  }
367  }
368 
369  if (FALSE == added)
370  {
371  (void) CFE_EVS_SendEvent(TO_CMD_ADD_MSG_FLOW_ERR_EID,
372  CFE_EVS_ERROR,
373  "Failed to add message flow. No unused slots available on channel %d.",
374  ChannelIdx);
375  }
376 
377  TO_Channel_UnlockByRef(channel);
378 
379  return added;
380 }

394  int32 status = CFE_SUCCESS;
395  TO_ChannelData_t *channel = NULL;
396 
397  /* First, check if the channel index is valid. */
398  if (ChannelIdx >= TO_MAX_CHANNELS)
399  {
400  (void) CFE_EVS_SendEvent(TO_CMD_REMOVE_MSG_FLOW_ERR_EID,
401  CFE_EVS_ERROR,
402  "Invalid channel (channel = %d, but max = %d).",
403  ChannelIdx, TO_MAX_CHANNELS);
404 
405  return FALSE;
406  }
407 
408  channel = &TO_AppData.ChannelData[ChannelIdx];
409 
410  if (NULL == channel->ConfigTblPtr)
411  {
412  (void) CFE_EVS_SendEvent(TO_MSG_FLOW_MISSING_TBL_ERR_EID,
413  CFE_EVS_ERROR,

415  ChannelIdx);
416 
417  return FALSE;
418  }
419 
420  TO_Channel_LockByRef(channel);
421 
422  /* Next, see if the channel is open. */
423  if (channel->State != TO_CHANNEL_OPENED)

446

448  * the entries.
449  */
450  status = CFE_SB_Unsubscribe(msgFlow->MsgId, channel->DataPipeId);
451 
452  if (status != CFE_SUCCESS)
453  {
454  (void) CFE_EVS_SendEvent(TO_CMD_REMOVE_MSG_FLOW_ERR_EID,
455  CFE_EVS_ERROR,
456  "Message flow failed to unsubscribe from 0x%08X on channel %d. (%ld)",
457  msgFlow->MsgId, ChannelIdx, status);
458 
459  TO_Channel_UnlockByRef(channel);
460  return FALSE;
461  }
462 
463  /* Now just clear the entries. */
464  channel->ConfigTblPtr->MessageFlow[msgFlowIndex].MsgId = 0;
465  channel->ConfigTblPtr->MessageFlow[msgFlowIndex].MsgLimit = 0;
466  channel->ConfigTblPtr->MessageFlow[msgFlowIndex].PQueueID = 0;
467  channel->DumpTbl.MessageFlow[msgFlowIndex].DroppedMsgCnt = 0;
468  channel->DumpTbl.MessageFlow[msgFlowIndex].QueuedMsgCnt = 0;
469  channel->DumpTbl.MessageFlow[msgFlowIndex].SBMsgCnt = 0;
470 
471  TO_Channel_UnlockByRef(channel);
472 
473  CFE_TBL_Modified(channel->ConfigTblHdl);
474 
475  return TRUE;
476 }
