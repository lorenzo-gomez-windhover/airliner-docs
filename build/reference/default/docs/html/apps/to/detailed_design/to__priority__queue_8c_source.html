72 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
73 /* */
74 /* Buildup all priority queues after a reconfiguration. */
75 /* */
76 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
77 int32 TO_PriorityQueue_BuildupAll(TO_ChannelData_t *channel)
78 {
79  uint32 i;
80  int32 status = CFE_SUCCESS;
81 
82  /* First, check if the channel NULL. */
83  if (NULL == channel)
84  {
85  (void) CFE_EVS_SendEvent(TO_PQUEUE_MISSING_TBL_ERR_EID,
86  CFE_EVS_ERROR,
87  "Channel is NULL.");
88  return TO_CHANNEL_PTR_NULL;
89  }
90 
91  if (NULL == channel->ConfigTblPtr)
92  {
93  (void) CFE_EVS_SendEvent(TO_PQUEUE_MISSING_TBL_ERR_EID,
94  CFE_EVS_ERROR,
95  "Failed to create priority queues on channel %d, missing table.",
96  channel->channelIdx);
97  return TO_NO_TABLE_ERR;
98  }
99 
100  /*
101  * Create the priority queues.
102  */
103  for (i = 0; i < TO_MAX_PRIORITY_QUEUES; ++i)
104  {
105  if (channel->ConfigTblPtr->PriorityQueue[i].State != TO_PQUEUE_UNUSED)
106  {
107  char QueueName[OS_MAX_API_NAME];
108  (void) CFE_PSP_MemSet(QueueName, 0, sizeof(QueueName));
109  snprintf(QueueName, OS_MAX_API_NAME, "TO_%s_%u", (const char*)&channel->ChannelName[0], (uint8)i);
110  status = OS_QueueCreate(
111  &channel->DumpTbl.PriorityQueue[i].OSALQueueID,
112  QueueName,
113  channel->ConfigTblPtr->PriorityQueue[i].MsgLimit,
114  sizeof(CFE_SB_Msg_t*),
115  0);
116  if (status != OS_SUCCESS)
117  {
118  /* For some reason, we failed to create the queue, but this is not a critical error.
119  * Report the failure, set the queue ID to OS_MAX_QUEUES, and continue. We set the
120  * ID to OS_MAX_QUEUES because the queue ID is a uint32 and 0 is a valid queue ID.
121  * OS_MAX_QUEUES is an invalid ID. That will prevent us from accidentally trying
122  * to queue messages into a nonexistent queue later.
123  */
124  (void) CFE_EVS_SendEvent(TO_PQUEUE_CREATE_ERR_EID,
125  CFE_EVS_ERROR,
126  "Failed to create '%s' priority queue #%u on channel %d. (%i)",
127  channel->ChannelName,
128  (unsigned int)i,
129  channel->channelIdx,
130  (int)status);
131 
132  channel->DumpTbl.PriorityQueue[i].OSALQueueID = OS_MAX_QUEUES;
133  }
134  }
135  }
136 
137  return CFE_SUCCESS;
138 }

144 /* Teardown all priority queues after a reconfiguration. */
145 /* */
146 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
147 int32 TO_PriorityQueue_TeardownAll(TO_ChannelData_t *channel)
148 {
149  uint32 i;
150  int32 status = 0;
151  int32 bytesReleased = 0;
152  uint32 bufferSize = 0;
153  void *buffer = NULL;
154 
155  if(NULL != channel->ConfigTblPtr)
156  {
157  for(i = 0; i < TO_MAX_PRIORITY_QUEUES; ++i)
158  {
159  if(channel->ConfigTblPtr->PriorityQueue[i].State != TO_PQUEUE_UNUSED)
160  {
161  if(channel->DumpTbl.PriorityQueue[i].OSALQueueID < OS_MAX_QUEUES)
162  {
163  buffer = NULL;
164  bufferSize = 0;
165  while(OS_SUCCESS == status)
166  {
167  status = OS_QueueGet(
168  channel->DumpTbl.PriorityQueue[i].OSALQueueID,
169  &buffer, sizeof(buffer), &bufferSize, OS_CHECK);
170  if(OS_SUCCESS == status)
171  {
172  bytesReleased = CFE_ES_PutPoolBuf(channel->MemPoolHandle, (uint32*)buffer);
173  if(bytesReleased < 0)
174  {
175  (void) CFE_EVS_SendEvent(TO_PQUEUE_TEARDOWN_ERR_EID,
176  CFE_EVS_ERROR,
177  "Failed to return message back to memory pool on tbl load for channel %d. (%ld)",
178  (unsigned int)channel->channelIdx,
179  (unsigned long int)bytesReleased);
180  } else {
181  TO_Channel_LockByRef(channel);
182  /* Since bytesReleased is positive, it is safe to cast */
183  channel->MemInUse -= (uint32)bytesReleased;
184  TO_Channel_UnlockByRef(channel);
185  }
186  }
187  }
188 
189  if(status != OS_QUEUE_EMPTY)
190  {
191  (void) CFE_EVS_SendEvent(TO_PQUEUE_TEARDOWN_ERR_EID,
192  CFE_EVS_ERROR,
193  "Message flow failed to pop all messages from pqueue %u for channel %d. (%d)",
194  (unsigned int)i,
195  (unsigned int)channel->channelIdx,
196  (int)status);
197  }
198 
199 
200  /* Queue may or may not be empty. If we ran into an error, just press forward and delete the
201  * queue.
202  */
203  status = OS_QueueDelete(
204  channel->DumpTbl.PriorityQueue[i].OSALQueueID);
205  if(status != OS_SUCCESS)
206  {
207  /* We ran into an error. Send an event, but just keep going. Nothing we can do. */
208  (void) CFE_EVS_SendEvent(TO_PQUEUE_TEARDOWN_ERR_EID,
209  CFE_EVS_ERROR,
210  "Failed to delete priority queue %u for channel %d. (%d)",
211  (unsigned int)i,
212  (unsigned int)channel->channelIdx,
213  (int)status);
214  }
215 
216  channel->DumpTbl.PriorityQueue[i].OSALQueueID = OS_MAX_QUEUES;
217  }
218  }
219  }
220  }
221  else
222  {
223  /* No table. Just keep going and try processing the next channel. */
224  (void) CFE_EVS_SendEvent(TO_PQUEUE_MISSING_TBL_ERR_EID,
225  CFE_EVS_ERROR,
226  "Failed to tear down priority queues on channel %d, missing table.",
227  (unsigned int)channel->channelIdx);
228  }
229 
230  return status;
231 }

234 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
235 /* */
236 /* Push a message on a priority queue. */
237 /* */
238 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
239 int32 TO_PriorityQueue_QueueMsg(
240  TO_ChannelData_t *channel,
241  CFE_SB_MsgPtr_t MsgPtr,
242  uint32 PQueueIndex)
243 {
244  int32 status = 0;
245  uint32 bufferSize = 0;
246  TO_PriorityQueueMetrics_t* pqueueDump = NULL;
247  void *TO_CopyBuffer = NULL;
248  CFE_SB_MsgId_t DataMsgID;
249 
250  if (NULL == channel->ConfigTblPtr)
251  {
252  (void) CFE_EVS_SendEvent(TO_PQUEUE_MISSING_TBL_ERR_EID, CFE_EVS_ERROR,
253  "Failed to queue message on channel %d, missing table!",
254  (unsigned int)channel->channelIdx);
255  return TO_NO_TABLE_ERR;
256  }
257 
258  pqueueDump = &channel->DumpTbl.PriorityQueue[PQueueIndex];
259 
260  DataMsgID = CFE_SB_GetMsgId(MsgPtr);
261  bufferSize = CFE_SB_GetTotalMsgLength(MsgPtr);
262 
263  /* We're going to be modifying the event message.
264  * Send minimum-sized event messages.
265  * Trim message before memory pool get.
266  */
267  if (CFE_EVS_EVENT_MSG_MID == DataMsgID)
268  {
269  bufferSize = TO_GetEventMsgLength((CFE_EVS_Packet_t *)MsgPtr);
270  }
271 
272  /* Allocate a chunk of memory from the memory pool to store the message
273  * copy.
274  */
275  status = CFE_ES_GetPoolBuf((uint32 **) &TO_CopyBuffer,
276  channel->MemPoolHandle, bufferSize);
277  if ((status < 0) || (TO_CopyBuffer == NULL))
278  {
279  /* The allocation failed. There's nothing we can do. Rather than
280  * treat this as a critical failure, we're going to keep going and try
281  * processing other messages. We also aren't going to send an event.
282  * Sending an event may make this problem worse since we're
283  * potentially adding more messages to an already full buffer. So,
284  * let's just increment a counter, and let the caller know the queue
285  * failed.
286  */
287  TO_Channel_LockByRef(channel);
288  channel->MemFullCount++;
289  pqueueDump->DroppedMsgCnt++;
290  TO_Channel_UnlockByRef(channel);
291  return TO_MEMORY_FULL_ERR;
292  }
293 
294  TO_Channel_LockByRef(channel);
295  /* Since status is positive, it is safe to cast */
296  channel->MemInUse += (uint32)status;
297  if (channel->MemInUse > channel->PeakMemInUse)
298  {
299  channel->PeakMemInUse = channel->MemInUse;
300  }
301  TO_Channel_UnlockByRef(channel);
302 
303  /* Copy the message into the temp buffer. */
304  (void) CFE_PSP_MemCpy(TO_CopyBuffer, MsgPtr, bufferSize);
305 
306  /* If the message is an event, set the corrected message size. */
307  if (CFE_EVS_EVENT_MSG_MID == DataMsgID)
308  {
309  CFE_SB_SetTotalMsgLength((CFE_SB_MsgPtr_t)TO_CopyBuffer, bufferSize);
310  }
311 
312  /* Queue the pointer to the message copy. */
313  status = OS_QueuePut(pqueueDump->OSALQueueID, &TO_CopyBuffer, sizeof(TO_CopyBuffer), 0);
314  if (OS_QUEUE_FULL == status)
315  {
316  /* This is ok. Just let the caller know the queue is full. But,
317  * deallocate the memory allocated first since we don't need it
318  * anymore.
319  */
320  status = CFE_ES_PutPoolBuf(channel->MemPoolHandle, (uint32*)TO_CopyBuffer);
321  if (status < 0)
322  {
323  pqueueDump->DroppedMsgCnt++;
324  (void) CFE_EVS_SendEvent(TO_PUT_POOL_ERR_EID,
325  CFE_EVS_ERROR,
326  "PutPoolBuf: channel=%d error=%i",
327  (unsigned int)channel->channelIdx,
328  (int)status);
329  return status;
330  }
331  TO_Channel_LockByRef(channel);
332  channel->MemInUse -= (uint32)status;
333 
334  TO_Channel_UnlockByRef(channel);
335  status = TO_PRIORITY_QUEUE_FULL_ERR;
336  }
337  else if (status != OS_SUCCESS)
338  {
339  pqueueDump->DroppedMsgCnt++;
340  (void) CFE_EVS_SendEvent(TO_OSQUEUE_PUT_ERROR_EID,
341  CFE_EVS_ERROR,
342  "OS_QueuePut failed: channel=%d size=%u error=%i",
343  (unsigned int)channel->channelIdx,
344  sizeof(TO_CopyBuffer),
345  (int)status);
346  return status;
347  }
348 
349  /* Update metrics. */
350  pqueueDump->DroppedMsgCnt++;
351  pqueueDump->QueuedMsgCnt++;
352  pqueueDump->CurrentlyQueuedCnt++;
353 
354  if (pqueueDump->HighwaterMark < pqueueDump->CurrentlyQueuedCnt)
355  {
356  pqueueDump->HighwaterMark = pqueueDump->CurrentlyQueuedCnt;
357  }
358 
359  return status;
360 }

366 /* Query a priority queue. */

368 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
369 osalbool TO_PriorityQueue_Query(uint16 ChannelIdx, uint16 PQueueIdx)
370 {
371  osalbool returnCode = FALSE;
372  TO_ChannelData_t *channel = NULL;
373 
374  /* First, check if the channel index is valid. */
375  if (ChannelIdx >= TO_MAX_CHANNELS)
376  {
377  (void) CFE_EVS_SendEvent(TO_PQUEUE_INFO_ERR_EID,
378  CFE_EVS_ERROR,
379  "Invalid channel index (index = %d, max = %d).",
380  ChannelIdx, TO_MAX_CHANNELS);
381  return returnCode;
382  }
383 
384  channel = &TO_AppData.ChannelData[ChannelIdx];
385 
386  if (NULL == channel->ConfigTblPtr)
387  {
388  (void) CFE_EVS_SendEvent(TO_PQUEUE_MISSING_TBL_ERR_EID,
389  CFE_EVS_ERROR,
390  "Channel missing valid table data (index = %d).",
391  ChannelIdx);
392  return returnCode;
393  }
394 
395  TO_Channel_LockByRef(channel);
396 
397  /* Next, see if the channel is open. */
398  if (channel->State != TO_CHANNEL_OPENED)
399  {
400  (void) CFE_EVS_SendEvent(TO_PQUEUE_INFO_ERR_EID,
401  CFE_EVS_ERROR,
402  "Channel %d not open.", ChannelIdx);
403  TO_Channel_UnlockByRef(channel);
404  return returnCode;
405  }
406 
407  if (PQueueIdx >= TO_MAX_PRIORITY_QUEUES)
408  {
409  (void) CFE_EVS_SendEvent(TO_PQUEUE_INFO_ERR_EID,
410  CFE_EVS_ERROR,
411  "PQueueIdx %u exceeds the largest available priority queue index, %d",
412  PQueueIdx,
413  TO_MAX_PRIORITY_QUEUES - 1);
414  }
415  else
416  {
417  (void) CFE_EVS_SendEvent(TO_PQUEUE_INFO_EID,
418  CFE_EVS_INFORMATION,
419  "CHANNEL=%d PQI=%u S=%u ML=%u QT=%u D=%lu Q=%lu CQ=%u HWM=%u",
420  ChannelIdx,
421  PQueueIdx,
422  channel->ConfigTblPtr->PriorityQueue[PQueueIdx].State,
423  channel->ConfigTblPtr->PriorityQueue[PQueueIdx].MsgLimit,
424  channel->ConfigTblPtr->PriorityQueue[PQueueIdx].QType,
425  channel->DumpTbl.PriorityQueue[PQueueIdx].DroppedMsgCnt,
426  channel->DumpTbl.PriorityQueue[PQueueIdx].QueuedMsgCnt,
427  channel->DumpTbl.PriorityQueue[PQueueIdx].CurrentlyQueuedCnt,
428  channel->DumpTbl.PriorityQueue[PQueueIdx].HighwaterMark);
429 
430  returnCode = TRUE;
431  }
432 
433  TO_Channel_UnlockByRef(channel);
434 
435  return returnCode;
436 }
