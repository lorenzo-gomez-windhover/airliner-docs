79 /* Initialize Config Table */
80 /* */
81 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
82 int32 TO_InitTables(TO_ChannelData_t *channel)
83 {
84  int32 status = CFE_SUCCESS;
85  int result;
86 
87  /* Register Config tables */
88  status = CFE_TBL_Register(&channel->ConfigTblHdl,
89  channel->ConfigTableName,
90  sizeof(TO_ChannelTbl_t),
91  CFE_TBL_OPT_DEFAULT,
92  TO_ValidateConfigTbl);
93 
94  if (status != CFE_SUCCESS)
95  {
96  /* Note, a critical table could return another nominal code. If this table is
97  * made critical this logic would have to change. */
98  (void) CFE_EVS_SendEvent(TO_INIT_CONFIG_ERR_EID, CFE_EVS_ERROR,
99  "Failed to register Config table (0x%08X) for channel %d",
100  (unsigned int)status,
101  (unsigned int)channel->channelIdx);
102  return status;
103  }
104 
105  /* Register Dump table */
106  status = CFE_TBL_Register(&channel->DumpTblHdl,
107  channel->DumpTableName,
108  sizeof(TO_ChannelDumpTbl_t),
109  CFE_TBL_OPT_USR_DEF_ADDR,
110  0);
111  if (status != CFE_SUCCESS)
112  {
113  /* Note, a critical table could return another nominal code. If this table is
114  * made critical this logic would have to change. */
115  (void) CFE_EVS_SendEvent(TO_INIT_CONFIG_ERR_EID,
116  CFE_EVS_ERROR,
117  "Failed to register Dump table (0x%08X) for channel %d",
118  (unsigned int)status,
119  (unsigned int)channel->channelIdx);
120  return status;
121  }
122 
123  /* Load Config table file */
124  status = CFE_TBL_Load(channel->ConfigTblHdl,
125  CFE_TBL_SRC_FILE,
126  channel->ConfigTableFileName);
127  if (status != CFE_SUCCESS)
128  {
129  /* Note, CFE_SUCCESS is for a successful full table load. If a partial table
130  load is desired then this logic would have to change. */
131  (void) CFE_EVS_SendEvent(TO_INIT_CONFIG_ERR_EID,
132  CFE_EVS_ERROR,
133  "Failed to load Config Table (0x%08X) for channel %d",
134  (unsigned int)status,
135  (unsigned int)channel->channelIdx);
136 
137  /* Load emergency backup table */
138  TO_LoadBackupConfigTable(channel);
139  }
140 
141  /* Load Dump table */
142  status = CFE_TBL_Load(channel->DumpTblHdl,
143  CFE_TBL_SRC_ADDRESS,
144  &channel->DumpTbl);
145  if (status != CFE_SUCCESS)
146  {
147  /* Note, CFE_SUCCESS is for a successful full table load. If a partial table
148  load is desired then this logic would have to change. */
149  (void) CFE_EVS_SendEvent(TO_INIT_CONFIG_ERR_EID,
150  CFE_EVS_ERROR,
151  "Failed to load Dump Table (0x%08X) for channel %d",
152  (unsigned int)status,
153  (unsigned int)channel->channelIdx);
154  return status;
155  }
156 
157  /* Force a manage on this channel's tables since this is the first time */
158  TO_ManageChannelTables(TRUE, channel->channelIdx);
159 
160  return (status);
161 }

196 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
197 /* */
198 /* Validate Config Table */
199 /* */
200 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
201 int32 TO_ValidateConfigTbl(void *configTblPtr)
202 {
203  int32 status = CFE_SUCCESS;
204  TO_ChannelTbl_t *TO_ConfigTblPtr = NULL;
205 
206  /* For validating PriorityQueues */
207  uint16 PQueueIdx;
208  uint16 unused_PQ_Count = 0;
209  uint16 valid_PQ_Count = 0;
210 
211  /* For validating MessageFlows/Ground Queues */
212  uint16 MFlowIdx;
213  uint16 unused_MF_Count = 0;
214  CCSDS_PriHdr_t CCSDSPriHdr;
215 
216  uint8 queueIdx = 0;
217 
218  /* Check if configTblPtr is NULL */
219  if (configTblPtr == NULL)
220  {
221  (void) CFE_EVS_SendEvent(TO_CONFIG_TABLE_NULL_PTR_EID,
222  CFE_EVS_ERROR,
223  "Config table pointer is NULL, table validation ended");
224 
225  return TO_CONFIG_TABLE_NULL_PTR_ERR;
226  }
227 
228  /* Pointer to the table data being verified */
229  TO_ConfigTblPtr = (TO_ChannelTbl_t *)configTblPtr;
230 
231  /* Check the Priority Queue section of the table */
232  for (PQueueIdx = 0; PQueueIdx < TO_MAX_PRIORITY_QUEUES; PQueueIdx++)
233  {
234  /* Check the Priority Queue State for each table entry */
235  if (TO_ConfigTblPtr->PriorityQueue[PQueueIdx].State != TO_PQUEUE_UNUSED &&
236  TO_ConfigTblPtr->PriorityQueue[PQueueIdx].State != TO_PQUEUE_ENA)
237  {
238  (void) CFE_EVS_SendEvent(TO_CONFIG_TABLE_PQUEUE_STATE_ERR_EID,
239  CFE_EVS_ERROR,
240  "Priority Queue State is invalid (%d), for Table ID (%lu) at table index (%d)",
241  TO_ConfigTblPtr->PriorityQueue[PQueueIdx].State,
242  TO_ConfigTblPtr->TableID,
243  PQueueIdx);
244 
245  return TO_CONFIG_TABLE_PQUEUE_STATE_ERR;
246  }
247  else if (TO_ConfigTblPtr->PriorityQueue[PQueueIdx].State == TO_PQUEUE_UNUSED)
248  {
249  /* Found unused priority queue. Increment counter */
250  unused_PQ_Count++;
251  }
252  else
253  {
254  /* Check individual Priority Queue's Qtype and MsgLimit for error.
255  * If the Priority Queue type is incorrect or MsgLimit out of range,
256  * the entire table is invalid.
257  *
258  * Check Priority Queue QType
259  */
260  if ((TO_ConfigTblPtr->PriorityQueue[PQueueIdx].QType != TO_PRIORITY_QUEUE_TYPE_SINGLE) &&
261  (TO_ConfigTblPtr->PriorityQueue[PQueueIdx].QType != TO_PRIORITY_QUEUE_TYPE_FIFO))
262 
263  {
264  (void) CFE_EVS_SendEvent(TO_CONFIG_TABLE_PQUEUE_QTYPE_ERR_EID,
265  CFE_EVS_ERROR,
266  "Priority Queue Qtype is invalid (%d) for Table ID (%lu) at table index (%d)",
267  TO_ConfigTblPtr->PriorityQueue[PQueueIdx].QType,
268  TO_ConfigTblPtr->TableID,
269  PQueueIdx);
270 
271  return TO_CONFIG_TABLE_PQUEUE_QTYPE_ERR;
272  }
273  /* Check Priority Queue MsgLimit */
274  if (TO_ConfigTblPtr->PriorityQueue[PQueueIdx].MsgLimit > TO_MAX_QUEUE_SIZE_LIMIT ||
275  TO_ConfigTblPtr->PriorityQueue[PQueueIdx].MsgLimit < 1)
276  {
277  (void) CFE_EVS_SendEvent(TO_CONFIG_TABLE_PQUEUE_MSG_LIMIT_ERR_EID,
278  CFE_EVS_ERROR,
279  "Priority Queue MsgLimit invalid (%d) for Table ID (%lu) at table index (%d)",
280  TO_ConfigTblPtr->PriorityQueue[PQueueIdx].MsgLimit,
281  TO_ConfigTblPtr->TableID,
282  PQueueIdx);
283 
284  return TO_CONFIG_TABLE_PQUEUE_MSG_LIMIT_ERR;
285  }
286 
287  valid_PQ_Count++;
288  }
289  }
290 
291  /* If all priority queues are unused or there's nothing in the table, the entire table is invalid.
292  * No need to check Message Flows. Stop table validation.
293  */
294  if (TO_MAX_PRIORITY_QUEUES == unused_PQ_Count)
295  {
296  (void) CFE_EVS_SendEvent(TO_CONFIG_TABLE_NO_PQUEUES_ERR_EID,
297  CFE_EVS_ERROR,
298  "No Priority Queues or all queue states are unused in the config table with Table ID (%lu)",
299  TO_ConfigTblPtr->TableID);
300 
301  return TO_CONFIG_TABLE_NO_PQUEUES_ERR;
302  }
303 
304  /* No issues found with Priority Queues. Check the Message Flows */
305  for (MFlowIdx = 0; MFlowIdx < TO_MAX_MESSAGE_FLOWS; MFlowIdx++)
306  {
307  if ((TO_ConfigTblPtr->MessageFlow[MFlowIdx].MsgId == 0x0000) &&
308  (TO_ConfigTblPtr->MessageFlow[MFlowIdx].MsgLimit == 0) &&
309  (TO_ConfigTblPtr->MessageFlow[MFlowIdx].PQueueID == 0))
310  {
311  /* Found unused queue increment count */
312  unused_MF_Count++;
313  }
314  else
315  {
316  /*
317  * Check each MsgId's the primary header for the correct CCSDS version,
318  * for the presence of the secondary header, and for packet completeness.
319  */
320 
321  /* Clear the primary header. */
322  CCSDS_CLR_PRI_HDR(CCSDSPriHdr);
323 
324  /* Set the stream ID */
325  CCSDSPriHdr.StreamId[0] = (uint8) (TO_ConfigTblPtr->MessageFlow[MFlowIdx].MsgId >> 8);
326  CCSDSPriHdr.StreamId[1] = (uint8) (TO_ConfigTblPtr->MessageFlow[MFlowIdx].MsgId & 0xff);
327 
328  /* Validate secondary header present */
329  if (CCSDS_RD_SHDR(CCSDSPriHdr) == CCSDS_NO_SEC_HDR)
330  {
331  (void) CFE_EVS_SendEvent(TO_CONFIG_TABLE_SHDR_ABSENT_EID,
332  CFE_EVS_ERROR,
333  "Secondary Header is absent for table index (%d), MsgId (0x%04X) in Table ID (%lu)",
334  MFlowIdx,
335  TO_ConfigTblPtr->MessageFlow[MFlowIdx].MsgId,
336  TO_ConfigTblPtr->TableID);
337 
338  return TO_CONFIG_TABLE_RETURN_NO_SECONDARY_HEADER;
339  }
340 
341  /* Validate CCSDS version */
342  if (CCSDS_RD_VERS(CCSDSPriHdr) != 0)
343  {
344  (void) CFE_EVS_SendEvent(TO_CONFIG_TABLE_CCSDS_VER_INVALID_EID,
345  CFE_EVS_ERROR,
346  "Invalid CCSDS Version for table index (%d), MsgId (0x%04X) in Table ID (%lu)",
347  MFlowIdx,
348  TO_ConfigTblPtr->MessageFlow[MFlowIdx].MsgId,
349  TO_ConfigTblPtr->TableID);
350 
351  return TO_CONFIG_TABLE_RETURN_INVALID_VERSION;
352  }
353 
354  /* Validate Message Flow message limit is in the proper range */
355  if (TO_ConfigTblPtr->MessageFlow[MFlowIdx].MsgLimit > TO_MAX_PQ_MSG_SIZE_LIMIT ||
356  TO_ConfigTblPtr->MessageFlow[MFlowIdx].MsgLimit < 1)
357  {
358  (void) CFE_EVS_SendEvent(TO_CONFIG_TABLE_MSG_FLOW_MSG_LIMIT_ERR_EID,
359  CFE_EVS_ERROR,
360  "Message Flow MsgLimit invalid (%u) in Table ID (%lu) at table index (%d) ",
361  (unsigned int)TO_ConfigTblPtr->MessageFlow[MFlowIdx].MsgLimit,
362  TO_ConfigTblPtr->TableID,
363  MFlowIdx);
364 
365  return TO_CONFIG_TABLE_MSG_FLOW_MSG_LIMIT_ERR;
366  }
367 
368  /* Validate that the Message Flow Priority Queue ID is within range */
369  if (TO_ConfigTblPtr->MessageFlow[MFlowIdx].PQueueID >= valid_PQ_Count)
370  {
371  (void) CFE_EVS_SendEvent(TO_CONFIG_TABLE_MSG_FLOW_PQ_ID_ERR_EID,
372  CFE_EVS_ERROR,
373  "Message Flow Priority Queue ID is invalid (%d) for Table ID (%lu) at table index (%d)",
374  TO_ConfigTblPtr->MessageFlow[MFlowIdx].PQueueID,
375  TO_ConfigTblPtr->TableID,
376  MFlowIdx);
377 
378  return TO_CONFIG_TABLE_MSG_FLOW_PQ_ID_ERR;
379 
380  }
381  }
382  }
383 
384  /* Check to see if no message flows found.
385  * Finding no message flows does not indicate a problem,
386  * but send an event to alert the user that there are no
387  * Message Flows.
388  */
389  if (TO_MAX_MESSAGE_FLOWS == unused_MF_Count)
390  {
391  (void) CFE_EVS_SendEvent(TO_CONFIG_TABLE_NO_MSG_FLOW_INF_EID,
392  CFE_EVS_INFORMATION,
393  "INFO: No Message Flow entries in the config table, Table Id (%lu)",
394  TO_ConfigTblPtr->TableID);
395  }
396 
397  return (status);
398 }

442 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
443 /* */
444 /* Manage all tables */
445 /* */
446 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
447 void TO_ManageChannelTables(osalbool initialManage, uint16 ChannelID)
448 {
449  int32 status = CFE_SUCCESS;
450 
451  osalbool dequeueStatus = TRUE;
452 
453  /* T=call CFE_TBL_Manage() on the config table, since CFE_TBL_GetStatus() might indicate
454  * there is no need. TRUE would only be expected when a table is initialized */
455  osalbool tableNeedsManaged;
456 
457  /* Override if this is the first time through to force a config table update. */
458  tableNeedsManaged = initialManage;
459 
460  TO_Channel_LockByIndex(ChannelID);
461 
462  if (TO_AppData.ChannelData[ChannelID].State != TO_CHANNEL_OPENED) {
463  TO_Channel_UnlockByIndex(ChannelID);
464  return;
465  }
466 
467  /* Always manage the dump table for this channel */
468  (void) CFE_TBL_Manage(TO_AppData.ChannelData[ChannelID].DumpTblHdl);
469 
470  status = CFE_TBL_GetStatus(TO_AppData.ChannelData[ChannelID].ConfigTblHdl);
471  if ((status & CFE_SEVERITY_BITMASK) == CFE_SEVERITY_ERROR)
472  {
473  (void) CFE_EVS_SendEvent(TO_CONFIG_TABLE_ERR_EID,
474  CFE_EVS_ERROR,
475  "Channel %u config table handle failure! (0x%08X)",
476  ChannelID,
477  (unsigned int)status);
478 
479  /* Null the table pointer, because we don't have confidence in the handle
480  * and the pointer might be to an invalid table address.
481  * Given this error the pointer MIGHT be set by CFE_TBL_GetAddress(), below,
482  * but perhaps not. */
483  TO_AppData.ChannelData[ChannelID].ConfigTblPtr = 0;
484  CFE_TBL_ReleaseAddress(TO_AppData.ChannelData[ChannelID].ConfigTblHdl);
485 
486  /* let's attempt Manage() below, in case we can get a valid table back */
487  tableNeedsManaged = TRUE;
488  }
489  else if (CFE_TBL_INFO_UPDATE_PENDING == status)
490  {
491  /* Teardown message flows and PQueues*/
492  (void) TO_MessageFlow_TeardownAll(&TO_AppData.ChannelData[ChannelID]);
493  (void) TO_PriorityQueue_TeardownAll(&TO_AppData.ChannelData[ChannelID]);
494 
495  /* Clear remaining messages from SB data pipe*/
496  dequeueStatus = TO_Channel_SBPipe_Dequeue_All(ChannelID);
497  if (FALSE == dequeueStatus)
498  {
499  (void) CFE_EVS_SendEvent(TO_CONFIG_TABLE_ERR_EID,
500  CFE_EVS_ERROR,
501  "Failed to clear SB data pipe for channel %u, (0x%08X)",
502  ChannelID,
503  (unsigned int)status);
504  }
505 
506  /* Now null the table pointer, because it is going to be updated. Also
507  * so that if something goes wrong it won't be set to an invalid address.
508  * It will be set by CFE_TBL_GetAddress(), below. */
509  TO_AppData.ChannelData[ChannelID].ConfigTblPtr = 0;
510  CFE_TBL_ReleaseAddress(TO_AppData.ChannelData[ChannelID].ConfigTblHdl);
511  tableNeedsManaged = TRUE;
512  }
513  else if (CFE_TBL_INFO_VALIDATION_PENDING == status)
514  {
515  tableNeedsManaged = TRUE;
516  }
517 
518  if (tableNeedsManaged)
519  {
520  /* Manage the config table */
521  status = CFE_TBL_Manage(TO_AppData.ChannelData[ChannelID].ConfigTblHdl);
522  if ((status != CFE_SUCCESS) && (status != CFE_TBL_INFO_UPDATED))
523  {
524  (void) CFE_EVS_SendEvent(TO_CONFIG_MANAGE_ERR_EID,
525  CFE_EVS_ERROR,
526  "Failed to manage Config table for channel %u, (0x%08X)",
527  ChannelID,
528  (unsigned int)status);
529  }
530  else
531  {
532  /*
533  ** Get a pointer to the table
534  */
535  status = CFE_TBL_GetAddress(
536  (void *) &TO_AppData.ChannelData[ChannelID].ConfigTblPtr,
537  TO_AppData.ChannelData[ChannelID].ConfigTblHdl);
538  if (CFE_TBL_INFO_UPDATED == status)
539  {
540  status = TO_ProcessNewConfigTbl(&TO_AppData.ChannelData[ChannelID]);
541  if (status != CFE_SUCCESS)
542  {
543  (void) CFE_EVS_SendEvent(TO_CONFIG_PROCESS_CONFIG_TBL_ERR_EID,
544  CFE_EVS_ERROR,
545  "Process New Config Table Failed for channel %u, (0x%08X)",
546  ChannelID,
547  (unsigned int)status);
548  }
549  }
550  else if (status != CFE_SUCCESS)
551  {
552  (void) CFE_EVS_SendEvent(TO_CONFIG_ADDR_ERR_EID,
553  CFE_EVS_ERROR,
554  "Failed to get Config table's address for channel %u, (0x%08X)",
555  ChannelID,
556  (unsigned int)status);
557  }
558  }
559  }
560 
561  TO_Channel_UnlockByIndex(ChannelID);
562 }
