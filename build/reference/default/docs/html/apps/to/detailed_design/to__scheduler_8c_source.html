44 void TO_Scheduler_Run(TO_ChannelData_t *channel)
45 {
46  uint32 i;
47  TO_PriorityQueue_t *pqueue = NULL;
48  TO_OutputQueue_t *oqueue = NULL;
49  int32 status = 0;
50  void *buffer = NULL;
51  uint32 nBytesCopied = 0;
52 
53  oqueue = &channel->OutputQueue;
54 
55  if (NULL == channel->ConfigTblPtr)
56  {
57  /* If there is no table data we can't access a priority queue. */
58  return;
59  }
60 
61  for (i = 0; i < TO_MAX_PRIORITY_QUEUES; ++i)
62  {
63  status = OS_SUCCESS;
64  pqueue = &channel->ConfigTblPtr->PriorityQueue[i];
65 
66  if (pqueue->State != TO_PQUEUE_UNUSED)
67  {
68  if (channel->DumpTbl.PriorityQueue[i].OSALQueueID != OS_MAX_QUEUES)
69  {
70  while ((OS_SUCCESS == status) && (oqueue->CurrentlyQueuedCnt < TO_OUTPUT_QUEUE_DEPTH))
71  {
72  status = OS_QueueGet(
73  channel->DumpTbl.PriorityQueue[i].OSALQueueID,
74  &buffer, sizeof(buffer), &nBytesCopied, OS_CHECK);
75 
76  if (OS_SUCCESS == status)
77  {
78  channel->DumpTbl.PriorityQueue[i].CurrentlyQueuedCnt--;
79  status = TO_OutputQueue_QueueMsg(channel, buffer);
80  if (CFE_SUCCESS == status)
81  {
82  oqueue->CurrentlyQueuedCnt++;
83  if (oqueue->HighwaterMark < oqueue->CurrentlyQueuedCnt)
84  {
85  oqueue->HighwaterMark = oqueue->CurrentlyQueuedCnt;
86  }
87  }
88  }
89  }
90  }
91  }
92  }
93 }
